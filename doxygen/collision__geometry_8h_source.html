48 [[nodiscard]] bool isApprox(const TaperedCapsuleT& other, const S& tol = Eps<S>(1e-4f, 1e-10))
49 const {
50 if (!transformation.isApprox(other.transformation, tol)) {
51 return false;

53

55 return false;
56 }
57

59 return false;
60 }
61
62 if (!::momentum::isApprox(length, other.length)) {
63 return false;
64 }
65
66 return true;
67 }
