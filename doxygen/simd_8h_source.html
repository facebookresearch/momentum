92template <size_t Alignment>
93[[nodiscard]] size_t computeOffset(const Eigen::Ref<Eigen::MatrixXf>& mat) {
94 constexpr size_t sizeOfScalar = sizeof(typename Eigen::MatrixXf::Scalar);
95 const size_t addressOffset =
96 Alignment / sizeOfScalar - (((size_t)mat.data() % Alignment) / sizeOfScalar);
97
98 // If the current alignment already meets the requirement, no offset is needed.
99 if (addressOffset == Alignment / sizeOfScalar) {
100 return 0;
101 }
102
103 return addressOffset;
104}

107template <size_t Alignment>
108void checkAlignment(const Eigen::Ref<Eigen::MatrixXf>& mat, size_t offset = 0) {
109 MT_THROW_IF(
110 (uintptr_t(mat.data() + offset)) % Alignment != 0,
111 "Matrix ({}x{}, ptr: {}) is not aligned ({}) correctly.",
112 mat.rows(),
113 mat.cols(),
114 uintptr_t(mat.data()),
115 Alignment);
116}
