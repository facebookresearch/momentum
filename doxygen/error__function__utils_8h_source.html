17template <typename T>
18void gradient_jointParams_to_modelParams(
19 const T& grad_jointParams,
20 const Eigen::Index iJointParam,
21 const ParameterTransform& parameterTransform,
22 Eigen::Ref<Eigen::VectorX<T>> gradient) {
23 // explicitly multiply with the parameter transform to generate parameter space gradients
24 for (auto index = parameterTransform.transform.outerIndexPtr()[iJointParam];
25 index < parameterTransform.transform.outerIndexPtr()[iJointParam + 1];
26 ++index) {
27 gradient[parameterTransform.transform.innerIndexPtr()[index]] +=
28 grad_jointParams * parameterTransform.transform.valuePtr()[index];
29 }
30}

101template <typename T, typename SinkFn>
102void accumulateChainDerivatives(
103 const SkeletonStateT<T>& state,
104 const Skeleton& skeleton,
105 const VectorX<bool>& activeJointParams,
106 const Vector3<T>& position,
107 const Vector3<T>& direction,
108 T weight,
109 size_t startJoint,
110 size_t stopJoint,
111 T scaleCorrection,
112 const SinkFn& sink) {
113 size_t jointIndex = startJoint;
114 while (jointIndex != kInvalidIndex && jointIndex != stopJoint) {
115 const auto& jointState = state.jointState[jointIndex];
116 const size_t paramIndex = jointIndex * kParametersPerJoint;
117 const Vector3<T> posd = position - jointState.translation();
118
119 for (size_t d = 0; d < 3; ++d) {
120 if (activeJointParams[paramIndex + d]) {
121 sink(direction.dot(jointState.getTranslationDerivative(d)) * weight, paramIndex + d);
122 }
123 if (activeJointParams[paramIndex + 3 + d]) {
124 sink(direction.dot(jointState.getRotationDerivative(d, posd)) * weight, paramIndex + 3 + d);
125 }
126 }
127 if (activeJointParams[paramIndex + 6]) {
128 sink(
129 (direction.dot(jointState.getScaleDerivative(posd)) + scaleCorrection) * weight,
130 paramIndex + 6);
131 }
132
133 jointIndex = skeleton.joints[jointIndex].parent;
134 }
135}
