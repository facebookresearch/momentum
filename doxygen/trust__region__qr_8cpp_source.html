49template <typename T>
50void TrustRegionQRT<T>::doIteration() {
51 MT_PROFILE_FUNCTION();
52
53 const auto sf = static_cast<SkeletonSolverFunctionT<T>*>(this->solverFunction_);
54
55 const auto skeleton = sf->getSkeleton();
56 const auto parameterTransform = sf->getParameterTransform();
57
58 {
59 MT_PROFILE_EVENT("JtJR - update state");
60 skeletonState_.set(parameterTransform->apply(this->parameters_), *skeleton);
61 }
62
63 const auto nFullParams = gsl::narrow_cast<Eigen::Index>(this->parameters_.size());
64
65 std::vector<Eigen::Index> enabledParameters;
66 enabledParameters.reserve(nFullParams);
67 for (Eigen::Index iFullParam = 0; iFullParam < nFullParams; ++iFullParam) {
68 if (this->activeParameters_.test(iFullParam)) {
69 enabledParameters.push_back(iFullParam);
70 }
71 }
72
73 const Eigen::Index nSubsetParams = gsl::narrow_cast<Eigen::Index>(enabledParameters.size());
74
75 // Add a tiny lambda just to make sure we don't divide by zero when back-substituting into R,
76 // this isn't intended to regularize the problem at all (searching for the actual lambda parameter
77 // that respects the trust region radius happens below).
78 T lambda = 1e-10;
79 this->qrSolver_.reset(nSubsetParams, lambda);
80
81 // Compute the QR factorization of the Jacobian:
82 double error_orig = 0.0;
83 for (auto errorFunction : sf->getErrorFunctions()) {
84 if (errorFunction->getWeight() <= 0) {
85 continue;
86 }
87
88 auto rows = errorFunction->getJacobianSize();
89
90 jacobian_.resizeAndSetZero(rows, nFullParams);
91 residual_.resizeAndSetZero(rows);
92
93 int usedRows = 0;
94 error_orig += errorFunction->getJacobian(
95 this->parameters_, skeletonState_, jacobian_.mat(), residual_.mat(), usedRows);
96 if (usedRows == 0) {
97 continue;
98 }
99
100 this->qrSolver_.addMutating(
101 ColumnIndexedMatrix<Eigen::MatrixX<T>>(
102 jacobian_.mat().topRows(usedRows), enabledParameters),
103 residual_.mat().topRows(usedRows));
104 }
105
106 this->error_ = error_orig;
107
108 // Because the error function is computed r^t*r, the gradient is
109 // 2*J^T*r.
110 this->gradientSub_ = T(2) * qrSolver_.At_times_b();
111
112 // Save the R matrix before we start adding lambdas:
113 this->Rmatrix_ = this->qrSolver_.R();
114
115 // The Hessian is approximated by 2*J^T*J.
116 // In the QR solver, we compute J = Q*R.
117 // 2*J^T*J = (Q*R)^T * (Q*R)
118 // = R^T * Q^T * Q * R
119 // = 2 * R^T * R
120 // To compute v^T * H * v, we can then compute
121 // 2 * ||R * v||^2
122
123 // This is the local approximation to the function used in the trust region solve:
124 auto evalQuadraticModel = [&](Eigen::Ref<const Eigen::VectorX<T>> p) -> T {
125 // MT_LOGI("In evalQuadraticModel(); orig error: {}", this->error_);
126 T result = this->error_;
127 result -= this->gradientSub_.dot(p);
128 // Normally in textbooks you'd see this multiplied by 1/2 for the proper Taylor series
129 // expansion but remember that our Hessian is actually 2*J^T*J:
130 result += (this->Rmatrix_.template triangularView<Eigen::Upper>() * p).squaredNorm();
131 return result;
132 };
133
134 auto subsetToFullVector = [&](Eigen::Ref<const Eigen::VectorX<T>> v_subset) -> Eigen::VectorX<T> {
135 Eigen::VectorX<T> result = Eigen::VectorX<T>::Zero(nFullParams);
136 for (Eigen::Index iSubsetParam = 0; iSubsetParam < nSubsetParams; ++iSubsetParam) {
137 const Eigen::Index iFullParam = enabledParameters[iSubsetParam];
138 result[iFullParam] = v_subset[iSubsetParam];
139 }
140 return result;
141 };
142
143 // TODO what is a good value for nu?
144 const T nu = 0;
145
146 for (size_t iTrustStep = 0; iTrustStep < 10; ++iTrustStep) {
147 Eigen::VectorX<T> searchDir_sub = this->qrSolver_.result();
148
149 // If even our most optimistic guess of how much we can decrease the error
150 // suggests that it's not going to change by a measurable amount (FLT_EPS times its current
151 // value), then stop trying to iterate. If we don't do that we run into issues below where the
152 // rho value is just capturing noise and our trust region will drop to 0.
153 if (searchDir_sub.dot(gradientSub_) < FLT_EPSILON * (T(1.0) + this->error_)) {
154 break;
155 }
156
157 // We use a standard Newton approach for quickly converging to the lambda that limits
158 // the step size to the trust region. This is described in section 4.3 of Nocedal/Wright,
159 // "Iterative solution of the subproblem."
160 //
161 // We will only allow the value of lambda to _increase_, which corresponds to shrinking the
162 // trust region. That is, we start with a trust region that is too large, and then we
163 // iteratively shrink it until it's close enough to the desired delta. This is because (1) in
164 // practice, this is what seems to happen anyways, given that we start from 0 and (2) it's a lot
165 // easier to add to lambda than subtract from lambda; the former can always be done by just
166 // adding some additional rows to the Jacobian while the latter requires "backing out" to a
167 // previous version of the QR solver. This is okay because it's always more acceptable (within
168 // reason) to take a too-small step than a too-large one:
169 for (size_t iIter = 0; iIter < 3; ++iIter) {
170 if (searchDir_sub.norm() < T(1.05) * this->curTrustRegionRadius_) {
171 break;
172 }
173
174 const Eigen::MatrixX<T>& Rmat = this->qrSolver_.R();
175 // This is equation 4.44 in Nocedal/Wright. Note that our error function is actually
176 // r^T*r which means gradientSub_ is 2*J^T*r, but the Hessian approximated inside the QR
177 // solver is J^T*J. Since we're applying the lambda within the QR solver also, we'll stay
178 // consistent with the J^T*J/J^T*r version of the world and multiply our gradientSub_ by 0.5
179 // on its way in.
180 Eigen::VectorX<T> p_l = Rmat.template triangularView<Eigen::Upper>().solve(
181 Rmat.template triangularView<Eigen::Upper>().transpose().solve(-T(0.5) * gradientSub_));
182 Eigen::VectorX<T> q_l = Rmat.template triangularView<Eigen::Upper>().transpose().solve(p_l);
183
184 const T p_l_norm2 = p_l.squaredNorm();
185 const T q_l_norm2 = q_l.squaredNorm();
186
187 if (q_l_norm2 < FLT_EPSILON) {
188 break;
189 }
190
191 const T p_l_norm = std::sqrt(p_l_norm2);
192 const T deltaLambda = (p_l_norm2 / q_l_norm2) *
193 ((p_l_norm - this->curTrustRegionRadius_) / (this->curTrustRegionRadius_));
194
195 // We can only easily _increase_ the lambda value, so stop if the change is negative:
196 if (deltaLambda <= 0) {
197 break;
198 }
199
200 {
201 // We want to find a y such that:
202 // lambda + y^2 = (lambda + deltaLambda)
203 // y = sqrt((lambda + deltaLambda) - lambda)
204 const T lambdaNew = lambda + deltaLambda;
205 const T y = std::sqrt(lambdaNew - lambda);
206
207 // Add the lambda terms:
208 // TODO make this faster than O(n^2):
209 this->lambdaDiag_.resize(nSubsetParams, nSubsetParams);
210 this->lambdaDiag_.setZero();
211 this->lambdaDiag_.diagonal().setConstant(y);
212
213 this->lambdaZero_.resize(nSubsetParams);
214 this->lambdaZero_.setZero();
215
216 this->qrSolver_.addMutating(this->lambdaDiag_, this->lambdaZero_);
217
218 lambda = lambdaNew;
219 }
220
221 searchDir_sub = this->qrSolver_.result();
222
223 MT_LOGI_IF(
224 this->verbose_,
225 " Iter: {}; Search dir cur norm: {}; delta: {}; lambda: {}",
226 iIter,
227 searchDir_sub.norm(),
228 this->curTrustRegionRadius_,
229 lambda);
230 }
231
232 const Eigen::VectorX<T> parameters_orig = this->parameters_;
233 this->solverFunction_->updateParameters(this->parameters_, subsetToFullVector(searchDir_sub));
234 const double error_new = this->solverFunction_->getError(this->parameters_);
235
236 // This is the decrease of the function relative to what we _expected_ the function to
237 // decrease by. It is eq. 4.4 in Nocedal and Wright.
238 const T quadraticModelEval = evalQuadraticModel(searchDir_sub);
239 const T rho = (this->error_ - error_new) / (this->error_ - quadraticModelEval);
240 MT_LOGI_IF(
241 this->verbose_,
242 "Error orig: {}; error new: {}; quadratic model: {}; rho: {}",
243 this->error_,
244 error_new,
245 quadraticModelEval,
246 rho);
247
248 if (rho < T(0.25)) {
249 // Model is no good, decrease the radius:
250 this->curTrustRegionRadius_ = T(0.25) * this->curTrustRegionRadius_;
251 } else if (rho > T(0.75) && lambda > 0) {
252 // Model seems good and we're pushing up against the trust region boundary, so increase it:
253 this->curTrustRegionRadius_ =
254 std::min(T(2) * this->curTrustRegionRadius_, this->maxTrustRegionRadius_);
255 }
256
257 if (rho > nu) {
258 break;
259 } else {
260 // Reject the step, reduce the trust region size and try again:
261 this->parameters_ = parameters_orig;
262 }
263 }
264}
