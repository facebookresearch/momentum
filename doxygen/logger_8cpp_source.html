48void logMesh(
49 const rerun::RecordingStream& rec,
50 const std::string& streamName,
51 const Mesh& mesh,
52 std::optional<rerun::Color> color) {
53 auto rerunMesh = rerun::Mesh3D(mesh.vertices).with_triangle_indices(mesh.faces);
54 if (color.has_value()) {
55 rerunMesh = std::move(rerunMesh).with_albedo_factor(rerun::AlbedoFactor(color.value()));
56 } else if (mesh.colors.size() == mesh.vertices.size()) {
57 rerunMesh = std::move(rerunMesh).with_vertex_colors(mesh.colors);
58 }
59
60 if (mesh.normals.size() == mesh.vertices.size()) {
61 rerunMesh = std::move(rerunMesh).with_vertex_normals(mesh.normals);
62 }
63
64 rec.log(streamName, rerunMesh);
65}

67void logJoints(
68 const rerun::RecordingStream& rec,
69 const std::string& streamName,
70 const Skeleton& skeleton,
71 const JointStateList& jointStates) {
72 const rerun::Color kGrey(180, 180, 180);
73 const auto names = skeleton.getJointNames();
74 std::vector<std::string> labels;
75 std::vector<std::vector<std::array<float, 3>>> lines;
76 labels.reserve(names.size());
77 lines.reserve(names.size());
78 for (size_t iJoint = 0; iJoint < jointStates.size(); ++iJoint) {
79 if (iJoint >= skeleton.joints.size()) {
80 break;
81 }
82 const size_t parentIdx = skeleton.joints[iJoint].parent;
83 if (parentIdx != kInvalidIndex && parentIdx < jointStates.size()) {
84 lines.push_back(
85 {toStdArray3f(jointStates[parentIdx].transform.translation),
86 toStdArray3f(jointStates[iJoint].transform.translation)});
87 labels.push_back(names[iJoint]);
88 }
89 rec.log(
90 streamName + "/" + names[iJoint],
91 rerun::Transform3D()
92 .with_mat3x3(
93 rerun::datatypes::Mat3x3(jointStates[iJoint].transform.toRotationMatrix().data()))
94 .with_translation(
95 rerun::datatypes::Vec3D(jointStates[iJoint].transform.translation.data()))
96 .with_axis_length(rerun::components::AxisLength(10)));
97 }
98 rec.log(
99 streamName,
100 rerun::LineStrips3D(lines).with_radii(0.2f).with_colors(kGrey).with_labels(labels));
101}

127void logLocators(
128 const rerun::RecordingStream& rec,
129 const std::string& streamName,
130 const LocatorList& locators,
131 const LocatorState& locatorState) {
132 if (locators.empty()) {
133 return;
134 }
135
136 const rerun::Color kGreen(100, 255, 100);
137 std::vector<std::string> labels;
138 std::vector<rerun::Position3D> points3d;
139 std::vector<rerun::Color> colors;
140 points3d.reserve(locatorState.position.size());
141 labels.reserve(locatorState.position.size());
142 colors.reserve(locatorState.position.size());
143
144 for (size_t i = 0; i < locatorState.position.size(); ++i) {
145 const auto& locator = locatorState.position[i];
146 points3d.push_back(toRerunPosition3D(locator));
147 labels.push_back(locators[i].name);
148 rerun::Color color{};
149 if (locators.at(i).name.find("Floor_") != std::string::npos) {
150 color = kGreen;
151 } else {
152 float intensity = 255.0 * locators[i].weight * 0.6;
153 color = rerun::Color(int8_t(intensity), int8_t(intensity), int8_t(intensity * 0.5));
154 }
155 colors.push_back(color);
156 }
157
158 // TODO: make radius and color configurable
159 rec.log(
160 streamName,
161 rerun::Points3D(points3d).with_radii(0.5f).with_colors(colors).with_labels(labels));
162}

164void logMarkerLocatorCorrespondence(
165 const rerun::RecordingStream& rec,
166 const std::string& streamName,
167 const std::map<std::string, size_t>& locatorLookup,
168 const LocatorState& locatorState,
169 gsl::span<const Marker> markers,
170 const float kPositionErrorThreshold) {
171 if (locatorLookup.empty()) {
172 // No correspondence provided.
173 return;
174 }
175 std::vector<std::string> labels;
176 std::vector<std::vector<std::array<float, 3>>> lines;
177 std::vector<rerun::components::Color> colors;
178 std::vector<rerun::components::Radius> radius;
179 lines.reserve(markers.size());
180 labels.reserve(markers.size());
181 colors.reserve(markers.size());
182 radius.reserve(markers.size());
183
184 const rerun::components::Color kGreenColor(50, 255, 128);
185 const rerun::components::Color kRedColor(255, 100, 100);
186 const rerun::components::Radius kDefaultRadius(0.1f);
187 const rerun::components::Radius kLargeRadius(0.5f);
188
189 for (const auto& marker : markers) {
190 if (!marker.occluded && (locatorLookup.count(marker.name) != 0)) {
191 const auto locatorIdx = locatorLookup.at(marker.name);
192 const auto locator = locatorState.position.at(locatorIdx);
193
194 lines.push_back({toStdArray3f(marker.pos), toStdArray3f(locator)});
195 labels.push_back(marker.name);
196
197 // TODO: expose marker error computation?
198 const auto error = (marker.pos.cast<float>() - locator).squaredNorm();
199 if (error > kPositionErrorThreshold) {
200 colors.push_back(kRedColor);
201 radius.push_back(kLargeRadius);
202 } else {
203 colors.push_back(kGreenColor);
204 radius.push_back(kDefaultRadius);
205 }
206 }
207
208 rec.log(
209 streamName,
210 rerun::LineStrips3D(lines).with_labels(labels).with_colors(colors).with_radii(radius));
211 }

289void logBvh(
290 const rerun::RecordingStream& rec,
291 const std::string& streamName,
292 const CollisionGeometry& collisionGeometry,
293 const SkeletonState& skeletonState) {
294 // Compute collision state
295 CollisionGeometryState collisionState;
296 collisionState.update(skeletonState, collisionGeometry);
297
298 // Compute AABBs
299 const auto n = collisionGeometry.size();
300 std::vector<axel::BoundingBoxf> aabbs(n);
301 for (size_t i = 0; i < n; ++i) {
302 auto& aabb = aabbs[i];
303 aabb.id = i;
304 updateAabb(
305 aabb, collisionState.origin[i], collisionState.direction[i], collisionState.radius[i]);
306 }
307
308 // Construct BVH
309 axel::Bvhf bvh;
310 bvh.setBoundingBoxes(aabbs);
311 const auto& bvs = bvh.getPrimitives();
312
313 // Log BVH
314 std::vector<rerun::Position3D> centers;
315 std::vector<rerun::HalfSize3D> halfSizes;
316 centers.reserve(bvs.size());
317 halfSizes.reserve(bvs.size());
318 for (const auto& bv : bvs) {
319 centers.push_back(toRerunPosition3D(bv.center()));
320 halfSizes.push_back(toRerunHalfSizes3D((bv.max() - bv.min()) / 2.0f));
321 }
322
323 // TODO: make radius configurable
324 // TODO: use different colors by the depth of BVH nodes
325 rec.log(
326 streamName,
327 rerun::Boxes3D::from_centers_and_half_sizes(centers, halfSizes)
328 .with_radii(0.1f)
329 .with_colors(rerun::Color(64, 128, 64)));
330}

332void logCollisionGeometry(
333 const rerun::RecordingStream& rec,
334 const std::string& streamName,
335 const CollisionGeometry& collisionGeometry,
336 const SkeletonState& skeletonState) {
337 // TODO: update collision geometry representation from box to capsule
338
339 std::vector<rerun::Position3D> translations;
340 std::vector<rerun::Quaternion> quaternions;
341 std::vector<float> lengths;
342 std::vector<float> radii;
343
344 translations.reserve(collisionGeometry.size());
345 quaternions.reserve(collisionGeometry.size());
346 lengths.reserve(collisionGeometry.size());
347 radii.reserve(collisionGeometry.size());
348
349 for (const auto& cg : collisionGeometry) {
350 const auto& js = skeletonState.jointState.at(cg.parent);
351
352 const Transform tf = js.transform * cg.transformation;
353 const Quaternionf& q = tf.rotation * Eigen::AngleAxisf(0.5f * pi(), Vector3f::UnitY());
354
355 translations.push_back(toRerunPosition3D(tf.translation));
356 quaternions.emplace_back(rerun::Quaternion::from_xyzw(q.x(), q.y(), q.z(), q.w()));
357 lengths.emplace_back(cg.length);
358 // TODO: Rerun doesn't support capsules with different radii (i.e. tapered capsule) yet
359 radii.emplace_back(cg.radius.maxCoeff());
360 }
361
362 // TODO: make radius and color configurable
363 rec.log(
364 streamName,
365 rerun::Capsules3D::from_lengths_and_radii(lengths, radii)
366 .with_translations(translations)
367 .with_quaternions(quaternions)
368 .with_colors(rerun::Color(128, 64, 64)));
369 // TODO: Switch to wireframe once available in Rerun
370
371 logBvh(rec, streamName + "/bvh", collisionGeometry, skeletonState);
372}

374void logCharacter(
375 const rerun::RecordingStream& rec,
376 const std::string& charStreamName,
377 const Character& character,
378 const CharacterState& characterState,
379 const rerun::Color& color) {
380 if (characterState.meshState != nullptr) {
381 logMesh(rec, charStreamName + "/mesh", *characterState.meshState, color);
382 }
383 if (!character.locators.empty()) {
384 logLocators(rec, charStreamName + "/locators", character.locators, characterState.locatorState);
385 }
386
387 logJoints(
388 rec, charStreamName + "/joints", character.skeleton, characterState.skeletonState.jointState);
389
390 if (const auto& collision = character.collision) {
391 logCollisionGeometry(
392 rec, charStreamName + "/collision_geometry", *collision, characterState.skeletonState);
393 }
394}
