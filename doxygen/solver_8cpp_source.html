47template <typename T>
48double SolverT<T>::solve(Eigen::VectorX<T>& params) {
49 MT_PROFILE_FUNCTION();
50
51 if (storeHistory) {
52 auto& parameterHistory = iterationHistory_["parameters"];
53 auto& errorHistory = iterationHistory_["error"];
54 auto& iterationCount = iterationHistory_["iterations"];
55 if (parameterHistory.rows() != gsl::narrow_cast<Eigen::Index>(numParameters_) ||
56 parameterHistory.cols() != gsl::narrow_cast<Eigen::Index>(maxIterations_)) {
57 parameterHistory.resize(numParameters_, maxIterations_);
58 }
59 if (errorHistory.rows() != gsl::narrow_cast<Eigen::Index>(maxIterations_) ||
60 errorHistory.cols() != 1) {
61 errorHistory.resize(maxIterations_, 1);
62 }
63 if (iterationCount.rows() != 1 || iterationCount.cols() != 1) {
64 iterationCount.resize(1, 1);
65 }
66
67 parameterHistory.setZero();
68 errorHistory.setZero();
69 iterationCount.setZero();
70 }
71
72 errorHistory_.clear();
73 errorHistory_.reserve(maxIterations_);
74
75 MT_CHECK(params.size() == gsl::narrow<Eigen::DenseIndex>(numParameters_));
76 parameters_ = params;
77
78 static_assert(
79 std::is_same<decltype(error_), decltype(lastError_)>::value,
80 "error and lastError_ should be of the same type");
81 error_ = std::numeric_limits<decltype(error_)>::max();
82 lastError_ = std::numeric_limits<decltype(lastError_)>::max();
83
84 initializeSolver();
85
86 iteration_ = 0;
87 for (; iteration_ < maxIterations_; iteration_++) {
88 // do actual iteration (iterations should update the error value)
89 doIteration();
90 errorHistory_.push_back(error_);
91
92 MT_LOGI_IF(this->verbose_, "Iteration: {}, error: {}", this->iteration_, error_);
93
94 // check for convergence
95 bool converged = false;
96 if (std::fabs(lastError_ - error_) / (std::fabs(error_) + std::numeric_limits<float>::min()) <=
97 threshold_ * std::numeric_limits<float>::epsilon()) {
98 converged = true;
99 }
100
101 // set history
102 if (storeHistory) {
103 auto& parameterHistory = iterationHistory_["parameters"];
104 auto& errorHistory = iterationHistory_["error"];
105 parameterHistory.col(iteration_) = parameters_;
106 errorHistory(iteration_, 0) = gsl::narrow_cast<float>(error_);
107
108 solverFunction_->storeHistory(iterationHistory_, iteration_, maxIterations_);
109 }
110
111 if (iteration_ >= minIterations_ && converged) {
112 break;
113 }
114
115 // update last error
116 lastError_ = error_;
117 }
118
119 if (storeHistory) {
120 auto& iterationCount = iterationHistory_["iterations"];
121 iterationCount(0, 0) = gsl::narrow_cast<float>(iteration_);
122 }
123
124 params = parameters_;
125 return error_;
126}
