25template <typename T>
26template <typename JetType>
27[[nodiscard]] JetType FullyDifferentiableProjectionErrorFunctionT<T>::constraintGradient_dot(
28 const momentum::ModelParametersT<T>& /*modelParams*/,
29 const momentum::SkeletonStateT<T>& skelState,
30 const int parentJointIndex_cons,
31 const Eigen::Vector3<JetType>& offset_cons,
32 const JetType& weight_cons,
33 const Eigen::Matrix<JetType, 3, 4>& projection_cons,
34 const Eigen::Vector2<JetType>& target_cons,
35 Eigen::Ref<const Eigen::VectorX<T>> vec) const {
36 auto result = JetType();
37
38 const auto& jsCons = skelState.jointState[parentJointIndex_cons];
39 const Eigen::Vector3<JetType> p_world_cm =
40 (jsCons.transform.toAffine3().template cast<T>() * offset_cons).eval();
41 const Eigen::Vector3<JetType> p_projected_cm =
42 (projection_cons * p_world_cm.homogeneous()).eval();
43
44 // Behind camera:
45 if (p_projected_cm.z().a < _nearClip) {
46 return result;
47 }
48
49 const Eigen::Vector2<JetType> p_res = p_projected_cm.hnormalized() - target_cons;
50
51 const JetType wgt = T(2) * weight_cons * kProjectionWeight * this->weight_;
52
53 const JetType& z = p_projected_cm(2);
54 const JetType zSqr = z * z;
55 const JetType x_zz = p_projected_cm(0) / zSqr;
56 const JetType y_zz = p_projected_cm(1) / zSqr;
57
58 // Add dError_dModelParametersT<T>^JetType * v =
59 // (dError/dJointParameter[jFullBodyDOF] *
60 // dJointParameter[jFullBodyDOF]/dModelParametersT<T>)^JetType * v
61 auto add_gradient_dot = [&](const size_t jFullBodyDOF,
62 const Eigen::Vector3<JetType>& d_p_world_cm) {
63 const Eigen::Vector3<JetType> d_p_projected =
64 (projection_cons.template topLeftCorner<3, 3>() * d_p_world_cm).eval();
65 const JetType& dx = d_p_projected(0);
66 const JetType& dy = d_p_projected(1);
67 const JetType& dz = d_p_projected(2);
68
69 const Eigen::Vector2<JetType> d_p_res(dx / z - x_zz * dz, dy / z - y_zz * dz);
70
71 // Error is E = w * (p_projected.xy / p_projected.z)^2
72 // Gradient of error is:
73 // dE/dJointParameter[jFullBodyDOF] = 2w*(p_window - p_target) *
74 // d(p_window)/ddJointParameter[jFullBodyDOF].
75 const JetType gradFull = d_p_res.dot(p_res);
76
77 // Multiply by the parameter transform, take the dot product with v, and add to the result:
78 result += times_parameterTransform_times_v(
79 gradFull * wgt, jFullBodyDOF, this->parameterTransform_, vec);
80 };
81
82 // loop over all joints the constraint is attached to and calculate gradient
83 size_t jointIndex = parentJointIndex_cons;
84 while (jointIndex != kInvalidIndex) {
85 // check for valid index
86 assert(jointIndex < this->skeleton_.joints.size());
87
88 const auto& jointState = skelState.jointState[jointIndex];
89 const size_t paramIndex = jointIndex * 7;
90 const Eigen::Vector3<JetType> posd_cm = p_world_cm - jointState.translation();
91
92 // calculate derivatives based on active joints
93 for (size_t d = 0; d < 3; d++) {
94 if (this->activeJointParams_[paramIndex + d]) {
95 add_gradient_dot(
96 paramIndex + d, jointState.getTranslationDerivative(d).template cast<JetType>());
97 }
98 if (this->activeJointParams_[paramIndex + 3 + d]) {
99 add_gradient_dot(
100 paramIndex + 3 + d,
101 getRotationDerivative(jointState, d, posd_cm).template cast<JetType>());
102 }
103 }
104 if (this->activeJointParams_[paramIndex + 6]) {
105 add_gradient_dot(
106 paramIndex + 6, getScaleDerivative(jointState, posd_cm).template cast<JetType>());
107 }
108
109 // go to the next joint
110 jointIndex = this->skeleton_.joints[jointIndex].parent;
111 }
112
113 return result;
114}

116template <typename T>
117Eigen::VectorX<T> FullyDifferentiableProjectionErrorFunctionT<T>::d_gradient_d_input_dot(
118 const std::string& inputName,
119 const ModelParametersT<T>& modelParams,
120 const SkeletonStateT<T>& state,
121 Eigen::Ref<const Eigen::VectorX<T>> inputVec) {
122 // Use automatic differentiation on the gradient function:
123 const auto nCons = this->constraints_.size();
124 if (inputName == kWeights) {
125 Eigen::VectorX<T> result = Eigen::VectorX<T>::Zero(nCons);
126 using JetType = ceres::Jet<T, 1>;
127 for (int iCons = 0; iCons < nCons; ++iCons) {
128 const auto dotProd = constraintGradient_dot<JetType>(
129 modelParams,
130 state,
131 this->constraints_[iCons].parent,
132 this->constraints_[iCons].offset.template cast<JetType>(),
133 JetType(this->constraints_[iCons].weight, 0),
134 this->constraints_[iCons].projection.template cast<JetType>(),
135 this->constraints_[iCons].target.template cast<JetType>(),
136 inputVec);
137 result(iCons) = dotProd.v[0];
138 }
139 return result;
140 } else if (inputName == kOffsets) {
141 Eigen::VectorX<T> result = Eigen::VectorX<T>::Zero(3 * nCons);
142
143 using JetType = ceres::Jet<T, 3>;
144 for (int iCons = 0; iCons < nCons; ++iCons) {
145 const auto dotProd = constraintGradient_dot<JetType>(
146 modelParams,
147 state,
148 this->constraints_[iCons].parent,
149 momentum::buildJetVec<T, 3>(this->constraints_[iCons].offset),
150 JetType(this->constraints_[iCons].weight),
151 this->constraints_[iCons].projection.template cast<JetType>(),
152 this->constraints_[iCons].target.template cast<JetType>(),
153 inputVec);
154 result.template segment<3>(3 * iCons) = dotProd.v;
155 }
156 return result;
157 } else if (inputName == kTargets) {
158 Eigen::VectorX<T> result = Eigen::VectorX<T>::Zero(2 * nCons);
159 using JetType = ceres::Jet<T, 2>;
160 for (int iCons = 0; iCons < nCons; ++iCons) {
161 const auto dotProd = constraintGradient_dot<JetType>(
162 modelParams,
163 state,
164 this->constraints_[iCons].parent,
165 this->constraints_[iCons].offset.template cast<JetType>(),
166 JetType(this->constraints_[iCons].weight),
167 this->constraints_[iCons].projection.template cast<JetType>(),
168 buildJetVec<T, 2>(this->constraints_[iCons].target),
169 inputVec);
170 result.template segment<2>(2 * iCons) = dotProd.v;
171 }
172 return result;
173 } else if (inputName == kProjections) {
174 Eigen::VectorX<T> result = Eigen::VectorX<T>::Zero(12 * nCons);
175 using JetType = ceres::Jet<T, 12>;
176 for (int iCons = 0; iCons < nCons; ++iCons) {
177 const auto dotProd = constraintGradient_dot<JetType>(
178 modelParams,
179 state,
180 this->constraints_[iCons].parent,
181 this->constraints_[iCons].offset.template cast<JetType>(),
182 JetType(this->constraints_[iCons].weight),
183 momentum::buildJetMat<T, 3, 4>(this->constraints_[iCons].projection),
184 this->constraints_[iCons].target.template cast<JetType>(),
185 inputVec);
186 result.template segment<12>(12 * iCons) = dotProd.v;
187 }
188 return result;
189 } else {
190 throw std::runtime_error(
191 "Unknown input name in FullyDifferentiablePositionErrorFunction::d_gradient_d_input_dot: " +
192 inputName);
193 }

203template <typename T>
204Eigen::Index FullyDifferentiableProjectionErrorFunctionT<T>::getInputSize(
205 const std::string& name) const {
206 if (name == kWeights) {
207 return this->constraints_.size();
208 } else if (name == kOffsets) {
209 return 3 * this->constraints_.size();
210 } else if (name == kTargets) {
211 return 2 * this->constraints_.size();
212 } else if (name == kProjections) {
213 return 3 * 4 * this->constraints_.size();
214 } else {
215 throw std::runtime_error("Unknown input for ProjectionErrorFunction: " + name);
216 }

219template <typename T>
220void FullyDifferentiableProjectionErrorFunctionT<T>::getInputImp(
221 const std::string& name,
222 Eigen::Ref<Eigen::VectorX<T>> result) const {
223 if (name == kWeights) {
224 for (size_t i = 0; i < this->constraints_.size(); ++i) {
225 result(i) = this->constraints_[i].weight;
226 }
227 } else if (name == kOffsets) {
228 for (size_t i = 0; i < this->constraints_.size(); ++i) {
229 result.template segment<3>(3 * i) = this->constraints_[i].offset;
230 }
231 } else if (name == kTargets) {
232 for (size_t i = 0; i < this->constraints_.size(); ++i) {
233 result.template segment<2>(2 * i) = this->constraints_[i].target;
234 }
235 } else if (name == kProjections) {
236 for (size_t i = 0; i < this->constraints_.size(); ++i) {
237 const auto& proj = this->constraints_[i].projection;
238
239 // Ensure row-major for compatibility with Torch:
240 for (int jRow = 0; jRow < 3; ++jRow) {
241 for (int kCol = 0; kCol < 4; ++kCol) {
242 result(12 * i + 4 * jRow + kCol) = proj(jRow, kCol);
243 }
244 }
245 }
246 } else {
247 throw std::runtime_error("Unknown input for ProjectionErrorFunction: " + name);
248 }
