28template <typename T>
29Vector3<T> parentOffsetFromWorldPoint(
30 const JointStateT<T>& parentState,
31 const Vector3<T>& worldPoint) {
32 const T parentScale = parentState.transform.scale;
33 const T minParentScale = Eps<T>(1e-8f, 1e-17);
34 MT_THROW_IF(
35 !std::isfinite(parentScale) || std::abs(parentScale) <= minParentScale,
36 "Support contact parent scale magnitude must be finite and greater than {}, got {}",
37 minParentScale,
38 parentScale);
39 return (parentState.transform.rotation.conjugate() *
40 (worldPoint - parentState.transform.translation)) /
41 parentScale;
42}

44template <typename T>
45void validateMatchingCollisionPlane(
46 const PlaneCollisionQueryT<T>& collisionQuery,
47 const SupportPlaneT<T>& supportPlane) {
48 const Vector3<T>& planeNormal = collisionQuery.getPlaneNormal();
49 const T planeOffset = collisionQuery.getPlaneOffset();
50 const T normalTolerance = Eps<T>(1e-5f, 1e-12);
51 const T offsetTolerance = normalTolerance *
52 std::max(T(1), std::max(std::abs(planeOffset), std::abs(supportPlane.offset)));
53 const T normalError = (planeNormal - supportPlane.normal).cwiseAbs().maxCoeff();
54 const T offsetError = std::abs(planeOffset - supportPlane.offset);
55 MT_THROW_IF(
56 !planeNormal.allFinite() || !std::isfinite(planeOffset) || normalError > normalTolerance ||
57 offsetError > offsetTolerance,
58 "Support contact collision function must use the same oriented support plane: "
59 "collision normal ({}, {}, {}), offset {}; support normal ({}, {}, {}), offset {}",
60 planeNormal.x(),
61 planeNormal.y(),
62 planeNormal.z(),
63 planeOffset,
64 supportPlane.normal.x(),
65 supportPlane.normal.y(),
66 supportPlane.normal.z(),
67 supportPlane.offset);
68}

70template <typename T, typename HandleFloorLocator>
71void visitFloorLocatorPositions(
72 const std::span<const JointStateT<T>> states,
73 const std::span<const Locator> locators,
74 const SupportPlaneT<T>& supportPlane,
75 HandleFloorLocator handleFloorLocator) {
76 for (const Locator& locator : locators) {
77 if (!isFloorLocatorName(locator.name)) {
78 continue;
79 }
80 MT_THROW_IF(
81 locator.parent >= states.size(),
82 "Floor locator '{}' parent {} is outside skeleton state with {} joints",
83 locator.name,
84 locator.parent,
85 states.size());
86
87 const JointStateT<T>& parentState = states[locator.parent];
88 const Vector3<T> position = parentState.transform * locator.offset.template cast<T>();
89 MT_THROW_IF(
90 !position.allFinite(),
91 "Floor locator '{}' position must contain only finite values",
92 locator.name);
93 const T signedDistance = supportPlane.signedDistance(position);
94 MT_THROW_IF(
95 !std::isfinite(signedDistance),
96 "Floor locator '{}' support-plane distance must be finite",
97 locator.name);
98 handleFloorLocator(locator, parentState, position, signedDistance);
99 }
100}

102template <typename T, typename HandleCollisionQuery>
103void withSupportCollisionQuery(
104 const Character& character,
105 PlaneCollisionQueryT<T>* collisionQuery,
106 const SupportPlaneT<T>& supportPlane,
107 HandleCollisionQuery handleCollisionQuery) {
108 if (collisionQuery != nullptr) {
109 validateMatchingCollisionPlane(*collisionQuery, supportPlane);
110 handleCollisionQuery(*collisionQuery);
111 return;
112 }
113
114 if (!character.collision || character.collision->empty()) {
115 return;
116 }
117
118 PlaneCollisionQueryT<T> defaultCollisionQuery(
119 *character.collision, supportPlane.normal, supportPlane.offset);
120 handleCollisionQuery(defaultCollisionQuery);
121}

125template <typename T>
126std::vector<SupportContactT<T>> computePlaneCollisionSupportContacts(
127 PlaneCollisionQueryT<T>& collisionQuery,
128 const std::span<const JointStateT<T>> states,
129 const T contactMargin) {
130 const std::vector<PlaneCollisionContactPointT<T>> contacts =
131 collisionQuery.getContactPointsByParentWithDetails(states, contactMargin);
132 std::vector<SupportContactT<T>> result;
133 result.reserve(contacts.size());
134 for (const PlaneCollisionContactPointT<T>& contact : contacts) {
135 MT_THROW_IF(
136 contact.parentJoint >= states.size(),
137 "Plane collision parent {} is outside skeleton state with {} joints",
138 contact.parentJoint,
139 states.size());
140 result.push_back(
141 {contact.parentJoint,
142 contact.position,
143 parentOffsetFromWorldPoint(states[contact.parentJoint], contact.position)});
144 }
145 return result;
146}

148template <typename T>
149SupportContactListT<T> computeFloorLocatorSupportContacts(
150 const std::span<const JointStateT<T>> states,
151 const std::span<const Locator> locators,
152 const T contactHeight,
153 const SupportPlaneT<T>& supportPlane) {
154 MT_THROW_IF(
155 !std::isfinite(contactHeight) || contactHeight < T(0),
156 "Floor locator contact height must be finite and non-negative, got {}",
157 contactHeight);
158
159 SupportContactListT<T> result;
160 result.contacts.reserve(locators.size());
161 result.floorLocatorPositions.reserve(locators.size());
162 visitFloorLocatorPositions<T>(
163 states,
164 locators,
165 supportPlane,
166 [&](const Locator& locator,
167 const JointStateT<T>& parentState,
168 const Vector3<T>& position,
169 const T signedDistance) {
170 result.floorLocatorPositions.push_back(position);
171 if (signedDistance > contactHeight) {
172 return;
173 }
174 result.contacts.push_back(
175 {locator.parent, position, parentOffsetFromWorldPoint(parentState, position)});
176 });
177 return result;
178}

180template <typename T>
181std::vector<Vector3<T>> computeSupportContactPositions(
182 const Character& character,
183 const std::span<const JointStateT<T>> states,
184 const T contactHeight,
185 PlaneCollisionQueryT<T>* collisionQuery,
186 const SupportPlaneT<T>& supportPlane) {
187 MT_THROW_IF(
188 !std::isfinite(contactHeight) || contactHeight < T(0),
189 "Support contact height must be finite and non-negative, got {}",
190 contactHeight);
191
192 std::vector<Vector3<T>> result;
193 result.reserve(character.locators.size());
194 visitFloorLocatorPositions<T>(
195 states,
196 character.locators,
197 supportPlane,
198 [&](const Locator& /*locator*/,
199 const JointStateT<T>& /*parentState*/,
200 const Vector3<T>& position,
201 const T signedDistance) {
202 if (signedDistance <= contactHeight) {
203 result.push_back(position);
204 }
205 });
206
207 withSupportCollisionQuery<T>(
208 character, collisionQuery, supportPlane, [&](PlaneCollisionQueryT<T>& query) {
209 const std::vector<Vector3<T>> collisionPositions =
210 query.getContactPointsByParent(states, contactHeight);
211 result.insert(result.end(), collisionPositions.begin(), collisionPositions.end());
212 });
213 return result;
214}

216template <typename T>
217SupportContactListT<T> computeSupportContacts(
218 const Character& character,
219 const std::span<const JointStateT<T>> states,
220 const T contactHeight,
221 PlaneCollisionQueryT<T>* collisionQuery,
222 const SupportPlaneT<T>& supportPlane) {
223 SupportContactListT<T> result =
224 computeFloorLocatorSupportContacts(states, character.locators, contactHeight, supportPlane);
225
226 // TODO: Floor locators and collision primitive support points are not identical contact
227 // signals. Compute a proper contact set instead of concatenating these proxies.
228 withSupportCollisionQuery<T>(
229 character, collisionQuery, supportPlane, [&](PlaneCollisionQueryT<T>& query) {
230 const std::vector<SupportContactT<T>> collisionContacts =
231 computePlaneCollisionSupportContacts(query, states, contactHeight);
232 result.contacts.insert(
233 result.contacts.end(), collisionContacts.begin(), collisionContacts.end());
234 });
235 return result;
236}
