55std::vector<size_t> sampleFrames(
56 momentum::Character& character,
57 const MatrixXf& initialMotion,
58 std::span<const std::vector<momentum::Marker>> markerData,
59 const ParameterSet& parameters,
60 const size_t frameStride,
61 const size_t numSamples) {
62 // sample frames so that we get the most variance from the initial input
63 const auto numFrames = static_cast<size_t>(initialMotion.cols());
64 size_t solvedFrames = (numFrames - 1) / frameStride + 1;
65
66 std::vector<size_t> frameIndices;
67 if (solvedFrames == 0) {
68 return frameIndices;
69 }
70 const size_t numActualSamples = std::min(numSamples, solvedFrames);
71
72 // get the indices of the parameters to be used
73 std::vector<size_t> usedParameters;
74 for (size_t i = 0; i < static_cast<size_t>(initialMotion.rows()); ++i) {
75 if (parameters.test(i)) {
76 usedParameters.push_back(i);
77 }
78 }
79
80 // calculate per frame error
81 const ParameterTransform& pt = character.parameterTransform;
82
83 // map locator name to its index
84 std::map<std::string, size_t> locatorLookup;
85 for (size_t i = 0; i < character.locators.size(); i++) {
86 locatorLookup[character.locators[i].name] = i;
87 }
88
89 std::vector<double> frameErrors(solvedFrames, 0.0f);
90 SkeletonState state;
91 for (size_t iFrame = 0, fi = 0; iFrame < numFrames; iFrame += frameStride, fi++) {
92 const auto jointParams = pt.apply(initialMotion.col(iFrame));
93 state.set(jointParams, character.skeleton, false);
94
95 const auto& markerList = markerData[iFrame];
96 for (const auto& jMarker : markerList) {
97 if (jMarker.occluded) {
98 continue;
99 }
100 auto query = locatorLookup.find(jMarker.name);
101 if (query == locatorLookup.end()) {
102 continue;
103 }
104 size_t locatorIdx = query->second;
105 if (locatorIdx >= character.locators.size()) {
106 continue;
107 }
108 const auto& locator = character.locators[locatorIdx];
109 if (locator.parent >= state.jointState.size()) {
110 continue;
111 }
112 const Vector3f locatorPos = state.jointState[locator.parent].transform * locator.offset;
113 const Vector3f diff = locatorPos - jMarker.pos.cast<float>();
114 const float markerError = diff.norm();
115 frameErrors[fi] += markerError;
116 }
117 }
118 std::vector<double> sortedErrors = frameErrors;
119 std::sort(sortedErrors.begin(), sortedErrors.end());
120
121 // do not use the worst 1/4 of the fitted errors, there's likely to be some outliers and
122 // errors due to the initialization
123 const double threshold = sortedErrors[(sortedErrors.size() * 3) / 4];
124
125 // get the motion of only the used parameters and then normalize
126 // by the mean and variance
127 MatrixXf normalized;
128 {
129 const MatrixXf subMotion =
130 initialMotion(usedParameters, Eigen::seq(0, numFrames - 1, frameStride));
131 const VectorXf mean = subMotion.rowwise().mean();
132 const MatrixXf centered = subMotion.colwise() - mean;
133 const VectorXf std = centered.array().square().rowwise().sum() / (numFrames - 1);
134
135 // normalize the motion by the sqrt of the variance, i.e. leave a little bit of scale in
136 normalized = centered.array().colwise() / (std.array().sqrt().sqrt().cwiseMax(1e-5f));
137 }
138
139 for (size_t i = 0; i < frameErrors.size(); ++i) {
140 if (frameErrors[i] > threshold) {
141 normalized.col(i) = VectorXf::Ones(normalized.rows()) * 1000.0f;
142 }
143 }
144
145 // calculate the distances from each frame to the already selected indices
146 frameIndices.push_back(0);
147 VectorXf distances = VectorXf::Zero(solvedFrames);
148 for (size_t fi = 0; fi < solvedFrames; fi++) {
149 distances[fi] = (normalized.col(frameIndices[0]) - normalized.col(fi)).norm();
150 }
151
152 // finally add additional samples with the largest distance
153 for (size_t i = frameIndices.size(); i < numActualSamples; ++i) {
154 size_t maxFrame = 0;
155 const float maxval = distances.maxCoeff(&maxFrame);
156 if (maxval < 1e-5f) {
157 break;
158 }
159 frameIndices.push_back(maxFrame);
160
161 for (size_t fi = 0; fi < solvedFrames; fi++) {
162 const float dist = (normalized.col(maxFrame) - normalized.col(fi)).cwiseAbs().maxCoeff();
163 distances[fi] = std::min(distances[fi], dist);
164 }
165 }
166
167 for (auto&& fi : frameIndices) {
168 fi *= frameStride;
169 }
170
171 return frameIndices;
172}

239Eigen::MatrixXf trackSequence(
240 const std::span<const std::vector<Marker>> markerData,
241 const Character& character,
242 const ParameterSet& globalParams,
243 const MatrixXf& initialMotion,
244 const TrackingConfig& config,
245 const std::vector<size_t>& frames,
246 float regularizer,
247 bool enforceFloorInFirstFrame,
248 const std::string& firstFramePoseConstraintSet,
249 float targetHeightCm) {
250 // sanity checks
251 const size_t numFrames = markerData.size();
252 MT_CHECK(numFrames > 0, "Input data is empty.");
253 MT_CHECK(
254 initialMotion.cols() >= numFrames,
255 "Number of frames in data {} doesn't match that of input motion {}",
256 numFrames,
257 initialMotion.cols());
258 MT_CHECK(
259 initialMotion.rows() == character.parameterTransform.numAllModelParameters(),
260 "Input motion parameters {} do not match character model parameters {}",
261 initialMotion.rows(),
262 character.parameterTransform.numAllModelParameters());
263
264 const ParameterTransform& pt = character.parameterTransform;
265 const size_t numMarkers = markerData[0].size();
266
267 // universal parameters include "scaling" and "locators" (if exists); pose parameters need to
268 // exclude "locators". universalParams is to indicate to the solver which parameters are "global"
269 // (ie. not time varying). The input globalParams indicate which parameters within universalParams
270 // we want to solve for. globalParams is either a subset or all of universalParams.
271 ParameterSet poseParams = pt.getPoseParameters();
272 ParameterSet universalParams = pt.getScalingParameters() | pt.getBlendShapeParameters();
273 const auto locatorSet =
274 pt.getParameterSet("locators", true) | pt.getParameterSet("skinnedLocators", true);
275 poseParams &= ~locatorSet;
276 universalParams |= locatorSet;
277
278 std::vector<momentum::SkinnedLocatorTriangleConstraintT<float>> skinnedLocatorMeshContraints;
279 if ((globalParams & pt.getBlendShapeParameters()).any() && !character.skinnedLocators.empty()) {
280 skinnedLocatorMeshContraints =
281 createSkinnedLocatorMeshConstraints(character, initialMotion.col(0), 1.0f);
282 }
283
284 // set up the solver function
285 std::vector<size_t> sortedFrames = frames;
286 std::sort(sortedFrames.begin(), sortedFrames.end());
287 size_t solvedFrames = sortedFrames.size();
288 auto solverFunc = SequenceSolverFunction(
289 character, character.parameterTransform, universalParams, solvedFrames);
290
291 // floor penetration constraints; we assume the world is y-up and floor is y=0 for mocap data.
292 const auto& floorConstraints = createFloorConstraints<float>(
293 "Floor_",
294 character.locators,
295 Vector3f::UnitY(),
296 /* y offset */ 0.0f,
297 /* weight */ 5.0f);
298
299 // marker constraints
300 const auto constrData = createConstraintData(markerData, character.locators);
301 const auto skinnedConstData = createSkinnedConstraintData(markerData, character.skinnedLocators);
302
303 // add per-frame constraint data to the solver
304 for (size_t solverFrame = 0; solverFrame < solvedFrames; ++solverFrame) {
305 const size_t& iFrame = sortedFrames[solverFrame];
306 if ((constrData.at(iFrame).size() + skinnedConstData.at(iFrame).size()) >
307 numMarkers * config.minVisPercent) {
308 auto posConstrWeight = PositionErrorFunction::kLegacyWeight;
309 if (solverFrame == 0 && (enforceFloorInFirstFrame || !firstFramePoseConstraintSet.empty())) {
310 posConstrWeight *= solvedFrames;
311 }
312
313 // prepare positional constraints
314 if (!constrData.at(iFrame).empty()) {
315 auto posConstrFunc = std::make_shared<PositionErrorFunction>(character, config.lossAlpha);
316 posConstrFunc->setConstraints(constrData.at(iFrame));
317 posConstrFunc->setWeight(posConstrWeight);
318 solverFunc.addErrorFunction(solverFrame, posConstrFunc);
319 }
320
321 if (!skinnedConstData.at(iFrame).empty()) {
322 auto skinnedConstrFunc = std::make_shared<SkinnedLocatorErrorFunction>(character);
323 skinnedConstrFunc->setConstraints(skinnedConstData.at(iFrame));
324 skinnedConstrFunc->setWeight(posConstrWeight);
325 solverFunc.addErrorFunction(solverFrame, skinnedConstrFunc);
326 }
327
328 if (!skinnedLocatorMeshContraints.empty() && solverFrame == 0) {
329 // Need to create a function to keep the skinned constraints close to
330 // the mesh.
331 auto skinnedTriangleConstrFunc =
332 std::make_shared<SkinnedLocatorTriangleErrorFunctionT<float>>(character);
333 skinnedTriangleConstrFunc->setConstraints(skinnedLocatorMeshContraints);
334 skinnedTriangleConstrFunc->setWeight(solvedFrames * posConstrWeight);
335 solverFunc.addErrorFunction(solverFrame, skinnedTriangleConstrFunc);
336 }
337
338 if (pt.getBlendShapeParameters().any() && solverFrame == 0) {
339 // regularize the blend shape parameters
340 auto blendShapeConstrFunc = std::make_shared<ModelParametersErrorFunction>(character);
341 blendShapeConstrFunc->setWeight(solvedFrames * 0.1f);
342 Eigen::VectorXf weights = Eigen::VectorXf::Zero(pt.numAllModelParameters());
343 for (Eigen::Index i = 0; i < pt.blendShapeParameters.size(); ++i) {
344 if (pt.blendShapeParameters[i] >= 0) {
345 weights[pt.blendShapeParameters[i]] = 1.0f;
346 }
347 }
348 blendShapeConstrFunc->setTargetParameters(
349 ModelParameters::Zero(pt.numAllModelParameters()), weights);
350 solverFunc.addErrorFunction(solverFrame, blendShapeConstrFunc);
351 }
352
353 // add height constraint if specified
354 if (targetHeightCm > 0.0f && solverFrame == 0) {
355 auto heightConstrFunc = std::make_shared<HeightErrorFunctionT<float>>(
356 character, targetHeightCm, Eigen::Vector3f::UnitY(), 10);
357 heightConstrFunc->setWeight(solvedFrames * 1.0f);
358 solverFunc.addErrorFunction(solverFrame, heightConstrFunc);
359 }
360
361 // prepare floor constraints
362 if (!floorConstraints.empty()) {
363 bool halfPlane = true;
364 float weightMultiplier = 1.0f;
365 if (enforceFloorInFirstFrame && solverFrame == 0) {
366 halfPlane = false;
367 weightMultiplier = solvedFrames;
368 }
369 auto halfPlaneConstrFunc =
370 std::make_shared<PlaneErrorFunction>(character, /*half plane*/ halfPlane);
371 halfPlaneConstrFunc->setConstraints(floorConstraints);
372 halfPlaneConstrFunc->setWeight(PlaneErrorFunction::kLegacyWeight * weightMultiplier);
373 solverFunc.addErrorFunction(solverFrame, halfPlaneConstrFunc);
374 }
375
376 // add pose constraint set if defined
377 if (!firstFramePoseConstraintSet.empty() && solverFrame == 0) {
378 // check if the constraint set is defined
379 if (character.parameterTransform.poseConstraints.count(firstFramePoseConstraintSet) > 0) {
380 // add parameter limits
381 const auto poseLimits = getPoseConstraintParameterLimits(
382 firstFramePoseConstraintSet, character.parameterTransform);
383 auto limitConstrFunc = std::make_shared<LimitErrorFunction>(character, poseLimits);
384 limitConstrFunc->setWeight(solvedFrames);
385 solverFunc.addErrorFunction(solverFrame, limitConstrFunc);
386 }
387 }
388 }
389 // Set per-frame initial value
390 solverFunc.setFrameParameters(solverFrame, initialMotion.col(iFrame));
391 }
392
393 // add parameter limits
394 auto limitConstrFunc = std::make_shared<LimitErrorFunction>(character);
395 limitConstrFunc->setWeight(0.1);
396 solverFunc.addErrorFunction(kAllFrames, limitConstrFunc);
397
398 // add collision error
399 if (config.collisionErrorWeight != 0 && character.collision != nullptr) {
400 auto collisionConstrFunc = std::make_shared<CollisionErrorFunctionStateless>(character);
401 collisionConstrFunc->setWeight(config.collisionErrorWeight);
402 solverFunc.addErrorFunction(kAllFrames, collisionConstrFunc);
403 }
404
405 // add a smoothness constraint in parameter space
406 if (config.smoothing != 0) {
407 auto smoothConstrFunc = std::make_shared<ModelParametersSequenceErrorFunction>(character);
408 smoothConstrFunc->setWeight(config.smoothing);
409 solverFunc.addSequenceErrorFunction(kAllFrames, smoothConstrFunc);
410 }
411
412 // minimize the change to global params
413 if (globalParams.count() > 0 && regularizer != 0) {
414 auto regularizerFunc = std::make_shared<ModelParametersErrorFunction>(character);
415 Eigen::VectorXf universalMask(pt.numAllModelParameters());
416 for (size_t i = 0; i < universalMask.size(); ++i) {
417 if (globalParams.test(i)) {
418 universalMask[i] = regularizer;
419 } else {
420 universalMask[i] = 0.0;
421 }
422 }
423 regularizerFunc->setTargetParameters(initialMotion.col(0), universalMask);
424 // Sufficient to add to the first frame since it won't change.
425 solverFunc.addErrorFunction(0, regularizerFunc);
426 }
427
428 // solver configration
429 SequenceSolverOptions solverOptions;
430 solverOptions.maxIterations = config.maxIter;
431 solverOptions.minIterations = 2;
432 solverOptions.progressBar = config.debug;
433 solverOptions.doLineSearch = false;
434 solverOptions.multithreaded = true;
435 solverOptions.verbose = config.debug;
436 solverOptions.threshold = 1.f;
437 solverOptions.regularization = config.regularization;
438
439 // solve the problem
440 SequenceSolver solver = SequenceSolver(solverOptions, &solverFunc);
441 solver.setEnabledParameters(poseParams | globalParams);
442 // returns all the dofs with initial values nicely packed into a vector
443 VectorXf dofs = solverFunc.getJoinedParameterVector();
444 solver.solve(dofs);
445 double error = solverFunc.getError(dofs);
446 MT_LOGI_IF(config.debug, "Solver residual: {}", error);
447
448 // set results to output
449 size_t sortedIndex = 0;
450 MatrixXf outMotion(pt.numAllModelParameters(), numFrames);
451 for (size_t fi = 0; fi < numFrames; fi++) {
452 if (sortedIndex < sortedFrames.size() - 1 && fi == sortedFrames[sortedIndex + 1]) {
453 sortedIndex++;
454 }
455 outMotion.col(fi) = solverFunc.getFrameParameters(sortedIndex).v;
456 }
457 return outMotion;
458}

468bool isGlobalTransformZero(const Eigen::VectorXf& dof, const ParameterSet& rigidParams) {
469 for (Eigen::Index i = 0; i < dof.size(); ++i) {
470 if (rigidParams.test(i) && dof[i] != 0.0f) {
471 return false;
472 }
473 }
474 return true;
475}

490Eigen::MatrixXf trackPosesPerframe(
491 const std::span<const std::vector<Marker>> markerData,
492 const Character& character,
493 const ModelParameters& globalParams,
494 const TrackingConfig& config,
495 const size_t frameStride) {
496 const size_t numFrames = markerData.size();
497 MT_CHECK(numFrames > 0, "Input data is empty.");
498
499 // Generate frame indices from stride
500 std::vector<size_t> frameIndices;
501 for (size_t iFrame = 0; iFrame < numFrames; iFrame += frameStride) {
502 frameIndices.push_back(iFrame);
503 }
504
505 // Convert globalParams to initial motion matrix
506 MatrixXf initialMotion(globalParams.v.size(), numFrames);
507 for (size_t i = 0; i < numFrames; ++i) {
508 initialMotion.col(i) = globalParams.v;
509 }
510
511 // Determine if tracking should be continuous (temporal coherence)
512 bool isContinuous = (frameStride < 5);
513
514 return trackPosesForFrames(
515 markerData, character, initialMotion, config, frameIndices, isContinuous);
516}

532Eigen::MatrixXf trackPosesForFrames(
533 const std::span<const std::vector<Marker>> markerData,
534 const Character& character,
535 const MatrixXf& initialMotion,
536 const TrackingConfig& config,
537 const std::vector<size_t>& frameIndices,
538 bool isContinuous) {
539 const size_t numFrames = markerData.size();
540 MT_CHECK(numFrames > 0, "Input data is empty.");
541 MT_CHECK(
542 initialMotion.rows() == character.parameterTransform.numAllModelParameters(),
543 "Input motion parameters {} do not match character model parameters {}",
544 initialMotion.rows(),
545 character.parameterTransform.numAllModelParameters());
546
547 const ParameterTransform& pt = character.parameterTransform;
548 const size_t numMarkers = markerData[0].size();
549
550 std::vector<size_t> sortedFrames = frameIndices;
551 std::sort(sortedFrames.begin(), sortedFrames.end());
552
553 // pose parameters need to exclude "locators"
554 ParameterSet poseParams = pt.getPoseParameters();
555 const auto& locatorSet = pt.parameterSets.find("locators");
556 if (locatorSet != pt.parameterSets.end()) {
557 poseParams &= ~locatorSet->second;
558 }
559
560 // set up the solver
561 auto solverFunc = SkeletonSolverFunction(character, pt);
562 GaussNewtonSolverQROptions solverOptions;
563 solverOptions.maxIterations = config.maxIter;
564 solverOptions.minIterations = 2;
565 solverOptions.doLineSearch = false;
566 solverOptions.verbose = config.debug;
567 solverOptions.threshold = 1.f;
568 solverOptions.regularization = config.regularization;
569 auto solver = GaussNewtonSolverQR(solverOptions, &solverFunc);
570 solver.setEnabledParameters(poseParams);
571
572 // parameter limits constraint
573 auto limitConstrFunc = std::make_shared<LimitErrorFunction>(character);
574 limitConstrFunc->setWeight(0.1);
575 solverFunc.addErrorFunction(limitConstrFunc);
576
577 // positional constraint function for markers
578 auto posConstrFunc = std::make_shared<PositionErrorFunction>(character, config.lossAlpha);
579 posConstrFunc->setWeight(PositionErrorFunction::kLegacyWeight);
580 solverFunc.addErrorFunction(posConstrFunc);
581
582 auto skinnedLocatorPosConstrFunc = std::make_shared<SkinnedLocatorErrorFunction>(character);
583 skinnedLocatorPosConstrFunc->setWeight(PositionErrorFunction::kLegacyWeight);
584 solverFunc.addErrorFunction(skinnedLocatorPosConstrFunc);
585
586 // floor penetration constraint data; we assume the world is y-up and floor is y=0 for mocap data.
587 const auto& floorConstraints = createFloorConstraints<float>(
588 "Floor_",
589 character.locators,
590 Vector3f::UnitY(),
591 /* y offset */ 0.0f,
592 /* weight */ 5.0f);
593 auto halfPlaneConstrFunc = std::make_shared<PlaneErrorFunction>(character, /*half plane*/ true);
594 halfPlaneConstrFunc->setConstraints(floorConstraints);
595 halfPlaneConstrFunc->setWeight(PlaneErrorFunction::kLegacyWeight);
596 solverFunc.addErrorFunction(halfPlaneConstrFunc);
597
598 // marker constraint data
599 auto constrData = createConstraintData(markerData, character.locators);
600 auto skinnedConstrData = createSkinnedConstraintData(markerData, character.skinnedLocators);
601
602 // smoothness constraint only for the joints and exclude global dofs because the global transform
603 // needs to be accurate (may not matter in practice?)
604 // Only use temporal smoothness if isContinuous is true
605 std::shared_ptr<ModelParametersErrorFunction> smoothConstrFunc;
606 if (isContinuous) {
607 smoothConstrFunc = std::make_shared<ModelParametersErrorFunction>(
608 character, poseParams & ~pt.getRigidParameters());
609 smoothConstrFunc->setWeight(config.smoothing);
610 solverFunc.addErrorFunction(smoothConstrFunc);
611 }
612
613 // add collision error
614 std::shared_ptr<CollisionErrorFunction> collisionErrorFunction;
615 if (config.collisionErrorWeight != 0 && character.collision != nullptr) {
616 collisionErrorFunction = std::make_shared<CollisionErrorFunction>(character);
617 collisionErrorFunction->setWeight(config.collisionErrorWeight);
618 solverFunc.addErrorFunction(collisionErrorFunction);
619 }
620
621 // initialize parameters to contain identity information
622 // the identity fields will be used but untouched during optimization
623 // globalParams could also be repurposed to pass in initial pose value
624 Eigen::VectorXf dof = initialMotion.col(sortedFrames.empty() ? 0 : sortedFrames[0]);
625 size_t solverFrame = 0;
626 double priorError = 0.0;
627 double error = 0.0;
628
629 // Use the initial global transform if it's not zero
630 bool needsInit = isGlobalTransformZero(dof, pt.getRigidParameters());
631
632 MatrixXf outMotion = initialMotion;
633 Eigen::Index outputIndex = 0;
634 { // scope the ProgressBar so it returns
635 ProgressBar progress("Tracking per-frame", sortedFrames.size());
636 for (const auto iFrame : sortedFrames) {
637 // For continuous tracking, keep the solved dof from previous frame (temporal coherence)
638 // For non-continuous tracking, always start from initial motion (independent solving)
639 if (!isContinuous) {
640 dof = initialMotion.col(iFrame);
641 needsInit = true;
642 }
643 // For continuous tracking, dof is preserved from previous iteration (or initial value)
644
645 if ((constrData.at(iFrame).size() + skinnedConstrData.at(iFrame).size()) >
646 numMarkers * config.minVisPercent) {
647 // add positional constraints
648 posConstrFunc->clearConstraints(); // clear constraint data from the previous frame
649 posConstrFunc->setConstraints(constrData.at(iFrame));
650
651 skinnedLocatorPosConstrFunc->clearConstraints();
652 skinnedLocatorPosConstrFunc->setConstraints(skinnedConstrData.at(iFrame));
653
654 // initialization
655 if (needsInit) { // solve only for the rigid parameters as preprocessing
656 MT_LOGI_IF(
657 config.debug && isContinuous,
658 "Solving for an initial rigid pose at frame {}",
659 iFrame);
660
661 // Set up different config for initialization
662 solverOptions.maxIterations = 50; // make sure it converges
663 solver.setOptions(solverOptions);
664 solver.setEnabledParameters(pt.getRigidParameters());
665 if (smoothConstrFunc) {
666 smoothConstrFunc->setWeight(0.0); // turn off smoothing - it doesn't affect rigid dofs
667 }
668
669 solver.solve(dof);
670
671 // Recover solver config
672 solverOptions.maxIterations = config.maxIter;
673 solver.setOptions(solverOptions);
674 solver.setEnabledParameters(poseParams);
675 if (smoothConstrFunc) {
676 smoothConstrFunc->setWeight(config.smoothing);
677 }
678
679 needsInit = false;
680 }
681
682 // set smoothness target as the last pose for continuous tracking
683 if (smoothConstrFunc) {
684 smoothConstrFunc->setTargetParameters(dof, smoothConstrFunc->getTargetWeights());
685 }
686
687 priorError += solverFunc.getError(dof);
688 error += solver.solve(dof);
689 ++solverFrame;
690 }
691
692 // store result
693 while (outputIndex <= iFrame) {
694 outMotion.col(outputIndex++) = dof;
695 }
696 progress.increment();
697 }
698
699 while (outputIndex < numFrames) {
700 outMotion.col(outputIndex++) = dof;
701 }
702 }
703 if (config.debug) {
704 if (solverFrame > 0) {
705 MT_LOGI("Pre optimization residual: {}", priorError / solverFrame);
706 MT_LOGI("Average per-frame residual: {}", error / solverFrame);
707 } else {
708 MT_LOGW("no valid frames to solve");
709 }
710 }
711 return outMotion;
712}

733 const ParameterTransform& pt = character.parameterTransform;
734
735 // Create a model parameters vector with scale and shape, but zero pose
736 ModelParameters neutralParams = identity;
737
738 // Zero out all pose parameters
739 const ParameterSet poseParams = pt.getPoseParameters();
740 for (size_t i = 0; i < neutralParams.v.size(); ++i) {
741 if (poseParams.test(i)) {
742 neutralParams.v[i] = 0.0f;
743 }
744 }
745
746 // Create skeleton state with the neutral (zero pose) parameters
747 const auto jointParams = pt.apply(neutralParams.v);
748 SkeletonState state;
749 state.set(jointParams, character.skeleton, false);
750
751 // Apply blend shape skinning to get the deformed mesh
752 Mesh skinnedMesh;
753 skinWithBlendShapes(character, state, neutralParams, skinnedMesh);
754
755 // Find min and max Y values across all vertices
756 if (skinnedMesh.vertices.empty()) {
757 MT_LOGW("Skinned mesh has no vertices, cannot compute height");
758 return 0.0f;
759 }
760
761 float minY = std::numeric_limits<float>::max();
762 float maxY = std::numeric_limits<float>::lowest();
763
764 for (const auto& vertex : skinnedMesh.vertices) {
765 minY = std::min(minY, vertex.y());
766 maxY = std::max(maxY, vertex.y());
767 }
768
769 const float height = maxY - minY;
770 return height;
771}

790void calibrateModel(

792 const CalibrationConfig& config,
793 Character& character,
794 ModelParameters& identity,
795 const std::array<float, 3>& regularizerWeights) {
796 MT_CHECK(
797 identity.v.size() == character.parameterTransform.numAllModelParameters(),

799 identity.v.size(),
800 character.parameterTransform.numAllModelParameters());

808 Character solvingCharacter = character;

814 // Extended quantities are for the solvingCharacter, which includes locators as bones
815 // w/o Extended quantities are for character with fixed locators

819 ParameterSet locatorSet = transformExtended.getParameterSet("locators", true) |
820 transformExtended.getParameterSet("skinnedLocators", true);

831 calibBodySet.set(paramIndex);

833 calibBodySetExtended = transformExtended.getScalingParameters();

841
842 // special trackingConfig for initialization: zero out smoothness and collision
843 TrackingConfig trackingConfig{
844 {config.minVisPercent, config.lossAlpha, config.maxIter, config.regularization, config.debug},
845 0.0,
846 0.0};

849 // identity information will be initialized and updated in the motion matrix throughout all the
850 // solves.

852 std::vector<size_t> frameIndices;

861 MT_LOGI_IF(config.debug, "Pre-solving for the first frame");
862 std::vector<size_t> firstFrame;
863 firstFrame.push_back(0);
864 motion.topRows(transform.numAllModelParameters()) = trackPosesForFrames(
865 markerData,
866 character,
867 motion.topRows(transform.numAllModelParameters()),
868 trackingConfig,
869 firstFrame,
870 false); // Not continuous for calibration keyframes
871 motion.topRows(transform.numAllModelParameters()) = trackSequence(
872 markerData,
873 character,
874 calibBodySet,
875 motion.topRows(transform.numAllModelParameters()),
876 trackingConfig,
877 firstFrame,
878 regularizerWeights.at(0),
879 config.enforceFloorInFirstFrame,
880 config.firstFramePoseConstraintSet,
881 config.targetHeightCm); // still solving a subset
882 std::tie(identity.v, character.locators, character.skinnedLocators) =
883 extractIdAndLocatorsFromParams(motion.col(0), solvingCharacter, character);
884
885 // track sequence with selected stride. we need to sample at least config.calibFrames frames
886 // so make sure we have a stride that allows that
887 size_t sampleStride = std::min(numFrames / config.calibFrames, config.greedySampling);
888
889 motion.topRows(transform.numAllModelParameters()) =
890 trackPosesPerframe(markerData, character, identity, trackingConfig, sampleStride);
891
892 const ParameterSet ps = transformExtended.getPoseParameters() &
893 ~transformExtended.getRigidParameters() & ~locatorSet;
894 frameIndices = sampleFrames(
895 character,
896 motion.topRows(transform.numAllModelParameters()),
897 markerData,
898 ps,
899 sampleStride,
900 config.calibFrames);
901
902 // then solve for identity and poses with fixed locators, initialized with solved poses
903 // this works using "character" because additional parameters for the locators are appended at
904 // the end, so the indices work out using topRows() without special treatment.
905 motion.topRows(transform.numAllModelParameters()) = trackSequence(
906 markerData,
907 character,
908 calibBodySet, // only solve for identity and not markers
909 motion.topRows(transform.numAllModelParameters()),
910 trackingConfig,
911 frameIndices,
912 regularizerWeights.at(0), // note: ideally allow large change at initialization with 0
913 // or low regularization weight
914 config.enforceFloorInFirstFrame,
915 config.firstFramePoseConstraintSet,
916 config.targetHeightCm);
917 } else {
918 motion.topRows(transform.numAllModelParameters()) =
919 trackPosesPerframe(markerData, character, identity, trackingConfig, frameStride);
920
921 // then solve for identity and poses with fixed locators, initialized with solved poses
922 // this works using "character" because additional parameters for the locators are appended at
923 // the end, so the indices work out using topRows() without special treatment.
924 motion.topRows(transform.numAllModelParameters()) = trackSequence(
925 markerData,
926 character,
927 calibBodySet, // only solve for identity and not markers
928 motion.topRows(transform.numAllModelParameters()),
929 trackingConfig,
930 regularizerWeights.at(0), // note: ideally allow large change at initialization with 0
931 // or low regularization weight
932 frameStride,
933 config.enforceFloorInFirstFrame,
934 config.firstFramePoseConstraintSet,
935 config.targetHeightCm);
936 }
937 }
938
939 // Solve everything together for a few iterations
940 for (size_t iIter = 0; iIter < config.majorIter; ++iIter) {
941 MT_LOGI_IF(config.debug, "Iteration {} of calibration", iIter);
942
943 if (config.greedySampling > 0) {
944 motion = trackSequence(
945 markerData,
946 solvingCharacter,
947 locatorSet | calibBodySetExtended,
948 motion,
949 trackingConfig,
950 frameIndices,
951 regularizerWeights.at(
952 1), // note: ideally use a small regularization to prevent too large a change
953 config.enforceFloorInFirstFrame,
954 config.firstFramePoseConstraintSet,
955 config.targetHeightCm); // still solving a subset
956 } else {
957 motion = trackSequence(
958 markerData,
959 solvingCharacter,
960 locatorSet | calibBodySetExtended,
961 motion,
962 trackingConfig,
963 regularizerWeights.at(
964 1), // note: ideally use a small regularization to prevent too large a change
965 frameStride,
966 config.enforceFloorInFirstFrame,
967 config.firstFramePoseConstraintSet,
968 config.targetHeightCm); // still solving a subset
969 }
970 // extract solving results to identity and character so we can pass them to trackPosesPerframe
971 // below.
972 std::tie(identity.v, character.locators, character.skinnedLocators) =
973 extractIdAndLocatorsFromParams(motion.col(0), solvingCharacter, character);
974 if (config.calibShape && solvingCharacter.blendShape && character.mesh) {
975 auto blendShapeParams =
976 extractBlendWeights(solvingCharacter.parameterTransform, ModelParameters(motion.col(0)));
977 MT_LOGI("Solved for blend shape coeffs: {}", blendShapeParams.v.transpose());

979 }
980
981 // The sequence solve above could get stuck with euler singularity but per-frame solve could get
982 // it out. Pass in the first frame from previous solve as a better initial guess than the zero
983 // pose.
984 if (config.greedySampling > 0) {
985 motion.topRows(transform.numAllModelParameters()) = trackPosesForFrames(
986 markerData,
987 character,
988 motion.topRows(transform.numAllModelParameters()),
989 trackingConfig,
990 frameIndices,
991 false); // Not continuous for calibration keyframes
992 } else {
993 const VectorXf initPose = motion.col(0).head(transform.numAllModelParameters());
994 motion.topRows(transform.numAllModelParameters()) =
995 trackPosesPerframe(markerData, character, initPose, trackingConfig, frameStride);
996 }
997 }
998
999 // Finally, fine tune marker offsets with fix identity.
1000 MT_LOGI_IF(config.debug, "Fine-tune marker offsets");
1001
1002 if (config.greedySampling > 0) {
1003 motion = trackSequence(
1004 markerData,
1005 solvingCharacter,
1006 locatorSet,
1007 motion,
1008 trackingConfig,
1009 frameIndices,
1010 regularizerWeights.at(
1011 2), // note: ideally use a higher regularizer weight to prevent too large a change
1012 config.enforceFloorInFirstFrame,
1013 config.firstFramePoseConstraintSet,
1014 config.targetHeightCm);
1015 } else {
1016 motion = trackSequence(
1017 markerData,
1018 solvingCharacter,
1019 locatorSet,
1020 motion,
1021 trackingConfig,
1022 regularizerWeights.at(
1023 2), // note: ideally use a higher regularizer weight to prevent too large a change
1024 frameStride,
1025 config.enforceFloorInFirstFrame,
1026 config.firstFramePoseConstraintSet,
1027 config.targetHeightCm);
1028 }
1029 std::tie(identity.v, character.locators, character.skinnedLocators) =
1030 extractIdAndLocatorsFromParams(motion.col(0), solvingCharacter, character);
1031
1032 // TODO: A hack to return the solved first frame as initialization for tracking later.
1033 identity.v = motion.col(0).head(transform.numAllModelParameters());
1034
1035 // Log the calibrated character height
1036 const float height = computeCharacterHeight(character, identity);
1037 MT_LOGI("Calibrated character height: {:.4f} cm", height);
1038}

1221std::pair<float, float> getLocatorError(
1222 std::span<const std::vector<momentum::Marker>> markerData,
1223 const MatrixXf& motion,
1224 momentum::Character& character) {
1225 const size_t numFrames = markerData.size();
1226 MT_CHECK(numFrames > 0, "Input data is empty.");
1227 MT_CHECK(
1228 motion.rows() == character.parameterTransform.numAllModelParameters(),
1229 "Input motion parameters {} do not match character model parameters {}",
1230 motion.rows(),
1231 character.parameterTransform.numAllModelParameters());
1232
1233 const ParameterTransform& pt = character.parameterTransform;
1234
1235 // map locator name to its index
1236 std::map<std::string, size_t> locatorLookup;
1237 for (size_t i = 0; i < character.locators.size(); i++) {
1238 locatorLookup[character.locators[i].name] = i;
1239 }
1240
1241 // map skinned locator name to its index
1242 std::unordered_map<std::string, size_t> skinnedLocatorLookup;
1243 for (size_t i = 0; i < character.skinnedLocators.size(); i++) {
1244 skinnedLocatorLookup[character.skinnedLocators[i].name] = i;
1245 }
1246
1247 SkeletonState state;
1248
1249 // go over all frames and pose the locators and compute the error
1250 double error = 0.0;
1251 double maxError = 0.0;
1252 size_t frameNum = 0.0;
1253 std::string markerName = "";
1254 for (size_t iFrame = 0; iFrame < numFrames; ++iFrame) {
1255 const auto jointParams = pt.apply(motion.col(iFrame));
1256 state.set(jointParams, character.skeleton, false);
1257
1258 double frameError = 0.0;
1259
1260 const auto& markerList = markerData[iFrame];
1261 size_t validMarkers = 0;
1262 for (const auto& jMarker : markerList) {
1263 if (jMarker.occluded) {
1264 continue;
1265 }
1266
1267 std::optional<Vector3f> locatorPos;
1268
1269 // First try regular locators
1270 auto query = locatorLookup.find(jMarker.name);
1271 if (query != locatorLookup.end()) {
1272 size_t locatorIdx = query->second;
1273 if (locatorIdx < character.locators.size()) {
1274 const auto& locator = character.locators[locatorIdx];
1275 if (locator.parent < state.jointState.size()) {
1276 locatorPos = state.jointState[locator.parent].transform * locator.offset;
1277 }
1278 }
1279 }
1280
1281 // If not found, try skinned locators
1282 if (!locatorPos.has_value()) {
1283 auto skinnedQuery = skinnedLocatorLookup.find(jMarker.name);
1284 if (skinnedQuery != skinnedLocatorLookup.end()) {
1285 size_t locatorIdx = skinnedQuery->second;
1286 if (locatorIdx < character.skinnedLocators.size()) {
1287 const auto& skinnedLocator = character.skinnedLocators[locatorIdx];
1288 locatorPos = getSkinnedLocatorPosition(
1289 skinnedLocator, skinnedLocator.position, character.inverseBindPose, state);
1290 }
1291 }
1292 }
1293
1294 if (!locatorPos.has_value()) {
1295 continue;
1296 }
1297
1298 const Vector3f diff = *locatorPos - jMarker.pos.cast<float>();
1299 const float markerError = diff.norm();
1300 frameError += markerError;
1301 if (markerError > maxError) {
1302 maxError = markerError;
1303 frameNum = iFrame;
1304 markerName = jMarker.name;
1305 }
1306 validMarkers++;
1307 }
1308
1309 if (validMarkers > 0) {
1310 error += frameError / validMarkers;
1311 }
1312 }
1313 MT_LOGI("Max marker error: {} at frame {} for marker {}", maxError, frameNum, markerName);
1314 return {static_cast<float>(error / numFrames), static_cast<float>(maxError)};
1315}
