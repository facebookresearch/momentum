51std::vector<size_t> sampleFrames(
52 momentum::Character& character,
53 const MatrixXf& initialMotion,
54 gsl::span<const std::vector<momentum::Marker>> markerData,
55 const ParameterSet& parameters,
56 const size_t frameStride,
57 const size_t numSamples) {
58 // sample frames so that we get the most variance from the initial input
59 const auto numFrames = static_cast<size_t>(initialMotion.cols());
60 size_t solvedFrames = (numFrames - 1) / frameStride + 1;
61
62 std::vector<size_t> frameIndices;
63 if (solvedFrames == 0) {
64 return frameIndices;
65 }
66 const size_t numActualSamples = std::min(numSamples, solvedFrames);
67
68 // get the indices of the parameters to be used
69 std::vector<size_t> usedParameters;
70 for (size_t i = 0; i < static_cast<size_t>(initialMotion.rows()); ++i) {
71 if (parameters.test(i)) {
72 usedParameters.push_back(i);
73 }
74 }
75
76 // calculate per frame error
77 const ParameterTransform& pt = character.parameterTransform;
78
79 // map locator name to its index
80 std::map<std::string, size_t> locatorLookup;
81 for (size_t i = 0; i < character.locators.size(); i++) {
82 locatorLookup[character.locators[i].name] = i;
83 }
84
85 std::vector<double> frameErrors(solvedFrames, 0.0f);
86 SkeletonState state;
87 for (size_t iFrame = 0, fi = 0; iFrame < numFrames; iFrame += frameStride, fi++) {
88 const auto jointParams = pt.apply(initialMotion.col(iFrame));
89 state.set(jointParams, character.skeleton, false);
90
91 const auto& markerList = markerData[iFrame];
92 for (const auto& jMarker : markerList) {
93 if (jMarker.occluded) {
94 continue;
95 }
96 auto query = locatorLookup.find(jMarker.name);
97 if (query == locatorLookup.end()) {
98 continue;
99 }
100 size_t locatorIdx = query->second;
101 if (locatorIdx >= character.locators.size()) {
102 continue;
103 }
104 const auto& locator = character.locators[locatorIdx];
105 if (locator.parent >= state.jointState.size()) {
106 continue;
107 }
108 const Vector3f locatorPos = state.jointState[locator.parent].transform * locator.offset;
109 const Vector3f diff = locatorPos - jMarker.pos.cast<float>();
110 const float markerError = diff.norm();
111 frameErrors[fi] += markerError;
112 }
113 }
114 std::vector<double> sortedErrors = frameErrors;
115 std::sort(sortedErrors.begin(), sortedErrors.end());
116
117 // do not use the worst 1/4 of the fitted errors, there's likely to be some outliers and
118 // errors due to the initialization
119 const double threshold = sortedErrors[(sortedErrors.size() * 3) / 4];
120
121 // get the motion of only the used parameters and then normalize
122 // by the mean and variance
123 MatrixXf normalized;
124 {
125 const MatrixXf subMotion =
126 initialMotion(usedParameters, Eigen::seq(0, numFrames - 1, frameStride));
127 const VectorXf mean = subMotion.rowwise().mean();
128 const MatrixXf centered = subMotion.colwise() - mean;
129 const VectorXf std = centered.array().square().rowwise().sum() / (numFrames - 1);
130
131 // normalize the motion by the sqrt of the variance, i.e. leave a little bit of scale in
132 normalized = centered.array().colwise() / (std.array().sqrt().sqrt().cwiseMax(1e-5f));
133 }
134
135 for (size_t i = 0; i < frameErrors.size(); ++i) {
136 if (frameErrors[i] > threshold) {
137 normalized.col(i) = VectorXf::Ones(normalized.rows()) * 1000.0f;
138 }
139 }
140
141 // calculate the distances from each frame to the already selected indices
142 frameIndices.push_back(0);
143 VectorXf distances = VectorXf::Zero(solvedFrames);
144 for (size_t fi = 0; fi < solvedFrames; fi++) {
145 distances[fi] = (normalized.col(frameIndices[0]) - normalized.col(fi)).norm();
146 }
147
148 // finally add additional samples with the largest distance
149 for (size_t i = frameIndices.size(); i < numActualSamples; ++i) {
150 size_t maxFrame = 0;
151 const float maxval = distances.maxCoeff(&maxFrame);
152 if (maxval < 1e-5f) {
153 break;
154 }
155 frameIndices.push_back(maxFrame);
156
157 for (size_t fi = 0; fi < solvedFrames; fi++) {
158 const float dist = (normalized.col(maxFrame) - normalized.col(fi)).cwiseAbs().maxCoeff();
159 distances[fi] = std::min(distances[fi], dist);
160 }
161 }
162
163 for (auto&& fi : frameIndices) {
164 fi *= frameStride;
165 }
166
167 return frameIndices;
168}

233Eigen::MatrixXf trackSequence(
234 const gsl::span<const std::vector<Marker>> markerData,
235 const Character& character,
236 const ParameterSet& globalParams,
237 const MatrixXf& initialMotion,
238 const TrackingConfig& config,
239 const std::vector<size_t>& frames,
240 float regularizer,
241 bool enforceFloorInFirstFrame,
242 const std::string& firstFramePoseConstraintSet) {
243 // sanity checks
244 const size_t numFrames = markerData.size();
245 MT_CHECK(numFrames > 0, "Input data is empty.");
246 MT_CHECK(
247 initialMotion.cols() >= numFrames,
248 "Number of frames in data {} doesn't match that of input motion {}",
249 numFrames,
250 initialMotion.cols());
251 MT_CHECK(
252 initialMotion.rows() == character.parameterTransform.numAllModelParameters(),
253 "Input motion parameters {} do not match character model parameters {}",
254 initialMotion.rows(),
255 character.parameterTransform.numAllModelParameters());
256
257 const ParameterTransform& pt = character.parameterTransform;
258 const size_t numMarkers = markerData[0].size();
259
260 // universal parameters include "scaling" and "locators" (if exists); pose parameters need to
261 // exclude "locators". universalParams is to indicate to the solver which parameters are "global"
262 // (ie. not time varying). The input globalParams indicate which parameters within universalParams
263 // we want to solve for. globalParams is either a subset or all of universalParams.
264 ParameterSet poseParams = pt.getPoseParameters();
265 ParameterSet universalParams = pt.getScalingParameters() | pt.getBlendShapeParameters();
266 const auto locatorSet =
267 pt.getParameterSet("locators", true) | pt.getParameterSet("skinnedLocators", true);
268 poseParams &= ~locatorSet;
269 universalParams |= locatorSet;
270
271 std::vector<momentum::SkinnedLocatorTriangleConstraintT<float>> skinnedLocatorMeshContraints;
272 if ((globalParams & pt.getBlendShapeParameters()).any() && !character.skinnedLocators.empty()) {
273 skinnedLocatorMeshContraints = createSkinnedLocatorMeshConstraints(character, 1.0f);
274 }
275
276 // set up the solver function
277 std::vector<size_t> sortedFrames = frames;
278 std::sort(sortedFrames.begin(), sortedFrames.end());
279 size_t solvedFrames = sortedFrames.size();
280 auto solverFunc = SequenceSolverFunction(
281 character, character.parameterTransform, universalParams, solvedFrames);
282
283 // floor penetration constraints; we assume the world is y-up and floor is y=0 for mocap data.
284 const auto& floorConstraints = createFloorConstraints<float>(
285 "Floor_",
286 character.locators,
287 Vector3f::UnitY(),
288 /* y offset */ 0.0f,
289 /* weight */ 5.0f);
290
291 // marker constraints
292 const auto constrData = createConstraintData(markerData, character.locators);
293 const auto skinnedConstData = createSkinnedConstraintData(markerData, character.skinnedLocators);
294
295 // add per-frame constraint data to the solver
296 for (size_t solverFrame = 0; solverFrame < solvedFrames; ++solverFrame) {
297 const size_t& iFrame = sortedFrames[solverFrame];
298 if ((constrData.at(iFrame).size() + skinnedConstData.at(iFrame).size()) >
299 numMarkers * config.minVisPercent) {
300 auto posConstrWeight = PositionErrorFunction::kLegacyWeight;
301 if (solverFrame == 0 && (enforceFloorInFirstFrame || !firstFramePoseConstraintSet.empty())) {
302 posConstrWeight *= solvedFrames;
303 }
304
305 // prepare positional constraints
306 if (!constrData.at(iFrame).empty()) {
307 auto posConstrFunc = std::make_shared<PositionErrorFunction>(character, config.lossAlpha);
308 posConstrFunc->setConstraints(constrData.at(iFrame));
309 posConstrFunc->setWeight(posConstrWeight);
310 solverFunc.addErrorFunction(solverFrame, posConstrFunc);
311 }
312
313 if (!skinnedConstData.at(iFrame).empty()) {
314 auto skinnedConstrFunc = std::make_shared<SkinnedLocatorErrorFunction>(character);
315 skinnedConstrFunc->setConstraints(skinnedConstData.at(iFrame));
316 skinnedConstrFunc->setWeight(posConstrWeight);
317 solverFunc.addErrorFunction(solverFrame, skinnedConstrFunc);
318 }
319
320 if (!skinnedLocatorMeshContraints.empty() && solverFrame == 0) {
321 // Need to create a function to keep the skinned constraints close to
322 // the mesh.
323 auto skinnedTriangleConstrFunc =
324 std::make_shared<SkinnedLocatorTriangleErrorFunctionT<float>>(character);
325 skinnedTriangleConstrFunc->setConstraints(skinnedLocatorMeshContraints);
326 skinnedTriangleConstrFunc->setWeight(solvedFrames * posConstrWeight);
327 solverFunc.addErrorFunction(solverFrame, skinnedTriangleConstrFunc);
328 }
329
330 if (pt.getBlendShapeParameters().any() && solverFrame == 0) {
331 // regularize the blend shape parameters
332 auto blendShapeConstrFunc = std::make_shared<ModelParametersErrorFunction>(character);
333 blendShapeConstrFunc->setWeight(solvedFrames * 0.1f);
334 Eigen::VectorXf weights = Eigen::VectorXf::Zero(pt.numAllModelParameters());
335 for (Eigen::Index i = 0; i < pt.blendShapeParameters.size(); ++i) {
336 if (pt.blendShapeParameters[i] >= 0) {
337 weights[pt.blendShapeParameters[i]] = 1.0f;
338 }
339 }
340 blendShapeConstrFunc->setTargetParameters(
341 ModelParameters::Zero(pt.numAllModelParameters()), weights);
342 solverFunc.addErrorFunction(solverFrame, blendShapeConstrFunc);
343 }
344
345 // prepare floor constraints
346 if (!floorConstraints.empty()) {
347 bool halfPlane = true;
348 float weightMultiplier = 1.0f;
349 if (enforceFloorInFirstFrame && solverFrame == 0) {
350 halfPlane = false;
351 weightMultiplier = solvedFrames;
352 }
353 auto halfPlaneConstrFunc =
354 std::make_shared<PlaneErrorFunction>(character, /*half plane*/ halfPlane);
355 halfPlaneConstrFunc->setConstraints(floorConstraints);
356 halfPlaneConstrFunc->setWeight(PlaneErrorFunction::kLegacyWeight * weightMultiplier);
357 solverFunc.addErrorFunction(solverFrame, halfPlaneConstrFunc);
358 }
359
360 // add pose constraint set if defined
361 if (!firstFramePoseConstraintSet.empty() && solverFrame == 0) {
362 // check if the constraint set is defined
363 if (character.parameterTransform.poseConstraints.count(firstFramePoseConstraintSet) > 0) {
364 // add parameter limits
365 const auto poseLimits = getPoseConstraintParameterLimits(
366 firstFramePoseConstraintSet, character.parameterTransform);
367 auto limitConstrFunc = std::make_shared<LimitErrorFunction>(character, poseLimits);
368 limitConstrFunc->setWeight(solvedFrames);
369 solverFunc.addErrorFunction(solverFrame, limitConstrFunc);
370 }
371 }
372 }
373 // Set per-frame initial value
374 solverFunc.setFrameParameters(solverFrame, initialMotion.col(iFrame));
375 }
376
377 // add parameter limits
378 auto limitConstrFunc = std::make_shared<LimitErrorFunction>(character);
379 limitConstrFunc->setWeight(0.1);
380 solverFunc.addErrorFunction(kAllFrames, limitConstrFunc);
381
382 // add collision error
383 if (config.collisionErrorWeight != 0 && character.collision != nullptr) {
384 auto collisionConstrFunc = std::make_shared<CollisionErrorFunctionStateless>(character);
385 collisionConstrFunc->setWeight(config.collisionErrorWeight);
386 solverFunc.addErrorFunction(kAllFrames, collisionConstrFunc);
387 }
388
389 // add a smoothness constraint in parameter space
390 if (config.smoothing != 0) {
391 auto smoothConstrFunc = std::make_shared<ModelParametersSequenceErrorFunction>(character);
392 smoothConstrFunc->setWeight(config.smoothing);
393 solverFunc.addSequenceErrorFunction(kAllFrames, smoothConstrFunc);
394 }
395
396 // minimize the change to global params
397 if (globalParams.count() > 0 && regularizer != 0) {
398 auto regularizerFunc = std::make_shared<ModelParametersErrorFunction>(character);
399 Eigen::VectorXf universalMask(pt.numAllModelParameters());
400 for (size_t i = 0; i < universalMask.size(); ++i) {
401 if (globalParams.test(i)) {
402 universalMask[i] = regularizer;
403 } else {
404 universalMask[i] = 0.0;
405 }
406 }
407 regularizerFunc->setTargetParameters(initialMotion.col(0), universalMask);
408 // Sufficient to add to the first frame since it won't change.
409 solverFunc.addErrorFunction(0, regularizerFunc);
410 }
411
412 // solver configration
413 SequenceSolverOptions solverOptions;
414 solverOptions.maxIterations = config.maxIter;
415 solverOptions.minIterations = 2;
416 solverOptions.progressBar = config.debug;
417 solverOptions.doLineSearch = false;
418 solverOptions.multithreaded = true;
419 solverOptions.verbose = config.debug;
420 solverOptions.threshold = 1.f;
421 solverOptions.regularization = config.regularization;
422
423 // solve the problem
424 SequenceSolver solver = SequenceSolver(solverOptions, &solverFunc);
425 solver.setEnabledParameters(poseParams | globalParams);
426 // returns all the dofs with initial values nicely packed into a vector
427 VectorXf dofs = solverFunc.getJoinedParameterVector();
428 solver.solve(dofs);
429 double error = solverFunc.getError(dofs);
430 MT_LOGI_IF(config.debug, "Solver residual: {}", error);
431
432 // set results to output
433 size_t sortedIndex = 0;
434 MatrixXf outMotion(pt.numAllModelParameters(), numFrames);
435 for (size_t fi = 0; fi < numFrames; fi++) {
436 if (sortedIndex < sortedFrames.size() - 1 && fi == sortedFrames[sortedIndex + 1]) {
437 sortedIndex++;
438 }
439 outMotion.col(fi) = solverFunc.getFrameParameters(sortedIndex).v;
440 }
441 return outMotion;
442}

457Eigen::MatrixXf trackPosesPerframe(
458 const gsl::span<const std::vector<Marker>> markerData,
459 const Character& character,
460 const ModelParameters& globalParams,
461 const TrackingConfig& config,
462 const size_t frameStride) {
463 const size_t numFrames = markerData.size();
464 MT_CHECK(numFrames > 0, "Input data is empty.");
465 MT_CHECK(
466 globalParams.v.size() == character.parameterTransform.numAllModelParameters(),
467 "Input model parameters {} do not match character model parameters {}",
468 globalParams.v.size(),
469 character.parameterTransform.numAllModelParameters());
470
471 const ParameterTransform& pt = character.parameterTransform;
472 const size_t numMarkers = markerData[0].size();
473
474 // pose parameters need to exclude "locators"
475 ParameterSet poseParams = pt.getPoseParameters();
476 const auto& locatorSet = pt.parameterSets.find("locators");
477 if (locatorSet != pt.parameterSets.end()) {
478 poseParams &= ~locatorSet->second;
479 }
480
481 // set up the solver
482 auto solverFunc = SkeletonSolverFunction(character, pt);
483 GaussNewtonSolverOptions solverOptions;
484 solverOptions.maxIterations = config.maxIter;
485 solverOptions.minIterations = 2;
486 solverOptions.doLineSearch = false;
487 solverOptions.verbose = config.debug;
488 solverOptions.threshold = 1.f;
489 solverOptions.regularization = config.regularization;
490 auto solver = GaussNewtonSolver(solverOptions, &solverFunc);
491 solver.setEnabledParameters(poseParams);
492
493 // parameter limits constraint
494 auto limitConstrFunc = std::make_shared<LimitErrorFunction>(character);
495 limitConstrFunc->setWeight(0.1);
496 solverFunc.addErrorFunction(limitConstrFunc);
497
498 // positional constraint function for markers
499 auto posConstrFunc = std::make_shared<PositionErrorFunction>(character, config.lossAlpha);
500 posConstrFunc->setWeight(PositionErrorFunction::kLegacyWeight);
501 solverFunc.addErrorFunction(posConstrFunc);
502
503 std::shared_ptr<SkinnedLocatorErrorFunction> skinnedLocatorPosConstrFunc =
504 std::make_shared<SkinnedLocatorErrorFunction>(character);
505 skinnedLocatorPosConstrFunc->setWeight(PositionErrorFunction::kLegacyWeight);
506 solverFunc.addErrorFunction(skinnedLocatorPosConstrFunc);
507
508 // floor penetration constraint data; we assume the world is y-up and floor is y=0 for mocap data.
509 const auto& floorConstraints = createFloorConstraints<float>(
510 "Floor_",
511 character.locators,
512 Vector3f::UnitY(),
513 /* y offset */ 0.0f,
514 /* weight */ 5.0f);
515 auto halfPlaneConstrFunc = std::make_shared<PlaneErrorFunction>(character, /*half plane*/ true);
516 halfPlaneConstrFunc->setConstraints(floorConstraints);
517 halfPlaneConstrFunc->setWeight(PlaneErrorFunction::kLegacyWeight);
518 solverFunc.addErrorFunction(halfPlaneConstrFunc);
519
520 // marker constraint data
521 auto constrData = createConstraintData(markerData, character.locators);
522 auto skinnedConstrData = createSkinnedConstraintData(markerData, character.skinnedLocators);
523
524 // smoothness constraint only for the joints and exclude global dofs because the global transform
525 // needs to be accurate (may not matter in practice?)
526 auto smoothConstrFunc = std::make_shared<ModelParametersErrorFunction>(
527 character, poseParams & ~pt.getRigidParameters());
528 smoothConstrFunc->setWeight(config.smoothing);
529 solverFunc.addErrorFunction(smoothConstrFunc);
530
531 // add collision error
532 std::shared_ptr<CollisionErrorFunction> collisionErrorFunction;
533 if (config.collisionErrorWeight != 0 && character.collision != nullptr) {
534 collisionErrorFunction = std::make_shared<CollisionErrorFunction>(character);
535 collisionErrorFunction->setWeight(config.collisionErrorWeight);
536 solverFunc.addErrorFunction(collisionErrorFunction);
537 }
538
539 MatrixXf motion(pt.numAllModelParameters(), numFrames);
540 // initialize parameters to contain identity information
541 // the identity fields will be used but untouched during optimization
542 // globalParams could also be repurposed to pass in initial pose value
543 Eigen::VectorXf dof = globalParams.v;
544 size_t solverFrame = 0;
545 double error = 0.0;
546 // Use the initial global transform is it's not zero
547 bool needsInit = dof.head(6).isZero(0); // TODO: assume first six dofs are global dofs
548
549 // When the frames are not continuous, we sometimes run into an issue when the desired joint
550 // rotation between two consecutive frames is large (eg. larger than 180). If we initialize from
551 // the previous result, the smaller rotation will be wrongly chosen, and we cannot recover from
552 // this mistake. To prevent this, we will solve each frame completely independently when they are
553 // not continuous.
554 bool continuous = (frameStride < 5);
555 if (!continuous) {
556 needsInit = true;
557 }
558
559 { // scope the ProgressBar so it returns
560 ProgressBar progress("", numFrames);
561 for (size_t iFrame = 0; iFrame < numFrames; iFrame += frameStride) {
562 // reinitialize if not continuous
563 if (!continuous) {
564 dof = globalParams.v;
565 }
566
567 if ((constrData.at(iFrame).size() + skinnedConstrData.at(iFrame).size()) >
568 config.minVisPercent * numMarkers) {
569 // add positional constraints
570 posConstrFunc->clearConstraints(); // clear constraint data from the previous frame
571 posConstrFunc->setConstraints(constrData.at(iFrame));
572
573 skinnedLocatorPosConstrFunc->clearConstraints();
574 skinnedLocatorPosConstrFunc->setConstraints(skinnedConstrData.at(iFrame));
575
576 // initialization
577 // TODO: run on first frame or tracking failure
578 if (needsInit) { // solve only for the rigid parameters as preprocessing
579 MT_LOGI_IF(
580 config.debug && continuous, "Solving for an initial rigid pose at frame {}", iFrame);
581
582 // Set up different config for initialization
583 solverOptions.maxIterations = 50; // make sure it converges
584 solver.setOptions(solverOptions);
585 solver.setEnabledParameters(pt.getRigidParameters());
586 smoothConstrFunc->setWeight(0.0); // turn off smoothing - it doesn't affect rigid dofs
587
588 solver.solve(dof);
589
590 // Recover solver config
591 solverOptions.maxIterations = config.maxIter;
592 solver.setOptions(solverOptions);
593 solver.setEnabledParameters(poseParams);
594 smoothConstrFunc->setWeight(config.smoothing);
595
596 if (continuous) {
597 needsInit = false;
598 }
599 }
600
601 // set smoothness target as the last pose -- dof holds parameter values from last (good)
602 // frame it will serve as a small regularization to rest pose for the first frame
603 // TODO: API needs improvement
604 smoothConstrFunc->setTargetParameters(dof, smoothConstrFunc->getTargetWeights());
605
606 error += solver.solve(dof);
607 ++solverFrame;
608 }
609
610 // set result to output; fill in frames within a stride
611 // note that dof contains complete parameter info with identity
612 for (size_t jDelta = 0; jDelta < frameStride && iFrame + jDelta < numFrames; ++jDelta) {
613 motion.col(iFrame + jDelta) = dof;
614 }
615 progress.increment(frameStride);
616 }
617 }
618 if (config.debug) {
619 if (solverFrame > 0) {
620 MT_LOGI("Average per-frame residual: {}", error / solverFrame);
621 } else {
622 MT_LOGW("no valid frames to solve");
623 }
624 }
625 return motion;
626}

641Eigen::MatrixXf trackPosesForFrames(
642 const gsl::span<const std::vector<Marker>> markerData,
643 const Character& character,
644 const MatrixXf& initialMotion,
645 const TrackingConfig& config,
646 const std::vector<size_t>& frameIndices) {
647 const size_t numFrames = markerData.size();
648 MT_CHECK(numFrames > 0, "Input data is empty.");
649 MT_CHECK(
650 initialMotion.rows() == character.parameterTransform.numAllModelParameters(),
651 "Input motion parameters {} do not match character model parameters {}",
652 initialMotion.rows(),
653 character.parameterTransform.numAllModelParameters());
654
655 const ParameterTransform& pt = character.parameterTransform;
656 const size_t numMarkers = markerData[0].size();
657
658 std::vector<size_t> sortedFrames = frameIndices;
659 std::sort(sortedFrames.begin(), sortedFrames.end());
660
661 // pose parameters need to exclude "locators"
662 ParameterSet poseParams = pt.getPoseParameters();
663 const auto& locatorSet = pt.parameterSets.find("locators");
664 if (locatorSet != pt.parameterSets.end()) {
665 poseParams &= ~locatorSet->second;
666 }
667
668 // set up the solver
669 auto solverFunc = SkeletonSolverFunction(character, pt);
670 GaussNewtonSolverOptions solverOptions;
671 solverOptions.maxIterations = config.maxIter;
672 solverOptions.minIterations = 2;
673 solverOptions.doLineSearch = false;
674 solverOptions.verbose = config.debug;
675 solverOptions.threshold = 1.f;
676 solverOptions.regularization = config.regularization;
677 auto solver = GaussNewtonSolver(solverOptions, &solverFunc);
678 solver.setEnabledParameters(poseParams);
679
680 // parameter limits constraint
681 auto limitConstrFunc = std::make_shared<LimitErrorFunction>(character);
682 limitConstrFunc->setWeight(0.1);
683 solverFunc.addErrorFunction(limitConstrFunc);
684
685 // positional constraint function for markers
686 auto posConstrFunc = std::make_shared<PositionErrorFunction>(character, config.lossAlpha);
687 posConstrFunc->setWeight(PositionErrorFunction::kLegacyWeight);
688 solverFunc.addErrorFunction(posConstrFunc);
689
690 auto skinnedLocatorPosConstrFunc = std::make_shared<SkinnedLocatorErrorFunction>(character);
691 skinnedLocatorPosConstrFunc->setWeight(PositionErrorFunction::kLegacyWeight);
692 solverFunc.addErrorFunction(skinnedLocatorPosConstrFunc);
693
694 // floor penetration constraint data; we assume the world is y-up and floor is y=0 for mocap data.
695 const auto& floorConstraints = createFloorConstraints<float>(
696 "Floor_",
697 character.locators,
698 Vector3f::UnitY(),
699 /* y offset */ 0.0f,
700 /* weight */ 5.0f);
701 auto halfPlaneConstrFunc = std::make_shared<PlaneErrorFunction>(character, /*half plane*/ true);
702 halfPlaneConstrFunc->setConstraints(floorConstraints);
703 halfPlaneConstrFunc->setWeight(PlaneErrorFunction::kLegacyWeight);
704 solverFunc.addErrorFunction(halfPlaneConstrFunc);
705
706 // marker constraint data
707 auto constrData = createConstraintData(markerData, character.locators);
708 auto skinnedConstrData = createSkinnedConstraintData(markerData, character.skinnedLocators);
709
710 // add collision error
711 std::shared_ptr<CollisionErrorFunction> collisionErrorFunction;
712 if (config.collisionErrorWeight != 0 && character.collision != nullptr) {
713 collisionErrorFunction = std::make_shared<CollisionErrorFunction>(character);
714 collisionErrorFunction->setWeight(config.collisionErrorWeight);
715 solverFunc.addErrorFunction(collisionErrorFunction);
716 }
717
718 // initialize parameters to contain identity information
719 // the identity fields will be used but untouched during optimization
720 // globalParams could also be repurposed to pass in initial pose value
721 std::vector<Eigen::VectorXf> poses(frameIndices.size());
722 Eigen::VectorXf dof = initialMotion.col(sortedFrames.empty() ? 0 : sortedFrames[0]);
723 size_t solverFrame = 0;
724 double priorError = 0.0;
725 double error = 0.0;
726
727 MatrixXf outMotion(pt.numAllModelParameters(), numFrames);
728 { // scope the ProgressBar so it returns
729 ProgressBar progress("", sortedFrames.size());
730 for (size_t fi = 0; fi < sortedFrames.size(); fi++) {
731 const size_t& iFrame = sortedFrames[fi];
732 dof = initialMotion.col(iFrame);
733
734 if ((constrData.at(iFrame).size() + skinnedConstrData.at(iFrame).size()) >
735 numMarkers * config.minVisPercent) {
736 // add positional constraints
737 posConstrFunc->clearConstraints(); // clear constraint data from the previous frame
738 posConstrFunc->setConstraints(constrData.at(iFrame));
739
740 skinnedLocatorPosConstrFunc->clearConstraints();
741 skinnedLocatorPosConstrFunc->setConstraints(skinnedConstrData.at(iFrame));
742
743 // initialization
744 solverOptions.maxIterations = 50; // make sure it converges
745 solver.setOptions(solverOptions);
746 solver.setEnabledParameters(pt.getRigidParameters());
747
748 solver.solve(dof);
749
750 // Recover solver config
751 solverOptions.maxIterations = config.maxIter;
752 solver.setOptions(solverOptions);
753 solver.setEnabledParameters(poseParams);
754
755 priorError += solverFunc.getError(dof);
756 error += solver.solve(dof);
757 ++solverFrame;
758 }
759
760 // store result
761 poses[fi] = dof;
762 progress.increment();
763 }
764
765 // set results to output
766 size_t sortedIndex = 0;
767 for (size_t fi = 0; fi < numFrames; fi++) {
768 if (sortedIndex < sortedFrames.size() - 1 && fi == sortedFrames[sortedIndex + 1]) {
769 sortedIndex++;
770 }
771 outMotion.col(fi) = poses[sortedIndex];
772 }
773 }
774 if (config.debug) {
775 if (solverFrame > 0) {
776 MT_LOGI("Pre optimization residual: {}", priorError / solverFrame);
777 MT_LOGI("Average per-frame residual: {}", error / solverFrame);
778 } else {
779 MT_LOGW("no valid frames to solve");
780 }
781 }
782 return outMotion;
783}

812void calibrateModel(
813 const gsl::span<const std::vector<Marker>> markerData,
814 const CalibrationConfig& config,
815 Character& character,
816 ModelParameters& identity) {
817 MT_CHECK(
818 identity.v.size() == character.parameterTransform.numAllModelParameters(),
819 "Input identity parameters {} do not match character parameters {}",
820 identity.v.size(),
821 character.parameterTransform.numAllModelParameters());
822
823 const size_t numFrames = markerData.size();
824 // uniformly sample frames for calibration
825 size_t frameStride = (numFrames - 1) / config.calibFrames;
826 frameStride = std::max(size_t(1), frameStride);
827
828 // create a solving character with markers as bones
829 Character solvingCharacter = character;
830 solvingCharacter = createLocatorCharacter(solvingCharacter, "locator_");
831 if (!solvingCharacter.skinnedLocators.empty()) {
832 solvingCharacter = addSkinnedLocatorParametersToTransform(solvingCharacter);
833 }
834
835 // Extended quantities are for the solvingCharacter, which includes locators as bones
836 // w/o Extended quantities are for character with fixed locators
837 const ParameterTransform& transformExtended = solvingCharacter.parameterTransform;
838 const ParameterTransform& transform = character.parameterTransform;
839
840 ParameterSet locatorSet = transformExtended.getParameterSet("locators", true) |
841 transformExtended.getParameterSet("skinnedLocators", true);
842 ParameterSet calibBodySetExtended;
843 ParameterSet calibBodySet;
844 if (config.globalScaleOnly) {
845 calibBodySetExtended.set(transformExtended.getParameterIdByName("scale_global"));
846 calibBodySet.set(transform.getParameterIdByName("scale_global"));
847 } else {
848 calibBodySetExtended = transformExtended.getScalingParameters();
849 calibBodySet = transform.getScalingParameters();
850
851 if (config.calibShape) {
852 calibBodySetExtended |= transformExtended.getBlendShapeParameters();
853 calibBodySet |= transform.getBlendShapeParameters();
854 }
855 }
856
857 // special trackingConfig for initialization: zero out smoothness and collision
858 TrackingConfig trackingConfig{
859 {config.minVisPercent, config.lossAlpha, config.maxIter, config.regularization, config.debug},
860 0.0,
861 0.0};
862
863 // only keep one motion; no need to duplicate.
864 // identity information will be initialized and updated in the motion matrix throughout all the
865 // solves.
866 MatrixXf motion = MatrixXf::Zero(transformExtended.numAllModelParameters(), numFrames);
867 std::vector<size_t> frameIndices;
868
869 { // Initialization
870 MT_LOGI_IF(config.debug, "Solving for an initial pose and skeleton");
871
872 // first solve for initial tracking poses with fixed identity and locators to default
873 // Because we are solving for poses only, use character to save compute.
874 if (config.greedySampling > 0) {
875 // first only track the first frame
876 MT_LOGI_IF(config.debug, "Pre-solving for the first frame");
877 std::vector<size_t> firstFrame;
878 firstFrame.push_back(0);
879 motion.topRows(transform.numAllModelParameters()) = trackPosesForFrames(
880 markerData,
881 character,
882 motion.topRows(transform.numAllModelParameters()),
883 trackingConfig,
884 firstFrame);
885 motion.topRows(transform.numAllModelParameters()) = trackSequence(
886 markerData,
887 character,
888 calibBodySet,
889 motion.topRows(transform.numAllModelParameters()),
890 trackingConfig,
891 firstFrame,
892 0.0,
893 config.enforceFloorInFirstFrame,
894 config.firstFramePoseConstraintSet); // still solving a subset
895 std::tie(identity.v, character.locators, character.skinnedLocators) =
896 extractIdAndLocatorsFromParams(motion.col(0), solvingCharacter, character);
897
898 // track sequence with selected stride. we need to sample at least config.calibFrames frames
899 // so make sure we have a stride that allows that
900 size_t sampleStride = std::min(numFrames / config.calibFrames, config.greedySampling);
901
902 motion.topRows(transform.numAllModelParameters()) =
903 trackPosesPerframe(markerData, character, identity, trackingConfig, sampleStride);
904
905 const ParameterSet ps = transformExtended.getPoseParameters() &
906 ~transformExtended.getRigidParameters() & ~locatorSet;
907 frameIndices = sampleFrames(
908 character,
909 motion.topRows(transform.numAllModelParameters()),
910 markerData,
911 ps,
912 sampleStride,
913 config.calibFrames);
914
915 // then solve for identity and poses with fixed locators, initialized with solved poses
916 // this works using "character" because additional parameters for the locators are appended at
917 // the end, so the indices work out using topRows() without special treatment.
918 motion.topRows(transform.numAllModelParameters()) = trackSequence(
919 markerData,
920 character,
921 calibBodySet, // only solve for identity and not markers
922 motion.topRows(transform.numAllModelParameters()),
923 trackingConfig,
924 frameIndices,
925 0.0,
926 /*regularizer*/ // allow large change at initialization without any regularization
927 config.enforceFloorInFirstFrame,
928 config.firstFramePoseConstraintSet);
929 } else {
930 motion.topRows(transform.numAllModelParameters()) =
931 trackPosesPerframe(markerData, character, identity, trackingConfig, frameStride);
932
933 // then solve for identity and poses with fixed locators, initialized with solved poses
934 // this works using "character" because additional parameters for the locators are appended at
935 // the end, so the indices work out using topRows() without special treatment.
936 motion.topRows(transform.numAllModelParameters()) = trackSequence(
937 markerData,
938 character,
939 calibBodySet, // only solve for identity and not markers
940 motion.topRows(transform.numAllModelParameters()),
941 trackingConfig,
942 0.0 /*regularizer*/, // allow large change at initialization without any regularization
943 frameStride,
944 config.enforceFloorInFirstFrame,
945 config.firstFramePoseConstraintSet);
946 }
947 }
948
949 // Solve everything together for a few iterations
950 for (size_t iIter = 0; iIter < config.majorIter; ++iIter) {
951 MT_LOGI_IF(config.debug, "Iteration {} of calibration", iIter);
952
953 if (config.greedySampling > 0) {
954 motion = trackSequence(
955 markerData,
956 solvingCharacter,
957 locatorSet | calibBodySetExtended,
958 motion,
959 trackingConfig,
960 frameIndices,
961 0.0, // TODO: use a small regularization to prevent too large a change
962 config.enforceFloorInFirstFrame,
963 config.firstFramePoseConstraintSet); // still solving a subset
964 } else {
965 motion = trackSequence(
966 markerData,
967 solvingCharacter,
968 locatorSet | calibBodySetExtended,
969 motion,
970 trackingConfig,
971 0.0, // TODO: use a small regularization to prevent too large a change
972 frameStride,
973 config.enforceFloorInFirstFrame,
974 config.firstFramePoseConstraintSet); // still solving a subset
975 }
976 // extract solving results to identity and character so we can pass them to trackPosesPerframe
977 // below.
978 std::tie(identity.v, character.locators, character.skinnedLocators) =
979 extractIdAndLocatorsFromParams(motion.col(0), solvingCharacter, character);
980 if (config.calibShape && solvingCharacter.blendShape && character.mesh) {
981 *character.mesh = extractBlendShapeFromParams(motion.col(0), solvingCharacter);
982 }
983
984 // The sequence solve above could get stuck with euler singularity but per-frame solve could get
985 // it out. Pass in the first frame from previous solve as a better initial guess than the zero
986 // pose.
987 if (config.greedySampling > 0) {
988 motion.topRows(transform.numAllModelParameters()) = trackPosesForFrames(
989 markerData,
990 character,
991 motion.topRows(transform.numAllModelParameters()),
992 trackingConfig,
993 frameIndices);
994 } else {
995 const VectorXf initPose = motion.col(0).head(transform.numAllModelParameters());
996 motion.topRows(transform.numAllModelParameters()) =
997 trackPosesPerframe(markerData, character, initPose, trackingConfig, frameStride);
998 }
999 }
1000
1001 // Finally, fine tune marker offsets with fix identity.
1002 MT_LOGI_IF(config.debug, "Fine-tune marker offsets");
1003
1004 // TODO: use a larger regularizer to prevent too large a change.
1005 if (config.greedySampling > 0) {
1006 motion = trackSequence(
1007 markerData,
1008 solvingCharacter,
1009 locatorSet,
1010 motion,
1011 trackingConfig,
1012 frameIndices,
1013 0.0,
1014 config.enforceFloorInFirstFrame,
1015 config.firstFramePoseConstraintSet);
1016 } else {
1017 motion = trackSequence(
1018 markerData,
1019 solvingCharacter,
1020 locatorSet,
1021 motion,
1022 trackingConfig,
1023 0.0,
1024 frameStride,
1025 config.enforceFloorInFirstFrame,
1026 config.firstFramePoseConstraintSet);
1027 }
1028 std::tie(identity.v, character.locators, character.skinnedLocators) =
1029 extractIdAndLocatorsFromParams(motion.col(0), solvingCharacter, character);
1030
1031 // TODO: A hack to return the solved first frame as initialization for tracking later.
1032 identity.v = motion.col(0).head(transform.numAllModelParameters());
1033}

1213std::pair<float, float> getLocatorError(
1214 gsl::span<const std::vector<momentum::Marker>> markerData,
1215 const MatrixXf& motion,
1216 momentum::Character& character) {
1217 const size_t numFrames = markerData.size();
1218 MT_CHECK(numFrames > 0, "Input data is empty.");
1219 MT_CHECK(
1220 motion.rows() == character.parameterTransform.numAllModelParameters(),
1221 "Input motion parameters {} do not match character model parameters {}",
1222 motion.rows(),
1223 character.parameterTransform.numAllModelParameters());
1224
1225 const ParameterTransform& pt = character.parameterTransform;
1226
1227 // map locator name to its index
1228 std::map<std::string, size_t> locatorLookup;
1229 for (size_t i = 0; i < character.locators.size(); i++) {
1230 locatorLookup[character.locators[i].name] = i;
1231 }
1232
1233 SkeletonState state;
1234
1235 // go over all frames and pose the locators and compute the error
1236 double error = 0.0;
1237 double maxError = 0.0;
1238 size_t frameNum = 0.0;
1239 std::string markerName = "";
1240 for (size_t iFrame = 0; iFrame < numFrames; ++iFrame) {
1241 const auto jointParams = pt.apply(motion.col(iFrame));
1242 state.set(jointParams, character.skeleton, false);
1243
1244 double frameError = 0.0;
1245
1246 const auto& markerList = markerData[iFrame];
1247 size_t validMarkers = 0;
1248 for (const auto& jMarker : markerList) {
1249 if (jMarker.occluded) {
1250 continue;
1251 }
1252 auto query = locatorLookup.find(jMarker.name);
1253 if (query == locatorLookup.end()) {
1254 continue;
1255 }
1256 size_t locatorIdx = query->second;
1257 if (locatorIdx >= character.locators.size()) {
1258 continue;
1259 }
1260 const auto& locator = character.locators[locatorIdx];
1261 if (locator.parent >= state.jointState.size()) {
1262 continue;
1263 }
1264 const Vector3f locatorPos = state.jointState[locator.parent].transform * locator.offset;
1265 const Vector3f diff = locatorPos - jMarker.pos.cast<float>();
1266 const float markerError = diff.norm();
1267 frameError += markerError;
1268 if (markerError > maxError) {
1269 maxError = markerError;
1270 frameNum = iFrame;
1271 markerName = jMarker.name;
1272 }
1273 validMarkers++;
1274 }
1275
1276 if (validMarkers > 0) {
1277 error += frameError / validMarkers;
1278 }
1279 }
1280 MT_LOGI("Max marker error: {} at frame {} for marker {}", maxError, frameNum, markerName);
1281 return {static_cast<float>(error / numFrames), static_cast<float>(maxError)};
1282}
