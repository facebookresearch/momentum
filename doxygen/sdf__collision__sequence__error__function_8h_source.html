205template <typename T, typename SdfColliderType>
206SDFCollisionSequenceErrorFunctionT<T, SdfColliderType>::SDFCollisionSequenceErrorFunctionT(
207 const Character& character,
208 const std::vector<SdfColliderType>& sdfColliders,
209 const std::vector<int>& participatingVertices,
210 const std::vector<T>& vertexWeights)
211 : SequenceErrorFunctionT<T>(character.skeleton, character.parameterTransform),
212 character_(character),
213 sdfColliders_(sdfColliders) {
214 MT_CHECK_NOTNULL(character_.mesh, "Character must have a mesh for SDF collision detection");
215 MT_CHECK_NOTNULL(
216 character_.skinWeights, "Character must have skin weights for SDF collision detection");
217
218 if (participatingVertices.empty()) {
219 participatingVertices_.resize(character_.mesh->vertices.size());
220 for (int i = 0; i < static_cast<int>(character_.mesh->vertices.size()); ++i) {
221 participatingVertices_[i] = i;
222 }
223 } else {
224 participatingVertices_ = participatingVertices;
225 }
226
227 for (const int vertexIndex : participatingVertices_) {
228 MT_CHECK(
229 vertexIndex >= 0 && vertexIndex < static_cast<int>(character_.mesh->vertices.size()),
230 "Invalid vertex index: {} (mesh has {} vertices)",
231 vertexIndex,
232 character_.mesh->vertices.size());
233 }
234
235 if (vertexWeights.empty()) {
236 vertexWeights_.assign(participatingVertices_.size(), T(1));
237 } else {
238 MT_CHECK(
239 vertexWeights.size() == participatingVertices_.size(),
240 "Weight count ({}) must match vertex count ({})",
241 vertexWeights.size(),
242 participatingVertices_.size());
243 vertexWeights_ = vertexWeights;
244 }

247template <typename T, typename SdfColliderType>
248T SDFCollisionSequenceErrorFunctionT<T, SdfColliderType>::refineCrossing(
249 const Eigen::Vector3<T>& localPos0,
250 const Eigen::Vector3<T>& delta,
251 const SdfColliderType& collider,
252 T sLo,
253 T sHi,
254 T dLo) const {
255 T lo = sLo;
256 T hi = sHi;
257 for (int i = 0; i < kBisectionIters; ++i) {
258 const T mid = T(0.5) * (lo + hi);
259 const Eigen::Vector3<T> pos = localPos0 + mid * delta;
260 const T dMid = collider.evaluate(pos);
261 // Keep the half-bracket that still straddles the sign change.
262 if ((dMid < T(0)) == (dLo < T(0))) {
263 lo = mid;
264 dLo = dMid;
265 } else {
266 hi = mid;
267 }
268 }
269 return T(0.5) * (lo + hi);
270}

272template <typename T, typename SdfColliderType>
273typename SDFCollisionSequenceErrorFunctionT<T, SdfColliderType>::SweptIntervalsT
274SDFCollisionSequenceErrorFunctionT<T, SdfColliderType>::findPenetratingIntervals(
275 const Eigen::Vector3<T>& localPos0,
276 const Eigen::Vector3<T>& delta,
277 T segLen,
278 const SdfColliderType& collider) const {
279 SweptIntervalsT intervals;
280
281 // Floor on the marching step (in s) so a far-outside query still makes progress; the sphere-trace
282 // takes larger steps when the surface is distant and shrinks to |sdf| near it, so a surface is
283 // never stepped over regardless of this floor.
284 const T minStepS = T(1) / static_cast<T>(kMaxMarchSteps);
285
286 T s = T(0);
287 T dPrev = collider.evaluate(Eigen::Vector3<T>(localPos0));
288 T intervalStart = T(-1); // invalid until an entry crossing is seen
289 int steps = 0;
290 while (s < T(1) && steps < 2 * kMaxMarchSteps) {
291 ++steps;
292 const T stepS = std::max(minStepS, std::abs(dPrev) / segLen);
293 const T sNext = std::min(T(1), s + stepS);
294 const Eigen::Vector3<T> posNext = localPos0 + sNext * delta;
295 const T dNext = collider.evaluate(posNext);
296
297 if (dPrev >= T(0) && dNext < T(0)) {
298 // Entry crossing: vertex passes from outside to inside.
299 intervalStart = refineCrossing(localPos0, delta, collider, s, sNext, dPrev);
300 } else if (dPrev < T(0) && dNext >= T(0) && intervalStart >= T(0)) {
301 // Exit crossing closing a previously-opened interval.
302 const T sOut = refineCrossing(localPos0, delta, collider, s, sNext, dPrev);
303 intervals.data[intervals.count++] = {intervalStart, sOut};
304 intervalStart = T(-1);
305 if (intervals.count >= kMaxIntervals) {
306 break;
307 }
308 }
309
310 s = sNext;
311 dPrev = dNext;
312 }
313
314 return intervals;
315}

317template <typename T, typename SdfColliderType>
318T SDFCollisionSequenceErrorFunctionT<T, SdfColliderType>::findDeepestPoint(
319 const Eigen::Vector3<T>& localPos0,
320 const Eigen::Vector3<T>& delta,
321 const SweptIntervalT& interval,
322 const SdfColliderType& collider) const {
323 // d(s) dips negative between the two crossings, so d'(s) = gradLocal(s) . delta runs from
324 // negative (still descending) to positive (ascending); bisect for the d' = 0 minimum.
325 T lo = interval.sIn;
326 T hi = interval.sOut;
327 for (int i = 0; i < kBisectionIters; ++i) {
328 const T mid = T(0.5) * (lo + hi);
329 const Eigen::Vector3<T> pos = localPos0 + mid * delta;
330 const auto [d, g] = collider.evaluateWithGradient(pos);
331 if (g.dot(delta) < T(0)) {
332 lo = mid; // d still decreasing: minimum is to the right
333 } else {
334 hi = mid; // d increasing: minimum is to the left
335 }
336 }
337 return T(0.5) * (lo + hi);
338}

340template <typename T, typename SdfColliderType>
341bool SDFCollisionSequenceErrorFunctionT<T, SdfColliderType>::sweptSegmentMayCollide(
342 const Eigen::Vector3<T>& localPos0,
343 const Eigen::Vector3<T>& localPos1,
344 const SdfColliderType& collider) const {
345 const auto bounds = collider.bounds();
346 const Eigen::Vector3<T> boundsMin = bounds.min().template cast<T>();
347 const Eigen::Vector3<T> boundsMax = bounds.max().template cast<T>();
348 const Eigen::Vector3<T> segMin = localPos0.cwiseMin(localPos1);
349 const Eigen::Vector3<T> segMax = localPos0.cwiseMax(localPos1);
350 return !(
351 (segMin.array() > boundsMax.array()).any() || (segMax.array() < boundsMin.array()).any());
352}

354template <typename T, typename SdfColliderType>
355double SDFCollisionSequenceErrorFunctionT<T, SdfColliderType>::getError(
356 std::span<const ModelParametersT<T>> /* modelParameters */,
357 std::span<const SkeletonStateT<T>> skelStates,
358 std::span<const MeshStateT<T>> meshStates) const {
359 MT_PROFILE_FUNCTION();
360 MT_CHECK(skelStates.size() == 2);
361 MT_CHECK(meshStates.size() == 2);
362 MT_CHECK_NOTNULL(meshStates[0].posedMesh_);
363 MT_CHECK_NOTNULL(meshStates[1].posedMesh_);
364
365 double totalError = 0.0;
366
367 for (const auto& collider : sdfColliders_) {
368 if (!collider.isValid()) {
369 continue;
370 }
371 const auto worldToCollider0 =
372 detail_sdf_collision::computeWorldToColliderTransform(collider, skelStates[0]);
373 const auto worldToCollider1 =
374 detail_sdf_collision::computeWorldToColliderTransform(collider, skelStates[1]);
375
376 for (size_t i = 0; i < participatingVertices_.size(); ++i) {
377 const int vertexIndex = participatingVertices_[i];
378 const T vertexWeight = vertexWeights_[i];
379 const Eigen::Vector3<T> localPos0 =
380 worldToCollider0 * meshStates[0].posedMesh_->vertices[vertexIndex];
381 const Eigen::Vector3<T> localPos1 =
382 worldToCollider1 * meshStates[1].posedMesh_->vertices[vertexIndex];
383 const Eigen::Vector3<T> delta = localPos1 - localPos0;
384 const T segLen = delta.norm();
385 if (segLen < std::numeric_limits<T>::epsilon()) {
386 continue; // no motion: static penetration is the per-frame function's job
387 }
388 if (!sweptSegmentMayCollide(localPos0, localPos1, collider)) {
389 continue;
390 }
391
392 const auto intervals = findPenetratingIntervals(localPos0, delta, segLen, collider);
393 for (const auto& interval : intervals) {
394 const T sStar = findDeepestPoint(localPos0, delta, interval, collider);
395 const Eigen::Vector3<T> pos = localPos0 + sStar * delta;
396 const T d = collider.evaluate(pos);
397 if (d >= T(0)) {
398 continue;
399 }
400 const T phi = -d;
401 totalError +=
402 static_cast<double>(vertexWeight * phi * phi * kSDFCollisionWeight * this->weight_);
403 }
404 }
405 }
406
407 return totalError;
408}

410template <typename T, typename SdfColliderType>
411double SDFCollisionSequenceErrorFunctionT<T, SdfColliderType>::getGradient(
412 std::span<const ModelParametersT<T>> /* modelParameters */,
413 std::span<const SkeletonStateT<T>> skelStates,
414 std::span<const MeshStateT<T>> meshStates,
415 Eigen::Ref<Eigen::VectorX<T>> gradient) const {
416 MT_PROFILE_FUNCTION();
417 MT_CHECK(skelStates.size() == 2);
418 MT_CHECK(meshStates.size() == 2);
419 MT_CHECK_NOTNULL(meshStates[0].posedMesh_);
420 MT_CHECK_NOTNULL(meshStates[0].restMesh_);
421 MT_CHECK_NOTNULL(meshStates[1].posedMesh_);
422 MT_CHECK_NOTNULL(meshStates[1].restMesh_);
423
424 const Eigen::Index np = this->parameterTransform_.numAllModelParameters();
425 double totalError = 0.0;
426
427 for (const auto& collider : sdfColliders_) {
428 if (!collider.isValid()) {
429 continue;
430 }
431 const std::array<TransformT<T>, 2> worldToCollider = {
432 detail_sdf_collision::computeWorldToColliderTransform(collider, skelStates[0]),
433 detail_sdf_collision::computeWorldToColliderTransform(collider, skelStates[1])};
434 const std::array<TransformT<T>, 2> colliderToWorld = {
435 worldToCollider[0].inverse(), worldToCollider[1].inverse()};
436
437 for (size_t i = 0; i < participatingVertices_.size(); ++i) {
438 const int vertexIndex = participatingVertices_[i];
439 const T vertexWeight = vertexWeights_[i];
440 const std::array<Eigen::Vector3<T>, 2> worldVertex = {
441 meshStates[0].posedMesh_->vertices[vertexIndex],
442 meshStates[1].posedMesh_->vertices[vertexIndex]};
443 const Eigen::Vector3<T> localPos0 = worldToCollider[0] * worldVertex[0];
444 const Eigen::Vector3<T> localPos1 = worldToCollider[1] * worldVertex[1];
445 const Eigen::Vector3<T> delta = localPos1 - localPos0;
446 const T segLen = delta.norm();
447 if (segLen < std::numeric_limits<T>::epsilon()) {
448 continue;
449 }
450 if (!sweptSegmentMayCollide(localPos0, localPos1, collider)) {
451 continue;
452 }
453
454 const size_t commonAncestor = detail_sdf_collision::findCommonAncestorForVertex(
455 *character_.skinWeights, this->skeleton_, vertexIndex, collider.parentJoint());
456
457 const auto intervals = findPenetratingIntervals(localPos0, delta, segLen, collider);
458 for (const auto& interval : intervals) {
459 const T sStar = findDeepestPoint(localPos0, delta, interval, collider);
460 const Eigen::Vector3<T> pos = localPos0 + sStar * delta;
461 const auto [d, localGradient] = collider.evaluateWithGradient(pos);
462 if (d >= T(0)) {
463 continue;
464 }
465 const T phi = -d;
466 totalError +=
467 static_cast<double>(vertexWeight * phi * phi * kSDFCollisionWeight * this->weight_);
468
469 // d(w*phi^2)/dp = 2*w*phi * d(phi)/dp; phi = -d(s*) with d'(s*) = 0, so the deepest point
470 // s* drifting with the parameters contributes nothing (envelope theorem).
471 const T wgt = T(2) * vertexWeight * kSDFCollisionWeight * this->weight_ * phi;
472
473 for (size_t f = 0; f < 2; ++f) {
474 const T alpha = (f == 0) ? (T(1) - sStar) : sStar;
475 const Eigen::Vector3<T> sdfGradient =
476 worldToCollider[f].scale * (colliderToWorld[f].rotation * localGradient);
477 Eigen::Ref<Eigen::VectorX<T>> gradBlock = gradient.segment(f * np, np);
478
479 SkinningWeightIteratorT<T> skinningIter(
480 character_, *meshStates[f].restMesh_, skelStates[f], vertexIndex);
481 while (!skinningIter.finished()) {
482 const auto [jointIndex, boneWeight, transformedVertex] = skinningIter.next();
483 if (jointIndex == commonAncestor) {
484 break;
485 }
486 if (std::abs(boneWeight) < std::numeric_limits<T>::epsilon()) {
487 continue;
488 }
489 const auto& jointState = skelStates[f].jointState[jointIndex];
490 const size_t paramIndex = jointIndex * kParametersPerJoint;
491 const Eigen::Vector3<T> posd = transformedVertex - jointState.translation();
492 detail_sdf_collision::accumulateJointGradient<T>(
493 jointState,
494 paramIndex,
495 posd,
496 -wgt * alpha * boneWeight * sdfGradient,
497 this->activeJointParams_,
498 this->parameterTransform_,
499 gradBlock);
500 }
501
502 if (collider.parentJoint() != kInvalidIndex) {
503 detail_sdf_collision::accumulateColliderHierarchyGradient<T>(
504 skelStates[f],
505 collider.parentJoint(),
506 commonAncestor,
507 worldVertex[f],
508 wgt * alpha * sdfGradient,
509 this->skeleton_,
510 this->activeJointParams_,
511 this->parameterTransform_,
512 gradBlock);
513 }
514 }
515 }
516 }
517 }
518
519 return totalError;
520}

522template <typename T, typename SdfColliderType>
523double SDFCollisionSequenceErrorFunctionT<T, SdfColliderType>::getJacobian(
524 std::span<const ModelParametersT<T>> /* modelParameters */,
525 std::span<const SkeletonStateT<T>> skelStates,
526 std::span<const MeshStateT<T>> meshStates,
527 Eigen::Ref<Eigen::MatrixX<T>> jacobian,
528 Eigen::Ref<Eigen::VectorX<T>> residual,
529 int& usedRows) const {
530 MT_PROFILE_FUNCTION();
531 MT_CHECK(skelStates.size() == 2);
532 MT_CHECK(meshStates.size() == 2);
533 MT_CHECK_NOTNULL(meshStates[0].posedMesh_);
534 MT_CHECK_NOTNULL(meshStates[0].restMesh_);
535 MT_CHECK_NOTNULL(meshStates[1].posedMesh_);
536 MT_CHECK_NOTNULL(meshStates[1].restMesh_);
537
538 const Eigen::Index np = this->parameterTransform_.numAllModelParameters();
539 MT_CHECK(jacobian.cols() == 2 * np, "Jacobian column count mismatch");
540
541 double totalError = 0.0;
542 int currentRow = 0;
543
544 for (const auto& collider : sdfColliders_) {
545 if (!collider.isValid()) {
546 continue;
547 }
548 const std::array<TransformT<T>, 2> worldToCollider = {
549 detail_sdf_collision::computeWorldToColliderTransform(collider, skelStates[0]),
550 detail_sdf_collision::computeWorldToColliderTransform(collider, skelStates[1])};
551 const std::array<TransformT<T>, 2> colliderToWorld = {
552 worldToCollider[0].inverse(), worldToCollider[1].inverse()};
553
554 for (size_t i = 0; i < participatingVertices_.size(); ++i) {
555 const int vertexIndex = participatingVertices_[i];
556 const T vertexWeight = vertexWeights_[i];
557 const std::array<Eigen::Vector3<T>, 2> worldVertex = {
558 meshStates[0].posedMesh_->vertices[vertexIndex],
559 meshStates[1].posedMesh_->vertices[vertexIndex]};
560 const Eigen::Vector3<T> localPos0 = worldToCollider[0] * worldVertex[0];
561 const Eigen::Vector3<T> localPos1 = worldToCollider[1] * worldVertex[1];
562 const Eigen::Vector3<T> delta = localPos1 - localPos0;
563 const T segLen = delta.norm();
564 if (segLen < std::numeric_limits<T>::epsilon()) {
565 continue;
566 }
567 if (!sweptSegmentMayCollide(localPos0, localPos1, collider)) {
568 continue;
569 }
570
571 const size_t commonAncestor = detail_sdf_collision::findCommonAncestorForVertex(
572 *character_.skinWeights, this->skeleton_, vertexIndex, collider.parentJoint());
573
574 const auto intervals = findPenetratingIntervals(localPos0, delta, segLen, collider);
575 for (const auto& interval : intervals) {
576 const T sStar = findDeepestPoint(localPos0, delta, interval, collider);
577 const Eigen::Vector3<T> pos = localPos0 + sStar * delta;
578 const auto [d, localGradient] = collider.evaluateWithGradient(pos);
579 if (d >= T(0)) {
580 continue;
581 }
582 const T phi = -d;
583 MT_CHECK(
584 currentRow < jacobian.rows() && currentRow < residual.rows(),
585 "Insufficient Jacobian/residual rows");
586
587 const T wgt = std::sqrt(vertexWeight * kSDFCollisionWeight * this->weight_);
588 residual[currentRow] = wgt * phi;
589 totalError +=
590 static_cast<double>(vertexWeight * phi * phi * kSDFCollisionWeight * this->weight_);
591 jacobian.row(currentRow).setZero();
592
593 for (size_t f = 0; f < 2; ++f) {
594 const T alpha = (f == 0) ? (T(1) - sStar) : sStar;
595 const Eigen::Vector3<T> sdfGradient =
596 worldToCollider[f].scale * (colliderToWorld[f].rotation * localGradient);
597
598 SkinningWeightIteratorT<T> skinningIter(
599 character_, *meshStates[f].restMesh_, skelStates[f], vertexIndex);
600 while (!skinningIter.finished()) {
601 const auto [jointIndex, boneWeight, transformedVertex] = skinningIter.next();
602 if (jointIndex == commonAncestor) {
603 break;
604 }
605 if (std::abs(boneWeight) < std::numeric_limits<T>::epsilon()) {
606 continue;
607 }
608 const auto& jointState = skelStates[f].jointState[jointIndex];
609 const size_t paramIndex = jointIndex * kParametersPerJoint;
610 const Eigen::Vector3<T> posd = transformedVertex - jointState.translation();
611 detail_sdf_collision::accumulateJointJacobian<T>(
612 jointState,
613 paramIndex,
614 posd,
615 -wgt * alpha * boneWeight * sdfGradient,
616 this->activeJointParams_,
617 this->parameterTransform_,
618 jacobian.block(currentRow, f * np, 1, np));
619 }
620
621 if (collider.parentJoint() != kInvalidIndex) {
622 detail_sdf_collision::accumulateColliderHierarchyJacobian<T>(
623 skelStates[f],
624 collider.parentJoint(),
625 commonAncestor,
626 worldVertex[f],
627 wgt * alpha * sdfGradient,
628 this->skeleton_,
629 this->activeJointParams_,
630 this->parameterTransform_,
631 jacobian.block(currentRow, f * np, 1, np));
632 }
633 }
634 ++currentRow;
635 }
636 }
637 }
638
639 usedRows = currentRow;
640 return totalError;
641}
