34[[nodiscard]] CharacterFormat parseCharacterFormat(const filesystem::path& filepath) noexcept {
35 std::string ext = filepath.extension().string();
36 if (ext.empty()) {
37 return CharacterFormat::Unknown;
38 }
39
40 std::transform(ext.begin(), ext.end(), ext.begin(), ::tolower);
41 if ((ext == ".glb") || (ext == ".gltf")) {
42 return CharacterFormat::Gltf;
43 } else if (ext == ".fbx") {
44 return CharacterFormat::Fbx;
45 } else if ((ext == ".usd") || (ext == ".usda") || (ext == ".usdc")) {
46 return CharacterFormat::Usd;
47 } else {
48 return CharacterFormat::Unknown;
49 }
50}

52[[nodiscard]] std::optional<Character> loadCharacterByFormat(
53 const CharacterFormat format,
54 const filesystem::path& filepath) {
55 if (format == CharacterFormat::Gltf) {
56 return loadGltfCharacter(filepath);
57 } else if (format == CharacterFormat::Fbx) {
58 return loadFbxCharacter(filepath, KeepLocators::Yes);
59 } else if (format == CharacterFormat::Usd) {
60#ifdef MOMENTUM_IO_WITH_USD
61 return loadUsdCharacter(filepath);
62#else
63 MT_THROW("USD format is not supported in this build");
64#endif
65 } else {
66 return {};
67 }

90Character loadFullCharacter(
91 const std::string& characterPath,
92 const std::string& parametersPath,
93 const std::string& locatorsPath) {
94 // Parse format
95 const auto format = parseCharacterFormat(characterPath);
96 MT_THROW_IF(
97 format == CharacterFormat::Unknown, "Unknown character format for path: {}", characterPath);
98
99 // Load character
100 auto character = loadCharacterByFormat(format, characterPath);
101 MT_THROW_IF(!character, "Failed to load buffered character");
102
103 // load parameter transform
104 if (!parametersPath.empty()) {
105 auto def = loadMomentumModel(parametersPath);
106 loadParameters(def, *character);
107 }
108
109 // Load locators
110 if (!locatorsPath.empty()) {
111 (*character).locators =
112 loadLocators(locatorsPath, (*character).skeleton, (*character).parameterTransform);
113 }
114
115 return character.value();
116}
