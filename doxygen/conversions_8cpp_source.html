16Unit toUnit(const std::string& unitStr) {
17 if (unitStr == "m" || unitStr == "M") {
18 return Unit::M;
19 } else if (unitStr == "dm" || unitStr == "DM") {
20 return Unit::DM;
21 } else if (unitStr == "cm" || unitStr == "CM") {
22 return Unit::CM;
23 } else if (unitStr == "mm" || unitStr == "MM") {
24 return Unit::MM;
25 } else {
26 return Unit::Unknown;
27 }
28}

48template <typename T>
49Vector3<T> toMomentumVector3(const Vector3<T>& vec, UpVector up, Unit unit) {
50 // Convert the input vector's units to centimeters
51 Vector3<T> vec_in_cm;
52 switch (unit) {
53 case Unit::M: // Meters
54 vec_in_cm = vec * (T)100;
55 break;
56 case Unit::DM: // Decimeter
57 vec_in_cm = vec * (T)10;
58 break;
59 case Unit::CM: // Centimeters
60 vec_in_cm = vec;
61 break;
62 case Unit::MM: // Millimeters
63 vec_in_cm = vec * (T)0.1;
64 break;
65 case Unit::Unknown: // Unknown units, default to centimeters
66 MT_LOGE(
67 "{}: Unknown unit '{}' found in the file. Use centimeters instead.",
68 __func__,
69 toString(unit));
70 vec_in_cm = vec;
71 break;
72 default:
73 vec_in_cm = vec;
74 break;
75 }
76
77 // Converting from the given UpVector coordinate system to the target coordinate system
78 Vector3<T> vec_in_momentum;
79 switch (up) {
80 case UpVector::X: // X-up
81 vec_in_momentum = Vector3<T>(vec_in_cm.y(), vec_in_cm.z(), vec_in_cm.x());
82 break;
83 case UpVector::Y: // Y-up
84 vec_in_momentum = Vector3<T>(vec_in_cm.x(), vec_in_cm.z(), -vec_in_cm.y());
85 break;
86 case UpVector::Z: // Z-up
87 vec_in_momentum = vec_in_cm;
88 break;
89 case UpVector::YNeg: // Negative Y-up
90 vec_in_momentum = Vector3<T>(vec_in_cm.x(), -vec_in_cm.z(), vec_in_cm.y());
91 break;
92 default:
93 vec_in_momentum = vec_in_cm;
94 break;
95 }
96
97 return vec_in_momentum;
98}
