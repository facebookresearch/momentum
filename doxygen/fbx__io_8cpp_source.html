44[[nodiscard]] ::fbxsdk::FbxAxisSystem::EUpVector toFbx(const FBXUpVector upVector) {
45 switch (upVector) {
46 case FBXUpVector::XAxis:
47 return ::fbxsdk::FbxAxisSystem::EUpVector::eXAxis;
48 case FBXUpVector::YAxis:
49 return ::fbxsdk::FbxAxisSystem::EUpVector::eYAxis;
50 case FBXUpVector::ZAxis:
51 return ::fbxsdk::FbxAxisSystem::EUpVector::eZAxis;
52 default:
53 MT_THROW("Unsupported up vector");
54 }
55}
56
57[[nodiscard]] ::fbxsdk::FbxAxisSystem::EFrontVector toFbx(const FBXFrontVector frontVector) {
58 switch (frontVector) {
59 case FBXFrontVector::ParityEven:
60 return ::fbxsdk::FbxAxisSystem::EFrontVector::eParityEven;
61 case FBXFrontVector::ParityOdd:
62 return ::fbxsdk::FbxAxisSystem::EFrontVector::eParityOdd;
63 default:
64 MT_THROW("Unsupported front vector");
65 }
66}
67
68[[nodiscard]] ::fbxsdk::FbxAxisSystem::ECoordSystem toFbx(const FBXCoordSystem coordSystem) {
69 switch (coordSystem) {
70 case FBXCoordSystem::RightHanded:
71 return ::fbxsdk::FbxAxisSystem::ECoordSystem::eRightHanded;
72 case FBXCoordSystem::LeftHanded:
73 return ::fbxsdk::FbxAxisSystem::ECoordSystem::eLeftHanded;
74 default:
75 MT_THROW("Unsupported coordinate system");
76 }
77}

81 toFbx(coordSystemInfo.upVector),

86void createLocatorNodes(
87 const Character& character,
88 ::fbxsdk::FbxScene* scene,
89 const std::vector<::fbxsdk::FbxNode*>& skeletonNodes) {
90 for (const auto& loc : character.locators) {
91 ::fbxsdk::FbxMarker* markerAttribute = ::fbxsdk::FbxMarker::Create(scene, loc.name.c_str());
92 markerAttribute->Look.Set(::fbxsdk::FbxMarker::ELook::eHardCross);
93
94 // create the node
95 ::fbxsdk::FbxNode* locatorNode = ::fbxsdk::FbxNode::Create(scene, loc.name.c_str());
96 locatorNode->SetNodeAttribute(markerAttribute);
97
98 // set translation offset
99 locatorNode->LclTranslation.Set(FbxVector4(loc.offset[0], loc.offset[1], loc.offset[2]));
100
101 // set parent if it has one
102 if (loc.parent != kInvalidIndex) {
103 skeletonNodes[loc.parent]->AddChild(locatorNode);
104 }
105 }

108void createCollisionGeometryNodes(
109 const Character& character,
110 ::fbxsdk::FbxScene* scene,
111 const std::vector<::fbxsdk::FbxNode*>& skeletonNodes) {
112 if (!character.collision) {
113 MT_LOGD(
114 "No collision geometry found in character, skipping creation of collision geometry nodes");
115 return;
116 }
117
118 const auto& collisions = *character.collision;
119 for (auto i = 0u; i < collisions.size(); ++i) {
120 const TaperedCapsule& collision = collisions[i];
121
122 ::fbxsdk::FbxNode* collisionNode =
123 ::fbxsdk::FbxNode::Create(scene, ("Collision " + std::to_string(i)).c_str());
124 auto* nullNodeAttr =
125 ::fbxsdk::FbxNull::Create(scene, "Null"); // TODO: Find a good node attribute
126 collisionNode->SetNodeAttribute(nullNodeAttr);
127
128 ::fbxsdk::FbxProperty::Create(collisionNode, ::fbxsdk::FbxBoolDT, "Col_Type").Set(true);
129 ::fbxsdk::FbxProperty::Create(collisionNode, ::fbxsdk::FbxFloatDT, "Length")
130 .Set(collision.length);
131 ::fbxsdk::FbxProperty::Create(collisionNode, ::fbxsdk::FbxFloatDT, "Rad_A")
132 .Set(collision.radius[0]);
133 ::fbxsdk::FbxProperty::Create(collisionNode, ::fbxsdk::FbxFloatDT, "Rad_B")
134 .Set(collision.radius[1]);
135
136 collisionNode->LclTranslation.Set(FbxVector4(
137 collision.transformation.translation.x(),
138 collision.transformation.translation.y(),
139 collision.transformation.translation.z()));
140 const Vector3f rot = rotationMatrixToEulerXYZ<float>(
141 collision.transformation.rotation.toRotationMatrix(), EulerConvention::Extrinsic);
142 collisionNode->LclRotation.Set(FbxDouble3(toDeg(rot.x()), toDeg(rot.y()), toDeg(rot.z())));
143 collisionNode->LclScaling.Set(FbxDouble3(1));
144
145 if (collision.parent != kInvalidIndex) {
146 skeletonNodes[collision.parent]->AddChild(collisionNode);
147 } else {
148 MT_LOGE("Found a collision node with no parent");
149 }
150 }

153void setFrameRate(::fbxsdk::FbxScene* scene, const double framerate) {
154 // enumerate common frame rates first, then resort to custom framerate
155 if (std::abs(framerate - 30.0) < 1e-6) {
156 scene->GetGlobalSettings().SetTimeMode(::fbxsdk::FbxTime::eFrames30);
157 } else if (std::abs(framerate - 24.0) < 1e-6) {
158 scene->GetGlobalSettings().SetTimeMode(::fbxsdk::FbxTime::eFrames24);
159 } else if (std::abs(framerate - 48.0) < 1e-6) {
160 scene->GetGlobalSettings().SetTimeMode(::fbxsdk::FbxTime::eFrames48);
161 } else if (std::abs(framerate - 50.0) < 1e-6) {
162 scene->GetGlobalSettings().SetTimeMode(::fbxsdk::FbxTime::eFrames50);
163 } else if (std::abs(framerate - 60.0) < 1e-6) {
164 scene->GetGlobalSettings().SetTimeMode(::fbxsdk::FbxTime::eFrames60);
165 } else if (std::abs(framerate - 72.0) < 1e-6) {
166 scene->GetGlobalSettings().SetTimeMode(::fbxsdk::FbxTime::eFrames72);
167 } else if (std::abs(framerate - 96.0) < 1e-6) {
168 scene->GetGlobalSettings().SetTimeMode(::fbxsdk::FbxTime::eFrames96);
169 } else if (std::abs(framerate - 100.0) < 1e-6) {
170 scene->GetGlobalSettings().SetTimeMode(::fbxsdk::FbxTime::eFrames100);
171 } else if (std::abs(framerate - 120.0) < 1e-6) {
172 scene->GetGlobalSettings().SetTimeMode(::fbxsdk::FbxTime::eFrames120);
173 } else if (std::abs(framerate - 1000.0) < 1e-6) {
174 scene->GetGlobalSettings().SetTimeMode(::fbxsdk::FbxTime::eFrames1000);
175 } else {
176 scene->GetGlobalSettings().SetTimeMode(::fbxsdk::FbxTime::eCustom);
177 scene->GetGlobalSettings().SetCustomFrameRate(framerate);
178 }
179}

181// jointValues: (numJointParameters x numFrames) matrix of joint values
182void createAnimationCurves(
183 const Character& character,
184 ::fbxsdk::FbxScene* scene,
185 const std::vector<::fbxsdk::FbxNode*>& skeletonNodes,
186 const MatrixXf& jointValues,
187 const double framerate,
188 const bool skipActiveJointParamCheck) {
189 // set the framerate
190 setFrameRate(scene, framerate);
191
192 const auto& aj = character.parameterTransform.activeJointParams;
193
194 // create animation stack
195 ::fbxsdk::FbxAnimStack* animStack =
196 ::fbxsdk::FbxAnimStack::Create(scene, "Skeleton Animation Stack");
197 ::fbxsdk::FbxAnimLayer* animBaseLayer = ::fbxsdk::FbxAnimLayer::Create(scene, "Layer0");
198 animStack->AddMember(animBaseLayer);
199
200 // create anim curves for each joint and store them in an array
201 std::vector<::fbxsdk::FbxAnimCurve*> animCurves(character.skeleton.joints.size() * 9, nullptr);
202 std::vector<size_t> animCurvesIndex;
203 for (size_t i = 0; i < character.skeleton.joints.size(); i++) {
204 const size_t jointIndex = i * kParametersPerJoint;
205 const size_t index = i * 9;
206 skeletonNodes[i]->LclTranslation.GetCurveNode(true);
207 if (skipActiveJointParamCheck || aj[jointIndex + 0]) {
208 animCurves[index + 0] = skeletonNodes[i]->LclTranslation.GetCurve(
209 animBaseLayer, FBXSDK_CURVENODE_COMPONENT_X, true);
210 animCurvesIndex.push_back(index + 0);
211 }
212 if (skipActiveJointParamCheck || aj[jointIndex + 1]) {
213 animCurves[index + 1] = skeletonNodes[i]->LclTranslation.GetCurve(
214 animBaseLayer, FBXSDK_CURVENODE_COMPONENT_Y, true);
215 animCurvesIndex.push_back(index + 1);
216 }
217 if (skipActiveJointParamCheck || aj[jointIndex + 2]) {
218 animCurves[index + 2] = skeletonNodes[i]->LclTranslation.GetCurve(
219 animBaseLayer, FBXSDK_CURVENODE_COMPONENT_Z, true);
220 animCurvesIndex.push_back(index + 2);
221 }
222 skeletonNodes[i]->LclRotation.GetCurveNode(true);
223 if (skipActiveJointParamCheck || aj[jointIndex + 3]) {
224 animCurves[index + 3] =
225 skeletonNodes[i]->LclRotation.GetCurve(animBaseLayer, FBXSDK_CURVENODE_COMPONENT_X, true);
226 animCurvesIndex.push_back(index + 3);
227 }
228 if (skipActiveJointParamCheck || aj[jointIndex + 4]) {
229 animCurves[index + 4] =
230 skeletonNodes[i]->LclRotation.GetCurve(animBaseLayer, FBXSDK_CURVENODE_COMPONENT_Y, true);
231 animCurvesIndex.push_back(index + 4);
232 }
233 if (skipActiveJointParamCheck || aj[jointIndex + 5]) {
234 animCurves[index + 5] =
235 skeletonNodes[i]->LclRotation.GetCurve(animBaseLayer, FBXSDK_CURVENODE_COMPONENT_Z, true);
236 animCurvesIndex.push_back(index + 5);
237 }
238 skeletonNodes[i]->LclScaling.GetCurveNode(true);
239 if (skipActiveJointParamCheck || aj[jointIndex + 6]) {
240 animCurves[index + 6] =
241 skeletonNodes[i]->LclScaling.GetCurve(animBaseLayer, FBXSDK_CURVENODE_COMPONENT_X, true);
242 animCurvesIndex.push_back(index + 6);
243 animCurves[index + 7] =
244 skeletonNodes[i]->LclScaling.GetCurve(animBaseLayer, FBXSDK_CURVENODE_COMPONENT_Y, true);
245 animCurvesIndex.push_back(index + 7);
246 animCurves[index + 8] =
247 skeletonNodes[i]->LclScaling.GetCurve(animBaseLayer, FBXSDK_CURVENODE_COMPONENT_Z, true);
248 animCurvesIndex.push_back(index + 8);
249 }
250 }
251
252 // calculate the actual motion and set the keyframes
253 ::fbxsdk::FbxTime time;
254 // now go over each animCurveIndex and generate the curve
255 for (const auto ai : animCurvesIndex) {
256 const size_t jointIndex = ai / 9;
257 const size_t jointOffset = ai % 9;
258 const size_t parameterIndex =
259 jointIndex * kParametersPerJoint + std::min(jointOffset, size_t(6));
260 if (!skipActiveJointParamCheck && aj[parameterIndex] == 0) {
261 continue;
262 }
263
264 animCurves[ai]->KeyModifyBegin();
265 for (size_t f = 0; f < jointValues.cols(); f++) {
266 // set keyframe time
267 time.SetSecondDouble(static_cast<double>(f) / framerate);
268
269 // get joint value
270 float jointVal = jointValues(parameterIndex, f);
271
272 // add translation offset for tx values
273 if (jointOffset < 3 && jointIndex < character.skeleton.joints.size()) {
274 jointVal += character.skeleton.joints[jointIndex].translationOffset[jointOffset];
275 }
276 // convert to degrees
277 else if (jointOffset >= 3 && jointOffset <= 5) {
278 jointVal = toDeg(jointVal);
279 }
280 // convert to non-exponential scaling
281 else {
282 jointVal = std::pow(2.0f, jointVal);
283 }
284
285 const auto keyIndex = animCurves[ai]->KeyAdd(time);
286 animCurves[ai]->KeySet(keyIndex, time, jointVal);
287 }
288 animCurves[ai]->KeyModifyEnd();
289 }
290}

292void saveFbxCommon(
293 const filesystem::path& filename,
294 const Character& character,
295 const MatrixXf& jointValues,
296 const double framerate,
297 const bool saveMesh,
298 const bool skipActiveJointParamCheck,
299 const FBXCoordSystemInfo& coordSystemInfo,
300 bool permissive) {
301 // ---------------------------------------------
302 // initialize FBX SDK and prepare for export
303 // ---------------------------------------------
304 auto* manager = ::fbxsdk::FbxManager::Create();
305 auto* ios = ::fbxsdk::FbxIOSettings::Create(manager, IOSROOT);
306 manager->SetIOSettings(ios);
307
308 // Create an exporter.
309 ::fbxsdk::FbxExporter* lExporter = ::fbxsdk::FbxExporter::Create(manager, "");
310
311 // Declare the path and filename of the file containing the scene.
312 // In this case, we are assuming the file is in the same directory as the executable.
313 // Going through string() because on windows, wchar_t (native filesystem path) are different from
314 // char https://en.cppreference.com/w/cpp/language/types This avoids a build error on windows
315 // only.
316 std::string sFilename = filename.string();
317 const char* lFilename = sFilename.c_str();
318
319 // Initialize the exporter.
320 bool lExportStatus = lExporter->Initialize(lFilename, -1, manager->GetIOSettings());
321
322 MT_THROW_IF(
323 !lExportStatus,
324 "Unable to initialize fbx exporter {}",
325 lExporter->GetStatus().GetErrorString());
326
327 // ---------------------------------------------
328 // create the scene
329 // ---------------------------------------------
330 ::fbxsdk::FbxScene* scene = ::fbxsdk::FbxScene::Create(manager, "momentum_scene");
331 ::fbxsdk::FbxNode* root = scene->GetRootNode();
332
333 // set the coordinate system
334 ::fbxsdk::FbxAxisSystem axis = toFbx(coordSystemInfo);
335 axis.ConvertScene(scene);
336
337 // ---------------------------------------------
338 // create the skeleton nodes
339 // ---------------------------------------------
340 std::vector<::fbxsdk::FbxNode*> skeletonNodes;
341 std::unordered_map<size_t, fbxsdk::FbxNode*> jointToNodeMap;
342
343 ::fbxsdk::FbxNode* skeletonRootNode = nullptr;
344
345 for (size_t i = 0; i < character.skeleton.joints.size(); i++) {
346 const auto& joint = character.skeleton.joints[i];
347
348 // create the node
349 ::fbxsdk::FbxNode* skeletonNode = ::fbxsdk::FbxNode::Create(scene, joint.name.c_str());
350 // create node attribute
351 ::fbxsdk::FbxSkeleton* skeletonAttribute =
352 ::fbxsdk::FbxSkeleton::Create(scene, joint.name.c_str());
353
354 if (joint.parent == kInvalidIndex) {
355 skeletonRootNode = skeletonNode;
356 skeletonAttribute->SetSkeletonType(::fbxsdk::FbxSkeleton::eRoot);
357 } else {
358 skeletonAttribute->SetSkeletonType(::fbxsdk::FbxSkeleton::eLimbNode);
359 }
360 skeletonNode->SetNodeAttribute(skeletonAttribute);
361 jointToNodeMap[i] = skeletonNode;
362
363 // set translation offset
364 skeletonNode->LclTranslation.Set(FbxDouble3(
365 joint.translationOffset[0], joint.translationOffset[1], joint.translationOffset[2]));
366
367 // set pre-rotation
368 const auto angles = rotationMatrixToEulerZYX(joint.preRotation.toRotationMatrix());
369 skeletonNode->SetPivotState(FbxNode::eSourcePivot, FbxNode::ePivotActive);
370 skeletonNode->SetRotationActive(true);
371 skeletonNode->SetPreRotation(
372 ::fbxsdk::FbxNode::eSourcePivot,
373 FbxDouble3(toDeg(angles[2]), toDeg(angles[1]), toDeg(angles[0])));
374
375 // add to list
376 skeletonNodes.emplace_back(skeletonNode);
377 }
378
379 // Second pass: handle the parenting, in case the parents are not in order
380 for (size_t i = 0; i < character.skeleton.joints.size(); i++) {
381 const auto& joint = character.skeleton.joints[i];
382
383 // set parent if it has one
384 auto* skeletonNode = jointToNodeMap[i];
385 if (joint.parent != kInvalidIndex) {
386 skeletonNodes[joint.parent]->AddChild(skeletonNode);
387 }
388 }
389
390 // ---------------------------------------------
391 // create the locator nodes
392 // ---------------------------------------------
393 createLocatorNodes(character, scene, skeletonNodes);
394
395 // ---------------------------------------------
396 // create the collision geometry nodes
397 // ---------------------------------------------
398 createCollisionGeometryNodes(character, scene, skeletonNodes);
399
400 // ---------------------------------------------
401 // create the mesh nodes
402 // ---------------------------------------------
403 if (saveMesh && character.mesh != nullptr) {
404 // Add the mesh
405 const int numVertices = character.mesh.get()->vertices.size();
406 const int numFaces = character.mesh.get()->faces.size();
407 ::fbxsdk::FbxNode* meshNode = ::fbxsdk::FbxNode::Create(scene, "body_mesh");
408 ::fbxsdk::FbxMesh* lMesh = ::fbxsdk::FbxMesh::Create(scene, "mesh");
409 lMesh->SetControlPointCount(numVertices);
410 lMesh->InitNormals(numVertices);
411 for (int i = 0; i < numVertices; i++) {
412 FbxVector4 point(
413 character.mesh.get()->vertices[i].x(),
414 character.mesh.get()->vertices[i].y(),
415 character.mesh.get()->vertices[i].z());
416 FbxVector4 normal(
417 character.mesh.get()->normals[i].x(),
418 character.mesh.get()->normals[i].y(),
419 character.mesh.get()->normals[i].z());
420 lMesh->SetControlPointAt(point, normal, i);
421 }
422 // Add polygons to lMesh
423 for (int iFace = 0; iFace < numFaces; iFace++) {
424 lMesh->BeginPolygon();
425 for (int i = 0; i < 3; i++) { // We have tris for models. This could be extended for
426 // supporting Quads or npoly if needed.
427 lMesh->AddPolygon(character.mesh.get()->faces[iFace][i]);
428 }
429 lMesh->EndPolygon();
430 }
431 lMesh->BuildMeshEdgeArray();
432 meshNode->SetNodeAttribute(lMesh);
433
434 // ---------------------------------------------
435 // add texture coordinates
436 // ---------------------------------------------
437 if (!character.mesh->texcoords.empty()) {
438 const fbxsdk::FbxLayerElement::EType uvType = fbxsdk::FbxLayerElement::eTextureDiffuse;
439
440 // Initialize UV set and indices first. Both functions must be called before adding UVs
441 // and UV indices.
442 lMesh->InitTextureUV(0, uvType);
443 lMesh->InitTextureUVIndices(
444 ::fbxsdk::FbxLayerElement::EMappingMode::eByPolygonVertex, uvType);
445
446 // Add UVs
447 for (const auto& texcoords : character.mesh->texcoords) {
448 // flip y back to fbx convention - refer to reading code
449 lMesh->AddTextureUV(::fbxsdk::FbxVector2(texcoords[0], 1.0f - texcoords[1]), uvType);
450 }
451
452 // Set UV indices for each face. We only have triangles.
453 int faceCount = 0;
454 for (const auto& texcoords : character.mesh->texcoord_faces) {
455 lMesh->SetTextureUVIndex(faceCount, 0, texcoords[0], uvType);
456 lMesh->SetTextureUVIndex(faceCount, 1, texcoords[1], uvType);
457 lMesh->SetTextureUVIndex(faceCount++, 2, texcoords[2], uvType);
458 }
459 }
460
461 // ---------------------------------------------
462 // create the skinning
463 // ---------------------------------------------
464 // Add the mesh skinning
465 // Momentum skinning is saved in two matrices: index and weight (size numvertices x
466 // not-ordered-joints). The index contains the joint index and the weight is the normalized
467 // weight the vertex for that joint.
468
469 MT_THROW_IF(
470 !permissive && !character.skinWeights,
471 " Failed to save the character '{}' to {}. The character has no skinning weights and permissive mode is not enabled. Only mesh-only characters are allowed in permissive mode.",
472 character.name,
473 filename.string());
474
475 if (character.skinWeights != nullptr) {
476 ::fbxsdk::FbxSkin* fbxskin = ::fbxsdk::FbxSkin::Create(scene, "meshskinning");
477 fbxskin->SetSkinningType(::fbxsdk::FbxSkin::eLinear);
478 fbxskin->SetGeometry(lMesh);
479 FbxAMatrix meshTransform;
480 meshTransform.SetIdentity();
481 for (const auto& jointNode : jointToNodeMap) {
482 size_t jointIdx = jointNode.first;
483 auto* fbxJointNode = jointNode.second;
484
485 std::ostringstream s;
486 s << "skinningcluster_" << jointIdx;
487 FbxCluster* pCluster = ::fbxsdk::FbxCluster::Create(scene, s.str().c_str());
488 pCluster->SetLinkMode(::fbxsdk::FbxCluster::ELinkMode::eNormalize);
489 pCluster->SetLink(fbxJointNode);
490
491 ::fbxsdk::FbxAMatrix globalMatrix = fbxJointNode->EvaluateLocalTransform();
492 ::fbxsdk::FbxNode* pParent = fbxJointNode->GetParent();
493 // TODO: should use inverse bind transform from character instead.
494 while (pParent != nullptr) {
495 globalMatrix = pParent->EvaluateLocalTransform() * globalMatrix;
496 pParent = pParent->GetParent();
497 }
498 pCluster->SetTransformLinkMatrix(globalMatrix);
499 pCluster->SetTransformMatrix(meshTransform);
500
501 for (int i = 0; i < character.skinWeights->index.rows(); i++) {
502 for (int j = 0; j < character.skinWeights->index.cols(); j++) {
503 auto boneIndex = character.skinWeights->index(i, j);
504 if (boneIndex == jointNode.first && character.skinWeights->weight(i, j) > 0) {
505 pCluster->AddControlPointIndex(i, character.skinWeights->weight(i, j));
506 }
507 }
508 }
509 fbxskin->AddCluster(pCluster);
510 }
511 lMesh->AddDeformer(fbxskin);
512 }
513 // Add the mesh under the root
514 root->AddChild(meshNode);
515 }
516
517 // ---------------------------------------------
518 // add the skeleton to the root
519 // ---------------------------------------------
520 if (!skeletonNodes.empty()) {
521 root->AddChild(skeletonRootNode);
522 }
523
524 // ---------------------------------------------
525 // create animation curves if we have motion
526 // ---------------------------------------------
527 if (jointValues.cols() != 0) {
528 if (jointValues.rows() == character.parameterTransform.numJointParameters()) {
529 createAnimationCurves(
530 character, scene, skeletonNodes, jointValues, framerate, skipActiveJointParamCheck);
531 } else {
532 MT_LOGE(
533 "Rows of joint values {} do not match joint parameter dimension {} so not saving any motion.",
534 jointValues.rows(),
535 character.parameterTransform.numJointParameters());
536 }
537 }
538
539 // ---------------------------------------------
540 // close the fbx exporter
541 // ---------------------------------------------
542
543 // finally export the scene
544 lExporter->Export(scene);
545 lExporter->Destroy();
546
547 // destroy the scene and the manager
548 if (scene != nullptr) {
549 scene->Destroy();
550 }
551 manager->Destroy();
552}

580void saveFbx(
581 const filesystem::path& filename,
582 const Character& character,
583 const MatrixXf& poses, // model parameters
584 const VectorXf& identity,
585 const double framerate,
586 const bool saveMesh,
587 const FBXCoordSystemInfo& coordSystemInfo,
588 bool permissive) {
589 CharacterParameters params;
590 if (identity.size() == character.parameterTransform.numJointParameters()) {
591 params.offsets = identity;
592 } else {
593 params.offsets = character.parameterTransform.bindPose();
594 }
595
596 // first convert model parameters to joint values
597 CharacterState state;
598 MatrixXf jointValues;
599 if (poses.cols() > 0) {
600 // Set the initial pose to initialize the state
601 params.pose = poses.col(0);
602 state.set(params, character, false, false, false);
603
604 // Resize the jointValues matrix based on the size of joint parameters and number of poses
605 jointValues.resize(state.skeletonState.jointParameters.v.size(), poses.cols());
606
607 // Store the joint parameters for the initial pose

616 }
617 }
