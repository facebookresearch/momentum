36std::shared_ptr<const Mppca> loadMppca(std::istream& inputStream) {
37 auto result = std::make_shared<Mppca>();
38
39 // try to load file
40 try {
41 MT_THROW_IF(!inputStream, "Error loading Mppca model: empty input stream.");
42
43 // load dimensions
44 uint64_t od = 0;
45 uint64_t op = 0;
46 inputStream.read((char*)&od, sizeof(od));
47 inputStream.read((char*)&op, sizeof(op));
48
49 // set data on result
50 result->d = od;
51 result->p = op;
52
53 // load names
54 result->names.resize(od);
55 for (size_t i = 0; i < od; i++) {
56 uint64_t count = 0;
57 inputStream.read((char*)&count, sizeof(uint64_t));
58 result->names[i].resize(count);
59 inputStream.read((char*)result->names[i].data(), count);
60 }
61
62 // prepare data structures
63 result->Rpre = VectorXf(op);
64 result->Cinv.resize(op);
65 result->L.resize(op);
66 result->mu.resize(op, od);
67
68 // load Rpre, Cinv, and mu
69 inputStream.read((char*)result->Rpre.data(), sizeof(float) * op);
70 for (size_t i = 0; i < op; i++) {
71 result->Cinv[i] = MatrixXf(od, od);
72 inputStream.read((char*)result->Cinv[i].data(), sizeof(float) * od * od);
73 result->L[i] = result->Cinv[i].llt().matrixL().transpose();
74 }
75 inputStream.read((char*)result->mu.data(), sizeof(float) * op * od);
76
77 return result;
78 } catch (std::exception& e) {
79 MT_THROW("Error loading Mppca model: {}", e.what());
80 }
81 return result;
82}

84void saveMppca(const Mppca& mppca, const std::string& name) {
85 // write output file
86 std::ofstream data(name, std::ios::out | std::ios::binary);
87 if (!data.is_open()) {
88 return;
89 }
90
91 // write dimensions
92 const uint64_t od = gsl::narrow<uint64_t>(mppca.d);
93 const uint64_t op = gsl::narrow<uint64_t>(mppca.p);
94 data.write((char*)&od, sizeof(od));
95 data.write((char*)&op, sizeof(op));
96
97 // write out names
98 MT_CHECK(mppca.names.size() >= od);
99 for (size_t i = 0; i < od; i++) {
100 // NOLINTNEXTLINE(facebook-hte-ParameterUncheckedArrayBounds)
101 const std::string& nm = mppca.names[i];
102 const uint64_t ln = nm.size();
103 data.write((const char*)&ln, sizeof(uint64_t));
104 data.write((const char*)nm.data(), ln);
105 }
106
107 // write pi, mu, W, sigma2
108 data.write((char*)mppca.Rpre.data(), sizeof(float) * op);
109 for (const auto& Cinv : mppca.Cinv) {
110 data.write((char*)Cinv.data(), sizeof(float) * od * od);
111 }
112 data.write((char*)mppca.mu.data(), sizeof(float) * op * od);
113
114 // close file
115 data.close();
116}
