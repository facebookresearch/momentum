17void saveMmo(
18 const std::string& filename,
19 std::span<const VectorXf> poses,
20 const VectorXf& scale,
21 const Character& character) {
22 if (poses.empty() ||
23 poses[0].size() !=
24 gsl::narrow<Eigen::Index>(character.parameterTransform.numAllModelParameters())) {
25 MT_LOGW("{}: Cannot save data: motion data is either empty or of wrong dimension", __func__);
26 return;
27 }
28
29 // make sure all poses have the right size
30 MatrixXf data(poses[0].size(), poses.size());
31 for (size_t i = 0; i < poses.size(); i++) {
32 if (poses[i].size() != poses[0].size()) {
33 MT_LOGW("{}: Cannot save data: not all poses are of the same dimension", __func__);
34 return;
35 }
36
37 data.col(i) = poses[i];
38 }
39
40 saveMmo(filename, data, scale, character);
41}

43void saveMmo(
44 const std::string& filename,
45 const MatrixXf& poses,
46 const VectorXf& scale,
47 const Character& character,
48 const MatrixXf& additionalParameters,
49 std::span<const std::string> additionalParameterNames) {
50 if (poses.rows() !=
51 gsl::narrow<Eigen::Index>(character.parameterTransform.numAllModelParameters())) {
52 MT_LOGW(
53 "{}: Cannot save data: poses are of wrong dimension. Expected: {}, actual: {}",
54 __func__,
55 character.parameterTransform.numAllModelParameters(),
56 poses.rows());
57 return;
58 }
59 if (scale.size() !=
60 gsl::narrow<Eigen::Index>(character.skeleton.joints.size() * kParametersPerJoint)) {
61 MT_LOGW(
62 "{}: Cannot save data: model scale is of wrong dimension. Expected: {}, actual: {}",
63 __func__,
64 character.skeleton.joints.size() * kParametersPerJoint,
65 scale.size());
66 return;
67 }
68 if (additionalParameters.rows() != gsl::narrow<Eigen::Index>(additionalParameterNames.size())) {
69 MT_LOGW(
70 "{}: Cannot save data: number of additional parameters don't match. Names: {}, data: {}",
71 __func__,
72 additionalParameterNames.size(),
73 additionalParameters.rows());
74 return;
75 }
76 if (additionalParameters.cols() != 0 && additionalParameters.cols() != poses.cols()) {
77 MT_LOGW(
78 "{}: Cannot save data: additional parameters don't match pose dimensions. Expected: {}, actual {}",
79 __func__,
80 poses.cols(),
81 additionalParameters.cols());
82 return;
83 }
84
85 // create enhance parameter list;
86 std::vector<std::string> parameterNames;
87 parameterNames.reserve(
88 character.parameterTransform.name.size() + additionalParameterNames.size());
89 for (const auto& name : character.parameterTransform.name) {
90 parameterNames.emplace_back(name);
91 }
92 for (const auto& name : additionalParameterNames) {
93 parameterNames.emplace_back("__" + name + "__");
94 }
95
96 // write out joint names
97 std::vector<std::string> jointNames;
98 for (const auto& joint : character.skeleton.joints) {
99 jointNames.emplace_back(joint.name);
100 }
101
102 // create joined pose matrix
103 MatrixXf pc = MatrixXf(poses.rows() + additionalParameters.rows(), poses.cols());
104 pc.topRows(poses.rows()) = poses;
105
106 if (additionalParameters.rows() > 0) {
107 pc.bottomRows(additionalParameters.rows()) = additionalParameters;
108 }
109
110 saveMmo(filename, pc, scale, parameterNames, jointNames);
111}

113void saveMmo(
114 const std::string& filename,
115 const MatrixXf& poses,
116 const VectorXf& scale,
117 std::span<const std::string> parameterNames,
118 std::span<const std::string> jointNames) {
119 if (poses.rows() != gsl::narrow<Eigen::Index>(parameterNames.size())) {
120 MT_LOGW(
121 "{}: Cannot save data: poses are of wrong dimension. Expected: {}, actual: {}",
122 __func__,
123 parameterNames.size(),
124 poses.rows());
125 return;
126 }
127 if (scale.size() != gsl::narrow<Eigen::Index>(jointNames.size() * kParametersPerJoint)) {
128 MT_LOGW(
129 "{}: Cannot save data: model scale is of wrong dimension. Expected: {}, actual: {}",
130 __func__,
131 jointNames.size() * kParametersPerJoint,
132 scale.size());
133 return;
134 }
135
136 std::ofstream fs(filename, std::ios::out | std::ios::binary);
137 // write out sizes
138 size_t data = 0;
139 data = parameterNames.size(); // number of parameters
140 fs.write((const char*)&data, sizeof(size_t));
141
142 data = jointNames.size(); // number of joints
143 fs.write((const char*)&data, sizeof(size_t));
144
145 data = poses.cols(); // number of frames
146 fs.write((const char*)&data, sizeof(size_t));
147
148 // write out parameters names
149 for (const auto& name : parameterNames) {
150 data = name.size();
151 fs.write((const char*)&data, sizeof(size_t));
152 fs.write((const char*)name.data(), data);
153 }
154
155 // write out joint names
156 for (const auto& name : jointNames) {
157 data = name.size();
158 fs.write((const char*)&data, sizeof(size_t));
159 fs.write((const char*)name.data(), data);
160 }
161
162 // write out offsets
163 fs.write((const char*)scale.data(), scale.size() * sizeof(float));
164
165 // write out frames
166 fs.write((const char*)poses.data(), poses.size() * sizeof(float));
167}

169std::tuple<MatrixXf, VectorXf> mapMotionToCharacter(
170 const MatrixXf& poses,
171 const VectorXf& offsets,
172 std::span<const std::string> parameterNames,
173 std::span<const std::string> jointNames,
174 const Character& character) {
175 std::tuple<MatrixXf, VectorXf> result;
176 auto& outPoses = std::get<0>(result);
177 auto& outOffset = std::get<1>(result);
178
179 // create mapping for parameters
180 std::vector<size_t> pMap;
181 for (const auto& name : parameterNames) {
182 auto iter = std::find(
183 character.parameterTransform.name.begin(), character.parameterTransform.name.end(), name);
184 size_t index = std::distance(character.parameterTransform.name.begin(), iter);
185 if (index >= character.parameterTransform.name.size()) {
186 index = kInvalidIndex;
187 }
188 pMap.push_back(index);
189 }
190
191 // create mapping for joints
192 std::vector<size_t> jMap;
193 jMap.reserve(jointNames.size());
194 for (const auto& name : jointNames) {
195 jMap.push_back(character.skeleton.getJointIdByName(name));
196 }
197
198 // read offsets and re-arrange them according to the map
199 outOffset.setZero(character.skeleton.joints.size() * kParametersPerJoint);
200 for (size_t i = 0; i < jMap.size(); i++) {
201 if (jMap[i] != kInvalidIndex) {
202 outOffset.middleRows<kParametersPerJoint>(jMap[i] * kParametersPerJoint) =
203 offsets.middleRows<kParametersPerJoint>(i * kParametersPerJoint);
204 }
205 }
206
207 // read frames and re-arrange them according to the map
208 outPoses.setZero(character.parameterTransform.numAllModelParameters(), poses.cols());
209 for (size_t i = 0; i < pMap.size(); i++) {
210 if (pMap[i] != kInvalidIndex) {
211 outPoses.row(pMap[i]) = poses.row(i);
212 }
213 }
214
215 return result;
216}

223// get auxiliary data from motion
224std::tuple<MatrixXf, std::vector<std::string>> getAuxiliaryDataFromMotion(
225 const MatrixXf& poses,
226 std::span<const std::string> parameterNames) {
227 std::tuple<MatrixXf, std::vector<std::string>> result;
228 auto& values = std::get<0>(result);
229 auto& names = std::get<1>(result);
230
231 std::vector<size_t> map;
232 for (size_t i = 0; i < parameterNames.size(); i++) {
233 const auto& n = parameterNames[i];
234 const auto l = n.length();
235 if (n.size() > 4 && n[0] == '_' && n[1] == '_' && n[l - 1] == '_' && n[l - 2] == '_') {
236 map.push_back(i);
237 names.push_back(n.substr(2, l - 4));
238 }
239 }
240
241 values.resize(gsl::narrow<Eigen::Index>(names.size()), poses.cols());
242 for (size_t i = 0; i < map.size(); i++) {
243 values.row(i) = poses.row(map[i]);
244 }
245
246 return result;
247}

249std::tuple<MatrixXf, VectorXf, std::vector<std::string>, std::vector<std::string>> loadMmo(
250 const std::string& filename) {
251 std::tuple<MatrixXf, VectorXf, std::vector<std::string>, std::vector<std::string>> result;
252 auto& poses = std::get<0>(result);
253 auto& scale = std::get<1>(result);
254 auto& parameterNames = std::get<2>(result);
255 auto& jointNames = std::get<3>(result);
256
257 std::ifstream fs(filename, std::ios::in | std::ios::binary);
258
259 if (!fs.is_open()) {
260 MT_LOGW("{}: Failed to open file {}", __func__, filename);
261 return result;
262 }
263
264 // read total number frames/joints/offsets
265 size_t nParams = 0;
266 size_t nJoints = 0;
267 size_t nFrames = 0;
268 fs.read((char*)&nParams, sizeof(size_t));
269 fs.read((char*)&nJoints, sizeof(size_t));
270 fs.read((char*)&nFrames, sizeof(size_t));
271
272 // load parameter names
273 for (size_t i = 0; i < nParams; i++) {
274 // read parameter name
275 size_t count = 0;
276 fs.read((char*)&count, sizeof(size_t));
277 std::string name;
278 name.resize(count);
279 fs.read((char*)name.data(), count);
280 parameterNames.push_back(name);
281 }
282
283 // load joint names
284 for (size_t i = 0; i < nJoints; i++) {
285 // read parameter name
286 size_t count = 0;
287 fs.read((char*)&count, sizeof(size_t));
288 std::string name;
289 name.resize(count);
290 fs.read((char*)name.data(), count);
291 jointNames.push_back(name);
292 }
293
294 // read offsets and re-arrange them according to the map
295 scale = VectorXf(nJoints * kParametersPerJoint);
296 fs.read((char*)scale.data(), scale.size() * sizeof(float));
297
298 // read frames and re-arrange them according to the map
299 poses = MatrixXf(nParams, nFrames);
300 fs.read((char*)poses.data(), poses.size() * sizeof(float));
301
302 return result;
303}
