28arvr::logging::LogResult callback(
29 int logLevel,
30 const char* /* channelName */,
31 size_t /* channelNameSizeInBytes */,
32 const char* message,
33 size_t /* messageSizeInBytes */,
34 bool /*isFatalLog*/,
35 arvr::logging::CustomUserData userData) {
36 const auto* rec = reinterpret_cast<const rerun::RecordingStream*>(userData);
37
38 rerun::TextLogLevel level;
39 switch (logLevel) {
40 case static_cast<int>(arvr::logging::Level::Error):
41 level = rerun::TextLogLevel::Error;
42 break;
43 case static_cast<int>(arvr::logging::Level::Warning):
44 level = rerun::TextLogLevel::Warning;
45 break;
46 case static_cast<int>(arvr::logging::Level::Info):
47 level = rerun::TextLogLevel::Info;
48 break;
49 case static_cast<int>(arvr::logging::Level::Debug):
50 level = rerun::TextLogLevel::Debug;
51 break;
52 case static_cast<int>(arvr::logging::Level::Trace):
53 level = rerun::TextLogLevel::Trace;
54 break;
55 default: {
56 return arvr::logging::LogResult::Filtered;
57 }
58 }
59
60 // TODO: Make logging channel configurable
61 rec->log("momentum", rerun::TextLog(message).with_level(level));
62
63 return arvr::logging::LogResult::Accepted;
64}

70bool redirectLogsToRerun(const rerun::RecordingStream& rec) {
71#if defined(MOMENTUM_WITH_XR_LOGGER)
72 static arvr::logging::LogSinkHandle handle = 0;
73 if (handle != 0) {
74 // If handle is not 0, it means the log sink has already been added, so return true to indicate
75 // that the log is already being redirected.
76 return true;
77 }
78
79 handle = arvr::logging::addSink(
80 callback, reinterpret_cast<arvr::logging::CustomUserData>(&rec), "Rerun logger");
81 return (handle != 0);
82#else
83 (void)rec;
84 MT_LOGW("Log redirection is not supported.\n");
85 return false;
86#endif
87}
