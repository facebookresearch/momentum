14template <typename T>
15UnionErrorFunctionT<T>::UnionErrorFunctionT(
16 const Skeleton& skel,
17 const ParameterTransform& pt,
18 const std::vector<std::shared_ptr<SkeletonErrorFunctionT<T>>>& errorFunctions)
19 : SkeletonErrorFunctionT<T>(skel, pt), errorFunctions_(errorFunctions) {
20 // Should this be allowed?
21 MT_THROW_IF(errorFunctions_.empty(), "No error functions in union?");
22
23 for (size_t i = 0; i < errorFunctions_.size(); ++i) {
24 auto diff_e =
25 std::dynamic_pointer_cast<FullyDifferentiableSkeletonErrorFunctionT<T>>(errorFunctions_[i]);
26 diffErrorFunctions_.push_back(diff_e);
27
28 if (diff_e) {
29 for (const auto& input : diff_e->inputs()) {
30 inputs_[input].push_back(i);
31 }
32 }
33 }
34
35 {
36 std::vector<std::string> names;
37 for (const auto& e : diffErrorFunctions_) {
38 if (e) {
39 names.push_back(e->name());
40 } else {
41 names.push_back("Unknown");
42 }
43 }
44
45 std::ostringstream name;
46 name << "Union[";
47 if (std::adjacent_find(std::begin(names), std::end(names), std::not_equal_to<>()) ==
48 std::end(names)) {
49 // All the same:
50 name << names.front() << " x " << names.size();
51 } else {
52 bool first = true;
53 for (const auto& n : names) {
54 if (!first) {
55 name << ", ";
56 }
57 name << n;
58 first = false;
59 }
60 }
61 name << "]";
62 name_ = name.str();
63 }
64}

90template <typename T>
91double UnionErrorFunctionT<T>::getJacobian(
92 const ModelParametersT<T>& params,
93 const SkeletonStateT<T>& skeletonState,
94 Eigen::Ref<Eigen::MatrixX<T>> jacobian,
95 Eigen::Ref<Eigen::VectorX<T>> residual,
96 int& usedRows) {
97 double error = 0;
98 size_t jacobianStart = 0;
99 for (auto& fun : errorFunctions_) {
100 const size_t memberJacobianSize = fun->getJacobianSize();
101
102 error += fun->getJacobian(
103 params,
104 skeletonState,
105 jacobian.block(jacobianStart, 0, memberJacobianSize, jacobian.cols()),
106 residual.segment(jacobianStart, memberJacobianSize),
107 usedRows);
108
109 jacobianStart += memberJacobianSize;
110 }
111
112 jacobian *= std::sqrt(this->weight_);
113 residual *= std::sqrt(this->weight_);
114
115 usedRows = static_cast<int>(jacobian.rows());
116
117 return this->weight_ * error;
118}

184template <typename T>
185Eigen::VectorX<T> UnionErrorFunctionT<T>::d_gradient_d_input_dot(
186 const std::string& name,
187 const ModelParametersT<T>& modelParams,
188 const SkeletonStateT<T>& state,
189 Eigen::Ref<const Eigen::VectorX<T>> inputVec) {
190 auto itr = inputs_.find(name);
191 MT_THROW_IF(
192 itr == inputs_.end(),
193 "Invalid input '{}' for UnionErrorFunction::d_gradient_d_input_dot",
194 name);
195
196 Eigen::Index currentOffset = 0;
197 Eigen::VectorX<T> result = Eigen::VectorX<T>::Zero(getInputSize(name));
198
199 for (const auto& iErrFun : itr->second) {
200 const auto errFun = diffErrorFunctions_[iErrFun];
201 const Eigen::Index sz = errFun->getInputSize(name);
202 result.segment(currentOffset, sz) =
203 errFun->d_gradient_d_input_dot(name, modelParams, state, inputVec);
204 currentOffset += sz;
205 }
206
207 result *= this->weight_;
208
209 return result;
210}
