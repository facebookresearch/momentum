27template <typename T>
28CharacterT<T>::CharacterT(
29 const Skeleton& s,
30 const ParameterTransform& pt,
31 const ParameterLimits& pl,
32 const LocatorList& l,
33 const Mesh* m,
34 const SkinWeights* sw,
35 const CollisionGeometry* cg,
36 const PoseShape* bs,
37 BlendShape_const_p blendShapes,
38 BlendShapeBase_const_p faceExpressionBlendShapes,
39 const std::string& nameIn,
40 const momentum::TransformationList& inverseBindPose_in,
41 const SkinnedLocatorList& skinnedLocators)
42 : skeleton(s),
43 parameterTransform(pt),
44 parameterLimits(pl),
45 locators(l),
46 skinnedLocators(skinnedLocators),
47 blendShape(std::move(blendShapes)),
48 faceExpressionBlendShape(std::move(faceExpressionBlendShapes)),
49 inverseBindPose(inverseBindPose_in),
50 name(nameIn) {
51 if (m) {
52 mesh = std::make_unique<Mesh>(*m);
53 // create skinweights copy only if both mesh and skinweights exist
54 if (sw) {
55 skinWeights = std::make_unique<SkinWeights>(*sw);
56 }
57 }
58
59 // copy pose blendshapes if present
60 if (bs) {
61 poseShapes = std::make_unique<PoseShape>(*bs);
62 }
63
64 // initialize bindpose inverse list
65 if (inverseBindPose.empty()) {
66 initInverseBindPose();
67 }
68 MT_CHECK(this->inverseBindPose.size() == skeleton.joints.size());
69
70 // copy collision geometry if it exists
71 if (cg) {
72 collision = std::make_unique<CollisionGeometry>(*cg);
73 }
74
75 // fill jointMap with identity
76 resetJointMap();
77}

146template <typename T>
147std::vector<bool> CharacterT<T>::parametersToActiveJoints(const ParameterSet& parameterSet) const {
148 // create a list of joints that are currently enabled
149 const auto nJoints = skeleton.joints.size();
150 std::vector<bool> result(nJoints, false);
151 for (int k = 0; k < parameterTransform.transform.outerSize(); ++k) {
152 for (SparseRowMatrixf::InnerIterator it(parameterTransform.transform, k); it; ++it) {
153 // row is influenced joint, col is parameter
154 const auto& row = it.row();
155 const auto& col = it.col();
156
157 // disable any joints that are not set
158 if (!parameterSet.test(col)) {
159 continue;
160 }
161
162 // set joint as enabled
163 const auto jointIndex = row / kParametersPerJoint;
164 MT_CHECK(jointIndex < nJoints, "{} vs {}", jointIndex, nJoints);
165 result[jointIndex] = true;
166 }
167 }
168
169 return result;
170}

172template <typename T>
173ParameterSet CharacterT<T>::activeJointsToParameters(const std::vector<bool>& activeJoints) const {
174 MT_CHECK(
175 activeJoints.size() == skeleton.joints.size(),
176 "{} is not {}",
177 activeJoints.size(),
178 skeleton.joints.size());
179
180 ParameterSet result;
181
182 // iterate over all non-zero entries of the matrix
183 for (int k = 0; k < parameterTransform.transform.outerSize(); ++k) {
184 for (SparseRowMatrixf::InnerIterator it(parameterTransform.transform, k); it; ++it) {
185 const auto globalParam = it.row();
186 const auto kParam = it.col();
187 MT_CHECK(
188 kParam < (Eigen::Index)parameterTransform.numAllModelParameters(),
189 "{} vs {}",
190 kParam,
191 parameterTransform.numAllModelParameters());
192 const auto jointIndex = globalParam / kParametersPerJoint;
193
194 if (activeJoints[jointIndex] != 0) {
195 result.set(kParam, true);
196 }
197 }
198 }
199
200 return result;
201}

203// create simplified skeleton for enabled parameters
204template <typename T>
205CharacterT<T> CharacterT<T>::simplifySkeleton(const std::vector<bool>& activeJoints) const {
206 MT_CHECK(
207 activeJoints.size() == skeleton.joints.size(),
208 "{} is not {}",
209 activeJoints.size(),
210 skeleton.joints.size());
211 MT_THROW_IF(
212 std::find(activeJoints.begin(), activeJoints.end(), true) == activeJoints.end(),
213 "In simplifySkeleton, no joints are enabled; resulting skeleton must have at least one valid joint.");
214
215 Skeleton simplifiedSkeleton;
216
217 // -------------------------------------------------------------------
218 // start by remapping the skeleton and parameter transform
219 // -------------------------------------------------------------------
220
221 // remember parameter mapping from skeleton to anim skeleton
222 size_t jointCount = 0;
223
224 // create a parameter transform with the correct offsets
225 auto zeroTransform = parameterTransform;
226
227 for (size_t jointIndex = 0; jointIndex < activeJoints.size(); ++jointIndex) {
228 // zero out all offsets related to joints that are enabled, keep the disabled ones set
229 if (activeJoints[jointIndex]) {
230 for (size_t o = 0; o < kParametersPerJoint; o++) {
231 zeroTransform.offsets(jointIndex * kParametersPerJoint + o) = 0.0f;
232 }
233 }
234 }
235
236 // some state for mapping joints to new joints
237 std::vector<size_t> intermediateJointMap(skeleton.joints.size(), kInvalidIndex);
238
239 // create a reference state for the model where all disabled joints keep their offset influence
240 const SkeletonState referenceState(
241 zeroTransform.apply(ModelParameters::Zero(zeroTransform.numAllModelParameters())),
242 skeleton,
243 false);
244
245 // create map from anim skeleton to new joints
246 std::vector<size_t> lastJoint(skeleton.joints.size(), kInvalidIndex);
247
248 // go over the anim skeleton, check which joints are active and convert them
249 std::vector<size_t> simplifiedJointMap(skeleton.joints.size(), kInvalidIndex);
250 for (size_t aIndex = 0; aIndex < skeleton.joints.size(); aIndex++) {
251 const Joint& j = skeleton.joints[aIndex];
252
253 // find the parent joint that belongs here
254 Vector3f offset = j.translationOffset;
255 size_t currentParent = kInvalidIndex;
256 size_t sIndex = aIndex;
257 if (j.parent != kInvalidIndex) {
258 // find the last not disabled parent joint
259 while (currentParent == kInvalidIndex && sIndex != kInvalidIndex) {
260 sIndex = skeleton.joints[sIndex].parent;
261 currentParent = lastJoint[sIndex];
262 }
263 // calculate the new offset of the joint in the new parent space
264 if (sIndex != kInvalidIndex) {
265 offset = referenceState.jointState[sIndex].transform.inverse() *
266 referenceState.jointState[aIndex].translation();
267 } else {
268 offset = referenceState.jointState[aIndex].translation();
269 }
270 }
271
272 // is joint enabled?
273 if (activeJoints[aIndex]) {
274 // create copy
275 Joint jt = j;
276
277 // set new parent
278 jt.parent = currentParent;
279
280 // set updated offset
281 jt.translationOffset = offset;
282
283 // update prerotation for the new parent joint by summing up
284 size_t parent = j.parent;
285 while (parent != sIndex && parent != kInvalidIndex) {
286 jt.preRotation = skeleton.joints[parent].preRotation * jt.preRotation;
287 parent = skeleton.joints[parent].parent;
288 }
289
290 // add it
291 simplifiedSkeleton.joints.push_back(jt);
292
293 // add jointmap entry
294 intermediateJointMap[aIndex] = jointCount;
295 jointCount++;
296 lastJoint[aIndex] = simplifiedSkeleton.joints.size() - 1;
297
298 // add joint to joint mapping
299 simplifiedJointMap[aIndex] = jointCount - 1;
300 } else {
301 size_t index = aIndex;
302 while (index != kInvalidIndex && intermediateJointMap[index] == kInvalidIndex) {
303 index = skeleton.joints[index].parent;
304 }
305 // Find the valid joint above this one in the hierarchy:
306 MT_THROW_IF(
307 index == kInvalidIndex,
308 "During skeleton simplification, inactive joint '{}' has no valid parent joint. "
309 "Every joint in the simplified skeleton must have at least one parent joint that is not disabled.",
310 skeleton.joints.at(aIndex).name);
311 simplifiedJointMap[aIndex] = intermediateJointMap[index];
312 }
313 }
314 MT_THROW_IF(
315 simplifiedSkeleton.joints.empty(),
316 "During skeleton simplification, resulting skeleton was empty. Simplified skeleton must have at least one valid joint.");
317
318 const ParameterTransform simplifiedTransform = mapParameterTransformJoints(
319 parameterTransform, simplifiedSkeleton.joints.size(), intermediateJointMap);
320
321 // create character result
322 CharacterT<T> result(simplifiedSkeleton, simplifiedTransform);
323
324 result.jointMap = simplifiedJointMap;
325
326 // -------------------------------------------------------------------
327 // remap the parameterlimits
328 // -------------------------------------------------------------------
329 result.parameterLimits = result.remapParameterLimits(parameterLimits, *this);
330
331 // -------------------------------------------------------------------
332 // remap the locators if we have any
333 // -------------------------------------------------------------------
334 result.locators = result.remapLocators(locators, *this);
335 result.skinnedLocators = result.remapSkinnedLocators(skinnedLocators, *this);
336
337 // -------------------------------------------------------------------
338 // remap the mesh and skinning if present
339 // -------------------------------------------------------------------
340
341 // create bind states for both source and target skeleton
342 const SkeletonState sourceBindState(parameterTransform.bindPose(), skeleton, false);
343 const SkeletonState targetBindState(result.parameterTransform.bindPose(), result.skeleton, false);
344
345 if (mesh && skinWeights) {
346 result.mesh = std::make_unique<Mesh>(*mesh);
347 result.skinWeights =
348 std::make_unique<SkinWeights>(result.remapSkinWeights(*skinWeights, *this));
349 }
350
351 // -------------------------------------------------------------------
352 // remap the collision if we have any
353 // -------------------------------------------------------------------
354 if (collision != nullptr) {
355 result.collision = std::make_unique<CollisionGeometry>(*collision);
356 for (auto&& c : *result.collision) {
357 const auto oldParent = c.parent;
358 c.parent = result.jointMap[c.parent];
359 c.transformation =
360 (targetBindState.jointState[c.parent].transform.inverse() *
361 sourceBindState.jointState[oldParent].transform * c.transformation);
362 }
363 }
364
365 // return the map
366 return result;
367}

393template <typename T>
394CharacterT<T> CharacterT<T>::simplify(const ParameterSet& activeParams) const {
395 auto activeJoints = parametersToActiveJoints(activeParams);
396 // always keep the root joint to ensure valid simplification.
397 // This is needed because while previously we generally parametrized the root joint with
398 // the root transform, modern skeletons sometimes have a body_world above b_root that is
399 // not parametrized, but if we throw it out we end up with an invalid character.
400 if (!activeJoints.empty()) {
401 activeJoints[0] = true;
402 }
403
404 return simplifySkeleton(activeJoints);
405}

407template <typename T>
408SkinWeights CharacterT<T>::remapSkinWeights(
409 const SkinWeights& inSkinWeights,
410 const CharacterT& originalCharacter) const {
411 MT_CHECK(
412 originalCharacter.parameterTransform.numAllModelParameters() ==
413 parameterTransform.numAllModelParameters(),
414 "{} is not {}",
415 originalCharacter.parameterTransform.numAllModelParameters(),
416 parameterTransform.numAllModelParameters());
417 MT_CHECK(
418 jointMap.size() == originalCharacter.skeleton.joints.size(),
419 "{} is not {}",
420 jointMap.size(),
421 originalCharacter.skeleton.joints.size());
422
423 SkinWeights result = inSkinWeights;
424
425 // go over all vertices
426 for (int v = 0; v < result.index.rows(); v++) {
427 // remap the parent bones according to the map
428 for (int i = 0; i < gsl::narrow_cast<int>(kMaxSkinJoints); i++) {
429 result.index(v, i) = gsl::narrow<uint32_t>(jointMap[result.index(v, i)]);
430 }
431
432 // join together all weights with the same parent
433 for (int i = 0; i < gsl::narrow_cast<int>(kMaxSkinJoints); i++) {
434 const auto& joint = result.index(v, i);
435 for (int j = i + 1; j < gsl::narrow_cast<int>(kMaxSkinJoints); j++) {
436 // if we have the same joint multiple times, add things up
437 if (result.index(v, j) == joint) {
438 result.weight(v, i) += result.weight(v, j);
439 result.weight(v, j) = 0.0f;
440 }
441 }
442 }
443
444 // do a clean up pass, sorting things by weight
445 for (int i = 0; i < gsl::narrow_cast<int>(kMaxSkinJoints); i++) {
446 for (int j = i + 1; j < gsl::narrow_cast<int>(kMaxSkinJoints); j++) {
447 if (result.weight(v, i) < result.weight(v, j)) {
448 std::swap(result.weight(v, i), result.weight(v, j));
449 std::swap(result.index(v, i), result.index(v, j));
450 }
451 }
452 }
453 }
454
455 return result;
456}

458template <typename T>
459ParameterLimits CharacterT<T>::remapParameterLimits(
460 const ParameterLimits& limits,
461 const CharacterT& originalCharacter) const {
462 MT_CHECK(
463 originalCharacter.parameterTransform.numAllModelParameters() ==
464 parameterTransform.numAllModelParameters(),
465 "{} is not {}",
466 originalCharacter.parameterTransform.numAllModelParameters(),
467 parameterTransform.numAllModelParameters());
468 MT_CHECK(
469 jointMap.size() == originalCharacter.skeleton.joints.size(),
470 "{} is not {}",
471 jointMap.size(),
472 originalCharacter.skeleton.joints.size());
473
474 ParameterLimits result;
475
476 // create bind states for both source and target skeleton
477 const SkeletonState sourceBindState(
478 originalCharacter.parameterTransform.bindPose(), originalCharacter.skeleton, false);
479 const SkeletonState targetBindState(parameterTransform.bindPose(), skeleton, false);
480
481 result = limits;
482 for (auto&& limit : result) {
483 // only need to remap limits that depend on joints
484 if (limit.type == MinMaxJoint || limit.type == MinMaxJointPassive) {
485 auto& data = limit.data.minMaxJoint;
486 data.jointIndex = jointMap[data.jointIndex];
487 } else if (limit.type == Ellipsoid) {
488 // ellipsoid limit requires re-targeting the ellipsoid to the new joint's local system
489 auto& data = limit.data.ellipsoid;
490 const auto sourceParent = data.parent;
491 const auto targetParent = jointMap[sourceParent];
492 MT_CHECK(
493 targetParent < targetBindState.jointState.size(),
494 "{} vs {}",
495 targetParent,
496 targetBindState.jointState.size());
497 data.parent = targetParent;
498
499 const auto targetTransformationInverse =
500 targetBindState.jointState[targetParent].transform.inverse();
501 const auto sourceTransformation = sourceBindState.jointState[sourceParent].transform;
502 data.offset = targetTransformationInverse * sourceTransformation * data.offset;
503
504 const auto sourceEllipsoidParent = data.ellipsoidParent;
505 data.ellipsoidParent = jointMap[data.ellipsoidParent];
506 const auto targetEllipsoidInverse =
507 targetBindState.jointState[data.ellipsoidParent].transform.inverse();
508 const auto sourceEllipsoid = sourceBindState.jointState[sourceEllipsoidParent].transform;
509 data.ellipsoid = targetEllipsoidInverse * sourceEllipsoid * data.ellipsoid;
510 data.ellipsoidInv = data.ellipsoid.inverse();
511 }
512 }
513
514 return result;
515}

517template <typename T>
518SkinnedLocatorList CharacterT<T>::remapSkinnedLocators(
519 const SkinnedLocatorList& locs,
520 const CharacterT& originalCharacter) const {
521 MT_CHECK(
522 originalCharacter.parameterTransform.numAllModelParameters() ==
523 parameterTransform.numAllModelParameters(),
524 "{} is not {}",
525 originalCharacter.parameterTransform.numAllModelParameters(),
526 parameterTransform.numAllModelParameters());
527 MT_CHECK(
528 jointMap.size() == originalCharacter.skeleton.joints.size(),
529 "{} is not {}",
530 jointMap.size(),
531 originalCharacter.skeleton.joints.size());
532
533 SkinnedLocatorList result = locs;
534
535 // create bind states for both source and target skeleton
536 const SkeletonState sourceBindState(
537 originalCharacter.parameterTransform.bindPose(), originalCharacter.skeleton, false);
538 const SkeletonState targetBindState(parameterTransform.bindPose(), skeleton, false);
539
540 // go over each skinned locator and remap its joint indices
541 for (auto& sl : result) {
542 // remap the joint indices according to the map
543 for (int i = 0; i < gsl::narrow_cast<int>(kMaxSkinJoints); i++) {
544 sl.parents(i) = gsl::narrow<uint32_t>(jointMap[sl.parents(i)]);
545 }
546
547 // join together all weights with the same joint
548 for (int i = 0; i < gsl::narrow_cast<int>(kMaxSkinJoints); i++) {
549 const auto& joint = sl.parents(i);
550 for (int j = i + 1; j < gsl::narrow_cast<int>(kMaxSkinJoints); j++) {
551 // if we have the same joint multiple times, add weights up
552 if (sl.parents(j) == joint) {
553 sl.skinWeights(i) += sl.skinWeights(j);
554 sl.skinWeights(j) = 0.0f;
555 }
556 }
557 }
558
559 // sort by weight, largest first
560 for (int i = 0; i < gsl::narrow_cast<int>(kMaxSkinJoints); i++) {
561 for (int j = i + 1; j < gsl::narrow_cast<int>(kMaxSkinJoints); j++) {
562 if (sl.skinWeights(i) < sl.skinWeights(j)) {
563 std::swap(sl.skinWeights(i), sl.skinWeights(j));
564 std::swap(sl.parents(i), sl.parents(j));
565 }
566 }
567 }
568 }
569
570 return result;
571}

573template <typename T>
574LocatorList CharacterT<T>::remapLocators(
575 const LocatorList& locs,
576 const CharacterT& originalCharacter) const {
577 MT_CHECK(
578 originalCharacter.parameterTransform.numAllModelParameters() ==
579 parameterTransform.numAllModelParameters(),
580 "{} is not {}",
581 originalCharacter.parameterTransform.numAllModelParameters(),
582 parameterTransform.numAllModelParameters());
583 MT_CHECK(
584 jointMap.size() == originalCharacter.skeleton.joints.size(),
585 "{} is not {}",
586 jointMap.size(),
587 originalCharacter.skeleton.joints.size());
588
589 LocatorList result;
590
591 // create bind states for both source and target skeleton
592 const SkeletonState sourceBindState(
593 originalCharacter.parameterTransform.bindPose(), originalCharacter.skeleton, false);
594 const SkeletonState targetBindState(parameterTransform.bindPose(), skeleton, false);
595
596 // go over each locator and remap it
597 for (const auto& sourceLocator : locs) {
598 result.emplace_back(sourceLocator);
599 auto& loc = result.back();
600 loc.parent = jointMap[loc.parent];
601 loc.offset = targetBindState.jointState[loc.parent].transform.inverse() *
602 sourceBindState.jointState[sourceLocator.parent].transform * sourceLocator.offset;
603 }
604
605 return result;
606}

629template <typename T>
630void CharacterT<T>::initInverseBindPose() {
631 // initialize bindpose inverse list
632 const SkeletonState bindState(parameterTransform.bindPose(), skeleton, false);
633 if (!inverseBindPose.empty()) {
634 inverseBindPose.clear();
635 }

638 inverseBindPose.push_back(t.transform.inverse().toAffine3());

642template <typename T>
643CharacterParameters CharacterT<T>::splitParameters(
644 const CharacterT& character,

646 const ParameterSet& parameterSet) {
647 auto parameterSplit = parameters;
648 auto offsetSplit = parameters;

651 parameterSplit.pose[j] = 0.0f;

669template <typename T>
670void CharacterT<T>::addBlendShape(
671 const BlendShape_const_p& blendShape_in,
672 Eigen::Index maxBlendShapes) {
673 MT_CHECK(this->mesh);
674 if (!blendShape_in) {
675 std::tie(parameterTransform, parameterLimits) =
676 addBlendShapeParameters(parameterTransform, parameterLimits, 0);
677 this->blendShape.reset();
678 return;
679 }
680
681 // TODO should we accommodate passing in a BlendShape with "extra" vertices (for the
682 // higher subdivision levels)?
683 MT_CHECK(
684 blendShape_in->modelSize() == this->mesh->vertices.size(),
685 "{} is not {}",
686 blendShape_in->modelSize(),
687 this->mesh->vertices.size());
688
689 blendShape = blendShape_in;
690
691 // Augment the parameter transform with blend shape parameters:
692 std::tie(parameterTransform, parameterLimits) = addBlendShapeParameters(
693 parameterTransform, parameterLimits, std::min(maxBlendShapes, blendShape_in->shapeSize()));
694}

751template <typename T>
752CharacterT<T> CharacterT<T>::bake(
753 const ModelParameters& modelParams,
754 bool bakeBlendShapes,
755 bool bakeScales) const {
756 // 1. Clone the character so we don't mutate the original.
757 CharacterT<T> result = *this;
758 MT_CHECK(result.mesh);
759
760 // 2. Blend-shape baking (you already had this, just guarded).
761 if (bakeBlendShapes && this->blendShape) {
762 const BlendWeights blendWeights = extractBlendWeights(result.parameterTransform, modelParams);
763 result.mesh->vertices = this->blendShape->template computeShape<float>(blendWeights);
764 }
765
766 // 3. Scale (or any joint transform) baking
767 if (bakeScales && this->skinWeights) {
768 // Evaluate the posed skeleton with modelParams.
769 const SkeletonState posed(
770 result.parameterTransform.apply(modelParams),
771 result.skeleton,
772 /*computeDeriv =*/false);
773 Mesh restMesh = *result.mesh;
774 applySSD(result.inverseBindPose, *result.skinWeights, restMesh, posed, *result.mesh);
775 }
776
777 result.mesh->updateNormals();
778
779 // 4. Strip baked parameters from the rig
780 ParameterSet baked = ParameterSet();
781 if (bakeBlendShapes) {
782 baked |= result.parameterTransform.getBlendShapeParameters();
783 }
784 if (bakeScales) {
785 baked |= result.parameterTransform.getScalingParameters();
786 }
787
788 std::tie(result.parameterTransform, result.parameterLimits) = subsetParameterTransform(
789 result.parameterTransform,
790 result.parameterLimits,
791 ~baked); // keep unbaked params only
792
793 return result;
794}
