17template <typename T>
18void FixedAxisDiffErrorFunctionT<T>::evalFunction(
19 const size_t constrIndex,
20 const JointStateT<T>& state,
21 Vector3<T>& f,
22 optional_ref<std::array<Vector3<T>, 1>> v,
23 optional_ref<std::array<Matrix3<T>, 1>> dfdv) const {
24 MT_PROFILE_FUNCTION();
25
26 const FixedAxisDataT<T>& constr = this->constraints_[constrIndex];
27 Vector3<T> vec = state.rotation() * constr.localAxis;
28 f = vec - constr.globalAxis;
29 if (v) {
30 v->get().at(0) = std::move(vec);
31 }
32 if (dfdv) {
33 dfdv->get().at(0).setIdentity();
34 }
35}

37template <typename T>
38void FixedAxisCosErrorFunctionT<T>::evalFunction(
39 const size_t constrIndex,
40 const JointStateT<T>& state,
41 Vector<T, 1>& f,
42 optional_ref<std::array<Vector3<T>, 1>> v,
43 optional_ref<std::array<Eigen::Matrix<T, 1, 3>, 1>> dfdv) const {
44 MT_PROFILE_FUNCTION();
45
46 const FixedAxisDataT<T>& constr = this->constraints_[constrIndex];
47 Vector3<T> vec = state.rotation() * constr.localAxis;
48 f[0] = 1 - vec.dot(constr.globalAxis);
49 if (v) {
50 v->get().at(0) = std::move(vec);
51 }
52 if (dfdv) {
53 dfdv->get().at(0) = -constr.globalAxis;
54 }
55}

57template <typename T>
58void FixedAxisAngleErrorFunctionT<T>::evalFunction(
59 const size_t constrIndex,
60 const JointStateT<T>& state,
61 Vector<T, 1>& f,
62 optional_ref<std::array<Vector3<T>, 1>> v,
63 optional_ref<std::array<Eigen::Matrix<T, 1, 3>, 1>> dfdv) const {
64 MT_PROFILE_FUNCTION();
65
66 const FixedAxisDataT<T>& constr = this->constraints_[constrIndex];
67 Vector3<T> vec = state.rotation() * constr.localAxis;
68 const T dot = vec.dot(constr.globalAxis);
69 f[0] = std::acos(std::clamp(dot, -T(1), T(1)));
70 if (v) {
71 v->get().at(0) = std::move(vec);
72 }
73 if (dfdv) {
74 // The derivative of d[acos(x)]/dx = -1/sqrt(1-x^2), where x is the cosine of the angle.
75 // When the angle is 0 or 180, x=+/-1.0, and d[acos(x)] is infinity. But because dx=sine(angle)
76 // is also zero, the final derivative will be zero as well.
77 // Comparing to the Cos version, this Acos version scales up the jacobian by the inverse of
78 // sine.
79 const T sine = sqrt(1 - dot * dot);
80 if (sine > 1e-9) {
81 dfdv->get().at(0) = -constr.globalAxis / sine;
82 } else {
83 dfdv->get().at(0).setZero();
84 }
85 }
86}
