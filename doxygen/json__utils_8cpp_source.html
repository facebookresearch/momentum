19void parameterSetsToJson(const Character& character, nlohmann::json& j) {
20 j = nlohmann::json::object();
21 for (const auto& set : character.parameterTransform.parameterSets) {
22 j[set.first] = nlohmann::json::array();
23 for (size_t i = 0; i < character.parameterTransform.name.size(); i++) {
24 if (set.second.test(i)) {
25 j[set.first].push_back(character.parameterTransform.name[i]);
26 }
27 }
28 }
29}

31ParameterSets parameterSetsFromJson(const Character& character, const nlohmann::json& j) {
32 ParameterSets result;
33
34 for (const auto& parameterSet : j.items()) {
35 auto& set = result[parameterSet.key()];
36 for (const auto& parameter : parameterSet.value().items()) {
37 const auto parameterIndex =
38 character.parameterTransform.getParameterIdByName(parameter.value());
39
40 if (parameterIndex != kInvalidIndex) {
41 set.set(parameterIndex);
42 }
43 }
44 }
45
46 return result;

61PoseConstraints poseConstraintsFromJson(const Character& character, const nlohmann::json& j) {
62 PoseConstraints result;
63
64 for (const auto& constraint : j.items()) {
65 auto& pose = result[constraint.key()];
66 for (const auto& constraintPair : constraint.value().items()) {
67 const auto parameterIndex =
68 character.parameterTransform.getParameterIdByName(constraintPair.key());
69
70 const float value = constraintPair.value();
71 if (parameterIndex != kInvalidIndex) {
72 pose.parameterIdValue.emplace_back(parameterIndex, value);
73 }
74 }
75 }
76
77 return result;

80void parameterTransformToJson(const Character& character, nlohmann::json& j) {
81 j["parameters"] = character.parameterTransform.name;
82
83 for (size_t jt = 0; jt < character.skeleton.joints.size(); jt++) {
84 const auto& jointName = character.skeleton.joints[jt].name;
85 const size_t paramIndex = jt * kParametersPerJoint;
86 for (size_t d = 0; d < kParametersPerJoint; d++) {
87 for (auto index = character.parameterTransform.transform.outerIndexPtr()[paramIndex + d];
88 index < character.parameterTransform.transform.outerIndexPtr()[paramIndex + d + 1];
89 ++index) {
90 const auto& parameterIndex = character.parameterTransform.transform.innerIndexPtr()[index];
91 const auto& parameterValue = character.parameterTransform.transform.valuePtr()[index];
92
93 if (parameterIndex < character.parameterTransform.name.size()) {
94 j["joints"][jointName][kJointParameterNames[d]]
95 [character.parameterTransform.name[parameterIndex]] = parameterValue;
96 }
97 }
98 }
99 }

102ParameterTransform parameterTransformFromJson(const Character& character, const nlohmann::json& j) {
103 ParameterTransform pt;
104 try {
105 pt.activeJointParams =
106 VectorX<bool>::Constant(character.skeleton.joints.size() * kParametersPerJoint, false);
107 pt.offsets = VectorXf::Zero(character.skeleton.joints.size() * kParametersPerJoint);
108
109 // load all the parameter names
110 auto parameterItr = j.find("parameters");
111 MT_THROW_IF(parameterItr == j.end(), "No 'parameters' found in parameter transform.");
112
113 for (const auto& p : *parameterItr) {
114 pt.name.push_back(p.get<std::string>());
115 }
116
117 // triplet list
118 std::vector<Eigen::Triplet<float>> triplets;
119
120 auto jointItr = j.find("joints");
121 MT_THROW_IF(jointItr == j.end(), "No 'joints' found in parameter transform.");
122
123 for (const auto& joint : jointItr->items()) {
124 const std::string& jointName = joint.key();
125
126 // get the right joint to modify
127 const size_t jointIndex = character.skeleton.getJointIdByName(jointName);
128 MT_THROW_IF(jointIndex == kInvalidIndex, "Unknown joint name in expression : {}", jointName);
129
130 for (const auto& jointParameter : joint.value().items()) {
131 const std::string& jointParameterName = jointParameter.key();
132
133 // the first pToken is the name of the joint and it's attribute type
134 size_t attributeIndex = kInvalidIndex;
135 for (size_t jt = 0; jt < kParametersPerJoint; jt++) {
136 if (jointParameterName == kJointParameterNames[jt]) {
137 attributeIndex = jt;
138 break;
139 }
140 }
141
142 // if we didn't find a right name exit with an error
143 MT_THROW_IF(
144 attributeIndex == kInvalidIndex,
145 "Unknown channel name in expression : {}",
146 jointParameterName);
147
148 // enable the attribute in the skeleton if we have a parameter controlling it
149 pt.activeJointParams[jointIndex * kParametersPerJoint + attributeIndex] = 1;
150
151 // split the parameter names
152 for (const auto& parameter : jointParameter.value().items()) {
153 // get weight and name
154 const float weight = parameter.value();
155 const std::string& parameterName = parameter.key();
156
157 auto parameterIndex = pt.getParameterIdByName(parameterName);
158
159 MT_THROW_IF(
160 parameterIndex == kInvalidIndex,
161 "Unknown parameter name in expression : {}",
162 parameterName);
163
164 // add triplet
165 triplets.emplace_back(
166 static_cast<int>(jointIndex) * kParametersPerJoint + static_cast<int>(attributeIndex),
167 static_cast<int>(parameterIndex),
168 weight);
169 }
170 }
171 }
172
173 // resize the Transform matrix to the correct size
174 pt.transform.resize(
175 static_cast<int>(character.skeleton.joints.size()) * kParametersPerJoint,
176 static_cast<int>(pt.name.size()));
177
178 // finish parameter setup by creating sparse matrix
179 pt.transform.setFromTriplets(triplets.begin(), triplets.end());
180 } catch (...) {
181 // create an identity parameter transform
182 pt = ParameterTransform::identity(character.skeleton.getJointNames());
183 }
184
185 return pt;
186}

188void parameterLimitsToJson(const Character& character, nlohmann::json& j) {
189 j = nlohmann::json::array();
190 for (const auto& lim : character.parameterLimits) {
191 j.push_back(nlohmann::json::object());
192 auto& li = j.back();
193 li["weight"] = lim.weight;
194
195 switch (lim.type) {
196 case MinMax:
197 li["type"] = "minmax";
198 li["parameter"] = character.parameterTransform.name[lim.data.minMax.parameterIndex];
199 li["limits"] = lim.data.minMax.limits;
200 break;
201 case MinMaxJoint:
202 li["type"] = "minmax_joint";
203 if (!character.skeleton.joints.empty() &&
204 lim.data.minMaxJoint.jointIndex < character.skeleton.joints.size()) {
205 li["jointIndex"] = character.skeleton.joints[lim.data.minMaxJoint.jointIndex].name;
206 }
207 li["jointParameter"] = kJointParameterNames[lim.data.minMaxJoint.jointParameter];
208 li["limits"] = lim.data.minMaxJoint.limits;
209 break;
210 case MinMaxJointPassive:
211 li["type"] = "minmax_joint_passive";
212 if (!character.skeleton.joints.empty() &&
213 lim.data.minMaxJoint.jointIndex < character.skeleton.joints.size()) {
214 li["jointIndex"] = character.skeleton.joints[lim.data.minMaxJoint.jointIndex].name;
215 }
216 li["jointParameter"] = kJointParameterNames[lim.data.minMaxJoint.jointParameter];
217 li["limits"] = lim.data.minMaxJoint.limits;
218 break;
219 case Linear:
220 li["type"] = "linear";
221 li["referenceParameter"] =
222 character.parameterTransform.name[lim.data.linear.referenceIndex];
223 li["targetParameter"] = character.parameterTransform.name[lim.data.linear.targetIndex];
224 li["scale"] = lim.data.linear.scale;
225 li["offset"] = lim.data.linear.offset;
226 if (lim.data.linear.rangeMin != -std::numeric_limits<float>::max()) {
227 li["rangeMin"] = lim.data.linear.rangeMin;
228 }
229 if (lim.data.linear.rangeMax != std::numeric_limits<float>::max()) {
230 li["rangeMax"] = lim.data.linear.rangeMax;
231 }
232 break;
233 case LinearJoint:
234 li["type"] = "linear_joint";
235 if (!character.skeleton.joints.empty() &&
236 lim.data.linearJoint.referenceJointIndex < character.skeleton.joints.size()) {
237 li["referenceJoint"] =
238 character.skeleton.joints[lim.data.linearJoint.referenceJointIndex].name;
239 }
240 li["referenceJointParameter"] = lim.data.linearJoint.referenceJointParameter;
241 if (!character.skeleton.joints.empty() &&
242 lim.data.linearJoint.targetJointIndex < character.skeleton.joints.size()) {
243 li["targetJoint"] = character.skeleton.joints[lim.data.linearJoint.targetJointIndex].name;
244 }
245 li["targetJointParameter"] = lim.data.linearJoint.targetJointParameter;
246 li["scale"] = lim.data.linearJoint.scale;
247 li["offset"] = lim.data.linearJoint.offset;
248 if (lim.data.linearJoint.rangeMin != -std::numeric_limits<float>::max()) {
249 li["rangeMin"] = lim.data.linearJoint.rangeMin;
250 }
251 if (lim.data.linearJoint.rangeMax != std::numeric_limits<float>::max()) {
252 li["rangeMax"] = lim.data.linearJoint.rangeMax;
253 }
254 break;
255 case HalfPlane:
256 li["type"] = "half_plane";
257 li["param1"] = character.parameterTransform.name[lim.data.halfPlane.param1];
258 li["param2"] = character.parameterTransform.name[lim.data.halfPlane.param2];
259 li["normal"] = lim.data.halfPlane.normal;
260 li["offset"] = lim.data.halfPlane.offset;
261 break;
262 case Ellipsoid: {
263 li["type"] = "ellipsoid";
264 if (!character.skeleton.joints.empty() &&
265 lim.data.ellipsoid.parent < character.skeleton.joints.size()) {
266 li["parent"] = character.skeleton.joints[lim.data.ellipsoid.parent].name;
267 }
268 if (!character.skeleton.joints.empty() &&
269 lim.data.ellipsoid.ellipsoidParent < character.skeleton.joints.size()) {
270 li["ellipsoidParent"] =
271 character.skeleton.joints[lim.data.ellipsoid.ellipsoidParent].name;
272 }
273 li["offset"] = lim.data.ellipsoid.offset * toM();
274 auto eli = lim.data.ellipsoid.ellipsoid;
275 eli.translation() *= toM();
276 toJson(eli.matrix(), li["ellipsoid"]);
277 break;
278 }
279 default: {
280 MT_LOGE(
281 "Unknown parameter limit type '{}' from character name '{}'",
282 toString(lim.type),
283 character.name);
284 break;
285 }
286 }
287 }

290ParameterLimits parameterLimitsFromJson(const Character& character, const nlohmann::json& j) {
291 ParameterLimits result;
292
293 for (const auto& element : j) {
294 const std::string type = element.value("type", "");
295 ParameterLimit l;
296 l.weight = element.value("weight", 0.0f);
297 if (type == "minmax") {
298 l.type = MinMax;
299 l.data.minMax.parameterIndex =
300 character.parameterTransform.getParameterIdByName(element.value("parameter", ""));
301 l.data.minMax.limits = fromJson<Vector2f>(element["limits"]);
302 } else if (type == "minmax_joint") {
303 l.type = MinMaxJoint;
304 l.data.minMaxJoint.jointIndex =
305 character.skeleton.getJointIdByName(element.value("jointIndex", ""));
306 const std::string attribute = element.value("jointParameter", "");
307 size_t attributeIndex = kInvalidIndex;
308 for (size_t t = 0; t < kParametersPerJoint; t++) {
309 if (attribute == kJointParameterNames[t]) {
310 attributeIndex = t;
311 break;
312 }
313 }
314 l.data.minMaxJoint.jointParameter = attributeIndex;
315 l.data.minMaxJoint.limits = fromJson<Vector2f>(element["limits"]);
316 } else if (type == "minmax_joint_passive") {
317 l.type = MinMaxJointPassive;
318 l.data.minMaxJoint.jointIndex =
319 character.skeleton.getJointIdByName(element.value("jointIndex", ""));
320 const std::string attribute = element.value("jointParameter", "");
321 size_t attributeIndex = kInvalidIndex;
322 for (size_t t = 0; t < kParametersPerJoint; t++) {
323 if (attribute == kJointParameterNames[t]) {
324 attributeIndex = t;
325 break;
326 }
327 }
328 l.data.minMaxJoint.jointParameter = attributeIndex;
329 l.data.minMaxJoint.limits = fromJson<Vector2f>(element["limits"]);
330 } else if (type == "linear") {
331 l.type = Linear;
332 l.data.linear.referenceIndex = character.parameterTransform.getParameterIdByName(
333 element.value("referenceParameter", ""));
334 l.data.linear.targetIndex =
335 character.parameterTransform.getParameterIdByName(element.value("targetParameter", ""));
336 l.data.linear.scale = element["scale"];
337 l.data.linear.offset = element["offset"];
338 l.data.linear.rangeMin = element.value("rangeMin", -std::numeric_limits<float>::max());
339 l.data.linear.rangeMax = element.value("rangeMax", std::numeric_limits<float>::max());
340 } else if (type == "linear_joint") {
341 l.type = LinearJoint;
342 l.data.linearJoint.referenceJointIndex =
343 character.skeleton.getJointIdByName(element.value("referenceJoint", ""));
344 l.data.linearJoint.targetJointIndex =
345 character.skeleton.getJointIdByName(element.value("targetJoint", ""));
346 l.data.linearJoint.referenceJointParameter = element["referenceJointParameter"].get<size_t>();
347 l.data.linearJoint.targetJointParameter = element["targetJointParameter"].get<size_t>();
348 l.data.linearJoint.scale = element["scale"];
349 l.data.linearJoint.offset = element["offset"];
350 l.data.linearJoint.rangeMin = element.value("rangeMin", -std::numeric_limits<float>::max());
351 l.data.linearJoint.rangeMax = element.value("rangeMax", std::numeric_limits<float>::max());
352 } else if (type == "half_plane") {
353 l.type = HalfPlane;
354 l.data.halfPlane.param1 =
355 character.parameterTransform.getParameterIdByName(element.value("param1", ""));
356 l.data.halfPlane.param2 =
357 character.parameterTransform.getParameterIdByName(element.value("param2", ""));
358 l.data.halfPlane.normal = fromJson<Vector2f>(element["normal"]);
359 l.data.halfPlane.offset = element["offset"];
360 } else if (type == "ellipsoid") {
361 l.type = Ellipsoid;
362 l.data.ellipsoid.parent = character.skeleton.getJointIdByName(element.value("parent", ""));
363 l.data.ellipsoid.ellipsoidParent =
364 character.skeleton.getJointIdByName(element.value("ellipsoidParent", ""));
365 l.data.ellipsoid.offset = fromJson<Vector3f>(element["offset"]);
366 l.data.ellipsoid.offset /= toM();
367 l.data.ellipsoid.ellipsoid.matrix() = fromJson<Matrix4f>(element["ellipsoid"]);
368 l.data.ellipsoid.ellipsoid.translation() /= toM();
369 l.data.ellipsoid.ellipsoidInv = l.data.ellipsoid.ellipsoid.inverse();
370 } else if (type == "elipsoid") {
371 // TODO: Remove once all the model files are migrated to ellipsoid
372 MT_LOGW_ONCE(
373 "Deprecated parameter limit type: {} (typo). Please use 'ellipsoid' instead.", type);
374 l.type = Ellipsoid;
375 l.data.ellipsoid.parent = character.skeleton.getJointIdByName(element.value("parent", ""));
376 l.data.ellipsoid.ellipsoidParent =
377 character.skeleton.getJointIdByName(element.value("elipsoidParent", ""));
378 l.data.ellipsoid.offset = fromJson<Vector3f>(element["offset"]);
379 l.data.ellipsoid.offset /= toM();
380 l.data.ellipsoid.ellipsoid.matrix() = fromJson<Matrix4f>(element["elipsoid"]);
381 l.data.ellipsoid.ellipsoid.translation() /= toM();
382 l.data.ellipsoid.ellipsoidInv = l.data.ellipsoid.ellipsoid.inverse();
383 } else {
384 MT_THROW("Unknown parameter limit type '{}' from character name '{}'.", type, character.name);
385 }
386 result.push_back(l);
387 }
388
389 return result;
390}
