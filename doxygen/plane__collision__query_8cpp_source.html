56template <typename T>
57void PlaneCollisionQueryT<T>::setPlane(const Vector3<T>& planeNormal, T planeOffset) {
58 MT_CHECK(planeNormal.allFinite(), "Plane collision normal must contain only finite values");
59 MT_CHECK(std::isfinite(planeOffset), "Plane collision offset must be finite");
60 const T norm = planeNormal.norm();
61 MT_CHECK(
62 norm > Eps<T>(1e-8f, 1e-17), "Plane collision normal must be non-zero, got norm {}", norm);
63 planeNormal_ = planeNormal / norm;
64 planeOffset_ = planeOffset / norm;
65}

67template <typename T>
68T PlaneCollisionQueryT<T>::primitiveParentSpaceBound(const size_t primitiveIndex) const {
69 const auto& primitive = collisionGeometry_[primitiveIndex];
70 const TransformT<T> primitiveTransform = primitive.transformation.cast<T>();
71 const Vector3<T>& origin = primitiveTransform.translation;
72
73 if (primitive.type == CollisionPrimitiveType::TaperedCapsule) {
74 const Vector3<T> endpoint =
75 origin + primitiveTransform.getAxisX() * primitiveTransform.scale * T(primitive.length);
76 const T radius = primitive.radius.cast<T>().cwiseAbs().maxCoeff();
77 return std::max(origin.norm(), endpoint.norm()) + radius;
78 }
79
80 if (primitive.type == CollisionPrimitiveType::Ellipsoid) {
81 return origin.norm() + primitive.ellipsoidRadii.cast<T>().cwiseAbs().maxCoeff();
82 }
83
84 if (primitive.type == CollisionPrimitiveType::Box) {
85 return origin.norm() + primitive.boxHalfExtents.cast<T>().cwiseAbs().norm();
86 }
87
88 MT_THROW(
89 "PlaneCollisionQuery: unsupported collision primitive type {}",
90 static_cast<int>(primitive.type));
91}

93template <typename T>
94void PlaneCollisionQueryT<T>::updatePrimitiveList() {
95 primitiveIndices_.clear();
96 primitiveIndices_.reserve(collisionGeometry_.size());
97 primitiveParentSpaceBounds_.assign(collisionGeometry_.size(), T(0));
98 for (size_t i = 0; i < collisionGeometry_.size(); ++i) {
99 if (collisionGeometry_[i].parent != kInvalidIndex) {
100 primitiveIndices_.push_back(i);
101 primitiveParentSpaceBounds_[i] = primitiveParentSpaceBound(i);
102 }
103 }
104 candidatePrimitiveIndices_.clear();
105 activeParentCollisions_.clear();
106}

108template <typename T>
109bool PlaneCollisionQueryT<T>::mayPrimitiveContactPlane(
110 const SkeletonStateT<T>& state,
111 const size_t primitiveIndex,
112 const T contactMargin) const {
113 const auto& primitive = collisionGeometry_[primitiveIndex];
114 const TransformT<T>& parentTransform = state.jointState[primitive.parent].transform;
115 const T parentSignedDistance = planeNormal_.dot(parentTransform.translation) - planeOffset_;
116 const T conservativeRadius =
117 std::abs(parentTransform.scale) * primitiveParentSpaceBounds_[primitiveIndex];
118 return parentSignedDistance - conservativeRadius <= contactMargin;
119}

142template <typename T>
143void PlaneCollisionQueryT<T>::update(const SkeletonStateT<T>& state, const T contactMargin) {
144 validateContactMargin(contactMargin);
145
146 MT_PROFILE_EVENT("PlaneCollision: updateCandidates");
147 collisionState_.resize(collisionGeometry_.size());
148 candidatePrimitiveIndices_.clear();
149 candidatePrimitiveIndices_.reserve(primitiveIndices_.size());
150
151 for (const auto primitiveIndex : primitiveIndices_) {
152 if (!mayPrimitiveContactPlane(state, primitiveIndex, contactMargin)) {
153 continue;
154 }
155 collisionState_.updatePrimitive(state, collisionGeometry_, primitiveIndex);
156 candidatePrimitiveIndices_.push_back(primitiveIndex);
157 }
158 activeParentCollisions_.clear();
159}

171template <typename T>
172bool PlaneCollisionQueryT<T>::checkCollision(size_t primitiveIndex, Result& result) const {
173 const auto type = collisionState_.type[primitiveIndex];
174 result.primitiveIndex = primitiveIndex;
175 result.parentJoint = collisionGeometry_[primitiveIndex].parent;
176
177 // `result.position` is the deepest surface point toward the plane (the contact point), offset
178 // from the primitive center/endpoint by the support radius along `-planeNormal_`. Differentiating
179 // at the contact point, rather than the center, lets the joint-chain walk capture orientation and
180 // scale dependence through that world-space support point.
181 if (type == CollisionPrimitiveType::TaperedCapsule) {
182 const Vector3<T> position0 = collisionState_.origin[primitiveIndex];
183 const Vector3<T> position1 =
184 collisionState_.origin[primitiveIndex] + collisionState_.direction[primitiveIndex];
185 const T signedDistance0 = planeNormal_.dot(position0) - planeOffset_;
186 const T signedDistance1 = planeNormal_.dot(position1) - planeOffset_;
187 const T radius0 = collisionState_.radius[primitiveIndex][0];
188 const T radius1 = collisionState_.radius[primitiveIndex][1];
189 const T surfaceDistance0 = signedDistance0 - radius0;
190 const T surfaceDistance1 = signedDistance1 - radius1;
191
192 if (surfaceDistance0 <= surfaceDistance1) {
193 result.signedDistance = signedDistance0;
194 result.radius = radius0;
195 result.position = position0 - radius0 * planeNormal_;
196 } else {
197 result.signedDistance = signedDistance1;
198 result.radius = radius1;
199 result.position = position1 - radius1 * planeNormal_;
200 }
201 } else if (isCenteredPrimitive(type)) {
202 const Vector3<T>& center = collisionState_.origin[primitiveIndex];
203 const Quaternion<T>& orientation = collisionState_.orientation[primitiveIndex];
204 result.signedDistance = planeNormal_.dot(center) - planeOffset_;
205 result.radius = centeredPrimitiveRadiusAlongDirection(
206 type,
207 orientation,
208 collisionState_.ellipsoidRadii[primitiveIndex],
209 collisionState_.boxHalfExtents[primitiveIndex],
210 planeNormal_);
211
212 const Vector3<T> localNormal = orientation.inverse() * planeNormal_;
213 Vector3<T> supportOffset;
214 if (type == CollisionPrimitiveType::Ellipsoid) {
215 const Vector3<T>& radii = collisionState_.ellipsoidRadii[primitiveIndex];
216 const T denom = result.radius > Eps<T>(1e-8f, 1e-16) ? result.radius : T(1);
217 supportOffset = orientation * (radii.cwiseProduct(radii).cwiseProduct(localNormal)) / denom;
218 } else {
219 const Vector3<T>& halfExtents = collisionState_.boxHalfExtents[primitiveIndex];
220 const Vector3<T> localSign =
221 localNormal.unaryExpr([](const T value) { return value >= T(0) ? T(1) : T(-1); });
222 supportOffset = orientation * halfExtents.cwiseAbs().cwiseProduct(localSign);
223 }
224 result.position = center - supportOffset;
225 } else {
226 MT_THROW(
227 "PlaneCollisionQuery: unsupported collision primitive type {}", static_cast<int>(type));
228 }
229
230 result.overlap = result.radius - result.signedDistance;
231 return result.overlap > T(0);
232}

234template <typename T>
235void PlaneCollisionQueryT<T>::updateActiveParentCollisions() {
236 activeParentCollisions_.clear();
237 activeParentCollisions_.reserve(candidatePrimitiveIndices_.size());
238
239 for (const auto primitiveIndex : candidatePrimitiveIndices_) {
240 Result result;
241 if (!checkCollision(primitiveIndex, result)) {
242 continue;
243 }
244
245 auto existing = std::ranges::find_if(activeParentCollisions_, [&](const Result& collision) {
246 return collision.parentJoint == result.parentJoint;
247 });
248 if (existing == activeParentCollisions_.end()) {
249 activeParentCollisions_.push_back(result);
250 continue;
251 }
252
253 if (result.overlap > existing->overlap) {
254 *existing = result;
255 }
256 }

291 const std::span<const JointStateT<T>> states,
292 const T contactMargin) {
293 update(states, contactMargin);
294
295 std::vector<Vector3<T>> points;

320template <typename T>
321std::vector<PlaneCollisionContactPointT<T>>
322PlaneCollisionQueryT<T>::getContactPointsByParentWithDetails(
323 const std::span<const JointStateT<T>> states,
324 const T contactMargin) {
325 update(states, contactMargin);
326
327 std::vector<PlaneCollisionContactPointT<T>> parentContacts;
328 parentContacts.reserve(candidatePrimitiveIndices_.size());
329 parentContactIndices_.assign(states.size(), kInvalidIndex);
330 parentSurfaceDistances_.assign(states.size(), T(0));
331 for (const auto primitiveIndex : candidatePrimitiveIndices_) {
332 Result result;
333 static_cast<void>(checkCollision(primitiveIndex, result));
334 const T surfaceDistance = result.surfaceDistance();
335 if (surfaceDistance > contactMargin) {
336 continue;
337 }
338
339 const size_t parent = result.parentJoint;
340 const size_t contactIndex = parentContactIndices_[parent];
341 if (contactIndex == kInvalidIndex) {
342 parentContactIndices_[parent] = parentContacts.size();
343 parentSurfaceDistances_[parent] = surfaceDistance;
344 parentContacts.push_back({parent, surfaceDistance, result.position});
345 continue;
346 }
347
348 if (surfaceDistance < parentSurfaceDistances_[parent]) {
349 parentSurfaceDistances_[parent] = surfaceDistance;
350 parentContacts[contactIndex] = {parent, surfaceDistance, result.position};
351 }
352 }
353 return parentContacts;
354}
