82template <typename T>
83void SkeletonStateT<T>::set(const Skeleton& referenceSkeleton, bool computeDeriv) {
84 MT_PROFILE_FUNCTION();
85 // get input joints
86 const JointListT<T>& joints = ::momentum::cast<T>(referenceSkeleton.joints);
87
88 // initialize array size variables
89 const size_t numJoints = joints.size();
90
91 // ensure that all variables are valid
92 MT_CHECK(
93 jointParameters.size() == gsl::narrow<Eigen::Index>(numJoints * kParametersPerJoint),
94 "Unexpected joint parameter size. Expected '{}' (# of joints '{}' X kParametersPerJoint '{}') but got '{}'.",
95 numJoints * kParametersPerJoint,
96 numJoints,
97 kParametersPerJoint,
98 jointParameters.size());
99
100 // go over all joint elements and calculate Transformation
101 for (size_t jointID = 0; jointID < numJoints; jointID++) {
102 const Eigen::Index parameterOffset = jointID * kParametersPerJoint;
103
104 // some reference for quick access
105 const JointT<T>& joint = joints[jointID];
106
107 // set joint-state based on parameters
108 // IMPORTANT: this all assumes that parent joints always appear before their children in the
109 // joint list, so their joint state will already be calculated when processing the children
110 if (joint.parent == kInvalidIndex) {
111 jointState[jointID].set(
112 joint, jointParameters.v.template middleRows<7>(parameterOffset), nullptr, computeDeriv);
113 } else {
114 jointState[jointID].set(
115 joint,
116 jointParameters.v.template middleRows<7>(parameterOffset),
117 &jointState[joint.parent],
118 computeDeriv);
119 }
120 }
121
122 // ensure arrays are valid
123 MT_CHECK(jointState.size() == numJoints, "{} is not {}", jointState.size(), numJoints);
124}

136template <typename T>
137StateSimilarity SkeletonStateT<T>::compare(
138 const SkeletonStateT<T>& state1,
139 const SkeletonStateT<T>& state2) {
140 // check that we're actually good to compare the states
141 MT_CHECK(
142 state1.jointParameters.size() == state2.jointParameters.size(),
143 "{} is not {}",
144 state1.jointParameters.size(),
145 state2.jointParameters.size());
146 MT_CHECK(
147 state1.jointState.size() == state2.jointState.size(),
148 "{} is not {}",
149 state1.jointState.size(),
150 state2.jointState.size());
151
152 StateSimilarity result;
153 result.positionError.resize(state1.jointState.size());
154 result.orientationError.resize(state1.jointState.size());
155
156 // calculate position error and orientation error in global coordinates per joint
157 for (size_t i = 0; i < state1.jointState.size(); i++) {
158 result.positionError[i] =
159 (state1.jointState[i].translation() - state2.jointState[i].translation()).norm();
160 const double dot = std::min(
161 std::max(
162 state1.jointState[i].rotation().normalized().dot(
163 state2.jointState[i].rotation().normalized()),
164 T(-1.0)),
165 T(1.0));
166 const double sgn = dot < 0.0 ? -1.0 : 1.0;
167 result.orientationError[i] = gsl::narrow_cast<float>(2.0 * std::acos(sgn * dot));
168 }
169
170 // derive RMSE/max values
171 result.positionRMSE = std::sqrt(
172 result.positionError.squaredNorm() / static_cast<float>(result.positionError.size()));
173 result.orientationRMSE = std::sqrt(
174 result.orientationError.squaredNorm() / static_cast<float>(result.positionError.size()));
175 result.positionMax = result.positionError.maxCoeff();
176 result.orientationMax = result.orientationError.maxCoeff();
177
178 return result;
179}

181// It turns out because the body is topologically sorted, there is a pretty simple algorithm for
182// computing the common ancestor.
183template <typename T>
184TransformT<T> transformAtoB(
185 size_t jointA,
186 size_t jointB,
187 const Skeleton& referenceSkeleton,
188 const SkeletonStateT<T>& skelState) {
189 size_t ancestorA = jointA;
190 size_t ancestorB = jointB;
191
192 TransformT<T> B_to_ancestorB;
193 TransformT<T> A_to_ancestorA;
194
195 while (true) {
196 // Note that we treat kInvalidIndex as if it is the _lowest_ index, as the world is the
197 // topological ancestor of everything.
198 if (ancestorB != kInvalidIndex && (ancestorA == kInvalidIndex || ancestorA < ancestorB)) {
199 // parentB can't possible be a parent of parentA, so move parentB up one level in the
200 // hierarchy.
201 B_to_ancestorB = skelState.jointState[ancestorB].localTransform * B_to_ancestorB;
202 if (ancestorB < referenceSkeleton.joints.size()) {
203 ancestorB = referenceSkeleton.joints[ancestorB].parent;
204 } else {
205 break;
206 }
207 } else if (
208 ancestorA != kInvalidIndex && (ancestorB == kInvalidIndex || ancestorB < ancestorA)) {
209 A_to_ancestorA = skelState.jointState[ancestorA].localTransform * A_to_ancestorA;
210 if (ancestorA < referenceSkeleton.joints.size()) {
211 ancestorA = referenceSkeleton.joints[ancestorA].parent;
212 } else {
213 break;
214 }
215 } else {
216 // Reached a common ancestor of A and B so we can stop.
217 break;
218 }
219 }
220
221 // At this point, ancestorA == ancestorB.
222 MT_CHECK(ancestorA == ancestorB, "{} is not {}", ancestorA, ancestorB);
223 return B_to_ancestorB.inverse() * A_to_ancestorA;
224}

228template <typename T>
229[[nodiscard]] Eigen::Vector<T, kParametersPerJoint> localTransformToJointParameters(
230 const Joint& joint,
231 const TransformT<T>& localTransform) {
232 Eigen::Vector<T, kParametersPerJoint> result = Eigen::Vector<T, 7>::Zero();
233 result.template segment<3>(0) =
234 localTransform.translation - joint.translationOffset.template cast<T>();
235
236 const Eigen::Quaternion<T> localRotation =
237 joint.preRotation.template cast<T>().inverse().template cast<T>() * localTransform.rotation;
238 result.template segment<3>(3) = quaternionToEuler(localRotation);
239 result(6) = std::log2(localTransform.scale);
240 return result;
241}

245template <typename T>
246JointParametersT<T> skeletonStateToJointParameters(
247 const SkeletonStateT<T>& state,
248 const Skeleton& skeleton) {
249 JointParametersT<T> result =
250 JointParametersT<T>::Zero(skeleton.joints.size() * kParametersPerJoint);
251 for (size_t i = 0; i < skeleton.joints.size(); ++i) {
252 const auto parentJoint = skeleton.joints[i].parent;
253 TransformT<T> parentXF;
254 if (parentJoint != kInvalidIndex) {
255 parentXF = state.jointState[parentJoint].transform;
256 }
257
258 // parentXF * localXF = jointXF
259 const TransformT<T> localXF = state.jointState[i].localTransform;
260 result.v.template segment<kParametersPerJoint>(i * kParametersPerJoint) =
261 localTransformToJointParameters(skeleton.joints[i], localXF);
262 }
263
264 return result;
265}

267template <typename T>
268JointParametersT<T> skeletonStateToJointParameters(
269 const TransformListT<T>& state,
270 const Skeleton& skeleton) {
271 JointParametersT<T> result =
272 JointParametersT<T>::Zero(skeleton.joints.size() * kParametersPerJoint);
273 for (size_t i = 0; i < skeleton.joints.size(); ++i) {
274 const auto parentJoint = skeleton.joints[i].parent;
275 TransformT<T> parentXF;
276 if (parentJoint != kInvalidIndex) {
277 parentXF = state[parentJoint];
278 }
279
280 // parentXF * localXF = jointXF
281 const TransformT<T> localXF = parentXF.inverse() * state[i];
282 result.v.template segment<kParametersPerJoint>(i * kParametersPerJoint) =
283 localTransformToJointParameters(skeleton.joints[i], localXF);
284 }
285 return result;
286}
