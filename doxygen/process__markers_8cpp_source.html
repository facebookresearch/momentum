23Eigen::MatrixXf processMarkers(
24 momentum::Character& character,
25 momentum::ModelParameters& identity,
26 const std::vector<std::vector<momentum::Marker>>& markerData,
27 const TrackingConfig& trackingConfig,
28 const CalibrationConfig& calibrationConfig,
29 bool calibrate,
30 size_t firstFrame,
31 size_t maxFrames) {
32 MT_THROW_IF(
33 firstFrame > markerData.size(),
34 "First frame {} can't exceed total frames {}",
35 firstFrame,
36 markerData.size());
37 const size_t lastFrame =
38 maxFrames > 0 ? std::min(firstFrame + maxFrames, markerData.size()) : markerData.size();
39 const gsl::span<const std::vector<momentum::Marker>> inputData(
40 markerData.data() + firstFrame, lastFrame - firstFrame);
41 // calibrate model and locators
42 if (calibrate) {
43 MT_CHECK(
44 !(calibrationConfig.globalScaleOnly & calibrationConfig.locatorsOnly),
45 "globalScaleOnly and locatorsOnly are exclusive; they cannot both be true.");
46
47 if (calibrationConfig.locatorsOnly) {
48 // The output locators will be written to character.
49 calibrateLocators(inputData, calibrationConfig, identity, character);
50 } else {
51 // The output locators will be written to character. The output identity will be saved in the
52 // identity variable.
53 calibrateModel(inputData, calibrationConfig, character, identity);
54 }
55 }
56
57 // track motion; identity parameters will be repeated for every frame in finalMotion.
58 Eigen::MatrixXf finalMotion = trackPosesPerframe(inputData, character, identity, trackingConfig);
59
60 if (trackingConfig.debug) {
61 auto errors = getLocatorError(inputData, finalMotion, character);
62 MT_LOGI("Average marker error: {}", errors.first);
63 MT_LOGI("Max marker error: {}", errors.second);
64 }
65 return finalMotion;
66}

68void processMarkerFile(
69 const std::string& inputMarkerFile,
70 const std::string& outputFile,
71 const TrackingConfig& trackingConfig,
72 const CalibrationConfig& calibrationConfig,
73 const ModelOptions& modelOptions,
74 bool calibrate,
75 size_t firstFrame,
76 size_t maxFrames) {
77 // validate output file extension and quit early
78 const auto outExtension = filesystem::path(outputFile).extension();
79 if (outExtension != ".fbx" && outExtension != ".gltf" && outExtension != ".glb") {
80 MT_LOGE(
81 "Invalid output file type {}; supported types are glb, gltf, and fbx",
82 outExtension.string());
83 return;
84 }
85
86 // read input character and optionally its identity parameters
87 auto [character, identity] = loadCharacterWithIdentity(modelOptions);
88
89 // read marker data
90 const auto actors = momentum::loadMarkers(inputMarkerFile);
91
92 // first try to see if we have an actor with the same name as the character
93 std::optional<MarkerSequence> actor;
94 for (const auto& a : actors) {
95 if (a.name == character.name) {
96 actor = a;
97 break;
98 }
99 }
100
101 // if not, get the main actor
102 if (!actor) {
103 const int index = findMainSubjectIndex(actors);
104 if (index >= 0) {
105 actor = actors[index];
106 }
107 }
108
109 // did we find an actor?
110 if (!actor) {
111 MT_LOGE("No marker sequence found in the marker file: {}", inputMarkerFile);
112 return;
113 }
114 try {
115 Eigen::MatrixXf finalMotion = processMarkers(
116 character,
117 identity,
118 actor->frames,
119 trackingConfig,
120 calibrationConfig,
121 calibrate,
122 firstFrame,
123 maxFrames);
124
125 // name the character after the actor name
126 character.name = actor->name;
127
128 // save results
129 const size_t lastFrame = maxFrames > 0 ? std::min(firstFrame + maxFrames, actor->frames.size())
130 : actor->frames.size();
131 saveMotion(
132 outputFile,
133 character,
134 identity,
135 finalMotion,
136 gsl::span<const std::vector<momentum::Marker>>(
137 actor->frames.data() + firstFrame, actor->frames.data() + lastFrame),
138 actor->fps);
139 MT_LOGI("{} saved", outputFile);
140 } catch (std::exception& e) {
141 MT_LOGE("Failed to track the markers from file {}, exception: {}", inputMarkerFile, e.what());
142 }
143}
