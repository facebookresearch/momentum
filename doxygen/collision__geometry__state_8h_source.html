111template <typename T>
112bool overlaps(
113 const Vector3<T>& originA,
114 const Vector3<T>& directionA,
115 const Vector2<T>& radiiA,
116 T deltaA,
117 const Vector3<T>& originB,
118 const Vector3<T>& directionB,
119 const Vector2<T>& radiiB,
120 T deltaB,
121 T& outDistance,
122 Vector2<T>& outClosestPoints,
123 T& outOverlap) {
124 // Use the sum of max radii as a proximity cutoff so closestPointsOnSegments can early-out
125 // when segments are guaranteed too far apart for the capsules to overlap.
126 const T maxRadiiSum = radiiA.maxCoeff() + radiiB.maxCoeff();
127
128 auto [success, closestDist, closestPoints] =
129 closestPointsOnSegments<T>(originA, directionA, originB, directionB, maxRadiiSum);
130
131 if (!success) {
132 return false;
133 }
134
135 outClosestPoints = closestPoints;
136
137 // Linearly interpolate each tapered radius at the closest-point parameter along its segment,
138 // then sum to get the combined radius the gap must clear for the capsules to be disjoint.
139 const T radiusAtClosestPoints =
140 radiiA[0] + closestPoints[0] * deltaA + radiiB[0] + closestPoints[1] * deltaB;
141
142 outOverlap = radiusAtClosestPoints - closestDist;
143 outDistance = closestDist;
144
145 // Reject coincident centerlines (closestDist below Eps) to avoid degenerate contact normals
146 // downstream when the two segments are effectively the same line.
147 return (outOverlap > T(0)) && (closestDist >= Eps<T>(1e-8, 1e-17));
148}

151template <typename T>
152[[nodiscard]] T closestPointOnSegmentParameter(
153 const Vector3<T>& origin,
154 const Vector3<T>& direction,
155 const Vector3<T>& point) {
156 const T directionNormSq = direction.squaredNorm();
157 if (directionNormSq <= Eps<T>(1e-8, 1e-17)) {
158 return T(0);
159 }
160
161 return std::clamp((point - origin).dot(direction) / directionNormSq, T(0), T(1));
162}

165template <typename T>
166[[nodiscard]] T ellipsoidRadiusAlongDirection(
167 const Quaternion<T>& orientation,
168 const Vector3<T>& radii,
169 const Vector3<T>& unitDirection) {
170 const Vector3<T> localDirection = orientation.inverse() * unitDirection;
171 return radii.cwiseAbs().cwiseProduct(localDirection).norm();
172}

175template <typename T>
176[[nodiscard]] T boxRadiusAlongDirection(
177 const Quaternion<T>& orientation,
178 const Vector3<T>& halfExtents,
179 const Vector3<T>& unitDirection) {
180 const Vector3<T> localDirection = orientation.inverse() * unitDirection;
181 return halfExtents.cwiseAbs().dot(localDirection.cwiseAbs());
182}

191[[nodiscard]] T centeredPrimitiveRadiusAlongDirection(

229template <typename T>
230[[nodiscard]] bool overlapsCapsuleCenteredPrimitive(
231 const Vector3<T>& capsuleOrigin,
232 const Vector3<T>& capsuleDirection,
233 const Vector2<T>& capsuleRadii,
234 T capsuleDelta,
235 CollisionPrimitiveType primitiveType,
236 const Vector3<T>& primitiveCenter,
237 const Quaternion<T>& primitiveOrientation,
238 const Vector3<T>& ellipsoidRadii,
239 const Vector3<T>& boxHalfExtents,
240 T& outDistance,
241 T& outCapsuleParam,
242 T& outOverlap,
243 T& outCapsuleRadius,
244 T& outPrimitiveRadius) {
245 outCapsuleParam =
246 closestPointOnSegmentParameter(capsuleOrigin, capsuleDirection, primitiveCenter);
247 const Vector3<T> capsulePoint = capsuleOrigin + capsuleDirection * outCapsuleParam;
248 const Vector3<T> separation = capsulePoint - primitiveCenter;
249 outDistance = separation.norm();
250 if (outDistance < Eps<T>(1e-8, 1e-17)) {
251 return false;
252 }
253
254 const Vector3<T> normal = separation / outDistance;
255 outCapsuleRadius = capsuleRadii[0] + outCapsuleParam * capsuleDelta;
256 outPrimitiveRadius = centeredPrimitiveRadiusAlongDirection(
257 primitiveType, primitiveOrientation, ellipsoidRadii, boxHalfExtents, normal);
258 outOverlap = outCapsuleRadius + outPrimitiveRadius - outDistance;
259 return outOverlap > T(0);
260}

274template <typename T>
275[[nodiscard]] bool overlaps(
276 const CollisionGeometryStateT<T>& collisionState,
277 const CollisionGeometry& collisionGeometry,
278 size_t indexA,
279 size_t indexB,
280 CollisionOverlapResultT<T>& result) {
281 const auto typeA = collisionGeometry[indexA].type;
282 const auto typeB = collisionGeometry[indexB].type;
283
284 if (typeA == CollisionPrimitiveType::TaperedCapsule &&
285 typeB == CollisionPrimitiveType::TaperedCapsule) {
286 if (!overlaps(
287 collisionState.origin[indexA],
288 collisionState.direction[indexA],
289 collisionState.radius[indexA],
290 collisionState.delta[indexA],
291 collisionState.origin[indexB],
292 collisionState.direction[indexB],
293 collisionState.radius[indexB],
294 collisionState.delta[indexB],
295 result.distance,
296 result.closestPoints,
297 result.overlap)) {
298 return false;
299 }
300
301 result.positionA =
302 collisionState.origin[indexA] + collisionState.direction[indexA] * result.closestPoints[0];
303 result.positionB =
304 collisionState.origin[indexB] + collisionState.direction[indexB] * result.closestPoints[1];
305 result.radiusA =
306 collisionState.radius[indexA][0] + result.closestPoints[0] * collisionState.delta[indexA];
307 result.radiusB =
308 collisionState.radius[indexB][0] + result.closestPoints[1] * collisionState.delta[indexB];
309 return true;
310 }
311
312 if (typeA == CollisionPrimitiveType::TaperedCapsule && isCenteredPrimitive(typeB)) {
313 if (!overlapsCapsuleCenteredPrimitive(
314 collisionState.origin[indexA],
315 collisionState.direction[indexA],
316 collisionState.radius[indexA],
317 collisionState.delta[indexA],
318 typeB,
319 collisionState.origin[indexB],
320 collisionState.orientation[indexB],
321 collisionState.ellipsoidRadii[indexB],
322 collisionState.boxHalfExtents[indexB],
323 result.distance,
324 result.closestPoints[0],
325 result.overlap,
326 result.radiusA,
327 result.radiusB)) {
328 return false;
329 }
330
331 result.closestPoints[1] = T(0);
332 result.positionA =
333 collisionState.origin[indexA] + collisionState.direction[indexA] * result.closestPoints[0];
334 result.positionB = collisionState.origin[indexB];
335 return true;
336 }
337
338 if (isCenteredPrimitive(typeA) && typeB == CollisionPrimitiveType::TaperedCapsule) {
339 if (!overlapsCapsuleCenteredPrimitive(
340 collisionState.origin[indexB],
341 collisionState.direction[indexB],
342 collisionState.radius[indexB],
343 collisionState.delta[indexB],
344 typeA,
345 collisionState.origin[indexA],
346 collisionState.orientation[indexA],
347 collisionState.ellipsoidRadii[indexA],
348 collisionState.boxHalfExtents[indexA],
349 result.distance,
350 result.closestPoints[1],
351 result.overlap,
352 result.radiusB,
353 result.radiusA)) {
354 return false;
355 }
356
357 result.closestPoints[0] = T(0);
358 result.positionA = collisionState.origin[indexA];
359 result.positionB =
360 collisionState.origin[indexB] + collisionState.direction[indexB] * result.closestPoints[1];
361 return true;
362 }
363
364 if (isCenteredPrimitive(typeA) && isCenteredPrimitive(typeB)) {
365 const Vector3<T> separation = collisionState.origin[indexA] - collisionState.origin[indexB];
366 result.distance = separation.norm();
367 if (result.distance < Eps<T>(1e-8, 1e-17)) {
368 return false;
369 }
370
371 const Vector3<T> normal = separation / result.distance;
372 result.radiusA = centeredPrimitiveRadiusAlongDirection(
373 typeA,
374 collisionState.orientation[indexA],
375 collisionState.ellipsoidRadii[indexA],
376 collisionState.boxHalfExtents[indexA],
377 normal);
378 result.radiusB = centeredPrimitiveRadiusAlongDirection(
379 typeB,
380 collisionState.orientation[indexB],
381 collisionState.ellipsoidRadii[indexB],
382 collisionState.boxHalfExtents[indexB],
383 normal);
384 result.overlap = result.radiusA + result.radiusB - result.distance;
385 result.positionA = collisionState.origin[indexA];
386 result.positionB = collisionState.origin[indexB];
387 result.closestPoints = Vector2<T>::Zero();
388 return result.overlap > T(0);
389 }
390
391 return false;
392}

400template <typename T>
401void updateAabb(
402 axel::BoundingBox<T>& aabb,
403 const Vector3<T>& originA,
404 const Vector3<T>& direction,
405 const Vector2<T>& radii) {
406 const Vector3<T> radius0 = Vector3<T>::Constant(radii[0]);
407 const Vector3<T> radius1 = Vector3<T>::Constant(radii[1]);
408 const Vector3<T> originB = originA + direction;
409
410 const Vector3<T> minA = originA - radius0;
411 const Vector3<T> maxA = originA + radius0;
412 const Vector3<T> minB = originB - radius1;
413 const Vector3<T> maxB = originB + radius1;
414
415 aabb.aabb.min() = minA.cwiseMin(minB);
416 aabb.aabb.max() = maxA.cwiseMax(maxB);
417}

430template <typename T>
431void updateEllipsoidAabb(
432 axel::BoundingBox<T>& aabb,
433 const Vector3<T>& center,
434 const Quaternion<T>& orientation,
435 const Vector3<T>& radii) {
436 const Eigen::Matrix<T, 3, 3> rotation = orientation.toRotationMatrix();
437 const Vector3<T> absRadii = radii.cwiseAbs();
438 Vector3<T> halfExtents;
439 for (int i = 0; i < 3; ++i) {
440 halfExtents[i] = rotation.row(i).cwiseProduct(absRadii.transpose()).norm();
441 }
442 aabb.aabb.min() = center - halfExtents;
443 aabb.aabb.max() = center + halfExtents;
444}

456template <typename T>
457void updateBoxAabb(
458 axel::BoundingBox<T>& aabb,
459 const Vector3<T>& center,
460 const Quaternion<T>& orientation,
461 const Vector3<T>& halfExtents) {
462 const Eigen::Matrix<T, 3, 3> absRotation = orientation.toRotationMatrix().cwiseAbs();
463 const Vector3<T> aabbHalfExtents = absRotation * halfExtents.cwiseAbs();
464 aabb.aabb.min() = center - aabbHalfExtents;
465 aabb.aabb.max() = center + aabbHalfExtents;
466}

470void updateAabb(
471 axel::BoundingBox<T>& aabb,
472 const CollisionGeometryStateT<T>& collisionState,
473 size_t index) {
474 if (collisionState.type[index] == CollisionPrimitiveType::TaperedCapsule) {
475 updateAabb(
476 aabb,
477 collisionState.origin[index],
478 collisionState.direction[index],
479 collisionState.radius[index]);
480 return;
481 }
482
483 if (collisionState.type[index] == CollisionPrimitiveType::Ellipsoid) {
484 updateEllipsoidAabb(
485 aabb,
486 collisionState.origin[index],
487 collisionState.orientation[index],
488 collisionState.ellipsoidRadii[index]);
489 return;
490 }
491
492 if (collisionState.type[index] == CollisionPrimitiveType::Box) {
493 updateBoxAabb(
494 aabb,
495 collisionState.origin[index],
496 collisionState.orientation[index],
497 collisionState.boxHalfExtents[index]);
498 }

516template <typename T, typename S>
517[[nodiscard]] bool isValidCollisionPair(
518 const CollisionGeometryStateT<T>& collisionState,
519 const CollisionGeometry& collisionGeometry,
520 const SkeletonT<S>& skeleton,
521 size_t i,
522 size_t j,
523 bool filterRestPoseOverlaps = true) {
524 const size_t p0 = collisionGeometry[i].parent;
525 const size_t p1 = collisionGeometry[j].parent;
526
527 // World-fixed primitives (kInvalidIndex parent):
528 // - If both are world-fixed, they can't collide with each other in a meaningful way.
529 // - If exactly one is world-fixed, it's always a valid collision pair (no rest-pose
530 // overlap or adjacency checks apply since the world geometry is independent of
531 // the character's skeleton).
532 if (p0 == kInvalidIndex || p1 == kInvalidIndex) {
533 return p0 != p1;
534 }
535
536 // Check for same or adjacent joints
537 if (skeleton.isSameOrAdjacentJoints(p0, p1)) {
538 return false;
539 }
540
541 if (!filterRestPoseOverlaps) {
542 return true;
543 }
544
545 // Check for rest-pose overlap (permanently overlapping pairs like adjacent body parts)
546 CollisionOverlapResultT<T> overlap;
547 return !overlaps(collisionState, collisionGeometry, i, j, overlap);
548}
