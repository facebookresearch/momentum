50template <typename T>
51void GaussNewtonSolverQRT<T>::doIteration() {
52 MT_PROFILE_FUNCTION();
53
54 const auto sf = static_cast<SkeletonSolverFunctionT<T>*>(this->solverFunction_);
55
56 const auto skeleton = sf->getSkeleton();
57 const auto parameterTransform = sf->getParameterTransform();
58
59 {
60 MT_PROFILE_EVENT("JtJR - update state");
61 skeletonState_->set(parameterTransform->apply(this->parameters_), *skeleton);
62 }
63
64 const auto nFullParams = gsl::narrow<Eigen::Index>(this->parameters_.size());
65
66 std::vector<Eigen::Index> enabledParameters;
67 enabledParameters.reserve(nFullParams);
68 for (int iFullParam = 0; iFullParam < nFullParams; ++iFullParam) {
69 if (this->activeParameters_.test(iFullParam)) {
70 enabledParameters.push_back(iFullParam);
71 }
72 }
73
74 const auto nSubsetParams = gsl::narrow_cast<int>(enabledParameters.size());
75
76 // momentum solves the problem (J^T*J + lambda*I) x = J^T*r;
77 // the QR solver wants the square root of that lambda.
78 qrSolver_.reset(nSubsetParams, std::sqrt(regularization_));
79
80 double error_orig = 0.0;
81 for (auto errorFunction : sf->getErrorFunctions()) {
82 if (errorFunction->getWeight() <= 0) {
83 continue;
84 }
85
86 const auto rows = gsl::narrow<Eigen::Index>(errorFunction->getJacobianSize());
87
88 jacobian_.resizeAndSetZero(rows, nFullParams);
89 residual_.resizeAndSetZero(rows);
90
91 int usedRows = 0;
92 error_orig += errorFunction->getJacobian(
93 this->parameters_, *skeletonState_, jacobian_.mat(), residual_.mat(), usedRows);
94 if (usedRows == 0) {
95 continue;
96 }
97
98 qrSolver_.addMutating(
99 ColumnIndexedMatrix<Eigen::MatrixX<T>>(
100 jacobian_.mat().topRows(usedRows), enabledParameters),
101 residual_.mat().topRows(usedRows));
102 }
103
104 this->error_ = error_orig;
105
106 const Eigen::VectorX<T> subsetDelta = qrSolver_.result();
107
108 Eigen::VectorX<T> searchDir = Eigen::VectorX<T>::Zero(nFullParams);
109 for (int iSubsetParam = 0; iSubsetParam < nSubsetParams; ++iSubsetParam) {
110 const auto iFullParam = enabledParameters[iSubsetParam];
111 searchDir(iFullParam) = subsetDelta(iSubsetParam);
112 }
113
114 if (doLineSearch_) {
115 const double innerProd = -qrSolver_.At_times_b().dot(subsetDelta);
116
117 // Line search:
118 const float c_1 = 1e-4f;
119 const float tau = 0.5f;
120 float alpha = 1.0f;
121
122 const Eigen::VectorX<T> parameters_orig = this->parameters_;
123 for (size_t kStep = 0; kStep < 10 && std::fpclassify(alpha) == FP_NORMAL; ++kStep) {
124 // update the this->parameters_
125 this->parameters_ = parameters_orig;
126 this->solverFunction_->updateParameters(this->parameters_, alpha * searchDir);
127
128 const double error_new = this->solverFunction_->getError(this->parameters_);
129
130 if ((error_orig - error_new) >= c_1 * alpha * -innerProd) {
131 break;
132 }
133
134 // Reduce step size:
135 alpha = alpha * tau;
136 }
137 } else {
138 this->solverFunction_->updateParameters(this->parameters_, searchDir);
139 }
140}
