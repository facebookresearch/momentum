44bool hasMomentumExtension(const fx::gltf::Document& model) {
45 bool hasExtension = model.extensionsAndExtras.count("extensions") != 0 &&
46 model.extensionsAndExtras["extensions"].count("FB_momentum") != 0;
47 return hasExtension;
48}

50nlohmann::json getMomentumExtension(const nlohmann::json& extensionsAndExtras) {
51 if (extensionsAndExtras.count("extensions") != 0 &&
52 extensionsAndExtras["extensions"].count("FB_momentum") != 0) {
53 return extensionsAndExtras["extensions"]["FB_momentum"];
54 } else {
55 return nlohmann::json::object();
56 }

59TaperedCapsule createCollisionCapsule(const fx::gltf::Node& node, const nlohmann::json& extension) {
60 TaperedCapsule tc;
61 tc.parent = kInvalidIndex;
62 tc.transformation = Transform();
63 tc.transformation.rotation = toMomentumQuaternionf(node.rotation);
64 tc.transformation.translation = toMomentumVec3f(node.translation);
65
66 try {
67 tc.length = extension["length"];
68 tc.radius = fromJson<Vector2f>(extension["radius"]);
69 } catch (const std::exception&) {
70 MT_THROW(
71 "Fail to parse json {} for collision capsule {} when loading character.",
72 node.name,
73 extension.dump());
74 }
75 return tc;
76}

78Locator createLocator(const fx::gltf::Node& node, const nlohmann::json& extension) {
79 Locator loc;
80 loc.parent = kInvalidIndex;
81 loc.name = node.name;
82 loc.offset = toMomentumVec3f(node.translation);
83
84 try {
85 loc.weight = extension["weight"];
86 if (extension.contains("limitOrigin")) {
87 loc.limitOrigin = fromJson<Vector3f>(extension["limitOrigin"]);
88 } else {
89 loc.limitOrigin = loc.offset;
90 }
91 loc.limitWeight = fromJson<Vector3f>(extension["limitWeight"]);
92 loc.locked = fromJson<Vector3i>(extension["locked"]);
93 } catch (const std::exception&) {
94 MT_THROW(
95 "Fail to parse json {} for locator {} when loading character.",
96 extension.dump(),
97 node.name);
98 }
99 return loc;
100}

111// Return whether or not the node is allowed in the skeleton hierarchy, which includes the
112// joints, locators and collision geometries.
113// The skinned mesh should not exist in the same hierarchy as the skeleton.
114// Note that markers have mesh, so we are allowing the nodes with mesh in the hierarchy

116 return (node.skin < 0) && (node.camera < 0);

119[[nodiscard]] std::vector<uint32_t> getAncestorsIncludingSelf(
120 uint32_t nodeId,
121 const std::vector<uint32_t>& parents) {
122 auto ancestors = std::vector<uint32_t>();
123 auto curNode = nodeId;
124 while (curNode != kInvalidNodeId) {
125 ancestors.push_back(curNode);
126 curNode = parents[curNode];
127 }
128 std::reverse(ancestors.begin(), ancestors.end());
129 return ancestors;
130}

132[[nodiscard]] uint32_t findClosestCommonAncestor(
133 const uint32_t node1,
134 const uint32_t node2,
135 const std::vector<uint32_t>& parents) {
136 const auto ancestors1 = getAncestorsIncludingSelf(node1, parents);
137 const auto ancestors2 = getAncestorsIncludingSelf(node2, parents);
138 const auto maxNumIter = std::min(ancestors1.size(), ancestors2.size());
139 auto commonAncestor = kInvalidNodeId;
140 for (auto i = 0; i < maxNumIter; i++) {
141 if (ancestors1[i] == ancestors2[i]) {
142 commonAncestor = ancestors1[i];
143 } else {
144 break;
145 }
146 }
147 return commonAncestor;
148}

150// Return a vector of roots for each skeleton. Each item represents a skeleton, and for each
151// skeleton multiple root nodes are allowed.
152std::vector<std::vector<uint32_t>> gatherSkeletonRoots(const fx::gltf::Document& model) {
153 MT_CHECK(model.scene < model.scenes.size(), "{}: {}", model.scene, model.scenes.size());
154 // If no skinned mesh is present, consider the scene root nodes as skeleton roots.
155 if (model.meshes.empty() || model.skins.empty()) {
156 std::vector<uint32_t> roots = model.scenes[model.scene].nodes;
157 return {roots};
158 }
159
160 // Build a parent map for ancestor query
161 auto parentMap = std::vector<uint32_t>(model.nodes.size(), kInvalidNodeId);
162 auto nodeStack = model.scenes[model.scene].nodes;
163 while (!nodeStack.empty()) {
164 std::vector<uint32_t> nextStack;
165 for (const auto& nodeId : nodeStack) {
166 // NOLINTNEXTLINE(facebook-hte-ParameterUncheckedArrayBounds)
167 const auto& node = model.nodes[nodeId];
168 nextStack.reserve(nextStack.size() + node.children.size());
169 nextStack.insert(nextStack.end(), node.children.begin(), node.children.end());
170 for (auto child : model.nodes[nodeId].children) {
171 parentMap[child] = nodeId;
172 }
173 }
174 nodeStack.swap(nextStack);
175 }
176
177 // Assumption: all the joints in the skin are directly connected (i.e. no non-joint node in
178 // between two joints) Merge skeleton within the same skin. If they are in the same hierarchy,
179 // they can be merged under their closest common ancestor.
180 auto skinSkeletonRoots =
181 std::vector<std::vector<uint32_t>>(model.skins.size(), std::vector<uint32_t>());
182 for (auto skinId = 0; skinId < model.skins.size(); skinId++) {
183 const auto& skin = model.skins[skinId];
184 auto& roots = skinSkeletonRoots[skinId];
185 for (const auto jointId : skin.joints) {
186 // NOLINTNEXTLINE(facebook-hte-LocalUncheckedArrayBounds)
187 const auto parentId = parentMap[jointId];
188 auto parentIter = std::find(skin.joints.begin(), skin.joints.end(), parentId);
189 if (parentIter == skin.joints.end()) {
190 bool foundSibling = false;
191 for (auto i = 0; i < roots.size(); i++) {
192 auto commonAncestor = findClosestCommonAncestor(jointId, roots[i], parentMap);
193 if (commonAncestor != kInvalidNodeId) {
194 roots[i] = commonAncestor;
195 foundSibling = true;
196 break;
197 }
198 }
199 if (!foundSibling) {
200 skinSkeletonRoots[skinId].push_back(jointId);
201 }
202 }
203 }
204 }
205
206 auto result = std::vector<std::vector<uint32_t>>(model.skins.size());
207 // Merge skeleton between different skins. If the skeleton root of one is an ancestor of another,
208 // the other one can be merged to the same root.
209 for (auto skinId = 0; skinId < model.skins.size(); skinId++) {
210 MT_CHECK(
211 !skinSkeletonRoots[skinId].empty(),
212 "id: {}, size: {}",
213 skinId,
214 skinSkeletonRoots[skinId].size());
215 if (skinSkeletonRoots[skinId].size() > 1) {
216 // NOLINTNEXTLINE(facebook-hte-LocalUncheckedArrayBounds)
217 auto& roots = result[skinId];
218 roots = skinSkeletonRoots[skinId];
219 } else {
220 auto ancestors = getAncestorsIncludingSelf(skinSkeletonRoots[skinId][0], parentMap);
221 bool isIndependent = true;
222 for (auto checkId = 0; checkId < model.skins.size(); checkId++) {
223 MT_CHECK(
224 !skinSkeletonRoots[checkId].empty(),
225 "id: {}, size: {}",
226 checkId,
227 skinSkeletonRoots[checkId].size());
228 if ((checkId == skinId) || skinSkeletonRoots[checkId].size() > 1) {
229 continue;
230 }
231 if (std::find(ancestors.begin(), ancestors.end(), skinSkeletonRoots[checkId][0]) !=
232 ancestors.end()) {
233 isIndependent = false;
234 break;
235 }
236 }
237 if (isIndependent) {
238 // NOLINTNEXTLINE(facebook-hte-LocalUncheckedArrayBounds)
239 auto& roots = result[skinId];
240 roots.push_back(skinSkeletonRoots[skinId][0]);
241 }
242 }
243 }
244 // Remove the empty skeletons that were skipped
245 result.erase(
246 std::remove_if(
247 result.begin(),
248 result.end(),
249 [](const std::vector<uint32_t>& roots) { return roots.empty(); }),
250 result.end());
251 return result;
252}

254// DFS the hierarchy graph to reconstruct it.
255// Along the way, save the joints, collision geometry and the locators, as well as a
256// nodeToObjectMap for query. nodeToObjectMap stores mapping from node index to joint index,
257// if the node is a joint. If it's a collision capsule or locator,
258// stores the index in the collision volume array or locator array.
259// If no extension is present in the glb file, right now we will load every not-skinned node as a
260// joint. #TODO: correctly load hierarchy from the skinning information if no momentum extension is
261// in the file.
262void loadHierarchyRecursive(
263 const fx::gltf::Document& model,
264 const int nodeId,
265 size_t parentJointId,
266 JointList& joints,
267 CollisionGeometry_u& collision,
268 LocatorList& locators,
269 std::vector<size_t>& nodeToObjectMap,
270 bool useExtension) {
271 MT_THROW_IF(
272 (nodeId < 0) || (nodeId > model.nodes.size()),
273 "Invalid node id found in the gltf hierarchy: {}",
274 nodeId);
275 const auto& node = model.nodes[nodeId];
276 if (!isHierarchyNode(node)) {
277 return;
278 }
279
280 const auto& extension = getMomentumExtension(node.extensionsAndExtras);
281 const std::string type = extension.value("type", "");
282 MT_CHECK(nodeId < nodeToObjectMap.size(), "id: {}, size: {}", nodeId, nodeToObjectMap.size());
283 MT_CHECK(
284 nodeToObjectMap.size() == model.nodes.size(),
285 "NodeMap: {}, nodes: {}",
286 nodeToObjectMap.size(),
287 model.nodes.size());
288 auto newParentJointId = parentJointId;
289 if (type == "collision_capsule") {
290 // Found collision geometry, should be the end node
291 MT_THROW_IF(
292 parentJointId == kInvalidIndex,
293 "Invalid collision capsule without a parent joint: {}",
294 node.name);
295 auto capsule = createCollisionCapsule(node, extension);
296 capsule.parent = parentJointId;
297 collision->push_back(capsule);
298 nodeToObjectMap[nodeId] = collision->size() - 1;
299 } else if (type == "locator") {
300 // Found locator, should be the end node
301 MT_THROW_IF(
302 parentJointId == kInvalidIndex, "Invalid locator without a parent joint: {}", node.name);
303 Locator loc = createLocator(node, extension);
304 loc.parent = parentJointId;
305 locators.push_back(loc);
306 nodeToObjectMap[nodeId] = locators.size() - 1;
307 } else if ((!useExtension && (node.mesh < 0)) || type == "skeleton_joint") {
308 Joint joint = createJoint(node);
309 joint.parent = parentJointId;
310 joints.push_back(joint);
311 nodeToObjectMap[nodeId] = joints.size() - 1;
312 newParentJointId = joints.size() - 1;
313 }
314 // #TODO: log skipped nodes
315
316 MT_CHECK(!model.nodes.empty());
317 for (auto childId : model.nodes[nodeId].children) {
318 MT_THROW_IF(
319 (childId < 0) || (childId > model.nodes.size()),
320 "Invalid node id found in the gltf hierarchy: {}",
321 childId);
322
323 loadHierarchyRecursive(
324 model,
325 childId,
326 newParentJointId,
327 joints,
328 collision,
329 locators,
330 nodeToObjectMap,
331 useExtension);
332 }
333}

335// Load the hierarchy from the file, including skeleton, collision geometry, locators and a mapping
336// between the nodes and the objects.
337std::tuple<std::string, JointList, CollisionGeometry_u, LocatorList, std::vector<size_t>>
338loadHierarchy(const fx::gltf::Document& model) {
339 JointList joints;
340 auto collision = std::make_unique<CollisionGeometry>();
341 LocatorList locators;
342 auto nodeToObjectMap = std::vector<size_t>(model.nodes.size(), kInvalidIndex);
343 const bool useExtension = hasMomentumExtension(model);
344
345 // Currently only the first found skeleton is used
346 // #TODO: support multiple characters in the file, and use the skin information stored in
347 // CharacterInfo.
348 auto skeletonRoots = gatherSkeletonRoots(model);
349 MT_CHECK(skeletonRoots.size() == 1, "{}", skeletonRoots.size());
350 for (const auto& rootNode : skeletonRoots[0]) {
351 loadHierarchyRecursive(
352 model, rootNode, kInvalidIndex, joints, collision, locators, nodeToObjectMap, useExtension);
353 if (!joints.empty()) {
354 break;
355 }
356 }
357 MT_THROW_IF(model.nodes.empty(), "No valid node found in the gltf file.");
358 std::string name = model.nodes.at(0).name;
359
360 std::sort(
361 collision->begin(), collision->end(), [](const TaperedCapsule& c1, const TaperedCapsule& c2) {
362 int c1Parent = c1.parent == kInvalidIndex ? -1 : static_cast<int>(c1.parent);
363 int c2Parent = c2.parent == kInvalidIndex ? -1 : static_cast<int>(c2.parent);
364 return c1Parent < c2Parent;
365 });
366 std::sort(locators.begin(), locators.end(), [](const Locator& l1, const Locator& l2) {
367 int l1Parent = l1.parent == kInvalidIndex ? -1 : static_cast<int>(l1.parent);
368 int l2Parent = l2.parent == kInvalidIndex ? -1 : static_cast<int>(l2.parent);
369 return l1Parent < l2Parent;
370 });
371 return std::make_tuple(name, joints, std::move(collision), locators, nodeToObjectMap);
372}

374// Return the number of vertices newly loaded
375size_t
376addMesh(const fx::gltf::Document& model, const fx::gltf::Primitive& primitive, Mesh_u& mesh) {
377 if (primitive.mode != fx::gltf::Primitive::Mode::Triangles) {
378 MT_LOGW("Mesh not loaded because it's not a triangle mesh");
379 return 0;
380 }
381
382 // load index buffer
383 auto idxDense = copyAccessorBuffer<uint32_t>(model, primitive.indices);
384 if (idxDense.empty()) {
385 // Try fallback with short indices.
386 auto a = copyAccessorBuffer<uint16_t>(model, primitive.indices);
387 for (const auto& ae : a) {
388 idxDense.push_back(ae);
389 }
390 }
391 MT_CHECK(idxDense.size() % 3 == 0, "{} % 3 = {}", idxDense.size(), idxDense.size() % 3);
392 std::vector<Vector3i> idx(idxDense.size() / 3);
393 std::copy_n(idxDense.data(), idxDense.size(), &idx[0][0]);
394
395 // load vertex position buffer
396 auto pos = copyAccessorBuffer<Vector3f>(model, primitive.attributes.at("POSITION"));
397 toMomentumVec3f(pos);
398
399 // if we have no points or indices, skip loading
400 if (idx.empty() || pos.empty()) {
401 return 0;
402 }
403
404 // load optional normal buffer
405 std::vector<Vector3f> nml;
406 const auto normId = primitive.attributes.find("NORMAL");
407 if (normId != primitive.attributes.end()) {
408 nml = copyAccessorBuffer<Vector3f>(model, normId->second);
409 }
410 MT_CHECK(nml.empty() || nml.size() == pos.size(), "nml: {}, pos: {}", nml.size(), pos.size());
411 MT_LOGT_IF(nml.empty(), "no vertex normal found");
412
413 // load optional color buffer
414 std::vector<Vector3b> col;
415 const auto colorId = primitive.attributes.find("COLOR_0");
416 if (colorId != primitive.attributes.end()) {
417 col = copyAccessorBuffer<Vector3b>(model, colorId->second);
418 }
419 MT_CHECK(col.empty() || col.size() == pos.size(), "col: {}, pos: {}", col.size(), pos.size());
420 MT_LOGT_IF(col.empty(), "no vertex color found");
421
422 // load optional texcoord buffer
423 std::vector<Vector2f> texcoord;
424 auto texcoordId = primitive.attributes.find("TEXCOORD_0");
425 if (texcoordId != primitive.attributes.end()) {
426 texcoord = copyAccessorBuffer<Vector2f>(model, texcoordId->second);
427 }
428 MT_CHECK(
429 texcoord.empty() || texcoord.size() == pos.size(),
430 "texcoord: {}, pos: {}",
431 texcoord.size(),
432 pos.size());
433 MT_LOGT_IF(texcoord.empty(), "no texture coords found");
434
435 const auto kVertexOffset = mesh->vertices.size();
436 // Update vertex indices of the faces!!!
437 if (kVertexOffset > 0) {
438 for (size_t iFace = 0; iFace < idx.size(); ++iFace) {
439 idx[iFace][0] += kVertexOffset;
440 idx[iFace][1] += kVertexOffset;
441 idx[iFace][2] += kVertexOffset;
442 }
443 }
444
445 // append new faces
446 mesh->faces.insert(mesh->faces.end(), idx.begin(), idx.end());
447 mesh->vertices.insert(mesh->vertices.end(), pos.begin(), pos.end());
448 mesh->normals.insert(mesh->normals.end(), nml.begin(), nml.end());
449 mesh->colors.insert(mesh->colors.end(), col.begin(), col.end());
450 mesh->texcoords.insert(mesh->texcoords.end(), texcoord.begin(), texcoord.end());
451 if (!mesh->texcoords.empty()) {
452 mesh->texcoord_faces = mesh->faces;
453 }
454
455 // make sure we have enough normals and colors
456 mesh->normals.resize(mesh->vertices.size(), Vector3f::Zero());
457 mesh->colors.resize(mesh->vertices.size(), Vector3b::Zero());
458 mesh->confidence.resize(mesh->vertices.size(), 1.0f);
459 return pos.size();
460}

462void addSkinWeights(
463 const fx::gltf::Document& model,
464 const fx::gltf::Skin& skin,
465 const fx::gltf::Primitive& primitive,
466 const std::vector<size_t>& nodeToObjectMap,
467 const size_t kNumVertices,
468 SkinWeights_u& skinWeights) {
469 const auto kVertexOffset = skinWeights->index.rows();
470 skinWeights->index.conservativeResize(kVertexOffset + kNumVertices, Eigen::NoChange);
471 skinWeights->weight.conservativeResize(kVertexOffset + kNumVertices, Eigen::NoChange);
472 skinWeights->index.bottomRows(kNumVertices).setZero();
473 skinWeights->weight.bottomRows(kNumVertices).setZero();
474
475 // load skinning in batches of 4 indices/weights at a time (up to 8)
476 for (size_t i = 0; i < 2; i++) {
477 // load skinning index buffer
478 auto jointAttribute = primitive.attributes.find(std::string("JOINTS_") + std::to_string(i));
479 std::vector<Vector4s> jointIndices;
480 if (jointAttribute != primitive.attributes.end()) {
481 jointIndices = copyAccessorBuffer<Vector4s>(model, jointAttribute->second);
482 if (jointIndices.empty()) {
483 // Try fallback with short indices.
484 auto jointIndicesShort = copyAccessorBuffer<Vector4b>(model, jointAttribute->second);
485 for (const auto& value : jointIndicesShort) {
486 jointIndices.push_back(value.cast<uint16_t>());
487 }
488 }
489 }
490
491 // load skinning weight buffer
492 auto weightsAttribute = primitive.attributes.find(std::string("WEIGHTS_") + std::to_string(i));
493 std::vector<Vector4f> weightsData;
494 if (weightsAttribute != primitive.attributes.end()) {
495 weightsData = copyAlignedAccessorBuffer<Vector4f>(model, weightsAttribute->second);
496 if (weightsData.empty()) {
497 MT_LOGW("No skinning weights read");
498 return;
499 }
500 } else {
501 MT_LOGW("No skinning weights stored on primitive");
502 return;
503 }
504
505 if (jointIndices.empty() || weightsData.empty() || jointIndices.size() != weightsData.size() ||
506 jointIndices.size() != kNumVertices) {
507 MT_LOGW_IF(jointIndices.empty() || weightsData.empty(), "Mesh is not skinned to any joint");
508 MT_LOGW_IF(
509 !jointIndices.empty() && !weightsData.empty() &&
510 jointIndices.size() != weightsData.size(),
511 "Inconsistent data: {} vertices are skinned but {} weights found",
512 jointIndices.size(),
513 weightsData.size());
514 MT_LOGW_IF(
515 jointIndices.size() != kNumVertices,
516 "Inconsistent data: {} vertices are skinned but mesh has {} vertices",
517 jointIndices.size(),
518 kNumVertices);
519 return;
520 }
521
522 // copy indices/vertices into our buffer
523 for (Eigen::Index v = 0; v < static_cast<int>(kNumVertices); v++) {
524 for (size_t d = 0; d < 4; d++) {
525 const auto& skinWeight = weightsData[v][d];
526 const auto& jointIdxInSkin = jointIndices[v][d];
527 MT_CHECK(
528 jointIdxInSkin >= 0 && jointIdxInSkin < skin.joints.size(),
529 "{}: {}",
530 jointIdxInSkin,
531 skin.joints.size());
532 const auto& nodeIdx = skin.joints[jointIdxInSkin];
533 const auto& jointIndex = nodeToObjectMap[nodeIdx];
534 if (jointIndex != kInvalidIndex) {
535 skinWeights->index(kVertexOffset + v, i * 4 + d) = (uint32_t)jointIndex;
536 skinWeights->weight(kVertexOffset + v, i * 4 + d) = skinWeight;
537 } else {
538 MT_LOGW("Invalid joint index encountered when reading skinning weights");
539 }
540 }
541 }
542 }
543}

545// Load mesh and skinning information together so we can validate correctness
546std::tuple<Mesh_u, SkinWeights_u> loadSkinnedMesh(
547 const fx::gltf::Document& model,
548 const std::vector<size_t>& meshNodes,
549 const std::vector<size_t>& nodeToObjectMap) {
550 std::vector<size_t> skinnedNodes;
551 skinnedNodes.reserve(meshNodes.size());
552 std::vector<size_t> unskinnedNodes;
553 unskinnedNodes.reserve(meshNodes.size());
554 for (auto nodeId : meshNodes) {
555 MT_CHECK(nodeId >= 0 && nodeId < model.nodes.size(), "{}: {}", nodeId, model.nodes.size());
556 const auto& node = model.nodes[nodeId];
557 if ((node.mesh >= 0) && (node.skin >= 0)) {
558 MT_CHECK(node.mesh < model.meshes.size(), "{}: {}", node.mesh, model.meshes.size());
559 MT_CHECK(node.skin < model.skins.size(), "{}: {}", node.skin, model.skins.size());
560 skinnedNodes.push_back(nodeId);
561 } else if (node.mesh >= 0) {
562 MT_CHECK(node.mesh < model.meshes.size(), "{}: {}", node.mesh, model.meshes.size());
563 unskinnedNodes.push_back(nodeId);
564 }
565 }
566
567 MT_LOGW_IF(
568 !skinnedNodes.empty() && !unskinnedNodes.empty(),
569 "Found both skinned meshes {} and meshes without skinning {}. Unskinned meshes will be ignored.",
570 skinnedNodes.size(),
571 unskinnedNodes.size());
572 auto resultMesh = std::make_unique<Mesh>();
573 auto skinWeights = std::make_unique<SkinWeights>();
574 if (!skinnedNodes.empty()) {
575 for (auto nodeId : skinnedNodes) {
576 // NOLINTNEXTLINE(facebook-hte-ParameterUncheckedArrayBounds)
577 const auto& node = model.nodes[nodeId];
578 const auto& mesh = model.meshes[node.mesh];
579 const auto& skin = model.skins[node.skin];
580 for (const auto& primitive : mesh.primitives) {
581 const auto kNumVertices = addMesh(model, primitive, resultMesh);
582 addSkinWeights(model, skin, primitive, nodeToObjectMap, kNumVertices, skinWeights);
583 MT_CHECK(
584 resultMesh->vertices.size() == skinWeights->index.rows(),
585 "vertices: {}, skinWeights: {}",
586 resultMesh->vertices.size(),
587 skinWeights->index.rows());
588 }
589 }
590 } else if (!unskinnedNodes.empty()) {
591 for (auto nodeId : unskinnedNodes) {
592 // NOLINTNEXTLINE(facebook-hte-ParameterUncheckedArrayBounds)
593 const auto& node = model.nodes[nodeId];
594 const auto& mesh = model.meshes[node.mesh];
595 for (const auto& primitive : mesh.primitives) {
596 addMesh(model, primitive, resultMesh);
597 }
598 }
599 return {std::move(resultMesh), nullptr};

706std::tuple<MotionParameters, IdentityParameters> getMotionFromModel(fx::gltf::Document& model) {
707 const auto& def = getMomentumExtension(model.extensionsAndExtras);
708 const auto motion = def.value("motion", nlohmann::json::object());
709
710 const size_t nframes = motion.value("nframes", 0);
711 const int32_t poseBuffer = motion.value("poses", int32_t(-1));
712 const int32_t offsetBuffer = motion.value("offsets", int32_t(-1));
713
714 if (nframes == 0 || (poseBuffer < 0 && offsetBuffer < 0)) {
715 return {};
716 }
717
718 // get values from gltf binary buffers
719 MatrixXf storedMotion;
720 VectorXf storedIdentity;
721 std::vector<std::string> parameterNames;
722 std::vector<std::string> jointNames;
723 if (poseBuffer >= 0) {
724 const auto poseValues = copyAccessorBuffer<float>(model, poseBuffer);
725 parameterNames = motion.value("parameterNames", std::vector<std::string>());
726
727 if (poseValues.size() != parameterNames.size() * nframes) {
728 return {};
729 }
730
731 storedMotion = Map<const MatrixXf>(poseValues.data(), parameterNames.size(), nframes);
732 }
733 if (offsetBuffer >= 0) {
734 const auto offsetValues = copyAccessorBuffer<float>(model, offsetBuffer);
735 jointNames = motion.value("jointNames", std::vector<std::string>());
736
737 if (offsetValues.size() != jointNames.size() * kParametersPerJoint) {
738 return {};
739 }
740
741 storedIdentity =
742 Map<const VectorXf>(offsetValues.data(), jointNames.size() * kParametersPerJoint);
743 }
744
745 return {{parameterNames, storedMotion}, {jointNames, storedIdentity}};
746}

748fx::gltf::Document loadModel(
749 const std::variant<filesystem::path, gsl::span<const std::byte>>& input) {
750 fx::gltf::Document model;
751 constexpr uint32_t kMax = std::numeric_limits<uint32_t>::max();
752 constexpr fx::gltf::ReadQuotas kQuotas = {8, kMax, kMax};
753
754 std::visit(
755 [&](auto&& arg) {
756 using T = std::decay_t<decltype(arg)>;
757 if constexpr (std::is_same_v<T, filesystem::path>) {
758 // Read first 4 chars for file sensing (https://docs.fileformat.com/3d/glb/)
759 // It is ASCII string "glTF", and can be used to identify data as Binary glTF
760 std::string magic;
761 std::ifstream infile(arg);
762 MT_THROW_IF(!infile.is_open(), "Unable to open: {}", arg.string());
763 std::copy_n(std::istreambuf_iterator<char>(infile.rdbuf()), 4, std::back_inserter(magic));
764 if (magic == "glTF") {
765 model = fx::gltf::LoadFromBinary(arg.string(), kQuotas);
766 } else {
767 model = fx::gltf::LoadFromText(arg.string(), kQuotas);
768 }
769 } else if constexpr (std::is_same_v<T, gsl::span<const std::byte>>) {
770 ispanstream inputStream(arg);
771 model = fx::gltf::LoadFromBinary(inputStream, "", kQuotas);
772 }
773 },
774 input);
775
776 return model;
777}

903// Function to load a glTF file and extract motion data stored in the native GLTF format
904std::tuple<Character, std::vector<SkeletonState>, std::vector<float>>
905loadCharacterWithSkeletonStatesCommon(
906 const std::variant<filesystem::path, gsl::span<const std::byte>>& input) {
907 const fx::gltf::Document model = loadModel(input);
908 std::vector<size_t> nodeToObjectMap;
909 Character character = loadGltfCharacterInternal(model, nodeToObjectMap);
910
911 // Iterate over each animation in the glTF document
912 std::vector<SkeletonState> skeletonStates;
913
914 if (model.animations.empty()) {
915 MT_LOGW("No animation found in gltf file");
916 return {character, skeletonStates, {}};
917 } else if (model.animations.size() > 1) {
918 MT_LOGW("Multiple animations found in gltf file. Only the first one will be used.");
919 }
920
921 const auto& animation = model.animations.front();
922 std::vector<float> timestamps;
923
924 std::vector<SkeletonState> states;
925
926 // Iterate over each channel in the animation
927 for (const auto& channel : animation.channels) {
928 // Get the target node for the channel
929 const auto& node = model.nodes[channel.target.node];
930 const auto& sampler = animation.samplers[channel.sampler];
931
932 if (channel.target.node >= nodeToObjectMap.size() ||
933 nodeToObjectMap.at(channel.target.node) == kInvalidIndex) {
934 continue;
935 }
936
937 const auto boneIdx = nodeToObjectMap[channel.target.node];
938 MT_THROW_IF(
939 boneIdx >= character.skeleton.joints.size(),
940 "Bone index {} is larger than the number of joints in the skeleton ({})",
941 boneIdx,
942 character.skeleton.joints.size());
943
944 MT_THROW_IF(
945 node.name != character.skeleton.joints[boneIdx].name,
946 "Bone mismatch between gltf and character; GLTF: {}, Character: {}",
947 node.name,
948 character.skeleton.joints[boneIdx].name);
949
950 // Check the type of transformation (translation, rotation, scale)
951 std::vector<Vector3f> translations_m;
952 std::vector<Quaternionf> rotations;
953 std::vector<Vector3f> scales;
954 if (channel.target.path == "translation") {
955 // Handle translation data
956 translations_m = copyAccessorBuffer<Vector3f>(model, sampler.output);
957 // Apply translations to the node or character skeleton
958 } else if (channel.target.path == "rotation") {
959 // Handle rotation data
960 rotations = copyAccessorBuffer<Quaternionf>(model, sampler.output);
961 // Apply rotations to the node or character skeleton
962 } else if (channel.target.path == "scale") {
963 // Handle scale data
964 scales = copyAccessorBuffer<Vector3f>(model, sampler.output);
965 // Apply scales to the node or character skeleton
966 } else {
967 continue;
968 }
969
970 auto timestampsCur = copyAccessorBuffer<float>(model, sampler.input);
971 if (timestamps.empty()) {
972 timestamps = timestampsCur;
973 } else {
974 if (timestamps.size() != timestampsCur.size()) {
975 MT_THROW(
976 "Timestamps do not match between channels, got {} and {}",
977 timestamps.size(),
978 timestampsCur.size());
979 }
980 }
981
982 if (states.empty()) {
983 states.resize(
984 timestamps.size(),
985 SkeletonState(
986 JointParameters::Zero(character.parameterTransform.numJointParameters()),
987 character.skeleton));
988 }
989
990 MT_THROW_IF(states.size() != timestamps.size(), "States and timestamps do not match in size");
991
992 if (translations_m.size() == states.size()) {
993 for (size_t i = 0; i < states.size(); ++i) {
994 states.at(i).jointState.at(boneIdx).localTransform.translation = toCm() * translations_m[i];
995 }
996 }
997 if (rotations.size() == states.size()) {
998 for (size_t i = 0; i < states.size(); ++i) {
999 states.at(i).jointState.at(boneIdx).localTransform.rotation = rotations[i];
1000 }
1001 }
1002 if (scales.size() == states.size()) {
1003 for (size_t i = 0; i < states.size(); ++i) {
1004 states.at(i).jointState.at(boneIdx).localTransform.scale = scales[i].x();
1005 }
1006 }
1007 }
1008
1009 for (auto& s : states) {
1010 for (size_t iJoint = 0; iJoint < character.skeleton.joints.size(); ++iJoint) {
1011 Transform parentTransform;
1012 const auto parentJoint = character.skeleton.joints[iJoint].parent;
1013 if (parentJoint != kInvalidIndex) {
1014 parentTransform = s.jointState[parentJoint].transform;
1015 }
1016 s.jointState[iJoint].transform = parentTransform * s.jointState[iJoint].localTransform;
1017 }
1018 }
1019
1020 return {character, states, timestamps};
1021}

1126MarkerSequence loadMarkerSequence(const filesystem::path& filename) {
1127 MarkerSequence result;
1128
1129 fx::gltf::Document model = loadModel(filename);
1130 if (model.animations.empty()) {
1131 return result;
1132 }
1133
1134 const auto& def = getMomentumExtension(model.extensionsAndExtras);
1135 const float fps = def.value("fps", 120.0f);
1136 result.fps = fps;
1137 // TODO: result.name = model.name;
1138
1139 const auto& animation = model.animations.front();
1140
1141 for (const auto& channel : animation.channels) {
1142 // make sure we're on a translation channel
1143 if (channel.target.path != "translation") {
1144 continue;
1145 }
1146
1147 // get the node for the channel
1148 const auto& node = model.nodes[channel.target.node];
1149
1150 // ignore skins and cameras; don't ignore meshes because a marker node may have a mesh for viz
1151 if (node.skin >= 0 || node.camera >= 0) {
1152 continue;
1153 }
1154
1155 const auto& extension = getMomentumExtension(node.extensionsAndExtras);
1156 const std::string type = extension.value("type", "");
1157 if (type == "marker") {
1158 // found a marker, get the sampler
1159 const auto& sampler = animation.samplers[channel.sampler];
1160
1161 // get the data from the sampler
1162 const auto timestamps = copyAccessorBuffer<float>(model, sampler.input);
1163 auto positions = copyAccessorBuffer<Vector3f>(model, sampler.output);
1164 toMomentumVec3f(positions);
1165 MT_CHECK(
1166 timestamps.size() == positions.size(),
1167 "timestamps: {}, positions: {}",
1168 timestamps.size(),
1169 positions.size());
1170
1171 // resize the output array if necessary
1172 const size_t length = static_cast<size_t>(timestamps.back() * fps + 0.5f) + 1;
1173 if (length > result.frames.size()) {
1174 result.frames.resize(length);
1175 }
1176
1177 // go over all data and enter into the output array
1178 for (size_t i = 0; i < timestamps.size(); i++) {
1179 const size_t index = static_cast<size_t>(timestamps[i] * fps + 0.5f);
1180 result.frames[index].emplace_back();
1181 auto& marker = result.frames[index].back();
1182
1183 marker.name = node.name;
1184 marker.occluded = false;
1185 marker.pos = positions[i].cast<double>();
1186 }
1187 }
1188 }
1189
1190 return result;
1191}
