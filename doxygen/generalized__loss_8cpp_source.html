20template <typename T>
21T L2Loss(const T& sqrError, const T& invC2) {
22 return sqrError * invC2;
23}

30template <typename T>
31T L1Loss(const T& sqrError, const T& invC2) {
32 // decided not to skip the -1 so the value itself is meaningful
33 return std::sqrt(L2Loss(sqrError, invC2) + T(1)) - T(1);
34}
35
36template <typename T>
37T derivL1Loss(const T& sqrError, const T& invC2) {
38 return T(0.5) * invC2 / std::sqrt(L2Loss(sqrError, invC2) + T(1));
39}
40
41template <typename T>
42T CauchyLoss(const T& sqrError, const T& invC2) {
43 return std::log(T(0.5) * L2Loss(sqrError, invC2) + T(1));
44}
45
46template <typename T>
47T derivCauchyLoss(const T& sqrError, const T& invC2) {
48 return invC2 / (invC2 * sqrError + T(2));
49}
50
51template <typename T>
52T WelschLoss(const T& sqrError, const T& invC2) {
53 return T(1) - std::exp(T(-0.5) * L2Loss(sqrError, invC2));
54}
55
56template <typename T>
57T derivWelschLoss(const T& sqrError, const T& invC2) {
58 return T(0.5) * invC2 * std::exp(T(-0.5) * L2Loss(sqrError, invC2));
59}

63template <typename T>
64GeneralizedLossT<T>::GeneralizedLossT(const T& a, const T& c) : alpha_(a), invC2_(T(1) / (c * c)) {
65 MT_CHECK(c > 0, "Parameter c should be positive but received {}", c);
66
67 // Use a threshold around special values for both stability and speed
68 if (alpha_ >= kL2 - kEps && alpha_ <= kL2 + kEps) {
69 lossType_ = LossType::L2;
70 } else if (alpha_ >= kL1 - kEps && alpha_ <= kL1 + kEps) {
71 lossType_ = LossType::L1;
72 } else if (alpha_ >= kCauchy - kEps && alpha_ <= kCauchy + kEps) {
73 lossType_ = LossType::Cauchy;
74 } else if (alpha_ <= kWelsch) {
75 lossType_ = LossType::Welsch;
76 } else {
77 lossType_ = LossType::General;
78 }
79}

81template <typename T>
82T GeneralizedLossT<T>::value(const T& sqrError) const {
83 switch (lossType_) {
84 case LossType::L2:
85 return L2Loss(sqrError, invC2_);
86 case LossType::L1:
87 return L1Loss(sqrError, invC2_);
88 case LossType::Cauchy:
89 return CauchyLoss(sqrError, invC2_);
90 case LossType::Welsch:
91 return WelschLoss(sqrError, invC2_);
92 case LossType::General:
93 // General case, slower
94 return (std::pow(L2Loss(sqrError, invC2_) / std::abs(alpha_ - T(2)) + T(1), T(0.5) * alpha_) -
95 T(1)) *
96 std::abs(alpha_ - T(2)) / alpha_;
97 default: {
98 MT_LOGE("Invalid lossType_ ({})", static_cast<int>(lossType_));
99 return {};
100 }
101 }
102}

104template <typename T>
105T GeneralizedLossT<T>::deriv(const T& sqrError) const {
106 switch (lossType_) {
107 case LossType::L2:
108 return derivL2Loss(sqrError, invC2_);
109 case LossType::L1:
110 return derivL1Loss(sqrError, invC2_);
111 case LossType::Cauchy:
112 return derivCauchyLoss(sqrError, invC2_);
113 case LossType::Welsch:
114 return derivWelschLoss(sqrError, invC2_);
115 case LossType::General:
116 // General case, slower
117 return T(0.5) * invC2_ *
118 std::pow(
119 L2Loss(sqrError, invC2_) / std::abs(alpha_ - T(2)) + T(1), T(0.5) * alpha_ - T(1));
120 default: {
121 MT_LOGE("Invalid lossType_ ({})", static_cast<int>(lossType_));
122 return {};
123 }
124 }
