44inline size_t findCommonAncestorForVertex(
45 const SkinWeights& skinWeights,
46 const Skeleton& skeleton,
47 size_t vertexIndex,
48 size_t sdfParent) {
49 if (sdfParent == kInvalidIndex) {
50 return kInvalidIndex;
51 }
52 size_t minSkinnedJoint = kInvalidIndex;
53 for (uint32_t k = 0; k < kMaxSkinJoints; ++k) {
54 const auto w = skinWeights.weight(vertexIndex, k);
55 if (w > 0) {
56 const auto joint = skinWeights.index(vertexIndex, k);
57 if (minSkinnedJoint == kInvalidIndex || joint < minSkinnedJoint) {
58 minSkinnedJoint = joint;
59 }
60 }
61 }
62 if (minSkinnedJoint == kInvalidIndex) {
63 return kInvalidIndex;
64 }
65 return skeleton.commonAncestor(minSkinnedJoint, sdfParent);
66}

72template <typename T, typename SdfColliderType>
73TransformT<T> computeWorldToColliderTransform(
74 const SdfColliderType& collider,
75 const SkeletonStateT<T>& state) {
76 TransformT<T> parentTransform;
77 const auto parentJoint = collider.parentJoint();
78 if (parentJoint != kInvalidIndex) {
79 MT_CHECK(parentJoint < state.jointState.size(), "Invalid parent joint index: {}", parentJoint);
80 parentTransform = state.jointState[parentJoint].transform;
81 }
82 auto colliderWorldTransform =
83 parentTransform * collider.localToParentTransform().template cast<T>();
84 return colliderWorldTransform.inverse();
85}

91template <typename T, typename SdfColliderType>
92axel::BoundingBox<T> computeColliderWorldBoundingBox(
93 const SdfColliderType& collider,
94 const SkeletonStateT<T>& state) {
95 if (!collider.isValid()) {
96 return axel::BoundingBox<T>();
97 }
98
99 TransformT<T> colliderWorldTransform;
100 const auto parentJoint = collider.parentJoint();
101 if (parentJoint == kInvalidIndex) {
102 colliderWorldTransform = collider.localToParentTransform().template cast<T>();
103 } else {
104 MT_CHECK(parentJoint < state.jointState.size(), "Invalid parent joint index: {}", parentJoint);
105 colliderWorldTransform = state.jointState[parentJoint].transform *
106 collider.localToParentTransform().template cast<T>();
107 }
108
109 const auto& localBounds = collider.bounds();
110 const auto& minCorner = localBounds.min();
111 const auto& maxCorner = localBounds.max();
112
113 std::array<Eigen::Vector3<T>, 8> corners = {
114 {{minCorner.x(), minCorner.y(), minCorner.z()},
115 {maxCorner.x(), minCorner.y(), minCorner.z()},
116 {minCorner.x(), maxCorner.y(), minCorner.z()},
117 {maxCorner.x(), maxCorner.y(), minCorner.z()},
118 {minCorner.x(), minCorner.y(), maxCorner.z()},
119 {maxCorner.x(), minCorner.y(), maxCorner.z()},
120 {minCorner.x(), maxCorner.y(), maxCorner.z()},
121 {maxCorner.x(), maxCorner.y(), maxCorner.z()}}};
122
123 const auto firstWorldCorner = colliderWorldTransform * corners[0];
124 axel::BoundingBox<T> worldBounds(firstWorldCorner, T(0));
125 for (size_t i = 1; i < corners.size(); ++i) {
126 worldBounds.extend(colliderWorldTransform * corners[i]);
127 }
128 return worldBounds;
129}

134template <typename T>
135axel::BoundingBox<T> computeBoneBoundingBox(
136 std::span<const size_t> vertexIndices,
137 const MeshT<T>& posedMesh) {
138 if (vertexIndices.empty()) {
139 return axel::BoundingBox<T>();
140 }
141 axel::BoundingBox<T> bbox(posedMesh.vertices[vertexIndices[0]], T(0));
142 for (size_t i = 1; i < vertexIndices.size(); ++i) {
143 bbox.extend(posedMesh.vertices[vertexIndices[i]]);
144 }
145 return bbox;
146}

153template <typename T>
154void accumulateJointGradient(
155 const JointStateT<T>& jointState,
156 size_t paramIndex,
157 const Eigen::Vector3<T>& posd,
158 const Eigen::Vector3<T>& direction,
159 const VectorX<bool>& activeJointParams,
160 const ParameterTransform& parameterTransform,
161 Eigen::Ref<Eigen::VectorX<T>> gradient) {
162 for (size_t d = 0; d < 3; d++) {
163 if (activeJointParams[paramIndex + d]) {
164 gradient_jointParams_to_modelParams(
165 direction.dot(jointState.getTranslationDerivative(d)),
166 paramIndex + d,
167 parameterTransform,
168 gradient);
169 }
170 if (activeJointParams[paramIndex + 3 + d]) {
171 gradient_jointParams_to_modelParams(
172 direction.dot(jointState.getRotationDerivative(d, posd)),
173 paramIndex + 3 + d,
174 parameterTransform,
175 gradient);
176 }
177 }
178 if (activeJointParams[paramIndex + 6]) {
179 gradient_jointParams_to_modelParams(
180 direction.dot(jointState.getScaleDerivative(posd)),
181 paramIndex + 6,
182 parameterTransform,
183 gradient);
184 }
185}

229template <typename T>
230void accumulateColliderHierarchyGradient(
231 const SkeletonStateT<T>& state,
232 size_t colliderParent,
233 size_t commonAncestor,
234 const Eigen::Vector3<T>& sdfSurfacePoint,
235 const Eigen::Vector3<T>& direction,
236 const Skeleton& skeleton,
237 const VectorX<bool>& activeJointParams,
238 const ParameterTransform& parameterTransform,
239 Eigen::Ref<Eigen::VectorX<T>> gradient) {
240 size_t jointIndex = colliderParent;
241 while (jointIndex != kInvalidIndex && jointIndex != commonAncestor) {
242 const auto& jointState = state.jointState[jointIndex];
243 const size_t paramIndex = jointIndex * kParametersPerJoint;
244 const Eigen::Vector3<T> posd = sdfSurfacePoint - jointState.translation();
245 accumulateJointGradient(
246 jointState, paramIndex, posd, direction, activeJointParams, parameterTransform, gradient);
247 jointIndex = skeleton.joints[jointIndex].parent;
248 }
249}
