25std::istringstream readOrThrow(const filesystem::path& filepath) {
26 std::ifstream file(filepath, std::ios::binary | std::ios::ate);
27 MT_THROW_IF(!file, "Failed to open the file.");
28
29 std::size_t size = static_cast<std::size_t>(file.tellg());
30 file.seekg(0, std::ios::beg);
31
32 std::string buffer(size, '\0');
33 MT_THROW_IF(!file.read(buffer.data(), size), "Failed to read the file.");
34
35 return std::istringstream(std::move(buffer));
36}

38void readShapeVectors(
39 std::istream& data,
40 MatrixXf& shapeVectors,
41 uint64_t numRows,
42 uint64_t numCols,
43 int expectedShapes,
44 int expectedVertices) {
45 // load shapeVectors
46 shapeVectors.resize(numRows, numCols);
47 data.read((char*)shapeVectors.data(), sizeof(float) * numRows * numCols);
48
49 // resize to few shapes
50 if (expectedShapes > 0) {
51 shapeVectors.conservativeResize(
52 shapeVectors.rows(),
53 std::min(gsl::narrow_cast<Eigen::Index>(expectedShapes), shapeVectors.cols()));
54 }
55 // resize vertices
56 if (expectedVertices > 0) {
57 shapeVectors.conservativeResize(
58 std::min(shapeVectors.rows(), gsl::narrow_cast<Eigen::Index>(expectedVertices * 3)),
59 shapeVectors.cols());
60 }
61}

94BlendShape loadBlendShape(std::istream& data, int expectedShapes, int expectedVertices) {
95 std::vector<Vector3f> baseShape;
96 MatrixXf shapeVectors;
97
98 // read dimensions
99 uint64_t numRows = 0;
100 uint64_t numCols = 0;
101 data.read((char*)&numRows, sizeof(numRows));
102 data.read((char*)&numCols, sizeof(numCols));
103
104 // read mean shape
105 baseShape.resize(numRows / 3);
106 data.read((char*)baseShape.data(), sizeof(float) * numRows);
107
108 if (expectedVertices > 0) {
109 baseShape.resize(std::min(baseShape.size(), gsl::narrow<size_t>(expectedVertices)));
110 }
111
112 // read shape vectors
113 readShapeVectors(data, shapeVectors, numRows, numCols, expectedShapes, expectedVertices);
114
115 BlendShape res(baseShape, shapeVectors.cols());
116 res.setShapeVectors(shapeVectors);
117 return res;
118}

128void saveBlendShape(std::ostream& os, const BlendShape& blendShape) {
129 // write dimensions
130 const uint64_t numRows = blendShape.getShapeVectors().rows();
131 const uint64_t numCols = blendShape.getShapeVectors().cols();
132 MT_CHECK(
133 numRows == blendShape.getBaseShape().size() * 3,
134 "{}, {}",
135 numRows,
136 blendShape.getBaseShape().size() * 3);
137
138 os.write((char const*)&numRows, sizeof(numRows));
139 os.write((char const*)&numCols, sizeof(numCols));
140
141 // write mean shape
142 os.write((char const*)blendShape.getBaseShape().data(), sizeof(float) * numRows);
143
144 // write coefficient matrix
145 os.write((char const*)blendShape.getShapeVectors().data(), sizeof(float) * numRows * numCols);
146}
