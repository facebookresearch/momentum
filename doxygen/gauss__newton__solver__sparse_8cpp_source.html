51template <typename T>
52void SparseGaussNewtonSolverT<T>::doIteration() {
53 MT_PROFILE_FUNCTION();
54
55 Eigen::VectorX<T> delta;
56
57 if (!directSparseJtJ_) { // make a dense matrix and sparsify later
58 if (useBlockJtJ_) {
59 // get JtJ and JtR pre-computed
60 {
61 MT_PROFILE_EVENT("Get sparse JtJ and JtR");
62 this->error_ = this->solverFunction_->getJtJR(this->parameters_, hessianApprox_, JtR_);
63 }
64
65 {
66 MT_PROFILE_EVENT("Sparsify JtJ");
67 // sparsify the system
68 JtJ_ = hessianApprox_.sparseView();
69 }
70 } else {
71 MT_PROFILE_EVENT("Sparse gauss newton step");
72
73 // get the jacobian and residual
74 size_t size = 0;
75 {
76 MT_PROFILE_EVENT("get Jacobian");
77 this->error_ =
78 this->solverFunction_->getJacobian(this->parameters_, jacobian_, residual_, size);
79 }
80
81 // sparsify the system
82 const Eigen::SparseMatrix<T> sjac =
83 jacobian_.topLeftCorner(size, this->actualParameters_).sparseView();
84 JtR_.noalias() = sjac.transpose() * residual_.head(size);
85
86 JtJ_ = sjac.transpose() * sjac;
87 }
88 } else { // directly sparse JtJ
89 MT_PROFILE_EVENT("Get sparse JtJ and JtR");
90 this->error_ = this->solverFunction_->getJtJR_Sparse(this->parameters_, JtJ_, JtR_);
91 }
92
93 if (D_.innerSize() != static_cast<Eigen::Index>(this->actualParameters_)) {
94 D_.resize(this->actualParameters_, this->actualParameters_);
95 D_.setIdentity();
96 }
97 JtJ_ += D_ * regularization_;
98
99 // Symbolic decomposition, only needed if the params pattern changed
100 if (this->newParameterPattern_) {
101 MT_PROFILE_EVENT("Sparse analyze");
102 lltSolver_.analyzePattern(JtJ_);
103 this->newParameterPattern_ = !directSparseJtJ_; // works fine for sparse matrix with explicit 0s
104 }
105
106 // Numerical update with the new coefficients
107 {
108 MT_PROFILE_EVENT("Sparse factorization");
109 lltSolver_.factorize(JtJ_);
110 }
111
112 // Solve, compute the gauss-newton step
113 {
114 MT_PROFILE_EVENT("Sparse solve");
115 delta.setZero(this->numParameters_);
116 delta.head(this->actualParameters_) = lltSolver_.solve(JtR_);
117 }
118
119 updateParameters(delta);
120
121 {
122 MT_PROFILE_EVENT("Store history");
123 if (this->storeHistory) {
124 this->iterationHistory_["solver_err"].setZero(1, 1);
125 if (lltSolver_.info() != Eigen::Success) {
126 this->iterationHistory_["solver_err"](0, 0) = 1.0;
127 }
128
129 this->iterationHistory_["jtr_norm"].resize(1, 1);
130 this->iterationHistory_["jtr_norm"](0, 0) = JtR_.norm();
131 }
132 }
133}

135template <typename T>
136void SparseGaussNewtonSolverT<T>::updateParameters(Eigen::VectorX<T>& delta) {
137 if (!doLineSearch_) {
138 MT_PROFILE_EVENT("Update params");
139 this->solverFunction_->updateParameters(this->parameters_, delta);
140 return;
141 }
142
143 MT_PROFILE_EVENT("Line search");
144
145 static constexpr T kC1 = 1e-3;
146 static constexpr T kTau = 0.5;
147 static constexpr size_t kMaxLineSearchSteps = 10;
148
149 const T scaledError = kC1 * this->error_;
150 const Eigen::VectorX<T> parametersOrig = this->parameters_;
151 T lineSearchScale = 1.0;
152
153 for (size_t i = 0; i < kMaxLineSearchSteps && std::isnormal(lineSearchScale); ++i) {
154 this->parameters_ = parametersOrig;
155
156 this->solverFunction_->updateParameters(this->parameters_, lineSearchScale * delta);
157
158 const double errorNew = this->solverFunction_->getError(this->parameters_);
159 if ((this->error_ - errorNew) >= lineSearchScale * scaledError) {
160 break;
161 }
162
163 // Reduce step size:
164 lineSearchScale *= kTau;
165 }
166}
