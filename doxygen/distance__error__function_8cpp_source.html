21template <typename T>
22double DistanceErrorFunctionT<T>::getError(
23 const ModelParametersT<T>& /*parameters*/,
24 const SkeletonStateT<T>& skeletonState,
25 const MeshStateT<T>& /* meshState */) {
26 double error = 0;
27
28 const auto& jointState = skeletonState.jointState;
29
30 for (const auto& cons : constraints_) {
31 const auto& js = jointState[cons.parent];
32 const Eigen::Vector3<T> p_world_cm = js.transform * cons.offset;
33 const Eigen::Vector3<T> diff_vec = p_world_cm - cons.origin;
34
35 const T distance = diff_vec.norm();
36 error += cons.weight * kDistanceWeight * this->weight_ * sqr(distance - cons.target);
37 }
38
39 return error;
40}

42template <typename T>
43double DistanceErrorFunctionT<T>::getGradient(
44 const ModelParametersT<T>& /*params*/,
45 const SkeletonStateT<T>& skeletonState,
46 const MeshStateT<T>& /* meshState */,
47 Eigen::Ref<Eigen::VectorX<T>> gradient) {
48 double error = 0;
49
50 for (size_t iCons = 0; iCons < constraints_.size(); ++iCons) {
51 const auto& cons = constraints_[iCons];
52
53 const auto& jsCons = skeletonState.jointState[cons.parent];
54 const Eigen::Vector3<T> p_world_cm = jsCons.transform * cons.offset;
55 const Eigen::Vector3<T> diff_vec = p_world_cm - cons.origin;
56
57 const T distance = diff_vec.norm();
58 if (distance == 0) {
59 // Gradients are undefined in this case, skip to avoid NaNs in gradient.
60 continue;
61 }
62
63 const Eigen::Vector3<T> diff_distance_diff_p_world = diff_vec / distance;
64
65 const T diff = distance - cons.target;
66 error += cons.weight * kDistanceWeight * this->weight_ * sqr(diff);
67 const T wgt = T(2) * cons.weight * kDistanceWeight * this->weight_;
68
69 auto addGradient = [&](const int jFullBodyDOF, const Eigen::Vector3<T>& diff_p_world_cm) {
70 // d(distance)/dTheta = d(distance)/d(p_world_cm) * d(p_world_cm)/dTheta
71 // distance is defined as ||p_world_cm - cons.origin||, so
72 // d(distance)/d(p_world_cm) =
73 // (p_world_cm - cons.origin) / ||p_world_cm - cons.origin|| . p_world_cm
74 const T diff_p_distance = diff_distance_diff_p_world.dot(diff_p_world_cm);
75 const auto gradFull = diff_p_distance * diff;
76
77 // multiply by the parameter transform:
78 for (momentum::SparseRowMatrixf::InnerIterator it(
79 this->parameterTransform_.transform, jFullBodyDOF);
80 it;
81 ++it) {
82 assert(it.row() == (Eigen::Index)jFullBodyDOF);
83 const auto jReducedDOF = it.col();
84 gradient(jReducedDOF) += it.value() * wgt * gradFull;
85 }
86 };
87
88 // loop over all joints the constraint is attached to and calculate gradient
89 size_t jointIndex = cons.parent;
90 while (jointIndex != kInvalidIndex) {
91 // check for valid index
92 assert(jointIndex < this->skeleton_.joints.size());
93
94 const auto& jointState = skeletonState.jointState[jointIndex];
95 const size_t paramIndex = jointIndex * 7;
96 const Eigen::Vector3<T> posd_cm = p_world_cm - jointState.translation();
97
98 // calculate derivatives based on active joints
99 for (size_t d = 0; d < 3; d++) {
100 if (this->activeJointParams_[paramIndex + d]) {
101 addGradient(paramIndex + d, jointState.getTranslationDerivative(d));
102 }
103 if (this->activeJointParams_[paramIndex + 3 + d]) {
104 addGradient(paramIndex + 3 + d, jointState.getRotationDerivative(d, posd_cm));
105 }
106 }
107 if (this->activeJointParams_[paramIndex + 6]) {
108 addGradient(paramIndex + 6, jointState.getScaleDerivative(posd_cm));
109 }
110
111 // go to the next joint
112 jointIndex = this->skeleton_.joints[jointIndex].parent;
113 }
114 }
115
116 return error;
117}

119template <typename T>
120double DistanceErrorFunctionT<T>::getJacobian(
121 const ModelParametersT<T>& /*parameters*/,
122 const SkeletonStateT<T>& skeletonState,
123 const MeshStateT<T>& /* meshState */,
124 Eigen::Ref<Eigen::MatrixX<T>> jacobian,
125 Eigen::Ref<Eigen::VectorX<T>> residual,
126 int& usedRows) {
127 double error = 0;
128
129 for (size_t iCons = 0; iCons < constraints_.size(); ++iCons) {
130 const auto& cons = constraints_[iCons];
131
132 const auto& jsCons = skeletonState.jointState[cons.parent];
133 const Eigen::Vector3<T> p_world_cm = jsCons.transform * cons.offset;
134 const Eigen::Vector3<T> diff_vec = p_world_cm - cons.origin;
135
136 const T distance = diff_vec.norm();
137 if (distance == 0) {
138 // Gradients are undefined in this case, skip to avoid NaNs in gradient.
139 continue;
140 }
141
142 const Eigen::Vector3<T> diff_vec_normalized = diff_vec / distance;
143
144 const T diff = distance - cons.target;
145 error += cons.weight * kDistanceWeight * this->weight_ * sqr(diff);
146 const T wgt = std::sqrt(cons.weight * kDistanceWeight * this->weight_);
147 residual(iCons) = wgt * diff;
148
149 auto addJacobian = [&](const size_t jFullBodyDOF, const Eigen::Vector3<T>& diff_p_world_cm) {
150 const T diff_p_distance = diff_vec_normalized.dot(diff_p_world_cm);
151
152 // multiply by the parameter transform:
153 for (momentum::SparseRowMatrixf::InnerIterator it(
154 this->parameterTransform_.transform, (int)jFullBodyDOF);
155 it;
156 ++it) {
157 assert(it.row() == (Eigen::Index)jFullBodyDOF);
158 const auto jReducedDOF = it.col();
159 jacobian(iCons, jReducedDOF) += it.value() * wgt * diff_p_distance;
160 }
161 };
162
163 // loop over all joints the constraint is attached to and calculate gradient
164 size_t jointIndex = cons.parent;
165 while (jointIndex != kInvalidIndex) {
166 // check for valid index
167 assert(jointIndex < this->skeleton_.joints.size());
168
169 const auto& jointState = skeletonState.jointState[jointIndex];
170 const size_t paramIndex = jointIndex * 7;
171 const Eigen::Vector3<T> posd_cm = p_world_cm - jointState.translation();
172
173 // calculate derivatives based on active joints
174 for (size_t d = 0; d < 3; d++) {
175 if (this->activeJointParams_[paramIndex + d]) {
176 addJacobian(paramIndex + d, jointState.getTranslationDerivative(d));
177 }
178 if (this->activeJointParams_[paramIndex + 3 + d]) {
179 addJacobian(paramIndex + 3 + d, jointState.getRotationDerivative(d, posd_cm));
180 }
181 }
182 if (this->activeJointParams_[paramIndex + 6]) {
183 addJacobian(paramIndex + 6, jointState.getScaleDerivative(posd_cm));
184 }
185
186 // go to the next joint
187 jointIndex = this->skeleton_.joints[jointIndex].parent;
188 }
189 }
190
191 usedRows = static_cast<int>(jacobian.rows());
192
193 return error;
194}
