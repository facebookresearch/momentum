14// Rerun SDK version compatibility layer
15// This header provides compatibility between rerun 0.23.x and 0.28.x+ APIs.
16//
17// Key API changes between versions:
18// - 0.24+: Scalar -> Scalars (takes batch of values)
19// - 0.24+: SeriesLine -> SeriesLines
20// - 0.24+: SeriesPoint -> SeriesPoints
21// - 0.26+: set_time_seconds deprecated -> use set_time_duration_secs
22// - 0.28+: Transform3D::with_axis_length removed (use Axes3D archetype instead)
23//
24// The RERUN_VERSION_GE macro is available since rerun 0.18.
25
26namespace momentum {
27
28// Helper function to log a single scalar value
29// In rerun 0.24+, Scalar was renamed to Scalars and takes a batch of values
30// This wrapper provides a consistent API across versions
31template <typename T>
32inline auto makeScalar([[maybe_unused]] T value) {
33#if defined(RERUN_VERSION_GE) && RERUN_VERSION_GE(0, 24, 0)
34 // Rerun 0.24+: Use Scalars with a single-element batch
35 return rerun::Scalars({static_cast<double>(value)});
36#else
37 // Rerun 0.23.x: Use Scalar with a single value
38 return rerun::Scalar(value);
39#endif
40}

42// Helper function to create a SeriesLine configuration
43// In rerun 0.24+, SeriesLine was renamed to SeriesLines

45#if defined(RERUN_VERSION_GE) && RERUN_VERSION_GE(0, 24, 0)

61// Helper function to set time in seconds on a recording stream
62// In rerun 0.26+, set_time_seconds was deprecated in favor of set_time_duration_secs
63inline void
64setTimeSeconds(const rerun::RecordingStream& rec, const std::string& timelineName, double seconds) {
65#if defined(RERUN_VERSION_GE) && RERUN_VERSION_GE(0, 26, 0)
66 rec.set_time_duration_secs(timelineName, seconds);
67#else
68 rec.set_time_seconds(timelineName, seconds);
69#endif
70}

72// Helper function to create a Transform3D with axis visualization
73// In rerun 0.28+, with_axis_length was removed from Transform3D
74// Instead, Axes3D archetype should be used for axis visualization
75// This wrapper returns the transform without axis_length for 0.28+
76// Callers should log a separate Axes3D archetype if axis visualization is needed
77template <typename... Args>
78inline auto makeTransform3D(Args&&... args) {
79 return rerun::Transform3D(std::forward<Args>(args)...);
