32// Function to convert a string_view to a numeric value.
33//
34// It removes leading whitespaces, and then uses std::from_chars to perform the conversion.
35// The base parameter is optional, and is used when converting to an integral type.
36template <typename T>
37[[nodiscard]] T from_chars_wrapping(std::string_view s, std::optional<int> base = std::nullopt) {
38 s.remove_prefix(std::min(s.find_first_not_of(" \t\n\r\f\v"), s.size()));
39 T value;
40
41 if constexpr (std::is_integral_v<T>) {
42 if (base.has_value()) {
43 std::from_chars(s.data(), s.data() + s.size(), value, base.value());
44 } else {
45 std::from_chars(s.data(), s.data() + s.size(), value);
46 }
47 } else {
48 value = std::stof(std::string(s));
49 }
50
51 return value;
52}

54// Function to convert a string_view to a numeric value of type T.
55template <typename T>
56[[nodiscard]] T svto(const std::string_view input, std::optional<int> base = std::nullopt) {
57 try {
58 return from_chars_wrapping<T>(input, base);
59 } catch (...) {
60 MT_THROW("Error parsing string to number");
61 }
62}

64// Convert a string_view to an int. Throws a runtime error if the conversion fails.
65[[nodiscard]] int svtoi(const std::string_view input) {
66 return svto<int>(input);

69// Convert a string_view to a float. Throws a runtime error if the conversion fails.
70[[nodiscard]] float svtof(const std::string_view input) {
71 return svto<float>(input);

74std::string firstDuplicate(const LocatorList& locators) {
75 std::unordered_set<std::string> locatorNames;
76 for (const auto& l : locators) {
77 auto itr = locatorNames.find(l.name);
78 if (itr != locatorNames.end()) {
79 return l.name;
80 }
81 locatorNames.insert(itr, l.name);
82 }
83 return "";
84}

88LocatorList loadLocators(
89 const filesystem::path& filename,
90 const Skeleton& skeleton,
91 const ParameterTransform& parameterTransform) {
92 std::ifstream instream(filename, std::ios::binary);
93 MT_THROW_IF(!instream.is_open(), "Cannot find file {}", filename.string());
94 instream.seekg(0, std::ios::end);
95 auto length = instream.tellg();
96 instream.seekg(0, std::ios::beg);
97
98 auto buffer = std::make_unique<char[]>(length);
99 instream.read((char*)buffer.get(), length);
100
101 return loadLocatorsFromBuffer(
102 gsl::make_span<const std::byte>(reinterpret_cast<const std::byte*>(buffer.get()), length),
103 skeleton,
104 parameterTransform);
105}

107LocatorList loadLocatorsFromBuffer(
108 gsl::span<const std::byte> rawData,
109 const Skeleton& skeleton,
110 const ParameterTransform& parameterTransform) {
111 const std::string_view input(reinterpret_cast<const char*>(rawData.data()), rawData.size());
112
113 LocatorList res;
114
115 const SkeletonState state(parameterTransform.bindPose(), skeleton);
116
117 const auto lines = tokenize(input, "\r\n");
118 if (lines.size() < 4 || lines[0] != "{" ||
119 trim(lines[1]).find("\"locators\":") == std::string::npos) {
120 return res;
121 }
122 for (size_t line = 2; line < lines.size(); line++) {
123 std::string_view current = trim(lines[line]);
124 if (current != "{") {
125 continue;
126 }
127
128 // start of locator
129 Locator l;
130 Vector3f global;
131 bool haveGlobal = false;
132 do {
133 current = trim(lines.at(++line));
134 if (current[0] == '}') {
135 break;
136 }
137 if (current[0] == '/') {
138 continue;
139 }
140 const auto tokens = tokenize(current, ":");
141 if (tokens.size() != 2) {
142 continue;
143 }
144 if (tokens[0] == "\"lockX\"") {
145 l.locked.x() = svtoi(tokens[1]);
146 } else if (tokens[0] == "\"lockY\"") {
147 l.locked.y() = svtoi(tokens[1]);
148 } else if (tokens[0] == "\"lockZ\"") {
149 l.locked.z() = svtoi(tokens[1]);
150 } else if (tokens[0] == "\"name\"") {
151 const auto first = tokens[1].find_first_of('"') + 1;
152 const auto last = tokens[1].find_last_of('"') - 1;
153 l.name = tokens[1].substr(first, last - first + 1);
154 } else if (tokens[0] == "\"parent\"") {
155 l.parent = svtoi(tokens[1]);
156 } else if (tokens[0] == "\"parentName\"") {
157 const auto first = tokens[1].find_first_of('"') + 1;
158 const auto last = tokens[1].find_last_of('"') - 1;
159 const std::string_view parent = tokens[1].substr(first, last - first + 1);
160 for (size_t k = 0; k < skeleton.joints.size(); k++) {
161 if (skeleton.joints[k].name == parent) {
162 l.parent = static_cast<int>(k);
163 break;
164 }
165 }
166 // invalid joint name
167 if (l.parent == kInvalidIndex) {
168 MT_LOGW("Invalid parent name: {}", parent);
169 }
170 } else if (tokens[0] == "\"weight\"") {
171 l.weight = svtof(tokens[1]);
172 } else if (tokens[0] == "\"offsetX\"") {
173 l.offset.x() = svtof(tokens[1]);
174 } else if (tokens[0] == "\"offsetY\"") {
175 l.offset.y() = svtof(tokens[1]);
176 } else if (tokens[0] == "\"offsetZ\"") {
177 l.offset.z() = svtof(tokens[1]);
178 } else if (tokens[0] == "\"globalX\"") {
179 global.x() = svtof(tokens[1]);
180 haveGlobal = true;
181 } else if (tokens[0] == "\"globalY\"") {
182 global.y() = svtof(tokens[1]);
183 haveGlobal = true;
184 } else if (tokens[0] == "\"globalZ\"") {
185 global.z() = svtof(tokens[1]);
186 haveGlobal = true;
187 } else if (tokens[0] == "\"limitWeightX\"") {
188 l.limitWeight[0] = svtof(tokens[1]);
189 } else if (tokens[0] == "\"limitWeightY\"") {
190 l.limitWeight[1] = svtof(tokens[1]);
191 } else if (tokens[0] == "\"limitWeightZ\"") {
192 l.limitWeight[2] = svtof(tokens[1]);
193 }
194 } while (line < lines.size() - 1 && (current != "}," || current != "}"));
195
196 // skip locator if it has no parent
197 if (l.parent == kInvalidIndex) {
198 continue;
199 }
200
201 if (haveGlobal) {
202 l.offset = state.jointState[l.parent].transform.inverse() * (global);
203 }
204 l.limitOrigin = l.offset;
205 res.push_back(l);
206 }
207
208 std::string dup = firstDuplicate(res);
209 MT_THROW_IF(!dup.empty(), "Duplicated locator {} found", dup);
210
211 return res;
212}

214void saveLocators(
215 const filesystem::path& filename,
216 const LocatorList& locators,
217 const Skeleton& skeleton,
218 const LocatorSpace space) {
219 const SkeletonState state(VectorXf::Zero(skeleton.joints.size() * kParametersPerJoint), skeleton);
220 const LocatorState lstate(state, locators);
221
222 std::ofstream o(filename);
223 o << "{\n\t\"locators\":[\n";
224 for (size_t i = 0; i < locators.size(); i++) {
225 o << "\t\t{\n";
226 o << "\t\t\t\"name\":\"" << locators[i].name << "\",\n";
227 if (space == LocatorSpace::Global) {
228 o << "\t\t\t\"globalX\":" << lstate.position[i].x() << ",\n";
229 o << "\t\t\t\"globalY\":" << lstate.position[i].y() << ",\n";
230 o << "\t\t\t\"globalZ\":" << lstate.position[i].z() << ",\n";
231 } else if (space == LocatorSpace::Local) {
232 o << "\t\t\t\"offsetX\":" << locators[i].offset.x() << ",\n";
233 o << "\t\t\t\"offsetY\":" << locators[i].offset.y() << ",\n";
234 o << "\t\t\t\"offsetZ\":" << locators[i].offset.z() << ",\n";
235 }
236 o << "\t\t\t\"lockX\":" << locators[i].locked.x() << ",\n";
237 o << "\t\t\t\"lockY\":" << locators[i].locked.y() << ",\n";
238 o << "\t\t\t\"lockZ\":" << locators[i].locked.z() << ",\n";
239 o << "\t\t\t\"weight\":" << locators[i].weight << ",\n";
240 if (locators[i].limitWeight[0] != 0.0f) {
241 o << "\t\t\t\"limitWeightX\":" << locators[i].limitWeight.x() << ",\n";
242 }
243 if (locators[i].limitWeight[1] != 0.0f) {
244 o << "\t\t\t\"limitWeightY\":" << locators[i].limitWeight.y() << ",\n";
245 }
246 if (locators[i].limitWeight[2] != 0.0f) {
247 o << "\t\t\t\"limitWeightZ\":" << locators[i].limitWeight.z() << ",\n";
248 }
249 MT_CHECK(
250 0 <= locators[i].parent && locators[i].parent < skeleton.joints.size(),
251 "Invalid joint index");
252 if (!skeleton.joints.empty() && locators[i].parent < skeleton.joints.size()) {
253 o << "\t\t\t\"parentName\":\"" << skeleton.joints[locators[i].parent].name << "\"\n";
254 }
255 if (i < locators.size() - 1) {
256 o << "\t\t},\n";
257 } else {
258 o << "\t\t}\n";
259 }
260 }
261 o << "\t]\n}\n";
262}
