29template <typename T>
30void SkeletonSolverFunctionT<T>::setEnabledParameters(const ParameterSet& ps) {
31 // find the last enabled parameter
32 this->actualParameters_ = 0;
33 for (size_t i = 0; i < this->numParameters_; i++) {
34 if (ps.test(i)) {
35 this->actualParameters_ = gsl::narrow_cast<int>(i + 1);
36 }
37 }
38 // set the enabled joints based on the parameter set
39 activeJointParams_ = parameterTransform_->computeActiveJointParams(ps);
40
41 // give data to helper functions
42 for (auto&& solvable : errorFunctions_) {
43 solvable->setActiveJoints(activeJointParams_);
44 solvable->setEnabledParameters(ps);
45 }
46}

48template <typename T>
49double SkeletonSolverFunctionT<T>::getError(const Eigen::VectorX<T>& parameters) {
50 // update the state according to the transformed parameters
51 state_->set(parameterTransform_->apply(parameters), *skeleton_, false);
52
53 double error = 0.0;
54
55 // sum up error for all solvables
56 for (auto&& solvable : errorFunctions_) {
57 if (solvable->getWeight() > 0.0f) {
58 error += (double)solvable->getError(parameters, *state_);
59 }
60 }
61
62 return (float)error;
63}

69 // update the state according to the transformed parameters

71
72 double error = 0.0;
73
74 // sum up error and gradients for all solvables
75 if (gradient.size() != parameters.size()) {

88template <typename T>
89std::pair<size_t, std::vector<size_t>> getDimensions(
90 const std::vector<std::shared_ptr<SkeletonErrorFunctionT<T>>>& error_func) {
91 MT_PROFILE_FUNCTION();
92 std::vector<size_t> offset(error_func.size());
93 size_t jacobianSize = 0;
94 for (size_t i = 0; i < error_func.size(); i++) {
95 const auto& solvable = error_func[i];
96
97 // ! offset values are only defined when the cost function is active
98 if (solvable->getWeight() > 0.0f) {
99 const auto jSize = solvable->getJacobianSize();
100 offset[i] = jSize;
101 jacobianSize += jSize;
102 }
103 }
104 jacobianSize += 8 - (jacobianSize % 8);
105 return std::make_pair(jacobianSize, std::move(offset));
106}

108template <typename T>
109double SkeletonSolverFunctionT<T>::getJacobian(
110 const Eigen::VectorX<T>& parameters,
111 Eigen::MatrixX<T>& jacobian,
112 Eigen::VectorX<T>& residual,
113 size_t& actualRows) {
114 MT_PROFILE_FUNCTION();
115
116 // update the state according to the transformed parameters
117 {
118 MT_PROFILE_EVENT("Set state");
119 state_->set(parameterTransform_->apply(parameters), *skeleton_);
120 }
121
122 double error = 0.0;
123
124 // calculate the jacobian size
125 const auto dimensions = getDimensions(errorFunctions_);
126 const auto& jacobianSize = dimensions.first;
127 const auto& offset = dimensions.second;
128
129 if (jacobianSize > static_cast<size_t>(jacobian.rows()) || parameters.size() != jacobian.cols()) {
130 MT_PROFILE_EVENT("ResizeJacobian");
131 jacobian.resize(jacobianSize, parameters.size());
132 residual.resize(jacobianSize);
133 }
134 {
135 MT_PROFILE_EVENT("InitializeJacobian");
136 jacobian.setZero();
137 residual.setZero();
138 }
139 actualRows = jacobianSize;
140
141 // add values to the jacobian
142 size_t position = 0;
143
144 MT_PROFILE_EVENT("Collect all jacobians");
145
146 for (size_t i = 0; i < errorFunctions_.size(); i++) {
147 int rows = 0;
148 const auto& solvable = errorFunctions_[i];
149 if (solvable->getWeight() > 0.0f) {
150 const auto& n = offset[i];
151 error += solvable->getJacobian(
152 parameters,
153 *state_,
154 jacobian.block(position, 0, n, parameters.size()),
155 residual.middleRows(position, n),
156 rows);
157 position += n;
158 }
159 }
160
161 return error;
162}

164/* override */
165template <typename T>
166double SkeletonSolverFunctionT<T>::getJtJR(
167 const Eigen::VectorX<T>& parameters,
168 Eigen::MatrixX<T>& JtJ,
169 Eigen::VectorX<T>& JtR) {
170 MT_PROFILE_FUNCTION();
171
172 // update the state according to the transformed parameters
173 {
174 MT_PROFILE_EVENT("JtJR - update state");
175 state_->set(parameterTransform_->apply(parameters), *skeleton_);
176 }
177
178 // calculate the jacobian size
179 const auto dimensions = getDimensions(errorFunctions_);
180 const auto& jacobianSize = dimensions.first;
181 const auto n_parameters = parameters.size();
182
183 if (dimensions.first > static_cast<size_t>(tJacobian_.rows()) ||
184 parameters.size() != tJacobian_.cols()) {
185 MT_PROFILE_EVENT("ResizetJacobian");
186 tJacobian_.resize(jacobianSize, n_parameters);
187 tResidual_.resize(jacobianSize);
188 }
189
190 if (JtJ.cols() != gsl::narrow_cast<Eigen::Index>(this->actualParameters_)) {
191 JtJ.resize(this->actualParameters_, this->actualParameters_);
192 JtR.resize(this->actualParameters_);
193 }
194
195 {
196 MT_PROFILE_EVENT("JtJR - set to Zero");
197
198 tJacobian_.topRows(jacobianSize).setZero();
199 tResidual_.head(jacobianSize).setZero();
200 JtJ.setZero();
201 JtR.setZero();
202 }
203
204 // block compute JtJ and JtR on the fly
205 size_t position = 0;
206 double error = 0.0;
207 for (size_t i = 0; i < errorFunctions_.size(); i++) {
208 const auto& solvable = errorFunctions_[i];
209
210 if (solvable->getWeight() > 0.0f) {
211 int rows = 0;
212 const auto& n = dimensions.second[i];
213
214 // This is timed through the corresponding solvable call
215 error += solvable->getJacobian(
216 parameters,
217 *state_,
218 tJacobian_.block(position, 0, n, n_parameters),
219 tResidual_.segment(position, n),
220 rows);
221 MT_CHECK(rows <= gsl::narrow_cast<int>(n));
222
223 // Update JtJ
224 if (rows > 0) {
225 MT_PROFILE_EVENT("Partial JtJ JtR");
226
227 // ! In truth, on the the "this->actualParameters_" leftmost block will be used
228 // We take advantage of this here and skip the other computations
229 const auto JtBlock =
230 tJacobian_.block(position, 0, rows, this->actualParameters_).transpose();
231
232 // Efficiently update JtJ (JtJ = J^T * J) using selfadjointView with rankUpdate,
233 // replacing triangularView to leverage symmetry and improve performance
234 JtJ.template selfadjointView<Eigen::Lower>().rankUpdate(JtBlock);
235
236 // Update JtR
237 JtR.noalias() += JtBlock * tResidual_.segment(position, rows);
238 }
239 position += n;
240 }
241 }
242
243 return error;
244}

247template <typename T>
248double SkeletonSolverFunctionT<T>::getSolverDerivatives(
249 const Eigen::VectorX<T>& parameters,
250 Eigen::MatrixX<T>& hess,
251 Eigen::VectorX<T>& grad) {
252 MT_PROFILE_FUNCTION();
253
254 // update the state according to the transformed parameters
255 {
256 MT_PROFILE_EVENT("UpdateState");
257 state_->set(parameterTransform_->apply(parameters), *skeleton_);
258 }
259
260 if (hess.cols() != gsl::narrow_cast<Eigen::Index>(this->actualParameters_)) {
261 hess.resize(this->actualParameters_, this->actualParameters_);
262 grad.resize(this->actualParameters_);
263 }
264
265 hess.setZero();
266 grad.setZero();
267
268 double error = 0.0;
269 for (size_t i = 0; i < errorFunctions_.size(); i++) {
270 const auto& solvable = errorFunctions_[i];
271 if (solvable->getWeight() > 0.0f) {
272 error +=
273 solvable->getSolverDerivatives(parameters, *state_, this->actualParameters_, hess, grad);
274 }
275 }
276
277 return error;
278}
