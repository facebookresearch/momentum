24template <typename T>
25SkeletonSolverFunctionT<T>::SkeletonSolverFunctionT(
26 const Character& character,
27 const ParameterTransformT<T>& parameterTransform,
28 std::span<const std::shared_ptr<SkeletonErrorFunctionT<T>>> errorFunctions)
29 : character_(character), parameterTransform_(parameterTransform), needsMeshState_(false) {
30 this->numParameters_ = parameterTransform_.numAllModelParameters();
31 this->actualParameters_ = this->numParameters_;
32 state_ = std::make_unique<SkeletonStateT<T>>(parameterTransform_.zero(), character_.skeleton);
33 activeJointParams_ = parameterTransform_.activeJointParams;
34 meshState_ = std::make_unique<MeshStateT<T>>();
35
36 for (auto& errf : errorFunctions) {
37 addErrorFunction(std::move(errf));
38 }
39}

44template <typename T>
45void SkeletonSolverFunctionT<T>::setEnabledParameters(const ParameterSet& ps) {
46 // find the last enabled parameter
47 this->actualParameters_ = 0;
48 for (size_t i = 0; i < this->numParameters_; i++) {
49 if (ps.test(i)) {
50 this->actualParameters_ = gsl::narrow_cast<int>(i + 1);
51 }
52 }
53 // set the enabled joints based on the parameter set
54 activeJointParams_ = parameterTransform_.computeActiveJointParams(ps);
55
56 // give data to helper functions
57 for (auto&& solvable : errorFunctions_) {
58 solvable->setActiveJoints(activeJointParams_);
59 solvable->setEnabledParameters(ps);
60 }
61}

63template <typename T>
64double SkeletonSolverFunctionT<T>::getError(const Eigen::VectorX<T>& parameters) {
65 // update the state according to the transformed parameters
66 state_->set(parameterTransform_.apply(parameters), character_.skeleton, false);
67
68 // Update mesh state if needed
69 if (needsMeshState()) {
70 updateMeshState(parameters, *state_);
71 }
72
73 double error = 0.0;
74
75 // sum up error for all solvables
76 for (auto&& solvable : errorFunctions_) {
77 if (solvable->getWeight() > 0.0f) {
78 error += (double)solvable->getError(parameters, *state_, *meshState_);
79 }
80 }
81
82 return (float)error;
83}

85template <typename T>
86double SkeletonSolverFunctionT<T>::getGradient(
87 const Eigen::VectorX<T>& parameters,
88 Eigen::VectorX<T>& gradient) {
89 // update the state according to the transformed parameters
90 state_->set(parameterTransform_.apply(parameters), character_.skeleton);
91
92 // Update mesh state if needed
93 if (needsMeshState()) {
94 updateMeshState(parameters, *state_);
95 }
96
97 double error = 0.0;
98
99 // sum up error and gradients for all solvables
100 if (gradient.size() != parameters.size()) {
101 gradient.resize(parameters.size());
102 }
103 gradient.setZero();
104 for (auto&& solvable : errorFunctions_) {
105 if (solvable->getWeight() > 0.0f) {
106 error += solvable->getGradient(parameters, *state_, *meshState_, gradient);
107 }
108 }
109
110 return error;
111}

114template <typename T>
115double SkeletonSolverFunctionT<T>::getSolverDerivatives(
116 const Eigen::VectorX<T>& parameters,
117 Eigen::MatrixX<T>& hess,
118 Eigen::VectorX<T>& grad) {
119 MT_PROFILE_FUNCTION();
120
121 // update the state according to the transformed parameters
122 {
123 MT_PROFILE_EVENT("UpdateState");
124 state_->set(parameterTransform_.apply(parameters), character_.skeleton);
125 }
126
127 // Update mesh state if needed
128 if (needsMeshState()) {
129 updateMeshState(parameters, *state_);
130 }
131
132 if (hess.cols() != gsl::narrow_cast<Eigen::Index>(this->actualParameters_)) {
133 hess.resize(this->actualParameters_, this->actualParameters_);
134 grad.resize(this->actualParameters_);
135 }
136
137 hess.setZero();
138 grad.setZero();
139
140 double error = 0.0;
141 for (size_t i = 0; i < errorFunctions_.size(); i++) {
142 const auto& solvable = errorFunctions_[i];
143 if (solvable->getWeight() > 0.0f) {
144 error += solvable->getSolverDerivatives(
145 parameters, *state_, *meshState_, this->actualParameters_, hess, grad);
146 }
147 }
148
149 return error;
150}

234template <typename T>
235double SkeletonSolverFunctionT<T>::computeJacobianBlock(
236 const Eigen::VectorX<T>& parameters,
237 size_t blockIndex,
238 Eigen::Ref<Eigen::MatrixX<T>> jacobianBlock,
239 Eigen::Ref<Eigen::VectorX<T>> residualBlock,
240 size_t& actualRows) {
241 MT_PROFILE_FUNCTION();
242
243 if (blockIndex >= errorFunctions_.size()) {
244 actualRows = 0;
245 return 0.0;
246 }
247
248 const auto& solvable = errorFunctions_[blockIndex];
249
250 if (solvable->getWeight() <= 0.0f) {
251 actualRows = 0;
252 return 0.0;
253 }
254
255 int rows = 0;
256 const double error =
257 solvable->getJacobian(parameters, *state_, *meshState_, jacobianBlock, residualBlock, rows);
258
259 actualRows = static_cast<size_t>(rows);
260 return error;
261}
