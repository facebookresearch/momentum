54template <typename T>
55TransformT<T> TransformT<T>::fromMatrix(const Matrix4<T>& other) {
56 TransformT<T> result;
57 result.translation = other.template topRightCorner<3, 1>();
58 // Calculate the scale by taking the norm of the first column, assuming uniform scaling
59 const auto& scaledR = other.template topLeftCorner<3, 3>();
60 result.scale = scaledR.col(0).norm();
61 MT_CHECK(result.scale >= Eps<T>(), "Scale is too small: {}", result.scale);
62 MT_CHECK(result.scale < T(1) / Eps<T>(), "Inverse scale is too small: {}", result.scale);
63 result.rotation = scaledR / result.scale;
64 return result;
65}

86template <typename T>
87TransformT<T> TransformT<T>::inverse() const {
88 // (translate(t) * rotation(R) * scale(s)).inv() =
89 // scale(s).inv() * rotation(R).inv() * translate(t).inv()
90 // = scale(1/s) * rotation(invR) * translate(-t)
91 // = translate(-invR*s*t) * rotation(invR) * scale(1/s)
92 const Quaternion<T> invRot = rotation.inverse();
93 const double invScale = T(1) / scale;
94 return TransformT<T>(-invScale * (invRot * translation), invRot, invScale);
95}
