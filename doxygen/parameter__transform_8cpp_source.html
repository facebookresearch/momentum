51template <typename T>
52ParameterTransformT<T> ParameterTransformT<T>::identity(gsl::span<const std::string> jointNames) {
53 ParameterTransformT<T> result;
54 const size_t nJoints = jointNames.size();
55 const size_t nJointParameters = nJoints * kParametersPerJoint;
56 result.offsets.setZero(nJointParameters);
57 result.activeJointParams.setConstant(nJointParameters, true);
58 result.transform.resize(nJointParameters, nJointParameters);
59 result.transform.setIdentity();
60 result.name.resize(nJointParameters);
61 for (size_t i = 0; i < nJoints; i++) {
62 for (size_t j = 0; j < kParametersPerJoint; ++j) {
63 result.name[i * kParametersPerJoint + j] =
64 fmt::format("{}_{}", jointNames[i], kJointParameterNames[j]);
65 }
66 }
67 return result;
68}

83// map model parameters to joint parameters using a linear transformation
84template <typename T>
85JointParametersT<T> ParameterTransformT<T>::apply(const ModelParametersT<T>& parameters) const {
86 MT_PROFILE_FUNCTION();
87 MT_CHECK(
88 parameters.size() == transform.cols(),
89 "Size mismatch: The size of the parameters vector '{}' does not match the number of columns in the transform matrix '{}'.",
90 parameters.size(),
91 transform.cols());
92 MT_CHECK(
93 offsets.size() == transform.rows(),
94 "Size mismatch: The size of the offsets vector '{}' does not match the number of rows in the transform matrix '{}'.",
95 offsets.size(),
96 transform.rows());
97 return transform * parameters.v + offsets;
98}

100// map model parameters to joint parameters using a linear transformation
101template <typename T>
102JointParametersT<T> ParameterTransformT<T>::apply(const CharacterParametersT<T>& parameters) const {
103 MT_PROFILE_FUNCTION();
104 MT_CHECK(
105 parameters.pose.size() == transform.cols(),
106 "{} is not {}",
107 parameters.pose.size(),
108 transform.cols());
109 JointParametersT<T> result = transform * parameters.pose.v;
110 if (parameters.offsets.size() > 0) {
111 MT_CHECK(
112 parameters.offsets.size() == transform.rows(),
113 "{} is not {}",
114 parameters.offsets.size(),
115 transform.rows());
116 result.v += parameters.offsets.v;
117 }
118 return result;
119}

205template <typename T>
206ParameterTransformT<T> mapParameterTransformJoints(
207 const ParameterTransformT<T>& parameterTransform,
208 size_t numTargetJoints,
209 const std::vector<size_t>& jointMapping) {
210 ParameterTransformT<T> mappedTransform;
211
212 // No change in the parameters:
213 mappedTransform.name = parameterTransform.name;
214
215 // resize the offset matrix to the right size
216 mappedTransform.offsets.setZero(numTargetJoints * kParametersPerJoint);
217 mappedTransform.activeJointParams.setConstant(numTargetJoints * kParametersPerJoint, false);
218
219 // map the offset from the anim skel to the new one
220 for (size_t i = 0; i < jointMapping.size(); i++) {
221 if (jointMapping[i] != kInvalidIndex) {
222 for (size_t d = 0; d < kParametersPerJoint; d++) {
223 mappedTransform.offsets[jointMapping[i] * kParametersPerJoint + d] =
224 parameterTransform.offsets(i * kParametersPerJoint + d);
225 }
226 }
227 }
228
229 // map the transform from the anim skel to the new one
230 std::vector<Eigen::Triplet<float>> triplets;
231 for (int k = 0; k < parameterTransform.transform.outerSize(); ++k) {
232 for (typename SparseRowMatrix<T>::InnerIterator it(parameterTransform.transform, k); it; ++it) {
233 // row is joint + type index, col is parameter
234 const int jIndex = static_cast<int>(it.row()) / kParametersPerJoint;
235 const int jOffset = static_cast<int>(it.row()) % kParametersPerJoint;
236
237 const auto mappedJoint = jointMapping[jIndex];
238 if (mappedJoint == kInvalidIndex) {
239 continue;
240 }
241
242 triplets.push_back(Eigen::Triplet<float>(
243 mappedJoint * kParametersPerJoint + jOffset, static_cast<int>(it.col()), it.value()));
244
245 // enable joint channels
246 mappedTransform.activeJointParams[mappedJoint * kParametersPerJoint + jOffset] = true;
247 }
248 }
249
250 // resize the Transform matrix to the correct size
251 mappedTransform.transform.resize(
252 static_cast<int>(numTargetJoints) * kParametersPerJoint,
253 static_cast<int>(mappedTransform.name.size()));
254
255 // create sparse matrix from triplet
256 mappedTransform.transform.setFromTriplets(triplets.begin(), triplets.end());
257
258 // copy over the parameterSets
259 mappedTransform.parameterSets = parameterTransform.parameterSets;
260 mappedTransform.poseConstraints = parameterTransform.poseConstraints;
261
262 return mappedTransform;
263}

265template <typename T>
266std::tuple<ParameterTransformT<T>, ParameterLimits> subsetParameterTransform(
267 const ParameterTransformT<T>& paramTransformOld,
268 const ParameterLimits& paramLimitsOld,
269 const ParameterSet& paramSet) {
270 const auto nOldParam = paramTransformOld.numAllModelParameters();
271
272 std::vector<size_t> oldParamToNewParam(nOldParam, kInvalidIndex);
273 std::vector<size_t> newParamToOldParam;
274
275 for (Eigen::Index iOldParam = 0; iOldParam < nOldParam; ++iOldParam) {
276 if (!paramSet.test(iOldParam)) {
277 continue;
278 }
279
280 oldParamToNewParam[iOldParam] = newParamToOldParam.size();
281 newParamToOldParam.push_back(iOldParam);
282 }
283
284 const auto nNewParam = newParamToOldParam.size();
285
286 ParameterTransformT<T> paramTransformNew;
287
288 paramTransformNew.name.reserve(nNewParam);
289 for (const auto iOldParam : newParamToOldParam) {
290 paramTransformNew.name.push_back(paramTransformOld.name[iOldParam]);
291 }
292
293 // Offsets are unchanged because they are the size #joints, not #param.
294 paramTransformNew.offsets = paramTransformOld.offsets;
295
296 paramTransformNew.activeJointParams.setConstant(paramTransformOld.numJointParameters(), false);
297
298 // Remap the transformation.
299 using IndexType = typename SparseRowMatrix<T>::Index;
300 std::vector<Eigen::Triplet<T>> tripletsNew;
301 for (int k = 0; k < paramTransformOld.transform.outerSize(); ++k) {
302 for (typename SparseRowMatrix<T>::InnerIterator it(paramTransformOld.transform, k); it; ++it) {
303 const auto iOldParam = it.col();
304 const auto iNewParam = oldParamToNewParam[iOldParam];
305
306 if (iNewParam == kInvalidIndex) {
307 continue;
308 }
309
310 tripletsNew.emplace_back((int)it.row(), (int)iNewParam, it.value());
311 paramTransformNew.activeJointParams[it.row()] = true;
312 }
313 }
314
315 paramTransformNew.transform.resize(paramTransformOld.transform.rows(), (IndexType)nNewParam);
316 paramTransformNew.transform.setFromTriplets(tripletsNew.begin(), tripletsNew.end());
317
318 for (const auto& [name, constraintsOld] : paramTransformOld.poseConstraints) {
319 PoseConstraint pcNew;
320 for (const auto& [idxOld, weight] : constraintsOld.parameterIdValue) {
321 auto idxNew = oldParamToNewParam[idxOld];
322 if (idxNew != kInvalidIndex) {
323 pcNew.parameterIdValue.emplace_back(idxNew, weight);
324 }
325 }
326
327 paramTransformNew.poseConstraints.insert(std::make_pair(name, pcNew));
328 }
329
330 paramTransformNew.blendShapeParameters =
331 Eigen::VectorXi::Constant(paramTransformOld.blendShapeParameters.size(), -1);
332 for (Eigen::Index iBlendShape = 0; iBlendShape < paramTransformOld.blendShapeParameters.size();
333 ++iBlendShape) {
334 auto paramOld = paramTransformOld.blendShapeParameters(iBlendShape);
335 if (paramOld >= 0) {
336 auto paramNew = oldParamToNewParam[paramOld];
337 if (paramNew != kInvalidIndex) {
338 paramTransformNew.blendShapeParameters(iBlendShape) = paramNew;
339 }
340 }
341 }
342
343 paramTransformNew.faceExpressionParameters =
344 Eigen::VectorXi::Constant(paramTransformOld.faceExpressionParameters.size(), -1);
345 for (Eigen::Index iBlendShape = 0;
346 iBlendShape < paramTransformOld.faceExpressionParameters.size();
347 ++iBlendShape) {
348 auto paramOld = paramTransformOld.faceExpressionParameters(iBlendShape);
349 if (paramOld >= 0) {
350 auto paramNew = oldParamToNewParam[paramOld];
351 if (paramNew != kInvalidIndex) {
352 paramTransformNew.faceExpressionParameters(iBlendShape) = paramNew;
353 }
354 }
355 }
356
357 paramTransformNew.skinnedLocatorParameters =
358 Eigen::VectorXi::Constant(paramTransformOld.skinnedLocatorParameters.size(), -1);
359 for (Eigen::Index iSkinnedLocator = 0;
360 iSkinnedLocator < paramTransformOld.skinnedLocatorParameters.size();
361 ++iSkinnedLocator) {
362 auto paramOld = paramTransformOld.skinnedLocatorParameters(iSkinnedLocator);
363 if (paramOld < 0) {
364 continue;
365 }
366
367 bool anyMapped = false;
368 for (int k = 0; k < 3; ++k) {
369 anyMapped = anyMapped || (oldParamToNewParam[paramOld + k] != kInvalidIndex);
370 }
371
372 if (anyMapped) {
373 auto paramNew = oldParamToNewParam[paramOld];
374 for (int k = 0; k < 3; ++k) {
375 if (oldParamToNewParam[paramOld + k] != paramNew + k) {
376 throw std::runtime_error(fmt::format(
377 "When mapping skinned locator parameters, must retain all 3 parameters for each locator. Locator {} is missing parameter {} ({}).",
378 iSkinnedLocator,
379 paramOld + k,
380 paramTransformOld.name[paramOld + k]));
381 }
382 }
383
384 paramTransformNew.skinnedLocatorParameters(iSkinnedLocator) = gsl::narrow_cast<int>(paramNew);
385 }
386 }
387
388 for (const auto& paramSetOld : paramTransformOld.parameterSets) {
389 ParameterSet paramSetNew;
390 paramSetNew.reset();
391
392 for (size_t iNewParam = 0; iNewParam < nNewParam; ++iNewParam) {
393 const auto iOldParam = newParamToOldParam[iNewParam];
394 paramSetNew.set(iNewParam, paramSetOld.second.test(iOldParam));
395 }
396
397 paramTransformNew.parameterSets.emplace(paramSetOld.first, paramSetNew);
398 }
399
400 ParameterLimits paramLimitsNew;
401 paramLimitsNew.reserve(paramLimitsOld.size());
402 for (const auto& limitOld : paramLimitsOld) {
403 auto limitNew = limitOld;
404 switch (limitOld.type) {
405 case MinMax: {
406 auto& data = limitNew.data.minMax;
407 data.parameterIndex = oldParamToNewParam[data.parameterIndex];
408 if (data.parameterIndex == kInvalidIndex) {
409 continue;
410 }
411 break;
412 }
413
414 case MinMaxJoint: {
415 break;
416 }
417
418 case Linear: {
419 auto& data = limitNew.data.linear;
420 data.referenceIndex = oldParamToNewParam[data.referenceIndex];
421 data.targetIndex = oldParamToNewParam[data.targetIndex];
422 if (data.referenceIndex == kInvalidIndex || data.targetIndex == kInvalidIndex) {
423 continue;
424 }
425 break;
426 }
427
428 case LinearJoint: {
429 break;
430 }
431
432 case HalfPlane: {
433 auto& data = limitNew.data.halfPlane;
434 data.param1 = oldParamToNewParam[data.param1];
435 data.param2 = oldParamToNewParam[data.param2];
436 if (data.param1 == kInvalidIndex || data.param2 == kInvalidIndex) {
437 continue;
438 }
439 break;
440 }
441
442 case Ellipsoid:
443 case MinMaxJointPassive:
444 case LimitTypeCount: {
445 // nothing to do here.
446 break;
447 }
448 }
449
450 paramLimitsNew.push_back(limitNew);
451 }
452
453 return std::make_tuple(paramTransformNew, paramLimitsNew);
454}

507std::tuple<ParameterTransform, ParameterLimits> addBlendShapeParameters(
508 ParameterTransform paramTransform,
509 ParameterLimits paramLimits,
510 Eigen::Index nBlendShapes) {
511 // First, strip out any existing blend shape parameters, to make sure this
512 // operation is idempotent.
513 std::tie(paramTransform, paramLimits) = subsetParameterTransform(
514 paramTransform, paramLimits, ~paramTransform.getBlendShapeParameters());
515
516 // Now add in the additional parameters:
517 paramTransform.blendShapeParameters.setZero(nBlendShapes);
518 for (Eigen::Index iBlend = 0; iBlend < nBlendShapes; ++iBlend) {
519 std::ostringstream oss;
520 oss << "blend_" << iBlend;
521 paramTransform.blendShapeParameters(iBlend) = (int)paramTransform.name.size();
522 paramTransform.name.push_back(oss.str());
523 }
524
525 paramTransform.transform.conservativeResize(
526 paramTransform.transform.rows(), paramTransform.name.size());
527 paramTransform.transform.makeCompressed();
528
529 return {paramTransform, paramLimits};
530}

557std::tuple<ParameterTransform, ParameterLimits> addSkinnedLocatorParameters(
558 ParameterTransform paramTransform,
559 ParameterLimits paramLimits,
560 const std::vector<bool>& activeLocators,
561 const std::vector<std::string>& locatorNames) {
562 // First, strip out any existing blend shape parameters, to make sure this
563 // operation is idempotent.
564 std::tie(paramTransform, paramLimits) = subsetParameterTransform(
565 paramTransform, paramLimits, ~paramTransform.getSkinnedLocatorParameters());
566
567 const auto nLocators = activeLocators.size();
568 MT_CHECK(locatorNames.empty() || locatorNames.size() == nLocators);
569 paramTransform.skinnedLocatorParameters.resize(nLocators);
570 paramTransform.skinnedLocatorParameters.setConstant(-1);
571
572 ParameterSet paramSet;
573
574 // Now add in the additional parameters:
575 for (size_t iLocator = 0; iLocator < activeLocators.size(); ++iLocator) {
576 if (!activeLocators[iLocator]) {
577 continue;
578 }
579
580 paramTransform.skinnedLocatorParameters(iLocator) = (int)paramTransform.name.size();
581
582 auto locatorName =
583 iLocator < locatorNames.size() ? locatorNames[iLocator] : std::to_string(iLocator);
584 std::array<const char*, 3> dirNames{"x", "y", "z"};
585 for (const auto& d : dirNames) {
586 paramSet.set(paramTransform.name.size());
587 std::ostringstream oss;
588 oss << "locator_" << locatorName << "_" << d;
589 paramTransform.name.push_back(oss.str());
590 }
591 }
592
593 paramTransform.parameterSets["skinnedLocators"] = paramSet;
594 paramTransform.transform.conservativeResize(
595 paramTransform.transform.rows(), paramTransform.name.size());
596 paramTransform.transform.makeCompressed();
597
598 return {paramTransform, paramLimits};
