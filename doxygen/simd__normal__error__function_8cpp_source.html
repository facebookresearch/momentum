30SimdNormalConstraints::SimdNormalConstraints(const Skeleton* skel) {
31 // resize all arrays to the number of joints
32 MT_CHECK(skel->joints.size() <= kMaxJoints);
33 numJoints = static_cast<int>(skel->joints.size());
34 const auto dataSize = kMaxConstraints * numJoints;
35 MT_CHECK(dataSize % kSimdAlignment == 0);
36
37 // create memory for all floats
38 constexpr size_t nBlocks = 10;
39 data = makeAlignedUnique<float, kSimdAlignment>(dataSize * nBlocks);
40 float* dataPtr = data.get();
41 std::fill_n(dataPtr, dataSize * nBlocks, 0.0f);
42
43 offsetX = dataPtr + dataSize * 0;
44 offsetY = dataPtr + dataSize * 1;
45 offsetZ = dataPtr + dataSize * 2;
46 normalX = dataPtr + dataSize * 3;
47 normalY = dataPtr + dataSize * 4;
48 normalZ = dataPtr + dataSize * 5;
49 targetX = dataPtr + dataSize * 6;
50 targetY = dataPtr + dataSize * 7;
51 targetZ = dataPtr + dataSize * 8;
52 weights = dataPtr + dataSize * 9;
53}

57bool SimdNormalConstraints::addConstraint(
58 const size_t jointIndex,
59 const Vector3f& offset,
60 const Vector3f& normal,
61 const Vector3f& target,
62 const float targetWeight) {
63 MT_CHECK(jointIndex < constraintCount.size());
64
65 // add the constraint to the corresponding arrays of the jointIndex if there's enough space
66 uint32_t index = 0;
67 while (true) {
68 index = constraintCount[jointIndex];
69 if (index == kMaxConstraints) {
70 return false;
71 }
72
73 if (constraintCount[jointIndex].compare_exchange_weak(index, index + (uint32_t)1)) {
74 break;
75 }
76 }
77
78 const auto finalIndex = jointIndex * kMaxConstraints + index;
79 offsetX[finalIndex] = offset.x();
80 offsetY[finalIndex] = offset.y();
81 offsetZ[finalIndex] = offset.z();
82 normalX[finalIndex] = normal.x();
83 normalY[finalIndex] = normal.y();
84 normalZ[finalIndex] = normal.z();
85 targetX[finalIndex] = target.x();
86 targetY[finalIndex] = target.y();
87 targetZ[finalIndex] = target.z();
88 weights[finalIndex] = targetWeight;
89 return true;
90}

118double SimdNormalErrorFunction::getError(
119 const ModelParameters& /* params */,
120 const SkeletonState& state) {
121 if (constraints_ == nullptr) {
122 return 0.0f;
123 }
124
125 MT_CHECK((size_t)constraints_->numJoints <= state.jointState.size());
126
127 // loop over all joints, as these are our base units
128 auto error = drjit::zeros<DoubleP>(); // use double to prevent rounding errors
129 for (int jointId = 0; jointId < constraints_->numJoints; jointId++) {
130 const size_t constraintCount = constraints_->constraintCount[jointId];
131 MT_CHECK(jointId < static_cast<int>(state.jointState.size()));
132 const auto& jointState = state.jointState[jointId];
133 const Eigen::Matrix3f jointRotMat = jointState.rotation().toRotationMatrix();
134
135 for (uint32_t index = 0; index < constraintCount; index += kSimdPacketSize) {
136 const auto constraintOffsetIndex = jointId * SimdNormalConstraints::kMaxConstraints + index;
137 const Vector3fP target{
138 drjit::load<FloatP>(&constraints_->targetX[constraintOffsetIndex]),
139 drjit::load<FloatP>(&constraints_->targetY[constraintOffsetIndex]),
140 drjit::load<FloatP>(&constraints_->targetZ[constraintOffsetIndex])};
141
142 const Vector3fP offset{
143 drjit::load<FloatP>(&constraints_->offsetX[constraintOffsetIndex]),
144 drjit::load<FloatP>(&constraints_->offsetY[constraintOffsetIndex]),
145 drjit::load<FloatP>(&constraints_->offsetZ[constraintOffsetIndex])};
146 const Vector3fP pos_world = jointState.transformation * offset;
147
148 const Vector3fP normal{
149 drjit::load<FloatP>(&constraints_->normalX[constraintOffsetIndex]),
150 drjit::load<FloatP>(&constraints_->normalY[constraintOffsetIndex]),
151 drjit::load<FloatP>(&constraints_->normalZ[constraintOffsetIndex])};
152 const Vector3fP normal_world = momentum::operator*(jointRotMat, normal);
153
154 const FloatP dist = dot(pos_world - target, normal_world);
155
156 const auto weight = drjit::load<FloatP>(&constraints_->weights[constraintOffsetIndex]);
157 error += weight * drjit::square(dist);
158 }
159 }
160
161 // sum the SIMD error register for final result
162 return static_cast<float>(drjit::sum(error) * kPlaneWeight * weight_);
163}

165double SimdNormalErrorFunction::getGradient(
166 const ModelParameters& /* params */,
167 const SkeletonState& state,
168 Ref<VectorXf> gradient) {
169 if (constraints_ == nullptr) {
170 return 0.0f;
171 }
172
173 auto error = drjit::zeros<DoubleP>(); // use double to prevent rounding errors
174 for (int jointId = 0; jointId < constraints_->numJoints; jointId++) {
175 const size_t constraintCount = constraints_->constraintCount[jointId];
176 if (jointId >= static_cast<int>(state.jointState.size())) {
177 break;
178 }
179 const Eigen::Matrix3f jointRotMat = state.jointState[jointId].rotation().toRotationMatrix();
180 const auto& jointState_cons = state.jointState[jointId];
181 for (uint32_t index = 0; index < constraintCount; index += kSimdPacketSize) {
182 const auto constraintOffsetIndex = jointId * SimdNormalConstraints::kMaxConstraints + index;
183
184 const Vector3fP offset{
185 drjit::load<FloatP>(&constraints_->offsetX[constraintOffsetIndex]),
186 drjit::load<FloatP>(&constraints_->offsetY[constraintOffsetIndex]),
187 drjit::load<FloatP>(&constraints_->offsetZ[constraintOffsetIndex])};
188 const Vector3fP pos_world = jointState_cons.transformation * offset;
189
190 const Vector3fP normal{
191 drjit::load<FloatP>(&constraints_->normalX[constraintOffsetIndex]),
192 drjit::load<FloatP>(&constraints_->normalY[constraintOffsetIndex]),
193 drjit::load<FloatP>(&constraints_->normalZ[constraintOffsetIndex])};
194 const Vector3fP normal_world = momentum::operator*(jointRotMat, normal);
195
196 const Vector3fP target{
197 drjit::load<FloatP>(&constraints_->targetX[constraintOffsetIndex]),
198 drjit::load<FloatP>(&constraints_->targetY[constraintOffsetIndex]),
199 drjit::load<FloatP>(&constraints_->targetZ[constraintOffsetIndex])};
200 const FloatP dist = dot(pos_world - target, normal_world);
201
202 const auto weight = drjit::load<FloatP>(&constraints_->weights[constraintOffsetIndex]);
203 const FloatP wgt = weight * (2.0f * kPlaneWeight);
204
205 error += weight * drjit::square(dist);
206
207 // loop over all joints the constraint is attached to and calculate gradient
208 size_t jointIndex = jointId;
209 while (jointIndex != kInvalidIndex) {
210 // check for valid index
211 const auto& jointState = state.jointState[jointIndex];
212
213 const size_t paramIndex = jointIndex * kParametersPerJoint;
214 const Vector3fP posd = momentum::operator-(pos_world, jointState.translation());
215
216 // calculate derivatives based on active joints
217 for (size_t d = 0; d < 3; d++) {
218 if (this->activeJointParams_[paramIndex + d]) {
219 // calculate joint gradient
220 const FloatP val =
221 dist * momentum::dot(normal_world, jointState.getTranslationDerivative(d)) * wgt;
222 // explicitly multiply with the parameter transform to generate parameter space
223 // gradients
224 for (auto ptIndex = parameterTransform_.transform.outerIndexPtr()[paramIndex + d];
225 ptIndex < parameterTransform_.transform.outerIndexPtr()[paramIndex + d + 1];
226 ++ptIndex) {
227 gradient[parameterTransform_.transform.innerIndexPtr()[ptIndex]] +=
228 drjit::sum(val) * parameterTransform_.transform.valuePtr()[ptIndex];
229 }
230 }
231 if (this->activeJointParams_[paramIndex + 3 + d]) {
232 // calculate joint gradient
233 const auto crossProd =
234 cross(normal_world, momentum::operator-(target, jointState.translation()));
235 const FloatP val =
236 -dist * momentum::dot(jointState.rotationAxis.col(d), crossProd) * wgt;
237 // explicitly multiply with the parameter transform to generate parameter space
238 // gradients
239 for (auto ptIndex = parameterTransform_.transform.outerIndexPtr()[paramIndex + d + 3];
240 ptIndex < parameterTransform_.transform.outerIndexPtr()[paramIndex + d + 3 + 1];
241 ++ptIndex) {
242 gradient[parameterTransform_.transform.innerIndexPtr()[ptIndex]] +=
243 drjit::sum(val) * parameterTransform_.transform.valuePtr()[ptIndex];
244 }
245 }
246 }
247 if (this->activeJointParams_[paramIndex + 6]) {
248 // calculate joint gradient
249 const FloatP val = dist * ln2() * dot(normal_world, posd) * wgt;
250 // explicitly multiply with the parameter transform to generate parameter space gradients
251 for (auto ptIndex = parameterTransform_.transform.outerIndexPtr()[paramIndex + 6];
252 ptIndex < parameterTransform_.transform.outerIndexPtr()[paramIndex + 6 + 1];
253 ++ptIndex) {
254 gradient[parameterTransform_.transform.innerIndexPtr()[ptIndex]] +=
255 drjit::sum(val) * parameterTransform_.transform.valuePtr()[ptIndex];
256 }
257 }
258
259 // go to the next joint
260 jointIndex = this->skeleton_.joints[jointIndex].parent;
261 }
262 }
263 }
264
265 return static_cast<float>(drjit::sum(error) * kPlaneWeight * weight_);
266}

268__vectorcall DRJIT_INLINE void jacobian_jointParams_to_modelParams(
269 const FloatP& jacobian_jointParams,
270 const Eigen::Index iJointParam,
271 const ParameterTransform& parameterTransform_,
272 Eigen::Ref<Eigen::MatrixX<float>> jacobian) {
273 // explicitly multiply with the parameter transform to generate parameter space gradients
274 for (auto index = parameterTransform_.transform.outerIndexPtr()[iJointParam];
275 index < parameterTransform_.transform.outerIndexPtr()[iJointParam + 1];
276 ++index) {
277 const auto modelParamIdx = parameterTransform_.transform.innerIndexPtr()[index];
278 float* jacPtr = jacobian.col(modelParamIdx).data();
279 drjit::store(
280 jacPtr,
281 drjit::load<FloatP>(jacPtr) +
282 parameterTransform_.transform.valuePtr()[index] * jacobian_jointParams);
283 }
284}

286double SimdNormalErrorFunction::getJacobian(
287 const ModelParameters& /* params */,
288 const SkeletonState& state,
289 Ref<MatrixXf> jacobian,
290 Ref<VectorXf> residual,
291 int& usedRows) {
292 MT_PROFILE_FUNCTION();
293 MT_CHECK(jacobian.cols() == static_cast<Eigen::Index>(parameterTransform_.transform.cols()));
294
295 if (constraints_ == nullptr) {
296 return 0.0f;
297 }
298
299 MT_CHECK((size_t)constraints_->numJoints <= jacobianOffset_.size());
300
301 // storage for joint errors
302 std::vector<double> jointErrors(constraints_->numJoints);
303
304 // loop over all joints, as these are our base units
305 auto dispensoOptions = dispenso::ParForOptions();
306 dispensoOptions.maxThreads = maxThreads_;
307 dispenso::parallel_for(
308 0,
309 constraints_->numJoints,
310 [&](const size_t jointId) {
311 const size_t constraintCount = constraints_->constraintCount[jointId];
312 const auto& jointState_cons = state.jointState[jointId];
313 const Eigen::Matrix3f jointRotMat = jointState_cons.rotation().toRotationMatrix();
314 auto jointError = drjit::zeros<DoubleP>(); // use double to prevent rounding errors
315
316 for (uint32_t index = 0; index < constraintCount; index += kSimdPacketSize) {
317 const auto constraintOffsetIndex =
318 jointId * SimdNormalConstraints::kMaxConstraints + index;
319 const auto jacobianOffsetCur = jacobianOffset_[jointId] + index;
320
321 const Vector3fP offset{
322 drjit::load<FloatP>(&constraints_->offsetX[constraintOffsetIndex]),
323 drjit::load<FloatP>(&constraints_->offsetY[constraintOffsetIndex]),
324 drjit::load<FloatP>(&constraints_->offsetZ[constraintOffsetIndex])};
325 const Vector3fP pos_world = jointState_cons.transformation * offset;
326
327 const Vector3fP normal{
328 drjit::load<FloatP>(&constraints_->normalX[constraintOffsetIndex]),
329 drjit::load<FloatP>(&constraints_->normalY[constraintOffsetIndex]),
330 drjit::load<FloatP>(&constraints_->normalZ[constraintOffsetIndex])};
331 const Vector3fP normal_world = momentum::operator*(jointRotMat, normal);
332
333 const Vector3fP target{
334 drjit::load<FloatP>(&constraints_->targetX[constraintOffsetIndex]),
335 drjit::load<FloatP>(&constraints_->targetY[constraintOffsetIndex]),
336 drjit::load<FloatP>(&constraints_->targetZ[constraintOffsetIndex])};
337 const FloatP dist = dot(pos_world - target, normal_world);
338
339 const auto weight = drjit::load<FloatP>(&constraints_->weights[constraintOffsetIndex]);
340 const FloatP wgt = drjit::sqrt(weight * kPlaneWeight * this->weight_);
341
342 jointError += weight * drjit::square(dist);
343
344 drjit::store(residual.segment(jacobianOffsetCur, kSimdPacketSize).data(), dist * wgt);
345
346 // loop over all joints the constraint is attached to and calculate gradient
347 size_t jointIndex = jointId;
348 while (jointIndex != kInvalidIndex) {
349 // check for valid index
350 const auto& jointState = state.jointState[jointIndex];
351 const size_t paramIndex = jointIndex * kParametersPerJoint;
352 // calculate derivatives based on active joints
353 for (size_t d = 0; d < 3; d++) {
354 if (this->activeJointParams_[paramIndex + d]) {
355 const FloatP jc =
356 momentum::dot(normal_world, jointState.getTranslationDerivative(d)) * wgt;
357 jacobian_jointParams_to_modelParams(
358 jc,
359 paramIndex + d,
360 parameterTransform_,
361 jacobian.middleRows(jacobianOffsetCur, kSimdPacketSize));
362 }
363 }
364
365 const Vector3fP tgtd = momentum::operator-(target, jointState.translation());
366 for (size_t d = 0; d < 3; ++d) {
367 if (this->activeJointParams_[paramIndex + 3 + d]) {
368 const Vector3fP crossProd = cross(normal_world, tgtd);
369 const Vector3f axis = jointState.rotationAxis.col(d);
370 const FloatP jc = -momentum::dot(axis, crossProd) * wgt;
371 jacobian_jointParams_to_modelParams(
372 jc,
373 paramIndex + d + 3,
374 parameterTransform_,
375 jacobian.middleRows(jacobianOffsetCur, kSimdPacketSize));
376 }
377 }
378
379 if (this->activeJointParams_[paramIndex + 6]) {
380 const Vector3fP posd = momentum::operator-(pos_world, jointState.translation());
381 const FloatP jc = dot(normal_world, posd) * (ln2() * wgt);
382 jacobian_jointParams_to_modelParams(
383 jc,
384 paramIndex + 6,
385 parameterTransform_,
386 jacobian.middleRows(jacobianOffsetCur, kSimdPacketSize));
387 }
388
389 // go to the next joint
390 jointIndex = this->skeleton_.joints[jointIndex].parent;
391 }
392 }
393
394 jointErrors[jointId] += drjit::sum(jointError);
395 },
396 dispensoOptions);
397
398 double error =
399 kPlaneWeight * weight_ * std::accumulate(jointErrors.begin(), jointErrors.end(), 0.0);
400
401 usedRows = gsl::narrow_cast<int>(jacobian.rows());
402
403 // sum the SIMD error register for final result
404 return static_cast<float>(error);
405}

407size_t SimdNormalErrorFunction::getJacobianSize() const {
408 if (constraints_ == nullptr) {
409 return 0;
410 }
411
412 jacobianOffset_.resize(constraints_->numJoints);
413 size_t num = 0;
414 for (int i = 0; i < constraints_->numJoints; i++) {
415 jacobianOffset_[i] = num;
416 const size_t count = constraints_->constraintCount[i];
417 const auto numPackets = (count + kSimdPacketSize - 1) / kSimdPacketSize;
418 num += numPackets * kSimdPacketSize;
419 }
420 return num;
421}

425inline double __vectorcall sum8(const __m256 x) {
426 // extract the higher and lower 4 floats
427 const __m128 high = _mm256_extractf128_ps(x, 1);
428 const __m128 low = _mm256_castps256_ps128(x);
429 // convert them to 256 bit doubles and add them up
430 const __m256d val = _mm256_add_pd(_mm256_cvtps_pd(high), _mm256_cvtps_pd(low));
431 // sum the 4 doubles up and return the result
432 const __m128d valupper = _mm256_extractf128_pd(val, 1);
433 const __m128d vallower = _mm256_castpd256_pd128(val);
434 _mm256_zeroupper();
435 const __m128d valval = _mm_add_pd(valupper, vallower);
436 const __m128d res = _mm_add_pd(_mm_permute_pd(valval, 1), valval);
437 return _mm_cvtsd_f64(res);
438}

440double SimdNormalErrorFunctionAVX::getJacobian(
441 const ModelParameters& /* params */,
442 const SkeletonState& state,
443 Ref<MatrixXf> jacobian,
444 Ref<VectorXf> residual,
445 int& usedRows) {
446 MT_PROFILE_FUNCTION();
447 MT_CHECK(jacobian.cols() == static_cast<Eigen::Index>(parameterTransform_.transform.cols()));
448
449 if (constraints_ == nullptr) {
450 return 0.0f;
451 }
452
453 MT_CHECK((size_t)constraints_->numJoints <= jacobianOffset_.size());
454
455 // storage for joint errors
456 std::vector<double> ets_error;
457
458 // need to make sure we're actually at a 32 byte data offset at the first offset for AVX access
459 const size_t addressOffset = computeOffset<kAvxAlignment>(jacobian);
460 checkAlignment<kAvxAlignment>(jacobian, addressOffset);
461
462 // loop over all joints, as these are our base units
463 auto dispensoOptions = dispenso::ParForOptions();
464 dispensoOptions.maxThreads = maxThreads_;
465 dispenso::parallel_for(
466 ets_error,
467 []() { return 0.0; },
468 0,
469 constraints_->numJoints,
470 [&](double& error_local, const size_t jointId) {
471 // get initial offset
472 const auto offset = jacobianOffset_[jointId] + addressOffset;
473
474 // pre-load some joint specific values
475 const auto& transformation = state.jointState[jointId].transformation;
476
477 __m256 posx;
478 __m256 posy;
479 __m256 posz;
480 __m256 nmlx;
481 __m256 nmly;
482 __m256 nmlz;
483 __m256 dist;
484
485 const auto jointOffset = jointId * SimdNormalConstraints::kMaxConstraints;
486
487 // loop over all constraints in increments of kAvxPacketSize
488 const uint32_t count = constraints_->constraintCount[jointId];
489 for (uint32_t index = 0; index < count; index += kAvxPacketSize) {
490 // transform offset by joint transformation : pos = transform * offset
491 // also transform normal by transformation : nml = (transform.linear() *
492 // normal).normalized()
493 const __m256 valx = _mm256_load_ps(&constraints_->offsetX[jointOffset + index]);
494 const __m256 nx = _mm256_load_ps(&constraints_->normalX[jointOffset + index]);
495 posx = _mm256_mul_ps(valx, _mm256_broadcast_ss(&transformation.data()[0]));
496 posy = _mm256_mul_ps(valx, _mm256_broadcast_ss(&transformation.data()[1]));
497 posz = _mm256_mul_ps(valx, _mm256_broadcast_ss(&transformation.data()[2]));
498 nmlx = _mm256_mul_ps(nx, _mm256_broadcast_ss(&transformation.data()[0]));
499 nmly = _mm256_mul_ps(nx, _mm256_broadcast_ss(&transformation.data()[1]));
500 nmlz = _mm256_mul_ps(nx, _mm256_broadcast_ss(&transformation.data()[2]));
501 const __m256 valy = _mm256_load_ps(&constraints_->offsetY[jointOffset + index]);
502 const __m256 ny = _mm256_load_ps(&constraints_->normalY[jointOffset + index]);
503 posx = _mm256_fmadd_ps(valy, _mm256_broadcast_ss(&transformation.data()[4]), posx);
504 posy = _mm256_fmadd_ps(valy, _mm256_broadcast_ss(&transformation.data()[5]), posy);
505 posz = _mm256_fmadd_ps(valy, _mm256_broadcast_ss(&transformation.data()[6]), posz);
506 nmlx = _mm256_fmadd_ps(ny, _mm256_broadcast_ss(&transformation.data()[4]), nmlx);
507 nmly = _mm256_fmadd_ps(ny, _mm256_broadcast_ss(&transformation.data()[5]), nmly);
508 nmlz = _mm256_fmadd_ps(ny, _mm256_broadcast_ss(&transformation.data()[6]), nmlz);
509 const __m256 valz = _mm256_load_ps(&constraints_->offsetZ[jointOffset + index]);
510 const __m256 nz = _mm256_load_ps(&constraints_->normalZ[jointOffset + index]);
511 posx = _mm256_fmadd_ps(valz, _mm256_broadcast_ss(&transformation.data()[8]), posx);
512 posy = _mm256_fmadd_ps(valz, _mm256_broadcast_ss(&transformation.data()[9]), posy);
513 posz = _mm256_fmadd_ps(valz, _mm256_broadcast_ss(&transformation.data()[10]), posz);
514 nmlx = _mm256_fmadd_ps(nz, _mm256_broadcast_ss(&transformation.data()[8]), nmlx);
515 nmly = _mm256_fmadd_ps(nz, _mm256_broadcast_ss(&transformation.data()[9]), nmly);
516 nmlz = _mm256_fmadd_ps(nz, _mm256_broadcast_ss(&transformation.data()[10]), nmlz);
517 posx = _mm256_add_ps(_mm256_broadcast_ss(&transformation.data()[12]), posx);
518 posy = _mm256_add_ps(_mm256_broadcast_ss(&transformation.data()[13]), posy);
519 posz = _mm256_add_ps(_mm256_broadcast_ss(&transformation.data()[14]), posz);
520
521 __m256 norm = _mm256_mul_ps(nmlx, nmlx);
522 norm = _mm256_fmadd_ps(nmly, nmly, norm);
523 norm = _mm256_fmadd_ps(nmlz, nmlz, norm);
524 norm = _mm256_rsqrt_ps(_mm256_max_ps(norm, _mm256_set1_ps(0.0001f)));
525 nmlx = _mm256_mul_ps(nmlx, norm);
526 nmly = _mm256_mul_ps(nmly, norm);
527 nmlz = _mm256_mul_ps(nmlz, norm);
528
529 // calculate distance of point to plane : dist = normal.dot(pos) -
530 // normal.dot(data->target)
531 const __m256 tgtx = _mm256_load_ps(&constraints_->targetX[jointOffset + index]);
532 const __m256 tgty = _mm256_load_ps(&constraints_->targetY[jointOffset + index]);
533 const __m256 tgtz = _mm256_load_ps(&constraints_->targetZ[jointOffset + index]);
534 dist = _mm256_mul_ps(posx, nmlx);
535 dist = _mm256_fmadd_ps(posy, nmly, dist);
536 dist = _mm256_fmadd_ps(posz, nmlz, dist);
537 dist = _mm256_fnmadd_ps(tgtx, nmlx, dist);
538 dist = _mm256_fnmadd_ps(tgty, nmly, dist);
539 dist = _mm256_fnmadd_ps(tgtz, nmlz, dist);
540
541 // calculate square-root of weight : wgt = sqrt(kPlaneWeight * weight
542 // * constraintWeight)
543 const float w = kPlaneWeight * weight_;
544 const __m256 wt = _mm256_mul_ps(
545 _mm256_broadcast_ss(&w), _mm256_load_ps(&constraints_->weights[jointOffset + index]));
546 const __m256 wgt = _mm256_sqrt_ps(wt);
547
548 // calculate residual : res = wgt * dist
549 const __m256 res = _mm256_mul_ps(wgt, dist);
550
551 // calculate error as squared residual : err = res * res
552 const __m256 err = _mm256_mul_ps(res, res);
553
554 // sum up the values and add to error
555 error_local += sum8(err);
556
557 // loop over all joints the constraint is attached to and calculate gradient
558 size_t jointIndex = jointId;
559 while (jointIndex != kInvalidIndex) {
560 // check for valid index
561 const auto& jointState = state.jointState[jointIndex];
562 const size_t paramIndex = jointIndex * kParametersPerJoint;
563
564 // calculate difference between constraint position and joint center : posd =
565 // pos - jointState.translation
566 const __m256 posdx = _mm256_sub_ps(posx, _mm256_broadcast_ss(&jointState.x()));
567 const __m256 posdy = _mm256_sub_ps(posy, _mm256_broadcast_ss(&jointState.y()));
568 const __m256 posdz = _mm256_sub_ps(posz, _mm256_broadcast_ss(&jointState.z()));
569
570 // calculate difference between target position and joint center : tgtd =
571 // tgt - jointState.translation
572 const __m256 tgtdx = _mm256_sub_ps(tgtx, _mm256_broadcast_ss(&jointState.x()));
573 const __m256 tgtdy = _mm256_sub_ps(tgty, _mm256_broadcast_ss(&jointState.y()));
574 const __m256 tgtdz = _mm256_sub_ps(tgtz, _mm256_broadcast_ss(&jointState.z()));
575
576 // calculate derivatives based on active joints
577 for (size_t d = 0; d < 3; d++) {
578 if (activeJointParams_[paramIndex + d]) {
579 // calculate jacobian with respect to joint : jac =
580 // normal.dot(axis) * wgt;
581 const Vector3f& axis = jointState.translationAxis.col(d);
582 const __m256 axisx = _mm256_broadcast_ss(&axis.x());
583 const __m256 axisy = _mm256_broadcast_ss(&axis.y());
584 const __m256 axisz = _mm256_broadcast_ss(&axis.z());
585 const __m256 jac = _mm256_mul_ps(
586 _mm256_fmadd_ps(
587 nmlz, axisz, _mm256_fmadd_ps(nmly, axisy, _mm256_mul_ps(nmlx, axisx))),
588 wgt);
589
590 // explicitly multiply with the parameter transform to generate parameter space
591 // jacobians
592 for (auto pIndex = parameterTransform_.transform.outerIndexPtr()[paramIndex + d];
593 pIndex < parameterTransform_.transform.outerIndexPtr()[paramIndex + d + 1];
594 ++pIndex) {
595 const float pw = parameterTransform_.transform.valuePtr()[pIndex];
596 const __m256 pjac = _mm256_mul_ps(_mm256_broadcast_ss(&pw), jac);
597 auto* const address = &jacobian(
598 offset + index, parameterTransform_.transform.innerIndexPtr()[pIndex]);
599 MT_CHECK((uintptr_t)address % 32 == 0);
600 const __m256 prev = _mm256_load_ps(address);
601 _mm256_store_ps(address, _mm256_add_ps(prev, pjac));
602 }
603 }
604 if (activeJointParams_[paramIndex + 3 + d]) {
605 // calculate jacobian with respect to joint : jac =
606 // axis.dot(tgtd.cross(normal)) * wgt;
607 const __m256 crossx = _mm256_fmsub_ps(tgtdy, nmlz, _mm256_mul_ps(tgtdz, nmly));
608 const __m256 crossy = _mm256_fmsub_ps(tgtdz, nmlx, _mm256_mul_ps(tgtdx, nmlz));
609 const __m256 crossz = _mm256_fmsub_ps(tgtdx, nmly, _mm256_mul_ps(tgtdy, nmlx));
610 const Vector3f& axis = jointState.rotationAxis.col(d);
611 const __m256 axisx = _mm256_broadcast_ss(&axis.x());
612 const __m256 axisy = _mm256_broadcast_ss(&axis.y());
613 const __m256 axisz = _mm256_broadcast_ss(&axis.z());
614 const __m256 jac = _mm256_mul_ps(
615 _mm256_fmadd_ps(
616 axisz,
617 crossz,
618 _mm256_fmadd_ps(axisy, crossy, _mm256_mul_ps(axisx, crossx))),
619 wgt);
620
621 // explicitly multiply with the parameter transform to generate parameter space
622 // jacobians
623 for (auto pIndex =
624 parameterTransform_.transform.outerIndexPtr()[paramIndex + d + 3];
625 pIndex < parameterTransform_.transform.outerIndexPtr()[paramIndex + d + 3 + 1];
626 ++pIndex) {
627 const float pw = parameterTransform_.transform.valuePtr()[pIndex];
628 const __m256 pjac = _mm256_mul_ps(_mm256_broadcast_ss(&pw), jac);
629 auto* const address = &jacobian(
630 offset + index, parameterTransform_.transform.innerIndexPtr()[pIndex]);
631 MT_CHECK((uintptr_t)address % 32 == 0);
632 const __m256 prev = _mm256_load_ps(address);
633 _mm256_store_ps(address, _mm256_add_ps(prev, pjac));
634 }
635 }
636 }
637 if (activeJointParams_[paramIndex + 6]) {
638 // calculate jacobian with respect to joint :
639 // jac = normal.dot(posd) * wgt * LN2;
640 constexpr double kLn2d = ln2<double>();
641 const __m256 jac = _mm256_mul_ps(
642 _mm256_mul_ps(
643 _mm256_fmadd_ps(
644 nmlz, posdz, _mm256_fmadd_ps(nmly, posdy, _mm256_mul_ps(nmlx, posdx))),
645 wgt),
646 _mm256_set_ps(kLn2d, kLn2d, kLn2d, kLn2d, kLn2d, kLn2d, kLn2d, kLn2d));
647
648 // explicitly multiply with the parameter transform to generate parameter space
649 // jacobians
650 for (auto pIndex = parameterTransform_.transform.outerIndexPtr()[paramIndex + 6];
651 pIndex < parameterTransform_.transform.outerIndexPtr()[paramIndex + 6 + 1];
652 ++pIndex) {
653 const float pw = parameterTransform_.transform.valuePtr()[pIndex];
654 const __m256 pjac = _mm256_mul_ps(_mm256_broadcast_ss(&pw), jac);
655 auto* const address = &jacobian(
656 offset + index, parameterTransform_.transform.innerIndexPtr()[pIndex]);
657 MT_CHECK((uintptr_t)address % 32 == 0);
658 const __m256 prev = _mm256_load_ps(address);
659 _mm256_store_ps(address, _mm256_add_ps(prev, pjac));
660 }
661 }
662
663 // go to the next joint
664 jointIndex = skeleton_.joints[jointIndex].parent;
665 }
666
667 // store the residual
668 _mm256_storeu_ps(&residual(offset + index), res);
669 }
670 },
671 dispensoOptions);
672
673 double error = std::accumulate(ets_error.begin(), ets_error.end(), 0.0);
674
675 usedRows = gsl::narrow_cast<int>(jacobian.rows());
676
677 // sum the AVX error register for final result
678 return static_cast<float>(error);
679}

681size_t SimdNormalErrorFunctionAVX::getJacobianSize() const {
682 if (constraints_ == nullptr) {
683 return 0;
684 }
685
686 jacobianOffset_.resize(constraints_->numJoints);
687 size_t num = 0;
688 for (int i = 0; i < constraints_->numJoints; i++) {
689 jacobianOffset_[i] = num;
690 const size_t count = constraints_->constraintCount[i];
691 const auto numPackets = (count + kAvxPacketSize - 1) / kAvxPacketSize;
692 num += numPackets * kAvxPacketSize;
693 }
694 return num;
695}
