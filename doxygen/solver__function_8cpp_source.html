21template <typename T>
22double SolverFunctionT<T>::getJacobian(
23 const VectorX<T>& parameters,
24 MatrixX<T>& jacobian,
25 VectorX<T>& residual,
26 size_t& actualRows) {
27 initializeJacobianComputation(parameters);
28
29 const size_t numBlocks = getJacobianBlockCount();
30
31 // Calculate total size
32 size_t totalRows = 0;
33 for (size_t i = 0; i < numBlocks; ++i) {
34 totalRows += getJacobianBlockSize(i);
35 }
36 // Add padding for alignment
37 totalRows = padToSimdAlignment(totalRows);
38
39 // Resize if needed
40 if (totalRows > static_cast<size_t>(jacobian.rows()) || parameters.size() != jacobian.cols()) {
41 jacobian.resize(totalRows, parameters.size());
42 residual.resize(totalRows);
43 }
44
45 jacobian.setZero();
46 residual.setZero();
47
48 double error = 0.0;
49 size_t position = 0;
50 actualRows = totalRows;
51
52 // Assemble blocks into full Jacobian
53 for (size_t i = 0; i < numBlocks; ++i) {
54 const size_t blockSize = getJacobianBlockSize(i);
55 if (blockSize == 0) {
56 continue;
57 }
58
59 size_t blockActualRows = 0;
60
61 auto jacBlock = jacobian.block(position, 0, blockSize, parameters.size());
62 auto resBlock = residual.segment(position, blockSize);
63
64 error += computeJacobianBlock(parameters, i, jacBlock, resBlock, blockActualRows);
65
66 position += blockSize;
67 }
68
69 finalizeJacobianComputation();
70 return error;
71}

73template <typename T>
74double SolverFunctionT<T>::getJtJR(const VectorX<T>& parameters, MatrixX<T>& jtj, VectorX<T>& jtr) {
75 initializeJacobianComputation(parameters);
76
77 const size_t numBlocks = getJacobianBlockCount();
78
79 // Calculate total Jacobian size
80 const size_t numParams = parameters.size();
81
82 jtj.setZero(actualParameters_, actualParameters_);
83 jtr.setZero(actualParameters_);
84
85 // Block compute JtJ and JtR on the fly
86 double error = 0.0;
87
88 for (size_t i = 0; i < numBlocks; ++i) {
89 const size_t blockSize = padToSimdAlignment(getJacobianBlockSize(i));
90 if (blockSize == 0) {
91 continue;
92 }
93
94 size_t blockActualRows = 0;
95
96 tJacobian_.resizeAndSetZero(blockSize, numParams);
97 tResidual_.resizeAndSetZero(blockSize);
98
99 auto jacBlock = tJacobian_.mat().block(0, 0, blockSize, numParams);
100 auto resBlock = tResidual_.vec().segment(0, blockSize);
101
102 error += computeJacobianBlock(parameters, i, jacBlock, resBlock, blockActualRows);
103
104 // Update JtJ and JtR
105 if (blockActualRows == 0) {
106 continue;
107 }
108
109 const auto JtBlock =
110 tJacobian_.mat().block(0, 0, blockActualRows, actualParameters_).transpose();
111
112 // Efficiently update JtJ using selfadjointView with rankUpdate
113 jtj.template selfadjointView<Eigen::Lower>().rankUpdate(JtBlock);
114
115 // Update JtR
116 jtr.noalias() += JtBlock * tResidual_.vec().segment(0, blockActualRows);
117 }
118
119 finalizeJacobianComputation();
120 return error;
121}
