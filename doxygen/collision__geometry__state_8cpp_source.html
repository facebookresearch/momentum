14template <typename T>
15void CollisionGeometryStateT<T>::update(
16 const SkeletonStateT<T>& skeletonState,
17 const CollisionGeometry& collisionGeometry) {
18 // resize all elements
19 const size_t numElements = collisionGeometry.size();
20 origin.resize(numElements);
21 direction.resize(numElements);
22 radius.resize(numElements);
23 delta.resize(numElements);
24
25 // calculate data from the geometry (can be parallelized)
26 for (size_t i = 0; i < numElements; ++i) {
27 const auto& tc = collisionGeometry[i];
28 const auto& js = skeletonState.jointState[tc.parent];
29 const TransformT<T> transform = js.transform * tc.transformation.cast<T>();
30 origin[i] = transform.translation;
31 direction[i].noalias() = transform.getAxisX() * transform.scale * tc.length;
32 radius[i].noalias() = tc.radius.cast<T>() * js.transform.scale;
33 delta[i] = radius[i][1] - radius[i][0];
34 }
35}
