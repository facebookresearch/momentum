19template <typename T>
20std::vector<Eigen::Index> computeSubsetParameterIndices(
21 const ParameterTransformT<T>& parameterTransform,
22 const ParameterSet& activeParams) {
23 std::vector<Eigen::Index> result;
24 for (size_t i = 0; i < parameterTransform.numAllModelParameters(); ++i) {
25 if (activeParams.test(i)) {
26 result.push_back(i);
27 }
28 }
29
30 return result;
31}

33template <typename T>
34Eigen::VectorX<T> extractSubsetVec(
35 Eigen::Ref<const Eigen::VectorX<T>> fullVec,
36 const std::vector<Eigen::Index>& subsetIndexToFullParamIndex) {
37 const Eigen::Index nSubset = subsetIndexToFullParamIndex.size();
38 Eigen::VectorX<T> result(nSubset);
39 for (Eigen::Index iSubset = 0; iSubset < nSubset; ++iSubset) {
40 result(iSubset) = fullVec(subsetIndexToFullParamIndex[iSubset]);
41 }
42 return result;
43}

45template <typename T>
46Eigen::MatrixX<T> extractSubsetMat(
47 Eigen::Ref<const Eigen::MatrixX<T>> fullMat,
48 const std::vector<Eigen::Index>& subsetIndexToFullParamIndex) {
49 const Eigen::Index nSubset = subsetIndexToFullParamIndex.size();
50 Eigen::MatrixX<T> result(fullMat.rows(), nSubset);
51 for (Eigen::Index iSubset = 0; iSubset < nSubset; ++iSubset) {
52 result.col(iSubset) = fullMat.col(subsetIndexToFullParamIndex[iSubset]);
53 }
54 return result;
55}

73template <typename T>
74Eigen::VectorX<T> hessianInverseTimes(
75 Eigen::Ref<const Eigen::MatrixX<T>> jacobian,
76 Eigen::Ref<const Eigen::VectorX<T>> v) {
77 // Use approximate Hessian:
78 // Hess ~ J^T * J
79 // We will use the SVD on J as follows:
80 // J = U * S * V^T
81 // Hess = 2 * J^T * J = (U * S * V^T)^T * (U * S * V^T)
82 // = 2 * V * S * U^T * U * S * V^T
83 // = 2 * V * S^2 * V^T
84 // This is equivalent to performing an eigenvalue decomposition
85 // on the Hessian, and lets us ensure that the Hessian is actually
86 // invertible.
87 //
88 // Note that the particular SVD used here is not terribly fast for large n;
89 // we should consider switching to the one in LAPACK (the divide-and-conquer
90 // version in Eigen is apparently not reliable with -ffast-math).
91 Eigen::JacobiSVD<Eigen::MatrixX<T>> svd(jacobian, Eigen::ComputeThinV);
92
93 // To get the Hess^{-1} * v, we need:
94 // 2 * V * S^2 * V^T * w = v
95 // w = 0.5 * V^T * v)
96 // V^T * w = 0.5 * (S^2)^{-1} * V^T * v
97 // w = 0.5 * V * (S^2)^{-1} * V^T * v
98 Eigen::VectorX<T> tmp = svd.matrixV().transpose() * v;
99 for (Eigen::Index i = 0; i < tmp.size(); ++i) {
100 const float sSqr = sqr(svd.singularValues()(i));
101 if (sSqr < 1e-5f) {
102 tmp(i) = 0;
103 } else {
104 tmp(i) /= sSqr;
105 }
106 }
107 return 0.5f * svd.matrixV() * tmp;
108}

110template <typename T>
111std::vector<ErrorFunctionDerivativesT<T>> d_modelParams_d_inputs(
112 const Skeleton& skeleton,
113 const ParameterTransformT<T>& parameterTransform,
114 const ParameterSet& activeParams,
115 const ModelParametersT<T>& modelParameters,
116 SkeletonSolverFunctionT<T>& solverFunction,
117 Eigen::Ref<const Eigen::VectorX<T>> dLoss_dModelParams,
118 T* gradientRmse) {
119 std::vector<ErrorFunctionDerivativesT<T>> result(solverFunction.getErrorFunctions().size());
120
121 const std::vector<Eigen::Index> subsetIndexToFullParamIndex =
122 computeSubsetParameterIndices(parameterTransform, activeParams);
123 const Eigen::Index nFullParams = parameterTransform.numAllModelParameters();
124
125 Eigen::MatrixX<T> jacobian_full;
126 Eigen::VectorX<T> residual_full;
127
128 // Some basic definitions:
129 // theta_j are the model parameters, basically the skeleton pose.
130 // w_j are the "other" inputs that affect the error function values. Here I
131 // am thinking of things like constraint weights, etc.
132 // E(theta_i, w_j) is the energy function our IK solver is minimizing.
133 //
134 // At the minimum, we have the gradient equal to zero:
135 // grad E(theta, w) = dE/dTheta = 0
136 //
137 // Let f(theta, w) = dE/dTheta.
138 // By the implicit function theorem, under certain assumptions about
139 // continuity etc., there exists a function g(w) such that f(theta = g(w), w)
140 // = 0; that is, the energy function is still minimized. Basically, g(w) maps
141 // from any new (but sufficiently close) value of w to a new value of theta.
142 //
143 // Now, what we want is the _gradient_ of g, this will tell us how theta
144 // changes as w varies. By the IFT again:
145 // dg/dw = (-df/dTheta)^{-1} * df/dw
146 // The first term (df/dTheta) is just the Hessian, but as is typical we will
147 // approximate it using the Gauss-Newton approximation.
148 // df/dw is how the gradient changes as a function of w,
149 // d/dw(dE/dTheta) = d^2E/(dTheta dw)
150 // This is going to be a big, very sparse matrix; typical values are like n ~
151 // 50 for the pose vector and m ~ 200 is the number of constraints, and it's
152 // likely that most constraints only affect a few of the pose components.
153 // So then we'd need to multiply that by the inverse Hessian.
154 //
155 // However, at this point it's worth remembering that we actually want to
156 // compute
157 // dLoss/dTheta * dTheta/dw
158 // for some dLoss_dTheta that was passed into this function (this is due to
159 // reverse-mode auto-differentiation). Thus, we can write
160 // dLoss/dTheta * (-df/dTheta)^{-1} * df/dw
161 // We can group the multiplications,
162 // (dLoss/dTheta * (-df/dTheta)^{-1}) * df/dw
163 // This first part is just (Hess^{-1} * dLoss/dTheta^T)^T. The result is a
164 // n-dimensional vector, and we can then multiply this into the df/dw
165 // directly. Thus no need to store any big sparse matrices.
166 size_t actualRows = 0;
167 solverFunction.getJacobian(modelParameters.v, jacobian_full, residual_full, actualRows);
168
169 const Eigen::MatrixX<T> jacobian_subset =
170 extractSubsetMat<T>(jacobian_full, subsetIndexToFullParamIndex);
171 const Eigen::VectorX<T> gradient_subset = 2.0f * jacobian_subset.transpose() * residual_full;
172 if (gradientRmse != nullptr) {
173 *gradientRmse = std::sqrt(gradient_subset.squaredNorm() / gradient_subset.size());
174 }
175
176 const Eigen::VectorX<T> Jinv_times_dLoss_dModelParams_subset = hessianInverseTimes<T>(
177 jacobian_subset, extractSubsetVec<T>(dLoss_dModelParams, subsetIndexToFullParamIndex));
178
179 const Eigen::VectorX<T> Jinv_times_dLoss_dModelParams_full = subsetVecToFullVec<T>(
180 Jinv_times_dLoss_dModelParams_subset, subsetIndexToFullParamIndex, nFullParams);
181
182 MT_LOGD("Jinv_times_dLoss_dModelParams = {}", Jinv_times_dLoss_dModelParams_full.transpose());
183
184 const JointParametersT<T> jointParameters = parameterTransform.apply(modelParameters);
185 SkeletonStateT<T> skelState(jointParameters, skeleton);
186 const auto& errorFunctions = solverFunction.getErrorFunctions();
187 for (size_t iErr = 0; iErr < errorFunctions.size(); ++iErr) {
188 const auto errf = errorFunctions[iErr];
189 Eigen::VectorX<T> dGrad_dWeight =
190 Eigen::VectorX<T>::Zero(parameterTransform.numAllModelParameters());
191 errf->getGradient(modelParameters, skelState, dGrad_dWeight);
192 if (errf->getWeight() != 0) {
193 dGrad_dWeight /= errf->getWeight();
194 }
195 MT_LOGD("dGrad_dWeight = {}", dGrad_dWeight.transpose());
196
197 /*
198 {
199 const auto w_init = errf->getWeight();
200
201 const float eps = 1e-3f;
202 Eigen::VectorX<T> grad_init =
203 Eigen::VectorX<T>::Zero(parameterTransform.numAllModelParameters());
204 errf->getGradient(modelParameters, skelState, grad_init);
205
206 errf->setWeight(w_init + eps);
207 Eigen::VectorX<T> grad_plus =
208 Eigen::VectorX<T>::Zero(parameterTransform.numAllModelParameters());
209 errf->getGradient(modelParameters, skelState, grad_plus);
210 errf->setWeight(w_init);
211
212 Eigen::VectorX<T> diff = (grad_plus - grad_init) / eps;
213 MT_LOGI("dGrad_dWeight (est) = {}", diff.transpose());
214 }
215 */
216
217 result[iErr].errorFunction = errf;
218 result[iErr].gradWeight = -dGrad_dWeight.dot(Jinv_times_dLoss_dModelParams_full);
219
220 auto differentiableErr =
221 std::dynamic_pointer_cast<FullyDifferentiableSkeletonErrorFunctionT<T>>(errf);
222 if (differentiableErr != nullptr) {
223 for (const auto& inputName : differentiableErr->inputs()) {
224 result[iErr].gradInputs.emplace(
225 inputName,
226 differentiableErr->d_gradient_d_input_dot(
227 inputName, modelParameters, skelState, -Jinv_times_dLoss_dModelParams_full));
228 }
229 }
230 }
231
232 return result;
233}
