17std::vector<Eigen::Vector3i> triangulate(
18 gsl::span<const uint32_t> indices,
19 gsl::span<const uint32_t> offsets) {
20 if (indices.empty()) {
21 return {};
22 }
23
24 const auto nFaces = offsets.size() - 1;
25
26 std::vector<Eigen::Vector3i> result;
27 result.reserve(2 * nFaces);
28
29 for (size_t iFace = 0; iFace < nFaces; ++iFace) {
30 const auto faceStart = offsets[iFace];
31 const auto faceEnd = offsets[iFace + 1];
32 const auto nv = faceEnd - faceStart;
33 MT_THROW_IF(nv < 3, "Invalid face with {} indices; expected at least 3.", nv);
34
35 for (size_t j = 1; j < (nv - 1); ++j) {
36 result.emplace_back(
37 indices[faceStart + 0], indices[faceStart + j], indices[faceStart + j + 1]);
38 }
39 }
40
41 return result;
42}

49std::string PolygonData::errorMessage(const size_t numVertices) const {
50 if (offsets.empty()) {
51 return "Empty offsets array";
52 }
53
54 if (offsets.front() != 0) {
55 return "Invalid first offset (expected 0)";
56 }
57
58 if (!textureIndices.empty() && textureIndices.size() != indices.size()) {
59 return "Mismatch between indices and textureIndices";
60 }
61
62 const auto nPoly = numPolygons();
63
64 for (size_t iFace = 0; iFace < nPoly; ++iFace) {
65 const auto faceBegin = offsets[iFace];
66 const auto faceEnd = offsets[iFace + 1];
67
68 if (faceBegin >= faceEnd) {
69 return "Non-monotonic offsets array";
70 }
71
72 if (faceEnd > indices.size()) {
73 return "Too-large offset";
74 }
75
76 const auto nv = faceEnd - faceBegin;
77 if (nv < 3) {
78 return "Too few vertices in face";
79 }
80 }
81
82 for (const auto& i : indices) {
83 if (i >= numVertices) {
84 return "Invalid too-large vertex";
85 }
86 }
87
88 if (offsets.back() != indices.size()) {
89 return "Unused trailing vertices";
90 }
91
92 return {};
93}
