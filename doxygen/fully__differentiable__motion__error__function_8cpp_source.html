44template <typename T>

46 const std::string& name,

48 if (name == kTargetParameters) {
49 const auto& targetParameters = this->getTargetParameters();
50 if (targetParameters.size() != this->parameterTransform_.numAllModelParameters()) {
51 value = Eigen::VectorX<T>::Zero(this->parameterTransform_.numAllModelParameters());
52 } else {
53 value = targetParameters.v;
54 }
55 } else if (name == kTargetWeights) {
56 const auto& targetWeights = this->getTargetWeights();
57 if (targetWeights.size() != this->parameterTransform_.numAllModelParameters()) {
58 value = Eigen::VectorX<T>::Zero(this->parameterTransform_.numAllModelParameters());
59 } else {
60 value = targetWeights;
61 }
62 } else {
63 MT_THROW("Unknown input to FullyDifferentiableMotionErrorFunctionT: {}", name);
64 }

80template <typename T>
81Eigen::VectorX<T> FullyDifferentiableMotionErrorFunctionT<T>::d_gradient_d_input_dot(
82 const std::string& inputName,
83 const ModelParametersT<T>& params,
84 const SkeletonStateT<T>& /* state */,
85 Eigen::Ref<const Eigen::VectorX<T>> inputVec) {
86 // The error function is just
87 // \sum_i (w_i * (param_i - targetParam_i))^2
88 // The gradient is then
89 // \sum_i 2 * w_i^2 * (param_i - targetParam_i)
90 //
91 // The two input gradients are thus:
92 // dGrad_dw_i = 4 * w_i * (param_i - targetParam_i)
93 // dGrad_dtargetParam_i = -2 * w_i^2;
94 const auto& targetParameters = this->getTargetParameters();
95 const auto& targetWeights = this->getTargetWeights();
96
97 if (targetWeights.size() != this->parameterTransform_.numAllModelParameters() ||
98 targetParameters.size() != this->parameterTransform_.numAllModelParameters()) {
99 return Eigen::VectorX<T>::Zero(this->parameterTransform_.numAllModelParameters());
100 }
101
102 if (inputName == kTargetParameters) {
103 return T(-2) * this->weight_ * ModelParametersErrorFunctionT<T>::kMotionWeight *
104 targetWeights.array().square() * inputVec.array();
105 } else if (inputName == kTargetWeights) {
106 return T(4) * this->weight_ * ModelParametersErrorFunctionT<T>::kMotionWeight *
107 targetWeights.array() * (params.v - targetParameters.v).array() * inputVec.array();
108 } else {
109 MT_THROW("Unknown input to FullyDifferentiableMotionErrorFunctionT: {}", inputName);
110 }
111}
