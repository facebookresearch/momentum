23Skeleton scale(const Skeleton& skel, float scale) {
24 Skeleton result = skel;
25 for (auto& j : result.joints) {
26 j.translationOffset *= scale;
27 }
28 return result;
29}

31std::unique_ptr<Mesh> scale(const std::unique_ptr<Mesh>& mesh, float scale) {
32 if (!mesh) {
33 return {};
34 }
35
36 auto result = std::make_unique<Mesh>(*mesh);
37 for (auto& v : result->vertices) {
38 v *= scale;
39 }
40 return result;

52ParameterLimits scale(const ParameterLimits& limits, float scale) {
53 ParameterLimits result = limits;
54 for (auto& limit : result) {
55 if (limit.type == LimitType::Ellipsoid) {
56 limit.data.ellipsoid.ellipsoid.translation() *= scale;
57 limit.data.ellipsoid.ellipsoidInv.translation() *= scale;
58 limit.data.ellipsoid.offset *= scale;
59 }
60 }
61
62 return result;
63}

65std::unique_ptr<CollisionGeometry> scale(
66 const std::unique_ptr<CollisionGeometry>& geom,
67 float scale) {
68 if (!geom) {
69 return {};
70 }
71
72 auto result = std::make_unique<CollisionGeometry>(*geom);
73 for (auto& capsule : (*result)) {
74 capsule.transformation.translation *= scale;
75 capsule.radius *= scale;
76 capsule.length *= scale;
77 }
78
79 return result;
80}

93template <typename T>
94std::vector<T> mapParents(const std::vector<T>& objects, const std::vector<size_t>& jointMapping) {
95 std::vector<T> result;
96 result.reserve(objects.size());
97 for (auto o : objects) {
98 MT_CHECK(
99 o.parent < jointMapping.size(),
100 "Parent {} exceeds joint mapping size {}",
101 o.parent,
102 jointMapping.size());
103 o.parent = jointMapping[o.parent];
104 if (o.parent != kInvalidIndex) {
105 result.push_back(o);
106 }
107 }
108 return result;
109}

111ParameterLimits mapParameterLimits(
112 const ParameterLimits& limits,
113 const std::vector<size_t>& jointMapping,
114 const std::vector<size_t>& parameterMapping) {
115 ParameterLimits result;
116 result.reserve(limits.size());
117 for (auto l : limits) {
118 switch (l.type) {
119 case LimitType::MinMax:
120 l.data.minMax.parameterIndex = parameterMapping[l.data.minMax.parameterIndex];
121 if (l.data.minMax.parameterIndex != kInvalidIndex) {
122 result.push_back(l);
123 }
124 break;
125 case LimitType::MinMaxJoint:
126 case LimitType::MinMaxJointPassive:
127 l.data.minMaxJoint.jointIndex = jointMapping[l.data.minMaxJoint.jointIndex];
128 if (l.data.minMaxJoint.jointIndex != kInvalidIndex) {
129 result.push_back(l);
130 }
131 break;
132 case LimitType::Linear:
133 l.data.linear.referenceIndex = parameterMapping[l.data.linear.referenceIndex];
134 l.data.linear.targetIndex = parameterMapping[l.data.linear.targetIndex];
135 if (l.data.linear.referenceIndex != kInvalidIndex &&
136 l.data.linear.targetIndex != kInvalidIndex) {
137 result.push_back(l);
138 }
139 break;
140 case LimitType::LinearJoint:
141 l.data.linearJoint.referenceJointIndex =
142 parameterMapping[l.data.linearJoint.referenceJointIndex];
143 l.data.linearJoint.targetJointIndex = parameterMapping[l.data.linearJoint.targetJointIndex];
144 if (l.data.linearJoint.referenceJointIndex != kInvalidIndex &&
145 l.data.linearJoint.targetJointIndex != kInvalidIndex) {
146 result.push_back(l);
147 }
148 break;
149 case LimitType::HalfPlane:
150 l.data.halfPlane.param1 = parameterMapping[l.data.halfPlane.param1];
151 l.data.halfPlane.param2 = parameterMapping[l.data.halfPlane.param2];
152 if (l.data.halfPlane.param1 != kInvalidIndex && l.data.halfPlane.param2 != kInvalidIndex) {
153 result.push_back(l);
154 }
155 break;
156 case LimitType::Ellipsoid:
157 l.data.ellipsoid.parent = jointMapping[l.data.ellipsoid.parent];
158 l.data.ellipsoid.ellipsoidParent = jointMapping[l.data.ellipsoid.ellipsoidParent];
159 if (l.data.ellipsoid.parent != kInvalidIndex &&
160 l.data.ellipsoid.ellipsoidParent != kInvalidIndex) {
161 result.push_back(l);
162 }
163 break;
164 case LimitType::LimitTypeCount: {
165 // nothing to do here.
166 break;
167 }
168 }
169 }
170 return result;
171}

173SkinWeights mapSkinWeights(SkinWeights weights, const std::vector<size_t>& jointMapping) {
174 SkinWeights result(weights);
175
176 for (Eigen::Index iRow = 0; iRow < weights.index.rows(); ++iRow) {
177 for (Eigen::Index jCol = 0; jCol < weights.index.cols(); ++jCol) {
178 if (weights.weight(iRow, jCol) == 0) {
179 continue;
180 }
181
182 const auto mapped = jointMapping[result.index(iRow, jCol)];
183 MT_CHECK(mapped != kInvalidIndex);
184 result.index(iRow, jCol) = (int)mapped;
185 }
186 }
187
188 return result;
189}

191template <typename T>
192std::vector<T> mergeVectors(const std::vector<T>& left, const std::vector<T>& right) {
193 std::vector<T> result;
194 result.reserve(left.size() + right.size());
195 std::copy(std::begin(left), std::end(left), std::back_inserter(result));
196 std::copy(std::begin(right), std::end(right), std::back_inserter(result));
197 return result;
198}

200std::vector<Eigen::Triplet<float>> toTriplets(const SparseRowMatrixf& mat) {
201 std::vector<Eigen::Triplet<float>> result;
202 for (int k = 0; k < mat.outerSize(); ++k) {
203 for (SparseRowMatrixf::InnerIterator it(mat, k); it; ++it) {
204 result.emplace_back(it.row(), it.col(), it.value());
205 }
206 }
207 return result;
208}

210std::vector<size_t> addMappedParameters(
211 const ParameterTransform& paramTransformOrig,
212 const std::vector<size_t>& jointMapping,
213 ParameterTransform& paramTransformResult) {
214 std::vector<bool> validParamsOrig(paramTransformOrig.numAllModelParameters());
215
216 // Pass 1: figure out which parameters we're going to keep:
217 for (Eigen::Index kJointParam = 0; kJointParam < paramTransformOrig.transform.outerSize();
218 ++kJointParam) {
219 const Eigen::Index jointIndex = kJointParam / kParametersPerJoint;
220 const size_t mappedJointIndex = jointMapping[jointIndex];
221 if (mappedJointIndex == kInvalidIndex) {
222 continue;
223 }
224
225 // Invalid joint index, so any model parameters are invalid too:
226 for (SparseRowMatrixf::InnerIterator it(paramTransformOrig.transform, kJointParam); it; ++it) {
227 validParamsOrig[it.col()] = true;
228 }
229 }
230
231 std::vector<size_t> origParamToNewParam(
232 static_cast<size_t>(paramTransformOrig.numAllModelParameters()), kInvalidIndex);
233 std::unordered_set<std::string> existingParams(
234 paramTransformResult.name.begin(), paramTransformResult.name.end());
235 for (Eigen::Index iParamOld = 0; iParamOld < paramTransformOrig.numAllModelParameters();
236 ++iParamOld) {
237 if (!validParamsOrig[iParamOld]) {
238 continue;
239 }
240
241 MT_THROW_IF(
242 existingParams.find(paramTransformOrig.name[iParamOld]) != existingParams.end(),
243 "Duplicate parameter {} found while merging parameter transforms.",
244 paramTransformOrig.name[iParamOld]);
245
246 origParamToNewParam[iParamOld] = paramTransformResult.name.size();
247 paramTransformResult.name.push_back(paramTransformOrig.name[iParamOld]);
248 }
249
250 std::vector<Eigen::Triplet<float>> tripletsNew = toTriplets(paramTransformResult.transform);
251 for (Eigen::Index kJointParam = 0; kJointParam < paramTransformOrig.transform.outerSize();
252 ++kJointParam) {
253 const Eigen::Index jointIndex = kJointParam / kParametersPerJoint;
254 const Eigen::Index offset = kJointParam % kParametersPerJoint;
255 const size_t mappedJointIndex = jointMapping[jointIndex];
256 if (mappedJointIndex == kInvalidIndex) {
257 continue;
258 }
259
260 const auto mappedJointParam = mappedJointIndex * kParametersPerJoint + offset;
261
262 for (SparseRowMatrixf::InnerIterator it(paramTransformOrig.transform, kJointParam); it; ++it) {
263 const auto iOldParam = it.col();
264 const auto iNewParam = origParamToNewParam[iOldParam];
265
266 if (iNewParam != SIZE_MAX) {
267 tripletsNew.emplace_back((int)mappedJointParam, (int)iNewParam, it.value());
268 }
269 }
270 }
271 paramTransformResult.transform.resize(
272 paramTransformResult.transform.rows(), paramTransformResult.name.size());
273 paramTransformResult.transform.setFromTriplets(tripletsNew.begin(), tripletsNew.end());
274
275 // Update the parameter sets:
276 for (const auto& paramSetOrig : paramTransformOrig.parameterSets) {
277 for (Eigen::Index iOrigParam = 0; iOrigParam < paramTransformOrig.numAllModelParameters();
278 ++iOrigParam) {
279 if (!paramSetOrig.second.test(iOrigParam)) {
280 continue;
281 }
282
283 const auto iNewParam = origParamToNewParam[iOrigParam];
284 if (iNewParam != kInvalidIndex) {
285 paramTransformResult.parameterSets[paramSetOrig.first].set(iNewParam);
286 }
287 }
288 }
289
290 return origParamToNewParam;
291}

293LocatorList removeDuplicateLocators(LocatorList locators, const LocatorList& toRemove) {
294 std::unordered_set<std::string> toRemoveNames;
295 for (const auto& l : toRemove) {
296 toRemoveNames.insert(l.name);
297 }
298
299 locators.erase(
300 std::remove_if(
301 locators.begin(),
302 locators.end(),

304 locators.end());
305

329Skeleton transformSkeleton(const Skeleton& skeleton, const Eigen::Affine3f& xform) {
330 const Eigen::Vector3f singularValues = xform.linear().jacobiSvd().singularValues();
331 for (Eigen::Index i = 0; i < 3; ++i) {
332 MT_CHECK(
333 singularValues(i) > 0.99 && singularValues(i) < 1.01,
334 "Transform should not include scale or shear.");
335 }
336
337 const Eigen::Quaternionf rotation(xform.linear());
338 const Eigen::Vector3f translation(xform.translation());
339
340 Skeleton result(skeleton);
341 MT_CHECK(!result.joints.empty());
342 MT_CHECK(!skeleton.joints.empty());
343 if (!result.joints.empty() && !skeleton.joints.empty()) {
344 result.joints.front().preRotation = rotation * skeleton.joints.front().preRotation;
345 result.joints.front().translationOffset =
346 rotation * skeleton.joints.front().translationOffset + translation;
347 }
348 return result;
349}

351BlendShape_const_p transformBlendShape(
352 const BlendShape_const_p& blendShape,
353 const Eigen::Affine3f& xform) {
354 if (!blendShape) {
355 return nullptr;
356 }
357
358 const Eigen::Vector3f singularValues = xform.linear().jacobiSvd().singularValues();
359 for (Eigen::Index i = 0; i < 3; ++i) {
360 MT_CHECK(
361 singularValues(i) > 0.99 && singularValues(i) < 1.01,
362 "Transform should not include scale or shear.");
363 }
364
365 const Eigen::Quaternionf rotation(xform.linear());
366 const Eigen::Vector3f translation(xform.translation());
367
368 momentum::BlendShape_p blendShapeTransformed =
369 std::make_unique<momentum::BlendShape>(*blendShape);
370
371 // apply the full transformation to the blendshape basis
372 std::vector<Vector3f> baseShape = blendShapeTransformed->getBaseShape();
373 for (auto& v : baseShape) {
374 v = xform * v;
375 }
376 blendShapeTransformed->setBaseShape(baseShape);
377
378 // apply only the rotation to the shape vectors
379 Eigen::MatrixXf shapeVectors = blendShapeTransformed->getShapeVectors();
380 // Apply rotation to each 3D displacement within each shape vector column
381 for (Eigen::Index iShape = 0; iShape < shapeVectors.cols(); ++iShape) {
382 for (Eigen::Index iVertex = 0; iVertex < shapeVectors.rows() / 3; ++iVertex) {
383 Eigen::Vector3f displacement = shapeVectors.block<3, 1>(iVertex * 3, iShape);
384 shapeVectors.block<3, 1>(iVertex * 3, iShape) = rotation * displacement;
385 }
386 }
387 blendShapeTransformed->setShapeVectors(shapeVectors);
388
389 return blendShapeTransformed;
390}

392std::unique_ptr<Mesh> transformMesh(const std::unique_ptr<Mesh>& mesh, const Eigen::Affine3f& xf) {
393 if (!mesh) {
394 return {};
395 }
396
397 auto result = std::make_unique<Mesh>(*mesh);
398 for (auto& v : result->vertices) {
399 v = xf * v;
400 }
401
402 for (auto& v : result->normals) {
403 v = xf.linear() * v;
404 }
405
406 return result;
407}

409TransformationList transformInverseBindPose(
410 const TransformationList& inverseBindPose,
411 const Eigen::Affine3f& xf) {
412 TransformationList result;
413 result.reserve(inverseBindPose.size());
414 const Eigen::Affine3f xfInv = xf.inverse();
415 for (const auto& m : inverseBindPose) {
416 result.push_back(m * xfInv);
417 }
418 return result;

439Character replaceSkeletonHierarchy(
440 const Character& srcCharacter,
441 const Character& tgtCharacter,
442 const std::string& srcRootJoint,
443 const std::string& tgtRootJoint) {
444 const Skeleton& srcSkeleton = srcCharacter.skeleton;
445 const Skeleton& tgtSkeleton = tgtCharacter.skeleton;
446
447 MT_THROW_IF(srcSkeleton.joints.empty(), "Trying to reparent empty skeleton.");
448
449 const auto tgtRoot = tgtSkeleton.getJointIdByName(tgtRootJoint);
450 MT_THROW_IF(
451 tgtRoot == kInvalidIndex,
452 "Unable to re-root skeleton, target root joint {} not found.",
453 tgtRootJoint);
454
455 const auto srcRoot = srcSkeleton.getJointIdByName(srcRootJoint);
456 MT_THROW_IF(
457 srcRoot == kInvalidIndex,
458 "Unable to re-root skeleton, source root joint {} not found.",
459 srcRootJoint);
460
461 Skeleton combinedSkeleton;
462
463 // Make sure we don't insert duplicates:
464 std::unordered_map<std::string, size_t> combinedSkeletonJointMapping;
465
466 auto addJoint =
467 [&](const Skeleton& skeleton, size_t jointIndex, std::vector<size_t>& jointMap) -> size_t {
468 const Joint& origJoint = skeleton.joints[jointIndex];
469 Joint combinedJoint = origJoint;
470 MT_THROW_IF(
471 combinedSkeletonJointMapping.find(combinedJoint.name) != combinedSkeletonJointMapping.end(),
472 "Duplicate joint '{}' found while reparenting.",
473 origJoint.name);
474 const size_t combinedIndex = combinedSkeleton.joints.size();
475 jointMap[jointIndex] = combinedIndex;
476 combinedSkeletonJointMapping.insert({combinedJoint.name, combinedIndex});
477
478 // Remap the parent:
479 if (origJoint.parent != kInvalidIndex) {
480 auto itr = combinedSkeletonJointMapping.find(skeleton.joints[origJoint.parent].name);
481 MT_CHECK(itr != combinedSkeletonJointMapping.end());
482 combinedJoint.parent = itr->second;
483 }
484
485 combinedSkeleton.joints.push_back(combinedJoint);
486 return combinedIndex;
487 };
488
489 std::vector<size_t> srcToCombinedJoints(srcCharacter.skeleton.joints.size(), kInvalidIndex);
490 std::vector<size_t> tgtToCombinedJoints(tgtCharacter.skeleton.joints.size(), kInvalidIndex);
491
492 for (size_t iTgtJoint = 0; iTgtJoint < tgtSkeleton.joints.size(); ++iTgtJoint) {
493 // Add the replacement joints just after the new parent rather than at the very end; this will
494 // improve coherency a bit.
495 if (iTgtJoint == tgtRoot) {
496 addJoint(tgtSkeleton, iTgtJoint, tgtToCombinedJoints);
497
498 // Copy the joints from the source skeleton over in place of the target root:
499 for (size_t kSrcJoint = srcRoot + 1; kSrcJoint < srcSkeleton.joints.size(); ++kSrcJoint) {
500 if (srcSkeleton.isAncestor(kSrcJoint, srcRoot)) {
501 addJoint(srcSkeleton, kSrcJoint, srcToCombinedJoints);
502 }
503 }
504 } else if (!tgtSkeleton.isAncestor(iTgtJoint, tgtRoot)) {
505 addJoint(tgtSkeleton, iTgtJoint, tgtToCombinedJoints);
506 }
507 }
508
509 // For locators, use the parent joint to decide whether to use the src locator or
510 // the target locator.
511 // For duplicate locators that are on both the src and tgt skeletons (but parented to different
512 // bones), we will always take the src locator. This is because we trust the locators on the
513 // primary skeleton's hands more than the secondary skeleton.
514 const LocatorList combinedLocators = [&]() {
515 const LocatorList remappedSrcLocators =
516 mapParents<Locator>(srcCharacter.locators, srcToCombinedJoints);
517 const LocatorList remappedTgtLocators =
518 mapParents<Locator>(tgtCharacter.locators, tgtToCombinedJoints);
519 return mergeVectors(
520 removeDuplicateLocators(remappedTgtLocators, remappedSrcLocators), remappedSrcLocators);
521 }();
522 {
523 std::unordered_set<std::string> locatorNames;
524 for (const auto& l : combinedLocators) {
525 MT_LOGW_IF(locatorNames.count(l.name) != 0, "Duplicate locator: {}", l.name);
526 locatorNames.insert(l.name);
527 }
528 }
529
530 CollisionGeometry combinedCollisionGeometry;
531 if (tgtCharacter.collision) {
532 combinedCollisionGeometry = mergeVectors(
533 combinedCollisionGeometry,
534 mapParents<TaperedCapsule>(*tgtCharacter.collision, tgtToCombinedJoints));
535 }
536 if (srcCharacter.collision) {
537 combinedCollisionGeometry = mergeVectors(
538 combinedCollisionGeometry,
539 mapParents<TaperedCapsule>(*srcCharacter.collision, srcToCombinedJoints));
540 }
541
542 // Build the merged parameter transform:
543 ParameterTransform combinedParamTransform;
544 combinedParamTransform.offsets =
545 VectorXf::Zero(combinedSkeleton.joints.size() * kParametersPerJoint);
546 combinedParamTransform.transform.resize(combinedSkeleton.joints.size() * kParametersPerJoint, 0);
547 const std::vector<size_t> tgtToCombinedParameters = addMappedParameters(
548 tgtCharacter.parameterTransform, tgtToCombinedJoints, combinedParamTransform);
549 const std::vector<size_t> srcToCombinedParameters = addMappedParameters(
550 srcCharacter.parameterTransform, srcToCombinedJoints, combinedParamTransform);
551 combinedParamTransform.activeJointParams = combinedParamTransform.computeActiveJointParams();
552
553 const ParameterLimits combinedParameterLimits = mergeVectors(
554 mapParameterLimits(
555 tgtCharacter.parameterLimits, tgtToCombinedJoints, tgtToCombinedParameters),
556 mapParameterLimits(
557 srcCharacter.parameterLimits, srcToCombinedJoints, srcToCombinedParameters));
558
559 std::unique_ptr<SkinWeights> skinWeightsCombined;
560 {
561 // Mapping for every joint in the target skeleton to a joint in the combined skeleton.
562 // The goal here is something that we can use to map e.g. the skinning weights.
563 // Rules are:
564 // 1. if the target joint is in the combined skeleton, then map it directly.
565 // 2. otherwise, find a name with a matching joint in the source skeleton.
566 // 3. if not, then go up one level of the hierarchy and try again.
567 std::vector<size_t> tgtToCombinedWithParents(
568 tgtCharacter.skeleton.joints.size(), kInvalidIndex);
569 for (size_t iTgtJoint = 0; iTgtJoint < tgtSkeleton.joints.size(); ++iTgtJoint) {
570 size_t tgtParent = iTgtJoint;
571 while (tgtParent != kInvalidIndex) {
572 const auto itr = combinedSkeletonJointMapping.find(tgtSkeleton.joints[tgtParent].name);
573 if (itr != combinedSkeletonJointMapping.end()) {
574 tgtToCombinedWithParents[iTgtJoint] = itr->second;
575 break;
576 }
577 tgtParent = tgtSkeleton.joints[iTgtJoint].parent;
578 }
579 MT_CHECK(tgtParent != kInvalidIndex);
580 }
581
582 if (tgtCharacter.skinWeights) {
583 skinWeightsCombined = std::make_unique<SkinWeights>(
584 mapSkinWeights(*tgtCharacter.skinWeights, tgtToCombinedWithParents));
585 }
586 }
587
588 return {
589 combinedSkeleton,
590 combinedParamTransform,
591 combinedParameterLimits,
592 combinedLocators,
593 tgtCharacter.mesh.get(),
594 skinWeightsCombined.get(),
595 combinedCollisionGeometry.empty() ? nullptr : &combinedCollisionGeometry,
596 tgtCharacter.poseShapes.get(),
597 tgtCharacter.blendShape};
598}

600Character removeJoints(const Character& character, gsl::span<const size_t> jointsToRemove) {
601 std::vector<bool> toRemove(character.skeleton.joints.size(), false);
602 for (const auto& j : jointsToRemove) {
603 MT_THROW_IF(j >= character.skeleton.joints.size(), "Invalid joint found in removeJoints.");
604 toRemove[j] = true;
605 }
606
607 // Remove all joints parented under the target joints as well:
608 const auto& srcSkeleton = character.skeleton;
609 std::vector<size_t> srcToResultJoints(srcSkeleton.joints.size(), kInvalidIndex);
610 Skeleton resultSkeleton;
611 for (size_t iJoint = 0; iJoint < srcSkeleton.joints.size(); ++iJoint) {
612 bool shouldRemove = false;
613
614 size_t parent = iJoint;
615 while (parent != kInvalidIndex) {
616 if (toRemove[parent]) {
617 shouldRemove = true;
618 break;
619 }
620 if (parent < character.skeleton.joints.size()) {
621 parent = character.skeleton.joints[parent].parent;
622 } else {
623 break;
624 }
625 }
626
627 if (shouldRemove) {
628 continue;
629 }
630
631 srcToResultJoints[iJoint] = resultSkeleton.joints.size();
632 auto joint = srcSkeleton.joints[iJoint];
633 if (joint.parent != kInvalidIndex) {
634 joint.parent = srcToResultJoints[joint.parent];
635 }
636 resultSkeleton.joints.push_back(joint);
637 }
638
639 ParameterTransform resultParamTransform =
640 ParameterTransform::empty(resultSkeleton.joints.size() * kParametersPerJoint);
641 const std::vector<size_t> srcToResultParameters =
642 addMappedParameters(character.parameterTransform, srcToResultJoints, resultParamTransform);
643
644 std::unique_ptr<SkinWeights> resultSkinWeights;
645 {
646 // Mapping for every joint in the target skeleton to a joint in the combined skeleton.
647 std::vector<size_t> srcToResultJointsWithParents(
648 character.skeleton.joints.size(), kInvalidIndex);
649 for (size_t iSrcJoint = 0; iSrcJoint < srcSkeleton.joints.size(); ++iSrcJoint) {
650 size_t srcParent = iSrcJoint;
651 // Anything skinned to a deleted joint should get skinned to its parent instead:
652 while (srcParent != kInvalidIndex) {
653 srcToResultJointsWithParents[iSrcJoint] = srcToResultJoints[srcParent];
654 if (srcToResultJointsWithParents[iSrcJoint] != kInvalidIndex) {
655 break;
656 }
657 srcParent = srcSkeleton.joints[srcParent].parent;
658 }
659 }
660
661 if (character.skinWeights) {

691MatrixXf mapMotionToCharacter(
692 const MotionParameters& inputMotion,
693 const Character& targetCharacter) {
694 // re-arrange poses according to the character names
695 const auto& parameterNames = std::get<0>(inputMotion);
696 const auto& motion = std::get<1>(inputMotion);
697
698 MT_CHECK(
699 static_cast<int>(parameterNames.size()) == motion.rows(),
700 "The number of parameter names {} does not match the number of rows {} in the motion.",
701 parameterNames.size(),
702 motion.rows());
703
704 MatrixXf result =
705 MatrixXf::Zero(targetCharacter.parameterTransform.numAllModelParameters(), motion.cols());
706 for (size_t i = 0; i < parameterNames.size(); i++) {
707 const auto index = targetCharacter.parameterTransform.getParameterIdByName(parameterNames[i]);
708 if (index != kInvalidIndex) {
709 result.row(index) = motion.row(i);
710 } else {
711 MT_LOGW("Model parameter {} not found in source motion", parameterNames[i]);
712 }
713 }
714
715 return result;
716}

718VectorXf mapIdentityToCharacter(
719 const IdentityParameters& inputIdentity,
720 const Character& targetCharacter) {
721 // re-arrange offsets according to the character names
722 const auto& jointNames = std::get<0>(inputIdentity);
723 const auto& identity = std::get<1>(inputIdentity);
724
725 MT_CHECK(
726 static_cast<int>(jointNames.size() * kParametersPerJoint) == identity.size(),
727 "The number of joint parameters {} does not match the identity vector size {}.",
728 jointNames.size() * kParametersPerJoint,
729 identity.size());
730
731 VectorXf result = VectorXf::Zero(targetCharacter.skeleton.joints.size() * kParametersPerJoint);
732 for (size_t i = 0; i < jointNames.size(); i++) {
733 const auto index = targetCharacter.skeleton.getJointIdByName(jointNames[i]);
734 if (index != kInvalidIndex) {
735 result.template middleRows<kParametersPerJoint>(index * kParametersPerJoint).noalias() =
736 identity.template middleRows<kParametersPerJoint>(i * kParametersPerJoint);
737 } else {
738 MT_LOGW("Joint {} not found in source identity", jointNames[i]);
739 }
740 }
741
742 return result;
743}
