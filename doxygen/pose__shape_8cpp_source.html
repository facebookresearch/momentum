15std::vector<Vector3f> PoseShape::compute(const SkeletonState& state) const {
16 MT_CHECK(
17 baseShape.size() == shapeVectors.rows(),
18 "{} is not {}",
19 baseShape.size(),
20 shapeVectors.rows());
21
22 // compute coefficients
23 VectorXf coefficients(shapeVectors.cols());
24
25 // calculate base transform
26 const Quaternionf base = (baseJoint < state.jointState.size())
27 ? baseRot * state.jointState.at(baseJoint).rotation().inverse()
28 : baseRot;
29
30 // set up pose shape coefficients
31 for (size_t i = 0; i < jointMap.size(); i++) {
32 const auto& jid = jointMap[i];
33 if (jid < state.jointState.size()) {
34 coefficients.segment<4>(i * 4) = (base * state.jointState.at(jid).rotation()).coeffs();
35 }
36 }
37
38 // use the coefficients to calculate the new base shape
39 std::vector<Vector3f> output(baseShape.size() / 3);
40 Map<VectorXf> outputVec(&output[0][0], output.size() * 3);
41
42 outputVec = baseShape + shapeVectors * coefficients;
43
44 return output;
45}
