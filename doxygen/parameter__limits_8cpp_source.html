39JointParameters applyPassiveJointParameterLimits(
40 const ParameterLimits& limits,
41 const JointParameters& jointParams) {
42 JointParameters res = jointParams;
43 for (const auto& limit : limits) {
44 if (limit.type != MinMaxJointPassive) {
45 continue;
46 }
47
48 const auto& data = limit.data.minMaxJoint;
49 const size_t parameterIndex = data.jointIndex * kParametersPerJoint + data.jointParameter;
50 MT_CHECK(
51 parameterIndex <= gsl::narrow<size_t>(jointParams.size()),
52 "{} vs {}",
53 parameterIndex,
54 jointParams.size());
55 if (res(parameterIndex) < data.limits[0]) {
56 res(parameterIndex) = data.limits[0];
57 }
58 if (res(parameterIndex) > data.limits[1]) {
59 res(parameterIndex) = data.limits[1];
60 }
61 }
62 return res;
63}

65ParameterLimits getPoseConstraintParameterLimits(
66 const std::string& name,
67 const ParameterTransform& pt,
68 const float weight) {
69 ParameterLimits res;
70
71 if (pt.poseConstraints.count(name) > 0) {
72 for (const auto& pc : pt.poseConstraints.at(name).parameterIdValue) {
73 ParameterLimit p;
74 p.type = MinMax;
75 p.data.minMax.parameterIndex = pc.first;
76 p.data.minMax.limits = Vector2f::Constant(pc.second);
77 p.weight = weight;
78 res.emplace_back(p);
79 }
80 }
81
82 return res;
83}

104bool isInRange(const LimitLinear& limit, float value) {
105 // Default initialized limits have all zeros in their members due to the std::fill_n above
106 // but we want in this case to have the limit apply across the entire range of values:
107 if (limit.rangeMin == 0 && limit.rangeMax == 0) {
108 return true;
109 }
110
111 return value >= limit.rangeMin && value < limit.rangeMax;
