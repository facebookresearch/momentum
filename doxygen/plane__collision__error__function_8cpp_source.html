73template <typename T>
74T PlaneCollisionErrorFunctionT<T>::primitiveParentSpaceBound(const size_t primitiveIndex) const {
75 const auto& primitive = collisionGeometry_[primitiveIndex];
76 const TransformT<T> primitiveTransform = primitive.transformation.cast<T>();
77 const Vector3<T>& origin = primitiveTransform.translation;
78
79 if (primitive.type == CollisionPrimitiveType::TaperedCapsule) {
80 const Vector3<T> endpoint =
81 origin + primitiveTransform.getAxisX() * primitiveTransform.scale * T(primitive.length);
82 const T radius = primitive.radius.cast<T>().cwiseAbs().maxCoeff();
83 return std::max(origin.norm(), endpoint.norm()) + radius;
84 }
85
86 if (primitive.type == CollisionPrimitiveType::Ellipsoid) {
87 return origin.norm() + primitive.ellipsoidRadii.cast<T>().cwiseAbs().maxCoeff();
88 }
89
90 if (primitive.type == CollisionPrimitiveType::Box) {
91 return origin.norm() + primitive.boxHalfExtents.cast<T>().cwiseAbs().norm();
92 }
93
94 MT_THROW(
95 "PlaneCollisionErrorFunction: unsupported collision primitive type {}",
96 static_cast<int>(primitive.type));
97}

99template <typename T>
100void PlaneCollisionErrorFunctionT<T>::updatePrimitiveList() {
101 primitiveIndices_.clear();
102 primitiveIndices_.reserve(collisionGeometry_.size());
103 primitiveParentSpaceBounds_.assign(collisionGeometry_.size(), T(0));
104 for (size_t i = 0; i < collisionGeometry_.size(); ++i) {
105 // This error function constrains character-attached primitives against a world plane.
106 if (collisionGeometry_[i].parent != kInvalidIndex) {
107 primitiveIndices_.push_back(i);
108 primitiveParentSpaceBounds_[i] = primitiveParentSpaceBound(i);
109 }
110 }
111 candidatePrimitiveIndices_.clear();
112 activeParentCollisions_.clear();
113}

115template <typename T>
116bool PlaneCollisionErrorFunctionT<T>::mayPrimitiveContactPlane(
117 const SkeletonStateT<T>& state,
118 const size_t primitiveIndex,
119 const T contactMargin) const {
120 const auto& primitive = collisionGeometry_[primitiveIndex];
121 const TransformT<T>& parentTransform = state.jointState[primitive.parent].transform;
122 const T parentSignedDistance = planeNormal_.dot(parentTransform.translation) - planeOffset_;
123 const T conservativeRadius =
124 std::abs(parentTransform.scale) * primitiveParentSpaceBounds_[primitiveIndex];
125 return parentSignedDistance - conservativeRadius <= contactMargin;
126}

147template <typename T>
148bool PlaneCollisionErrorFunctionT<T>::checkCollision(
149 size_t primitiveIndex,
150 PlaneCollisionResult& result) const {
151 const auto type = collisionState_.type[primitiveIndex];
152
153 // `result.position` is the deepest surface point toward the plane (the contact point), offset
154 // from the primitive center/endpoint by the support radius along `-planeNormal_`. Differentiating
155 // at the contact point, rather than the center, lets the joint-chain walk capture orientation and
156 // scale dependence through that world-space support point.
157 if (type == CollisionPrimitiveType::TaperedCapsule) {
158 const Vector3<T> position0 = collisionState_.origin[primitiveIndex];
159 const Vector3<T> position1 =
160 collisionState_.origin[primitiveIndex] + collisionState_.direction[primitiveIndex];
161 const T signedDistance0 = planeNormal_.dot(position0) - planeOffset_;
162 const T signedDistance1 = planeNormal_.dot(position1) - planeOffset_;
163 const T radius0 = collisionState_.radius[primitiveIndex][0];
164 const T radius1 = collisionState_.radius[primitiveIndex][1];
165 const T surfaceDistance0 = signedDistance0 - radius0;
166 const T surfaceDistance1 = signedDistance1 - radius1;
167
168 if (surfaceDistance0 <= surfaceDistance1) {
169 result.signedDistance = signedDistance0;
170 result.radius = radius0;
171 result.position = position0 - radius0 * planeNormal_;
172 } else {
173 result.signedDistance = signedDistance1;
174 result.radius = radius1;
175 result.position = position1 - radius1 * planeNormal_;
176 }
177 } else if (isCenteredPrimitive(type)) {
178 const Vector3<T>& center = collisionState_.origin[primitiveIndex];
179 const Quaternion<T>& orientation = collisionState_.orientation[primitiveIndex];
180 result.signedDistance = planeNormal_.dot(center) - planeOffset_;
181 result.radius = centeredPrimitiveRadiusAlongDirection(
182 type,
183 orientation,
184 collisionState_.ellipsoidRadii[primitiveIndex],
185 collisionState_.boxHalfExtents[primitiveIndex],
186 planeNormal_);
187
188 const Vector3<T> localNormal = orientation.inverse() * planeNormal_;
189 Vector3<T> supportOffset;
190 if (type == CollisionPrimitiveType::Ellipsoid) {
191 const Vector3<T>& radii = collisionState_.ellipsoidRadii[primitiveIndex];
192 const T denom = result.radius > Eps<T>(1e-8f, 1e-16) ? result.radius : T(1);
193 supportOffset = orientation * (radii.cwiseProduct(radii).cwiseProduct(localNormal)) / denom;
194 } else {
195 const Vector3<T>& halfExtents = collisionState_.boxHalfExtents[primitiveIndex];
196 const Vector3<T> localSign =
197 localNormal.unaryExpr([](const T value) { return value >= T(0) ? T(1) : T(-1); });
198 supportOffset = orientation * halfExtents.cwiseAbs().cwiseProduct(localSign);
199 }
200 result.position = center - supportOffset;
201 } else {
202 MT_THROW(
203 "PlaneCollisionErrorFunction: unsupported collision primitive type {}",
204 static_cast<int>(type));
205 }
206
207 result.overlap = result.radius - result.signedDistance;
208 return result.overlap > T(0);
209}

211template <typename T>
212void PlaneCollisionErrorFunctionT<T>::updateActiveParentCollisions() {
213 activeParentCollisions_.clear();
214 activeParentCollisions_.reserve(candidatePrimitiveIndices_.size());
215
216 for (const auto primitiveIndex : candidatePrimitiveIndices_) {
217 PlaneCollisionResult result;
218 if (!checkCollision(primitiveIndex, result)) {
219 continue;
220 }
221
222 const size_t parent = collisionGeometry_[primitiveIndex].parent;
223 auto existing =
224 std::ranges::find_if(activeParentCollisions_, [&](const ActivePlaneCollision& collision) {
225 return collisionGeometry_[collision.primitiveIndex].parent == parent;
226 });
227 if (existing == activeParentCollisions_.end()) {
228 activeParentCollisions_.push_back({primitiveIndex, result});
229 continue;
230 }
231
232 if (result.overlap > existing->result.overlap) {
233 *existing = {primitiveIndex, result};
234 }
235 }
236}

320template <typename T>
321void PlaneCollisionErrorFunctionT<T>::prepareContactPointQuery(
322 const std::span<const JointStateT<T>> states,
323 const T contactMargin) {
324 MT_THROW_IF(
325 !std::isfinite(contactMargin) || contactMargin < T(0),
326 "Plane collision contact margin must be finite and non-negative, got {}",
327 contactMargin);
328 for (const auto primitiveIndex : primitiveIndices_) {
329 const size_t parent = collisionGeometry_[primitiveIndex].parent;
330 MT_THROW_IF(
331 parent >= states.size(),
332 "Plane collision primitive {} parent joint {} is outside the skeleton state",
333 primitiveIndex,
334 parent);
335 }
336
337 SkeletonStateT<T> state;
338 state.jointState.assign(states.begin(), states.end());
339 updateCandidateCollisionState(state, contactMargin);
340}

342template <typename T>
343std::vector<Vector3<T>> PlaneCollisionErrorFunctionT<T>::getContactPoints(
344 const std::span<const JointStateT<T>> states,
345 const T contactMargin) {
346 prepareContactPointQuery(states, contactMargin);
347
348 std::vector<Vector3<T>> points;
349 for (const auto primitiveIndex : candidatePrimitiveIndices_) {
350 PlaneCollisionResult result;
351 static_cast<void>(checkCollision(primitiveIndex, result));
352 const T surfaceDistance = result.signedDistance - result.radius;
353 if (surfaceDistance <= contactMargin) {
354 points.push_back(result.position);
355 }
356 }

360template <typename T>
361std::vector<Vector3<T>> PlaneCollisionErrorFunctionT<T>::getContactPointsByParent(
362 const std::span<const JointStateT<T>> states,
363 const T contactMargin) {
364 prepareContactPointQuery(states, contactMargin);
365
366 struct ParentContact {
367 size_t parent = kInvalidIndex;
368 T surfaceDistance = T(0);
369 Vector3<T> position = Vector3<T>::Zero();
370 };
371
372 std::vector<ParentContact> parentContacts;
373 parentContacts.reserve(candidatePrimitiveIndices_.size());
374 for (const auto primitiveIndex : candidatePrimitiveIndices_) {
375 PlaneCollisionResult result;
376 static_cast<void>(checkCollision(primitiveIndex, result));
377 const T surfaceDistance = result.signedDistance - result.radius;
378 if (surfaceDistance > contactMargin) {
379 continue;
380 }
381
382 const size_t parent = collisionGeometry_[primitiveIndex].parent;
383 auto existing = std::ranges::find_if(parentContacts, [parent](const ParentContact& contact) {
384 return contact.parent == parent;
385 });
386 if (existing == parentContacts.end()) {
387 parentContacts.push_back({parent, surfaceDistance, result.position});
388 continue;
389 }
390
391 if (surfaceDistance < existing->surfaceDistance) {
392 *existing = {parent, surfaceDistance, result.position};
393 }
394 }
395
396 std::vector<Vector3<T>> points;
397 points.reserve(parentContacts.size());
398 for (const ParentContact& contact : parentContacts) {
399 points.push_back(contact.position);
400 }
401 return points;
402}

404template <typename T>
405double PlaneCollisionErrorFunctionT<T>::getError(
406 const ModelParametersT<T>& /* params */,
407 const SkeletonStateT<T>& state,
408 const MeshStateT<T>& /* meshState */) {
409 if (state.jointState.empty()) {
410 return 0.0;
411 }
412
413 MT_CHECK(
414 state.jointParameters.size() ==
415 gsl::narrow<Eigen::Index>(this->skeleton_.joints.size() * kParametersPerJoint));
416
417 updateCandidateCollisionState(state, T(0));
418 updateActiveParentCollisions();
419
420 double error = 0;
421 for (const ActivePlaneCollision& collision : activeParentCollisions_) {
422 error += sqr(collision.result.overlap) * kCollisionWeight * this->weight_;
423 }
424
425 MT_LOGD(
426 "PlaneCollisionError: {} active parents / {} candidates / {} primitives, error={:.8f}, weight={:.4f}",
427 activeParentCollisions_.size(),
428 candidatePrimitiveIndices_.size(),
429 primitiveIndices_.size(),
430 error,
431 this->weight_);
432
433 return error;
434}

436template <typename T>
437double PlaneCollisionErrorFunctionT<T>::getGradient(
438 const ModelParametersT<T>& /* params */,
439 const SkeletonStateT<T>& state,
440 const MeshStateT<T>& /* meshState */,
441 Ref<VectorX<T>> gradient) {
442 if (state.jointState.empty()) {
443 return 0.0;
444 }
445
446 MT_CHECK(
447 state.jointParameters.size() ==
448 gsl::narrow<Eigen::Index>(this->skeleton_.joints.size() * kParametersPerJoint));
449
450 updateCandidateCollisionState(state, T(0));
451 updateActiveParentCollisions();
452
453 double error = 0;
454 for (const ActivePlaneCollision& collision : activeParentCollisions_) {
455 const PlaneCollisionResult& result = collision.result;
456 error += sqr(result.overlap) * kCollisionWeight * this->weight_;
457
458 const T wgt = -T(2) * kCollisionWeight * this->weight_ * result.overlap;
459 accumulateGradientAlongChain(
460 state,
461 result.position,
462 planeNormal_,
463 wgt,
464 collisionGeometry_[collision.primitiveIndex].parent,
465 gradient,
466 /*scaleCorrection=*/T(0));
467 }
468
469 return error;
470}

472template <typename T>
473double PlaneCollisionErrorFunctionT<T>::getJacobian(
474 const ModelParametersT<T>& /* params */,
475 const SkeletonStateT<T>& state,
476 const MeshStateT<T>& /* meshState */,
477 Ref<MatrixX<T>> jacobian,
478 Ref<VectorX<T>> residual,
479 int& usedRows) {
480 MT_PROFILE_FUNCTION();
481 MT_CHECK(
482 jacobian.rows() >= gsl::narrow<Eigen::Index>(getJacobianSize()),
483 "Jacobian rows mismatch: Actual {}, Expected {}",
484 jacobian.rows(),
485 gsl::narrow<Eigen::Index>(getJacobianSize()));
486 MT_CHECK(
487 residual.rows() >= gsl::narrow<Eigen::Index>(getJacobianSize()),
488 "Residual rows mismatch: Actual {}, Expected {}",
489 residual.rows(),
490 gsl::narrow<Eigen::Index>(getJacobianSize()));
491
492 if (state.jointState.empty()) {
493 usedRows = 0;
494 return 0.0;
495 }
496
497 MT_CHECK(
498 state.jointParameters.size() ==
499 gsl::narrow<Eigen::Index>(this->skeleton_.joints.size() * kParametersPerJoint));
500
501 updateCandidateCollisionState(state, T(0));
502 updateActiveParentCollisions();
503
504 const T wgt = std::sqrt(kCollisionWeight * this->weight_);
505 int pos = 0;
506 double error = 0;
507
508 for (const ActivePlaneCollision& collision : activeParentCollisions_) {
509 const PlaneCollisionResult& result = collision.result;
510 error += sqr(result.overlap) * kCollisionWeight * this->weight_;
511
512 const int row = pos++;
513 accumulateJacobianAlongChain(
514 state,
515 result.position,
516 planeNormal_,
517 -wgt,
518 collisionGeometry_[collision.primitiveIndex].parent,
519 jacobian,
520 row,
521 /*scaleCorrection=*/T(0));
522 residual(row) = result.overlap * wgt;
523 }
524
525 usedRows = pos;
526 return error;
527}
