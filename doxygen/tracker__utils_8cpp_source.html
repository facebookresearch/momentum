26std::vector<std::vector<PositionData>> createConstraintData(
27 const gsl::span<const std::vector<Marker>> markerData,
28 const LocatorList& locators) {
29 std::vector<std::vector<PositionData>> results(markerData.size());
30
31 // map locator name to its index
32 std::map<std::string, size_t> locatorLookup;
33 for (size_t i = 0; i < locators.size(); i++) {
34 locatorLookup[locators[i].name] = i;
35 }
36
37 // create a list of position constraints per frame
38 for (size_t iFrame = 0; iFrame < markerData.size(); ++iFrame) {
39 const auto& markerList = markerData[iFrame];
40 for (const auto& jMarker : markerList) {
41 if (jMarker.occluded) {
42 continue;
43 }
44 auto query = locatorLookup.find(jMarker.name);
45 if (query == locatorLookup.end()) {
46 continue;
47 }
48 size_t locatorIdx = query->second;
49
50 results.at(iFrame).emplace_back(PositionData(
51 locators.at(locatorIdx).offset,
52 jMarker.pos.cast<float>(),
53 locators.at(locatorIdx).parent,
54 locators.at(locatorIdx).weight));
55 }
56 }
57
58 return results;
59}

96std::pair<Eigen::Vector<uint32_t, kMaxSkinJoints>, Eigen::Vector<float, kMaxSkinJoints>>
97averageTriangleSkinWeights(
98 const Character& character,
99 int triangleIndex,
100 const Eigen::Vector3f& barycentric) {
101 Eigen::VectorXf skinWeights = Eigen::VectorXf::Zero(character.skeleton.joints.size());
102
103 const auto& mesh = *character.mesh;
104 const auto& skin = *character.skinWeights;
105
106 // get the triangle
107 const auto& triangle = mesh.faces[triangleIndex];
108 for (int kTriVert = 0; kTriVert < 3; ++kTriVert) {
109 for (int kSkinWeight = 0; kSkinWeight < skin.index.cols(); ++kSkinWeight) {
110 const auto& skinIndex = skin.index(triangle(kTriVert), kSkinWeight);
111 const auto& skinWeight = skin.weight(triangle(kTriVert), kSkinWeight);
112 skinWeights[skinIndex] += skinWeight * barycentric[kTriVert];
113 }
114 }
115
116 std::vector<std::pair<float, uint32_t>> sortedWeights;
117 for (Eigen::Index i = 0; i < skinWeights.size(); ++i) {
118 if (skinWeights[i] > 0.0f) {
119 sortedWeights.emplace_back(skinWeights[i], static_cast<uint32_t>(i));
120 }
121 }
122 std::sort(sortedWeights.begin(), sortedWeights.end(), std::greater<>());
123
124 Eigen::Vector<float, kMaxSkinJoints> resultWeights = Eigen::Vector<float, kMaxSkinJoints>::Zero();
125 Eigen::Vector<uint32_t, kMaxSkinJoints> resultIndices =
126 Eigen::Vector<uint32_t, kMaxSkinJoints>::Zero();
127 for (size_t i = 0; i < kMaxSkinJoints; ++i) {
128 if (i < sortedWeights.size()) {
129 resultWeights[i] = sortedWeights[i].first;
130 resultIndices[i] = sortedWeights[i].second;
131 }
132 }
133
134 resultWeights /= resultWeights.sum();
135
136 return {resultIndices, resultWeights};
137}

150ClosestPointOnMeshResult closestPointOnMeshMatchingParent(
151 const momentum::Mesh& mesh,
152 const momentum::SkinWeights& skin,
153 const Eigen::Vector3f& p_world,
154 uint32_t parentIdx,
155 float cutoffWeight = 0.1) {
156 ClosestPointOnMeshResult result;
157 for (size_t iTri = 0; iTri < mesh.faces.size(); ++iTri) {
158 const auto& f = mesh.faces[iTri];
159
160 float skinWeight = 0;
161 for (int kTriVert = 0; kTriVert < 3; ++kTriVert) {
162 for (int kSkinWeight = 0; kSkinWeight < skin.index.cols(); ++kSkinWeight) {
163 if (skin.index(f(kTriVert), kSkinWeight) == parentIdx) {
164 skinWeight += skin.weight(f(kTriVert), kSkinWeight);
165 }
166 }
167 }
168 skinWeight /= 3.0f;
169
170 if (skinWeight < cutoffWeight) {
171 continue;
172 }
173
174 // Note: the return value from the does _not_ have anything to do with
175 // whether the projection is valid, it just indicates whether the projection is
176 // inside the triangle or not, so we can ignore it and just use the projected point directly.
177 Eigen::Vector3f p_tri;
178 Eigen::Vector3f bary;
179 axel::projectOnTriangle(
180 p_world, mesh.vertices[f(0)], mesh.vertices[f(1)], mesh.vertices[f(2)], p_tri, &bary);
181 const float dist = (p_tri - p_world).norm();
182 if (!result.valid || dist < result.distance) {
183 result = {true, iTri, bary, dist};
184 }
185 }
186
187 return result;
188}

190momentum::Character locatorsToSkinnedLocators(
191 const momentum::Character& sourceCharacter,
192 float maxDistance) {
193 if (!sourceCharacter.mesh || !sourceCharacter.skinWeights) {
194 return sourceCharacter;
195 }
196
197 const JointParameters restJointParams =
198 JointParameters::Zero(sourceCharacter.parameterTransform.numJointParameters());
199 const SkeletonState restState(restJointParams, sourceCharacter.skeleton);
200
201 SkinnedLocatorList skinnedLocators;
202 LocatorList locators;
203 for (size_t i = 0; i < sourceCharacter.locators.size(); ++i) {
204 const auto& locator = sourceCharacter.locators[i];
205 const auto& offset = locator.offset;
206 const auto& parent = locator.parent;
207 const Eigen::Vector3f p_world = restState.jointState[parent].transformation * offset;
208
209 // Find the closest point on the mesh to the locator.
210 const auto closestPointResult = closestPointOnMeshMatchingParent(
211 *sourceCharacter.mesh,
212 *sourceCharacter.skinWeights,
213 p_world,
214 gsl::narrow_cast<uint32_t>(parent));
215 if (!closestPointResult.valid || closestPointResult.distance > maxDistance) {
216 locators.push_back(locator);
217 continue;
218 }
219
220 SkinnedLocator skinnedLocator;
221 skinnedLocator.name = locator.name;
222 skinnedLocator.position = p_world;
223 skinnedLocator.weight = locator.weight;
224
225 // Set the locator's parent to the closest face.
226 std::tie(skinnedLocator.parents, skinnedLocator.skinWeights) = averageTriangleSkinWeights(
227 sourceCharacter, closestPointResult.triangleIdx, closestPointResult.baryCoords);
228 skinnedLocators.push_back(skinnedLocator);
229 }
230
231 // Strip out regular locators and replace with skinned locators:
232 return {
233 sourceCharacter.skeleton,
234 sourceCharacter.parameterTransform,
235 sourceCharacter.parameterLimits,
236 locators,
237 sourceCharacter.mesh.get(),
238 sourceCharacter.skinWeights.get(),
239 sourceCharacter.collision.get(),
240 sourceCharacter.poseShapes.get(),
241 sourceCharacter.blendShape,
242 sourceCharacter.faceExpressionBlendShape,
243 sourceCharacter.name,
244 sourceCharacter.inverseBindPose,
245 skinnedLocators};
246}

248Character createLocatorCharacter(const Character& sourceCharacter, const std::string& prefix) {
249 // create a copy
250 Character character = sourceCharacter;
251
252 auto& newSkel = character.skeleton;
253 auto& newTransform = character.parameterTransform;
254 auto& newLocators = character.locators;
255 auto& newLimits = character.parameterLimits;
256 auto& newInvBindPose = character.inverseBindPose;
257
258 ParameterSet locatorSet;
259
260 // create a new additional joint for each locator
261 std::vector<Eigen::Triplet<float>> triplets;
262 for (auto& newLocator : newLocators) {
263 // ignore locator if it is fixed
264 if (newLocator.locked.all()) {
265 continue;
266 }
267
268 // create a new joint for the locator
269 Joint joint;
270 joint.name = std::string(prefix + newLocator.name);
271 joint.parent = newLocator.parent;
272 joint.translationOffset = newLocator.offset;
273
274 // insert joint
275 const size_t id = newSkel.joints.size();
276 newSkel.joints.push_back(joint);
277 newInvBindPose.push_back(Affine3f::Identity());
278
279 // create parameter for the added joint
280 static const std::array<std::string, 3> tNames{"_tx", "_ty", "_tz"};
281 for (size_t j = 0; j < 3; j++) {
282 if (newLocator.locked[j] != 0) {
283 continue;
284 }
285 const std::string jname = joint.name + tNames[j];
286 triplets.emplace_back(
287 static_cast<int>(id) * kParametersPerJoint + static_cast<int>(j),
288 static_cast<int>(newTransform.name.size()),
289 1.0f);
290 MT_CHECK(
291 newTransform.name.size() < kMaxModelParams,
292 "Number of model parameters reached the {} limit.",
293 kMaxModelParams);
294 locatorSet.set(newTransform.name.size());
295 newTransform.name.push_back(jname);
296
297 // check if we need joint limit
298 if (newLocator.limitWeight[j] > 0.0f) {
299 ParameterLimit p;
300 p.type = MinMaxJoint;
301 p.data.minMaxJoint.jointIndex = id;
302 p.data.minMaxJoint.jointParameter = j;
303 const float referencePosition = newLocator.limitOrigin[j] - newLocator.offset[j];
304 p.data.minMaxJoint.limits = Vector2f(referencePosition, referencePosition);
305 p.weight = newLocator.limitWeight[j];
306 newLimits.push_back(p);
307 }
308 }
309 newTransform.offsets.conservativeResize(newTransform.offsets.size() + kParametersPerJoint);
310 newTransform.activeJointParams.conservativeResize(
311 newTransform.activeJointParams.size() + kParametersPerJoint);
312 for (size_t j = newTransform.offsets.size() - gsl::narrow<int>(kParametersPerJoint);
313 j < newTransform.offsets.size();
314 ++j) {
315 newTransform.offsets(j) = 0.0;
316 }
317
318 for (size_t j = newTransform.activeJointParams.size() - gsl::narrow<int>(kParametersPerJoint);
319 j < newTransform.activeJointParams.size();
320 ++j) {
321 newTransform.activeJointParams(j) = true;
322 }
323
324 // reattach locator to new joint
325 newLocator.parent = id;
326 newLocator.offset.setZero();
327 }
328 newTransform.parameterSets["locators"] = locatorSet;
329
330 // update parameter transform
331 const int newRows = static_cast<int>(newSkel.joints.size()) * kParametersPerJoint;
332 const int newCols = static_cast<int>(newTransform.name.size());
333 newTransform.transform.conservativeResize(newRows, newCols);
334 SparseRowMatrixf additionalTransforms(newRows, newCols);
335 additionalTransforms.setFromTriplets(triplets.begin(), triplets.end());
336 newTransform.transform += additionalTransforms;
337
338 // return the new character
339 return character;
340}

342LocatorList extractLocatorsFromCharacter(
343 const Character& locatorCharacter,
344 const CharacterParameters& calibParams) {
345 const SkeletonState state(
346 locatorCharacter.parameterTransform.apply(calibParams), locatorCharacter.skeleton);
347 const LocatorList& locators = locatorCharacter.locators;
348 const auto& skeleton = locatorCharacter.skeleton;
349
350 LocatorList result = locators;
351
352 // Check if locators is empty
353 if (locators.empty()) {
354 return result;
355 }
356
357 // revert each locator to the original attachment and add an offset
358 for (size_t i = 0; i < locators.size(); i++) {
359 // only map locators back that are not fixed
360 if (locators[i].locked.all()) {
361 continue;
362 }
363
364 // joint id
365 const size_t jointId = locators[i].parent;
366
367 // Check bounds for jointState access
368 if (jointId >= state.jointState.size()) {
369 MT_LOGW(
370 "Joint ID {} is out of bounds for jointState (size: {})",
371 jointId,
372 state.jointState.size());
373 continue;
374 }
375
376 // get global locator position
377 const Vector3f pos = state.jointState[jointId].transformation * locators[i].offset;
378
379 // change attachment to original joint
380 result[i].parent = skeleton.joints[jointId].parent;
381
382 // Check bounds for parent joint access
383 if (result[i].parent >= state.jointState.size()) {
384 MT_LOGW(
385 "Parent joint ID {} is out of bounds for jointState (size: {})",
386 result[i].parent,
387 state.jointState.size());
388 continue;
389 }
390
391 // calculate new offset
392 const Vector3f offset = state.jointState[result[i].parent].transformation.inverse() * pos;
393
394 // change offset to current state
395 result[i].offset = offset;
396 }
397
398 // return
399 return result;
400}

454void fillIdentity(
455 const ParameterSet& idSet,
456 const ModelParameters& identity,
457 Eigen::MatrixXf& motionNoId) {
458 MT_CHECK(identity.v.size() == motionNoId.rows());
459
460 const size_t numParams = motionNoId.rows();
461 const size_t numFrames = motionNoId.cols();
462
463 for (size_t iParam = 0; iParam < numParams; ++iParam) {
464 if (!idSet.test(iParam)) {
465 continue;
466 }
467 for (size_t jFrame = 0; jFrame < numFrames; ++jFrame) {
468 motionNoId(iParam, jFrame) += identity.v(iParam);
469 }
470 }
471}

490ModelParameters jointIdentityToModelIdentity(
491 const Character& c,
492 const ParameterSet& idSet,
493 const JointParameters& jointIdentity) {
494 // convert from joint to model parameters using only the parameters active in idSet
495 auto scalingTransform = c.parameterTransform.simplify(idSet);
496 momentum::ModelParameters scaleParameters =
497 InverseParameterTransform(scalingTransform).apply(jointIdentity).pose;
498 momentum::ModelParameters identity =
499 ModelParameters::Zero(c.parameterTransform.numAllModelParameters());
500 size_t iScale = 0;
501 for (size_t iParam = 0; iParam < c.parameterTransform.numAllModelParameters(); iParam++) {
502 if (idSet.test(iParam)) {
503 identity[iParam] = scaleParameters[iScale];
504 ++iScale;
505 }
506 }
507
508 return identity;
509}

511std::vector<std::vector<Marker>> extractMarkersFromMotion(
512 const Character& character,
513 const Eigen::MatrixXf& motion) {
514 const size_t nFrames = motion.cols();
515 std::vector<std::vector<Marker>> result(nFrames);
516 for (size_t iFrame = 0; iFrame < nFrames; ++iFrame) {
517 const JointParameters params(motion.col(iFrame));
518 const auto& states = SkeletonState(params, character.skeleton).jointState;
519
520 std::vector<Marker>& markers = result.at(iFrame);
521 for (const auto& loc : character.locators) {
522 const Vector3d pos = (states.at(loc.parent).transformation * loc.offset).cast<double>();
523 markers.emplace_back(Marker{loc.name, pos, false});
524 }
525 }
526
527 return result;
528}
