21template <typename T>
22Vector3<T> fallbackSupportPlaneAxis(const Vector3<T>& normal) {
23 Vector3<T> axis = Vector3<T>::UnitX();
24 T axisAlignment = std::abs(normal.x());
25 if (const T yAlignment = std::abs(normal.y()); yAlignment < axisAlignment) {
26 axis = Vector3<T>::UnitY();
27 axisAlignment = yAlignment;
28 }
29 if (std::abs(normal.z()) < axisAlignment) {
30 axis = Vector3<T>::UnitZ();
31 }
32 return (axis - normal * axis.dot(normal)).normalized();
33}

35bool supportPointFinite(const Vector2f& point) {
36 return std::isfinite(point.x()) && std::isfinite(point.y());

39bool supportPointLess(const Vector2f& lhs, const Vector2f& rhs) {
40 if (lhs.x() == rhs.x()) {
41 return lhs.y() < rhs.y();
42 }
43 return lhs.x() < rhs.x();
44}

46bool supportPointEqual(const Vector2f& lhs, const Vector2f& rhs) {
47 return lhs.x() == rhs.x() && lhs.y() == rhs.y();

50void appendSupportHullPoint(std::vector<Vector2f>& hull, const Vector2f& point) {
51 while (hull.size() >= 2) {
52 if (cross2d(hull.at(hull.size() - 2), hull.back(), point) > 0.0f) {
53 break;
54 }
55 hull.pop_back();
56 }
57 hull.push_back(point);
58}

60void removeLastSupportHullPoint(std::vector<Vector2f>& hull) {
61 if (!hull.empty()) {
62 hull.pop_back();

68template <typename T>
69SupportPlaneT<T>::SupportPlaneT(
70 const Vector3<T>& planeNormal,
71 const T planeOffset,
72 const Vector3<T>& uAxisHint) {
73 MT_THROW_IF(!planeNormal.allFinite(), "Support plane normal must contain only finite values");
74 MT_THROW_IF(!std::isfinite(planeOffset), "Support plane offset must be finite");
75 const T normalNorm = planeNormal.norm();
76 MT_THROW_IF(normalNorm <= Eps<T>(1e-8f, 1e-17), "Support plane normal must be non-zero");
77
78 normal = planeNormal / normalNorm;
79 offset = planeOffset / normalNorm;
80
81 MT_THROW_IF(!uAxisHint.allFinite(), "Support plane axis hint must contain only finite values");
82 Vector3<T> tangent = uAxisHint - normal * uAxisHint.dot(normal);
83 const T tangentNorm = tangent.norm();
84 if (tangentNorm > Eps<T>(1e-8f, 1e-17)) {
85 uAxis = tangent / tangentNorm;
86 } else {
87 uAxis = fallbackSupportPlaneAxis(normal);
88 }
89 vAxis = uAxis.cross(normal).normalized();
90}

118float cross2d(const Vector2f& origin, const Vector2f& a, const Vector2f& b) {
119 const Vector2f originToA = a - origin;
120 const Vector2f originToB = b - origin;
121 return originToA.x() * originToB.y() - originToA.y() * originToB.x();
122}

124std::vector<Vector2f> computeConvexHull2d(const std::span<const Vector2f> points) {
125 std::vector<Vector2f> sortedPoints;
126 sortedPoints.reserve(points.size());
127 for (const Vector2f& point : points) {
128 MT_THROW_IF(
129 !supportPointFinite(point), "Support polygon point must contain only finite values");
130 sortedPoints.push_back(point);
131 }
132
133 std::ranges::sort(sortedPoints, supportPointLess);
134 sortedPoints.erase(
135 std::unique(sortedPoints.begin(), sortedPoints.end(), supportPointEqual), sortedPoints.end());
136 if (sortedPoints.size() <= 2) {
137 return sortedPoints;
138 }
139
140 std::vector<Vector2f> lower;
141 lower.reserve(sortedPoints.size());
142 for (const Vector2f& point : sortedPoints) {
143 appendSupportHullPoint(lower, point);
144 }
145
146 std::vector<Vector2f> upper;
147 upper.reserve(sortedPoints.size());
148 for (size_t i = sortedPoints.size(); i-- > 0;) {
149 appendSupportHullPoint(upper, sortedPoints.at(i));
150 }
151
152 removeLastSupportHullPoint(lower);
153 removeLastSupportHullPoint(upper);
154 lower.insert(lower.end(), upper.begin(), upper.end());
155 return lower;
156}
