16template <typename T>
17void MeshT<T>::updateNormals() {
18 // calculate normals
19 normals.resize(vertices.size());
20 std::fill(normals.begin(), normals.end(), Eigen::Vector3<T>::Zero());
21 const auto verticesNum = static_cast<int>(vertices.size());
22 for (const auto& face : faces) {
23 // Skip faces with out-of-boundaries indexes
24 if (std::any_of(face.begin(), face.end(), [verticesNum](int idx) {
25 return idx < 0 || idx >= verticesNum;
26 })) {
27 continue;
28 }
29 // calculate normal and add for each vertex
30 const Eigen::Vector3<T> normal =
31 (vertices[face[1]] - vertices[face[0]]).cross(vertices[face[2]] - vertices[face[0]]);
32 if (IsNanNoOpt(normal[0])) {
33 continue;
34 }
35 for (const auto& faceIdx : face) {
36 normals[faceIdx] += normal;
37 }
38 }
39 // re-normalize normals
40 for (size_t i = 0; i < normals.size(); i++) {
41 const T len = normals[i].norm();
42 // avoid divide-by-zero:
43 if (len != 0) {
44 normals[i] /= len;
45 }
46 }
47}
