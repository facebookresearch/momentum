52template <typename T>
53double FullyDifferentiableOrientationErrorFunctionT<T>::getError(
54 const ModelParametersT<T>& /*params*/,
55 const SkeletonStateT<T>& state) {
56 // loop over all constraints and calculate the error
57 double error = 0.0;
58
59 for (size_t i = 0; i < constraints_.size(); ++i) {
60 const OrientationConstraintT<T>& ct = constraints_[i];
61
62 // calculate orientation error
63 // Note: |R0 - RT|² is a valid norm on SO3, it doesn't have the same slope as the squared
64 // angle difference
65 // but it's derivative doesn't have a singularity at the minimum, so is more stable
66 const Eigen::Quaternion<T> rot = state.jointState[ct.parent].rotation() * ct.offset;
67 const Eigen::Matrix3<T> rotDiff = rot.toRotationMatrix() - ct.target.toRotationMatrix();
68 error += ct.weight * rotDiff.squaredNorm() * kOrientationWeight;
69 }
70
71 // return error
72 return error * this->weight_;
73}

75template <typename T>
76double FullyDifferentiableOrientationErrorFunctionT<T>::getGradient(
77 const ModelParametersT<T>& /*params*/,
78 const SkeletonStateT<T>& state,
79 Ref<Eigen::VectorX<T>> gradient) {
80 const auto& parameterTransform = this->parameterTransform_;
81
82 // storage for joint gradients
83 jointGrad_.setZero(parameterTransform.numAllModelParameters());
84
85 // loop over all constraints and calculate the error
86 double error = 0.0;
87
88 for (size_t i = 0; i < constraints_.size(); ++i) {
89 const OrientationConstraintT<T>& ct = constraints_[i];
90 if (ct.weight == 0) {
91 continue;
92 }
93
94 error += calculateOrientationGradient(state, constraints_[i], jointGrad_);
95 }
96
97 // convert joint gradient to parameter gradient and return
98 gradient.noalias() += jointGrad_ * this->weight_;
99
100 // return error
101 return error * this->weight_;
102}

104template <typename T>
105double FullyDifferentiableOrientationErrorFunctionT<T>::getJacobian(
106 const ModelParametersT<T>& params,
107 const SkeletonStateT<T>& state,
108 Ref<Eigen::MatrixX<T>> jacobian,
109 Ref<Eigen::VectorX<T>> residual,
110 int& usedRows) {
111 // loop over all constraints and calculate the error
112 double error = 0.0;
113 std::atomic_int offset(0);
114
115 for (size_t i = 0; i < constraints_.size(); ++i) {
116 const OrientationConstraintT<T>& ct = constraints_[i];
117 if (ct.weight == 0) {
118 continue;
119 }
120
121 const auto currentOffset = offset.fetch_add(9, std::memory_order_relaxed);
122 error += calculateOrientationJacobian(
123 state,
124 constraints_[i],
125 jacobian.block(currentOffset, 0, 9, params.size()),
126 residual.middleRows(currentOffset, 9));
127 }
128
129 usedRows = gsl::narrow_cast<int>(jacobian.rows());
130
131 return error;
132}

134template <typename T>
135T FullyDifferentiableOrientationErrorFunctionT<T>::calculateOrientationJacobian(
136 const SkeletonStateT<T>& state,
137 const OrientationConstraintT<T>& constr,
138 Ref<Eigen::MatrixX<T>> jac,
139 Ref<Eigen::VectorX<T>> res) const {
140 const auto& parameterTransform = this->parameterTransform_;
141
142 // calculate orientation error
143 const Eigen::Quaternion<T> rot = state.jointState[constr.parent].rotation() * constr.offset;
144 const Eigen::Quaternion<T>& target = constr.target;
145 const Eigen::Matrix3<T> rotDiff = rot.toRotationMatrix() - target.toRotationMatrix();
146 const T wgt = kOrientationWeight * constr.weight * this->weight_;
147 const T swgt = std::sqrt(wgt);
148
149 // loop over all joints the constraint is attached to and calculate gradient
150 size_t jointIndex = constr.parent;
151 while (jointIndex != kInvalidIndex) {
152 // check for valid index
153 MT_CHECK(jointIndex < this->skeleton_.joints.size());
154
155 const size_t paramIndex = jointIndex * kParametersPerJoint;
156
157 // calculate derivatives based on active joints
158 for (size_t d = 0; d < 3; d++) {
159 if (this->activeJointParams_[paramIndex + 3 + d]) {
160 const Eigen::Vector3<T> axis = state.jointState[jointIndex].rotationAxis.col(d);
161 const Eigen::Matrix3<T> rotD = crossProductMatrix(axis) * rot * swgt;
162 const auto ja = Map<const Eigen::VectorX<T>>(rotD.data(), rotD.size());
163
164 // explicitly multiply with the parameter transform to generate parameter space gradients
165 for (auto index = parameterTransform.transform.outerIndexPtr()[paramIndex + 3 + d];
166 index < parameterTransform.transform.outerIndexPtr()[paramIndex + d + 3 + 1];
167 ++index) {
168 jac.col(parameterTransform.transform.innerIndexPtr()[index]) +=
169 ja * parameterTransform.transform.valuePtr()[index];
170 }
171 }
172 }
173
174 // go to the next joint
175 jointIndex = this->skeleton_.joints[jointIndex].parent;
176 }
177
178 res = Map<const Eigen::VectorX<T>>(rotDiff.data(), rotDiff.size()) * swgt;
179
180 return rotDiff.squaredNorm() * wgt;
181}

215template <typename T>
216void FullyDifferentiableOrientationErrorFunctionT<T>::setInputImp(
217 const std::string& name,
218 Eigen::Ref<const Eigen::VectorX<T>> value) {
219 // Normalization is a tricky topic here. When we set the values, we must normalize normals and
220 // quaternions etc because the rest of the code won't work correctly without this (for example
221 // computing the angle between unnormalized vectors requires normalizing them first). However,
222 // if we do normalize the values that are passed in, then the derivatives of those values are
223 // no longer correct. For example, imagine that we pass in v = (1000, 0, 0) as the normal to a
224 // plane. Clearly we must turn this into (1, 0, 0) so the IK solve does something reasonable, but
225 // when we do we find that our dv/dTheta is off by a factor of 1000. It's hard to address that
226 // inside this code, the right way to do it is to require that the inputs be normalized inside
227 // Pytorch before getting passed in, which will ensure all the derivatives work out correctly.
228 if (name == kWeights) {
229 for (size_t i = 0; i < constraints_.size(); ++i) {
230 constraints_[i].weight = static_cast<float>(value[i]);
231 }
232 } else if (name == kOffsets) {
233 for (size_t i = 0; i < constraints_.size(); ++i) {
234 constraints_[i].offset = Eigen::Quaternion<T>(value.template segment<4>(4 * i)).normalized();
235 }
236 } else if (name == kTargets) {
237 for (size_t i = 0; i < constraints_.size(); ++i) {
238 constraints_[i].target = Eigen::Quaternion<T>(value.template segment<4>(4 * i)).normalized();
239 }
240 } else {
241 MT_THROW(
242 "Unknown input to FullyDifferentiableOrientationErrorFunctionT<T>::getInput: {}", name);
243 }

246template <typename T>
247T FullyDifferentiableOrientationErrorFunctionT<T>::calculateOrientationGradient(
248 const SkeletonStateT<T>& state,
249 const OrientationConstraintT<T>& constr,
250 Eigen::VectorX<T>& jGrad) const {
251 const auto& parameterTransform = this->parameterTransform_;
252
253 // calculate orientation error
254 const Eigen::Quaternion<T> rot = state.jointState[constr.parent].rotation() * constr.offset;
255 const Eigen::Matrix3<T> rotDiff = rot.toRotationMatrix() - constr.target.toRotationMatrix();
256 const T wgt = kOrientationWeight * constr.weight;
257
258 // loop over all joints the constraint is attached to and calculate gradient
259 size_t jointIndex = constr.parent;
260 while (jointIndex != kInvalidIndex) {
261 // check for valid index
262 MT_CHECK(jointIndex < this->skeleton_.joints.size());
263
264 const size_t paramIndex = jointIndex * kParametersPerJoint;
265
266 // calculate derivatives based on active joints
267 for (size_t d = 0; d < 3; d++) {
268 if (this->activeJointParams_[paramIndex + 3 + d]) {
269 // calculate joint gradient
270 const Eigen::Vector3<T> axis = state.jointState[jointIndex].rotationAxis.col(d);
271 const auto rotD = crossProductMatrix(axis) * rot;
272 const T val = 2.0f * wgt * rotD.cwiseProduct(rotDiff).sum();
273 // explicitly multiply with the parameter transform to generate parameter space gradients
274 for (auto index = parameterTransform.transform.outerIndexPtr()[paramIndex + 3 + d];
275 index < parameterTransform.transform.outerIndexPtr()[paramIndex + d + 3 + 1];
276 ++index) {
277 jGrad[parameterTransform.transform.innerIndexPtr()[index]] +=
278 val * parameterTransform.transform.valuePtr()[index];
279 }
280 }
281 }
282
283 // go to the next joint
284 jointIndex = this->skeleton_.joints[jointIndex].parent;
285 }
286
287 return rotDiff.squaredNorm() * wgt;
288}

290template <typename T>
291template <typename JetType>
292[[nodiscard]] JetType
293FullyDifferentiableOrientationErrorFunctionT<T>::calculateOrientationGradient_dot(
294 const SkeletonStateT<T>& state,
295 const size_t constrParent,
296 const JetType& constrWeight,
297 const Eigen::Quaternion<JetType>& constrOrientationOffset,
298 const Eigen::Quaternion<JetType>& constrOrientationTarget,
299 Eigen::Ref<const Eigen::VectorX<T>> vec) const {
300 JetType result;
301
302 // calculate orientation error
303 const Eigen::Quaternion<JetType> rot =
304 (state.jointState[constrParent].rotation().template cast<JetType>() *
305 constrOrientationOffset);
306 const Eigen::Matrix3<JetType> rotDiff =
307 rot.toRotationMatrix() - constrOrientationTarget.toRotationMatrix();
308 const JetType wgt = this->weight_ * this->kOrientationWeight * constrWeight;
309
310 // loop over all joints the constraint is attached to and calculate gradient
311 size_t jointIndex = constrParent;
312 while (jointIndex != kInvalidIndex) {
313 // check for valid index
314 assert(jointIndex < this->skeleton_.joints.size());
315
316 const size_t paramIndex = jointIndex * kParametersPerJoint;
317
318 // calculate derivatives based on active joints
319 for (size_t d = 0; d < 3; d++) {
320 if (this->activeJointParams_[paramIndex + 3 + d]) {
321 // calculate joint gradient
322 const Eigen::Vector3<T> axis = state.jointState[jointIndex].rotationAxis.col(d);
323 const Eigen::Matrix3<JetType> rotD = crossProductMatrix(axis) * rot;
324 const JetType val = T(2) * wgt * rotD.cwiseProduct(rotDiff).sum();
325 result += times_parameterTransform_times_v(
326 val, paramIndex + 3 + d, this->parameterTransform_, vec);
327 }
328 }
329
330 // go to the next joint
331 jointIndex = this->skeleton_.joints[jointIndex].parent;
332 }
333
334 return result;
335}

337template <typename T>
338Eigen::VectorX<T> FullyDifferentiableOrientationErrorFunctionT<T>::d_gradient_d_input_dot(
339 const std::string& inputName,
340 const ModelParametersT<T>& /*modelParams*/,
341 const SkeletonStateT<T>& state,
342 Eigen::Ref<const Eigen::VectorX<T>> inputVec) {
343 // Use automatic differentiation on the gradient function:
344 if (inputName == kWeights) {
345 Eigen::VectorX<T> result = Eigen::VectorX<T>::Zero(constraints_.size());
346 using JetType = ceres::Jet<T, 1>;
347 for (size_t iCons = 0; iCons < constraints_.size(); ++iCons) {
348 const auto& cons = constraints_[iCons];
349 result(iCons) = calculateOrientationGradient_dot<JetType>(
350 state,
351 cons.parent,
352 JetType(cons.weight, 0),
353 cons.offset.template cast<JetType>(),
354 cons.target.template cast<JetType>(),
355 inputVec)
356 .v[0];
357 }
358 return result;
359 } else if (inputName == kOffsets) {
360 Eigen::VectorX<T> result = Eigen::VectorX<T>::Zero(4 * constraints_.size());
361 using JetType = ceres::Jet<T, 4>;
362 for (size_t iCons = 0; iCons < constraints_.size(); ++iCons) {
363 const auto& cons = constraints_[iCons];
364 const auto dotProd = calculateOrientationGradient_dot<JetType>(
365 state,
366 cons.parent,
367 JetType(cons.weight),
368 Eigen::Quaternion<JetType>(buildJetVec<T, 4>(cons.offset.coeffs())).normalized(),
369 cons.target.template cast<JetType>(),
370 inputVec);
371 result.template segment<4>(4 * iCons) = dotProd.v;
372 }
373 return result;
374 } else if (inputName == kTargets) {
375 Eigen::VectorX<T> result = Eigen::VectorX<T>::Zero(4 * constraints_.size());
376 using JetType = ceres::Jet<T, 4>;
377 for (size_t iCons = 0; iCons < constraints_.size(); ++iCons) {
378 const auto& cons = constraints_[iCons];
379 const auto dotProd = calculateOrientationGradient_dot<JetType>(
380 state,
381 cons.parent,
382 JetType(cons.weight),
383 cons.offset.template cast<JetType>(),
384 Eigen::Quaternion<JetType>(buildJetVec<T, 4>(cons.target.coeffs())).normalized(),
385 inputVec);
386 result.template segment<4>(4 * iCons) = dotProd.v;
387 }
388 return result;
389 } else {
390 MT_THROW(
391 "Unknown input name in FullyDifferentiableOrientationErrorFunctionT<T>::d_gradient_d_input_dot: {}",
392 inputName);
393 }
