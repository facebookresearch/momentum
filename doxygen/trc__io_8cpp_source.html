18MarkerSequence loadTrc(const std::string& filename, UpVector up) {
19 MarkerSequence res;
20 res.fps = 0.f;
21
22 std::ifstream infile(filename);
23 if (!infile.is_open()) {
24 return res;
25 }
26
27 std::string line;
28
29 // parse header
30 GetLineCrossPlatform(infile, line);
31 if (line.find("PathFileType\t4\t(X/Y/Z)") == std::string::npos) {
32 return res;
33 }
34
35 GetLineCrossPlatform(infile, line);
36 if (line.find(
37 "DataRate\tCameraRate\tNumFrames\tNumMarkers\tUnits\tOrigDataRate\tOrigDataStartFrame\tOrigNumFrames") ==
38 std::string::npos) {
39 return res;
40 }
41
42 GetLineCrossPlatform(infile, line);
43 auto tokens = tokenize(line, " \t\r\n");
44 if (tokens.size() != 8) {
45 return res;
46 }
47 const double frameRate = std::stod(tokens[0]);
48 const size_t numMarkers = std::stoi(tokens[3]);
49 res.fps = static_cast<float>(frameRate);
50
51 // get line with marker names and parse it
52 GetLineCrossPlatform(infile, line);
53 tokens = tokenize(line, " \t\r\n");
54 if (tokens.size() != numMarkers + 2) {
55 res.fps = 0.0f;
56 return res;
57 }
58 std::vector<std::string> markerNames;
59 for (size_t i = 2; i < tokens.size(); i++) {
60 markerNames.push_back(tokens[i]);
61 }
62
63 // ignore next line
64 GetLineCrossPlatform(infile, line);
65
66 while (GetLineCrossPlatform(infile, line)) {
67 tokens = tokenize(line, "\t", false);
68 // check for right size
69 if (tokens.size() != numMarkers * 3 + 2) {
70 continue;
71 }
72
73 // create actor structure for every frame
74 std::vector<Marker> markers(numMarkers);
75
76 // go over all markers
77 for (size_t i = 0; i < numMarkers; i++) {
78 const size_t pos = i * 3 + 2;
79 markers[i].name = markerNames[i];
80 if (tokens[pos + 0].empty() || tokens[pos + 1].empty() || tokens[pos + 2].empty()) {
81 markers[i].occluded = true;
82 } else {
83 markers[i].occluded = false;
84 Vector3d p{
85 std::stod(tokens[pos + 0]), std::stod(tokens[pos + 1]), std::stod(tokens[pos + 2])};
86 p = toMomentumVector3(p, up, tokens[4]);
87 markers[i].pos = p;
88 }
89 }
90
91 // store frame
92 res.frames.push_back(markers);
93 }
94
95 return res;
96}
