30template <typename T>
31double ModelParametersSequenceErrorFunctionT<T>::getError(
32 gsl::span<const ModelParametersT<T>> modelParameters,
33 gsl::span<const SkeletonStateT<T>> /* skelStates */) const {
34 // ignore if we don't have any reasonable data
35 if (targetWeights_.size() !=
36 gsl::narrow_cast<Eigen::Index>(this->parameterTransform_.numAllModelParameters())) {
37 return 0.0;
38 }
39
40 const auto np = gsl::narrow_cast<Eigen::Index>(this->parameterTransform_.numAllModelParameters());
41
42 MT_CHECK(modelParameters.size() == 2);
43 const auto& prevParams = modelParameters[0];
44 const auto& nextParams = modelParameters[1];
45
46 // calculate difference between parameters and desired parameters
47 double error = 0;
48 for (Eigen::Index i = 0; i < np; ++i) {
49 if (this->enabledParameters_.test(i) && targetWeights_(i) > 0) {
50 const auto pdiff = targetWeights_(i) * (nextParams(i) - prevParams(i));
51 error += pdiff * pdiff;
52 }
53 }
54
55 // return error
56 return error * this->weight_ * kMotionWeight;
57}

59template <typename T>
60double ModelParametersSequenceErrorFunctionT<T>::getGradient(
61 gsl::span<const ModelParametersT<T>> modelParameters,
62 gsl::span<const SkeletonStateT<T>> /* skelStates */,
63 Eigen::Ref<Eigen::VectorX<T>> gradient) const {
64 // ignore if we don't have any reasonable data
65 if (targetWeights_.size() !=
66 gsl::narrow_cast<Eigen::Index>(this->parameterTransform_.numAllModelParameters())) {
67 return 0.0;
68 }
69
70 const auto np = gsl::narrow_cast<Eigen::Index>(this->parameterTransform_.numAllModelParameters());
71
72 MT_CHECK(modelParameters.size() == 2);
73 const auto& prevParams = modelParameters[0];
74 const auto& nextParams = modelParameters[1];
75
76 Eigen::Ref<Eigen::VectorX<T>> prevGrad = gradient.segment(0, np);
77 Eigen::Ref<Eigen::VectorX<T>> nextGrad = gradient.segment(np, np);
78
79 double error = 0;
80 for (Eigen::Index i = 0; i < np; ++i) {
81 if (this->enabledParameters_.test(i) && targetWeights_(i) > 0) {
82 const auto pdiff = targetWeights_(i) * (nextParams(i) - prevParams(i));
83 prevGrad(i) -= 2.0f * targetWeights_(i) * pdiff * this->weight_ * kMotionWeight;
84 nextGrad(i) += 2.0f * targetWeights_(i) * pdiff * this->weight_ * kMotionWeight;
85 error += pdiff * pdiff;
86 }
87 }
88
89 // return error
90 return error * this->weight_ * kMotionWeight;
91}

98template <typename T>
99double ModelParametersSequenceErrorFunctionT<T>::getJacobian(
100 gsl::span<const ModelParametersT<T>> modelParameters,
101 gsl::span<const SkeletonStateT<T>> /* skelStates */,
102 Eigen::Ref<Eigen::MatrixX<T>> jacobian,
103 Eigen::Ref<Eigen::VectorX<T>> residual,
104 int& usedRows) const {
105 MT_PROFILE_FUNCTION();
106 MT_CHECK(
107 jacobian.cols() ==
108 gsl::narrow_cast<Eigen::Index>(
109 numFrames() * this->parameterTransform_.numAllModelParameters()));
110
111 usedRows = 0;
112
113 const auto np = gsl::narrow_cast<Eigen::Index>(this->parameterTransform_.numAllModelParameters());
114
115 // ignore if we don't have any reasonable data
116 if (targetWeights_.size() !=
117 gsl::narrow_cast<Eigen::Index>(this->parameterTransform_.numAllModelParameters())) {
118 return 0.0;
119 }
120
121 const float sWeight = std::sqrt(this->weight_ * kMotionWeight);
122
123 MT_CHECK(modelParameters.size() == 2);
124 const auto& prevParams = modelParameters[0];
125 const auto& nextParams = modelParameters[1];
126 Eigen::Ref<Eigen::MatrixX<T>> prevJac = jacobian.topLeftCorner(jacobian.rows(), np);
127 Eigen::Ref<Eigen::MatrixX<T>> nextJac = jacobian.block(0, np, jacobian.rows(), np);
128
129 Eigen::Index out = 0;
130
131 // calculate difference between parameters and desired parameters
132 double error = 0;
133 for (Eigen::Index i = 0; i < np; ++i) {
134 if (this->enabledParameters_.test(i) && targetWeights_(i) > 0) {
135 const auto pdiff = targetWeights_(i) * (nextParams(i) - prevParams(i));
136 error += pdiff * pdiff;
137 residual(out) = pdiff * sWeight;
138 prevJac(out, i) = -sWeight * targetWeights_(i);
139 nextJac(out, i) = sWeight * targetWeights_(i);
140 ++out;
141 }
142 }
143
144 MT_CHECK(out <= residual.rows() && out <= jacobian.rows());
145
146 usedRows = gsl::narrow_cast<int>(out);
147
148 // return error
149 return error * this->weight_ * kMotionWeight;
150}
