22template <typename T>
23void BlendShape::computeShape(
24 const BlendWeightsT<T>& coefficients,
25 std::vector<Eigen::Vector3<T>>& output) const {
26 MT_CHECK(
27 gsl::narrow<Eigen::Index>(baseShape_.size()) * 3 == shapeVectors_.rows(),
28 "{} is not {}",
29 baseShape_.size() * 3,
30 shapeVectors_.rows());

32 output.resize(baseShape_.size());
33 Eigen::Map<Eigen::VectorX<T>> outputVec(&output[0][0], output.size() * 3);
34 const Eigen::Map<const Eigen::VectorXf> baseVec(&baseShape_[0][0], baseShape_.size() * 3);

36 outputVec = baseVec.template cast<T>() + computeDeltas<T>(coefficients);

39VectorXf BlendShape::estimateCoefficients(
40 gsl::span<const Vector3f> vertices,
41 const float regularization,
42 const VectorXf& weights) const {
43 MT_CHECK(
44 gsl::narrow<Eigen::Index>(baseShape_.size()) * 3 == shapeVectors_.rows(),
45 "{} is not {}",
46 baseShape_.size() * 3,
47 shapeVectors_.rows());
48 MT_CHECK(
49 vertices.size() == baseShape_.size(), "{} is not {}", vertices.size(), baseShape_.size());
50
51 const Map<const VectorXf> baseVec(&baseShape_[0][0], baseShape_.size() * 3);
52 const Map<const VectorXf> vertexVec(&vertices[0][0], vertices.size() * 3);
53
54 // if we're not using weights, use default pipeline
55 if (weights.size() != gsl::narrow<Eigen::Index>(vertices.size())) {
56 // check if factorization is valid
57 if (!factorizationValid_) {
58 factorization_.compute(shapeVectors_, Eigen::ComputeThinU | Eigen::ComputeThinV);
59 factorizationValid_ = true;
60 }
61
62 // estimate the shapeVectors for the given vertices
63 return factorization_.solve(vertexVec - baseVec);
64 } else {
65 // if we have weights, we definitely have to compute a new factorization that incorporates the
66 // weights
67 VectorXf wts(weights.size() * 3);
68 Eigen::Map<VectorXf, 0, Eigen::Stride<1, 3>>(&wts(0), weights.size()) = weights;
69 Eigen::Map<VectorXf, 0, Eigen::Stride<1, 3>>(&wts(1), weights.size()) = weights;
70 Eigen::Map<VectorXf, 0, Eigen::Stride<1, 3>>(&wts(2), weights.size()) = weights;
71
72 // calculate weighted coefficient matrix and rhs
73 const auto rows = shapeVectors_.rows();
74 const auto cols = shapeVectors_.cols();
75 MatrixXf A(rows + cols, cols);
76 for (int i = 0; i < wts.size(); i++) {
77 A.row(i) = shapeVectors_.row(i) * wts(i);
78 }
79 A.block(rows, 0, cols, cols) = MatrixXf::Identity(cols, cols) * regularization;
80
81 VectorXf b = VectorXf::Zero(rows + cols);
82 b.topRows(rows) = (vertexVec - baseVec).cwiseProduct(wts);
83
84 // solve normal equations to get results
85 return (A.transpose() * A).ldlt().solve(A.transpose() * b);
86 }
87}

101bool BlendShape::isApprox(const BlendShape& blendShape) const {
102 if (baseShape_.size() != blendShape.getBaseShape().size()) {
103 return false;
104 }
105 for (size_t i = 0; i < baseShape_.size(); i++) {
106 if (!baseShape_[i].isApprox(blendShape.getBaseShape()[i])) {
107 return false;
108 }
109 }
110
111 return shapeVectors_.isApprox(blendShape.getShapeVectors()) &&
112 (factorizationValid_ == blendShape.getFactorizationValid());
113}
