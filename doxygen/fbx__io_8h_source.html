18// UpVector Specifies which canonical axis represents up in the system
19// (typically Y or Z). Maps to fbxsdk::FbxAxisSystem::EUpVector
20enum class FBXUpVector { XAxis = 1, YAxis = 2, ZAxis = 3 };
21
22// FrontVector Vector with origin at the screen pointing toward the camera.
23// This is a subset of enum EUpVector because axis cannot be repeated.
24// We use the system of "parity" to define this vector because its value (X,Y or
25// Z axis) really depends on the up-vector. The EPreDefinedAxisSystem list the
26// up-vector, parity and coordinate system values for the predefined systems.
27// Maps to fbxsdk::FbxAxisSystem::EFrontVector
28enum class FBXFrontVector { ParityEven = 1, ParityOdd = 2 };
29
30// CoordSystem Specifies the third vector of the system.
31// Maps to fbxsdk::FbxAxisSystem::ECoordSystem
32enum class FBXCoordSystem { RightHanded, LeftHanded };
33
34// KeepLocators Specifies whether Nulls in the transform hierarchy should be turned into Locators.

37// LoadBlendShapes Specifies whether blendshapes should be loaded or not
