18PoseShape loadPoseShape(const std::string& filename, const Character& character) {
19 PoseShape result;
20
21 MT_CHECK(character.mesh);
22
23 std::ifstream data(filename, std::ios::in | std::ios::binary);
24 if (!data.is_open()) {
25 return result;
26 }
27
28 // read dimensions
29 uint64_t numRows = 0;
30 uint64_t numJoints = 0;
31 data.read((char*)&numRows, sizeof(numRows));
32 data.read((char*)&numJoints, sizeof(numJoints));
33 const uint64_t numCols = numJoints * 4;
34
35 MT_CHECK(
36 character.mesh->vertices.size() * 3 == numRows,
37 "{}, {}",
38 character.mesh->vertices.size() * 3,
39 numRows);
40
41 uint64_t count = 0;
42 data.read((char*)&count, sizeof(uint64_t));
43 std::string base;
44 base.resize(count);
45 data.read((char*)base.data(), count);
46 result.baseJoint = character.skeleton.getJointIdByName(base);
47 MT_CHECK(result.baseJoint != kInvalidIndex);
48 MT_CHECK(
49 0 <= result.baseJoint && result.baseJoint < character.skeleton.joints.size(),
50 "Invalid joint index");
51 // NOLINTNEXTLINE(facebook-hte-ParameterUncheckedArrayBounds)
52 result.baseRot = character.skeleton.joints[result.baseJoint].preRotation;
53
54 // load names
55 std::vector<std::string> names(numJoints);
56 for (size_t i = 0; i < numJoints; i++) {
57 data.read((char*)&count, sizeof(uint64_t));
58 names[i].resize(count);
59 data.read((char*)names[i].data(), count);
60 }
61
62 // read mean shape
63 result.baseShape.resize(numRows);
64 data.read((char*)result.baseShape.data(), sizeof(float) * numRows);
65
66 // add character vertices
67 const Map<const VectorXf> mesh(
68 &character.mesh->vertices[0][0], character.mesh->vertices.size() * 3);
69 result.baseShape += mesh;
70
71 // load shapeVectors
72 result.shapeVectors.resize(numRows, numCols);
73 data.read((char*)result.shapeVectors.data(), sizeof(float) * numRows * numCols);
74
75 // generate mapping from names
76 result.jointMap.resize(numJoints);
77 for (size_t i = 0; i < names.size(); i++) {
78 result.jointMap[i] = character.skeleton.getJointIdByName(names[i]);
79 MT_CHECK(result.jointMap[i] != kInvalidIndex);
80 }
81
82 return result;
83}
