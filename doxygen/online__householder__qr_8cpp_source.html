16template <typename T>
17T sqr(T val) {
18 return val * val;

21// Apply the Householder transformation H*y, where
22// H = (I - beta*v*v^T)
23// We use v_2m and y_2m to indicate that these vectors contain the
24// 2...m entries of the vector.
25//
26// Thus, the y vector is divided up:
27// y = [ y_1 y_2m... ]^T
28// and v is implicitly assumed to have 1 stored in its first entry,
29// v = [ 1 v_2m ]^T
30template <typename T>
31void applyHouseholderTransformation(
32 T beta,
33 const Eigen::Ref<const Eigen::Matrix<T, Eigen::Dynamic, 1>>& v_2m,
34 T& y_1,
35 Eigen::Ref<Eigen::Matrix<T, Eigen::Dynamic, 1>> y_2m) {
36 MT_CHECK(v_2m.size() == y_2m.size());
37
38 // Compute v^T * y. Note that v_1 is assumed to be 1 here.
39 const T dotProd = y_1 + v_2m.dot(y_2m);
40 const T scalar = dotProd * beta;
41
42 // Subtract off (beta * v * v^T * A)
43 y_1 -= scalar;
44 y_2m.noalias() -= scalar * v_2m;
45}
46
47// Computes the Householder reflection vector for the column [x1 x2...xm]^T
48// (x1 is passed in separately because it comes from the R matrix rather than A).
49// Uses the stable algorithm from Golub and van Loan.
50// The result of applying the Householder transform (I - beta*v*v^T) is a vector of the form
51// [mu 0 0 0 ... 0]^T
52// Returns the pair [beta, mu].
53template <typename T>
54std::pair<T, T> computeHouseholderVec(
55 T x1,
56 Eigen::Ref<Eigen::Matrix<T, Eigen::Dynamic, 1>>
57 vec) // NOLINT(facebook-hte-ConstantArgumentPassByValue)
58{
59 const T sigma = vec.squaredNorm();
60 if (sigma == 0) {
61 // No need to apply a transform if the vector is already zeroed out.
62 return {0, x1};
63 }
64
65 const T mu = std::sqrt(sqr(x1) + sigma);
66 const T v1 = (x1 <= 0) ? (x1 - mu) : (-sigma / (x1 + mu));
67 const T beta = T(2) * sqr(v1) / (sigma + sqr(v1));
68 vec /= v1;
69
70 return std::make_pair(beta, mu);
71}

120template <typename T>
121void OnlineHouseholderQR<T>::addMutating(
122 ColumnIndexedMatrix<MatrixType> A,
123 Eigen::Ref<VectorType> b) {
124 MT_PROFILE_FUNCTION();
125
126 const Eigen::Index n = R_.rows();
127 MT_CHECK(A.rows() == b.rows());
128 MT_CHECK(A.cols() == n);
129 for (Eigen::Index iCol = 0; iCol < n; ++iCol) {
130 // At this point, we have zeroed out all the columns to the
131 // left of iCol. The current matrix looks like this.
132 // iCol
133 // \|/
134 // [ r r r r ]
135 // [ 0 r r r ]
136 // [ 0 0 r r ]
137 // [ 0 0 0 r ]
138 // [ 0 0 0 0 ]
139 // [ ... ]
140 // [ 0 0 a a ]
141 // [ 0 0 a a ]
142 // We will apply a Householder reflection (I - beta*v*v^T) * A
143 // This will zero out the ith column.
144 const auto [beta, mu] = computeHouseholderVec<T>(R_(iCol, iCol), A.col(iCol));
145 if (beta == 0) {
146 continue;
147 }
148
149 // computeHouseholderVec guarantees that applying the transform to iCol
150 // results in a vector [mu 0 ... 0].
151 R_(iCol, iCol) = mu;
152
153 // Apply the Householder vector to the remaining columns.
154 const auto& v2m = A.col(iCol);
155 for (Eigen::Index jCol = iCol + 1; jCol < n; ++jCol) {
156 applyHouseholderTransformation<T>(beta, v2m, R_(iCol, jCol), A.col(jCol));
157 }
158
159 // Apply the Householder vector to the rhs.
160 applyHouseholderTransformation<T>(beta, v2m, y_(iCol), b);
161 }
162}

164template <typename T>
165typename OnlineHouseholderQR<T>::VectorType OnlineHouseholderQR<T>::At_times_b() const {
166 // We defined R and y as
167 // R = Q*A y = Q*b
168 // Therefore
169 // R^T*y = (Q*A)^T*(Q*b)
170 // = A^T * Q^T * Q * b =
171 // = A^T*b
172 return R_.transpose() * y_;
173}

193template <typename T>
194void OnlineBlockHouseholderQR<T>::addMutating(
195 size_t iBlock,
196 ColumnIndexedMatrix<MatrixType> A_diag,
197 ColumnIndexedMatrix<MatrixType> A_common,
198 Eigen::Ref<VectorType> b) {
199 const Eigen::Index n_diag = A_diag.cols();
200 const Eigen::Index n_common = R_nn_.cols();
201 MT_CHECK(A_diag.rows() == b.rows());
202 MT_CHECK(A_common.rows() == b.rows());
203 MT_CHECK(A_common.cols() == n_common);
204
205 while (R_ii_.size() <= iBlock) {
206 R_ii_.push_back(MatrixType());
207 }
208 while (y_i_.size() <= iBlock) {
209 y_i_.push_back(VectorType());
210 }
211 while (R_in_.size() <= iBlock) {
212 R_in_.push_back(MatrixType());
213 }
214
215 if (R_ii_[iBlock].rows() == 0) {
216 R_ii_[iBlock] = MatrixType::Zero(n_diag, n_diag);
217 R_ii_[iBlock].diagonal().setConstant(lambda_);
218
219 y_i_[iBlock] = VectorType::Zero(n_diag);
220
221 R_in_[iBlock] = MatrixType::Zero(n_diag, n_common);
222 } else {
223 MT_CHECK(R_ii_[iBlock].cols() == A_diag.cols());
224 MT_CHECK(y_i_[iBlock].rows() == A_diag.cols());
225 }
226
227 auto& R_ii = R_ii_[iBlock];
228 auto& y_i = y_i_[iBlock];
229 auto& R_in = R_in_[iBlock];
230 auto& R_nn = R_nn_;
231
232 for (Eigen::Index iCol = 0; iCol < n_diag; ++iCol) {
233 if (A_diag.col(iCol).isZero()) {
234 continue;
235 }
236
237 // At this point, we have zeroed out all the columns to the
238 // left of iCol. The current matrix looks like this.
239 // iCol
240 // \|/
241 // [ r r r r ]
242 // [ 0 r r r ]
243 // [ 0 0 r r ]
244 // [ 0 0 0 r ]
245 // [ 0 0 0 0 ]
246 // [ ... ]
247 // [ 0 0 a a ]
248 // [ 0 0 a a ]
249 // We will apply a Householder reflection (I - beta*v*v^T) * A
250 // This will zero out the ith column.
251 const auto [beta, mu] = computeHouseholderVec<T>(R_ii(iCol, iCol), A_diag.col(iCol));
252 if (beta == 0) {
253 continue;
254 }
255
256 // computeHouseholderVec guarantees that applying the transform to iCol
257 // results in a vector [mu 0 ... 0].
258 R_ii(iCol, iCol) = mu;
259
260 // Apply the Householder vector to the remaining columns.
261 const auto& v2m = A_diag.col(iCol);
262 for (Eigen::Index jCol_diag = iCol + 1; jCol_diag < n_diag; ++jCol_diag) {
263 applyHouseholderTransformation<T>(beta, v2m, R_ii(iCol, jCol_diag), A_diag.col(jCol_diag));
264 }
265
266 for (Eigen::Index jCol_common = 0; jCol_common < n_common; ++jCol_common) {
267 applyHouseholderTransformation<T>(
268 beta, v2m, R_in(iCol, jCol_common), A_common.col(jCol_common));
269 }
270
271 // Apply the Householder vector to the rhs.
272 applyHouseholderTransformation<T>(beta, v2m, y_i(iCol), b);
273 }
274
275 // Now for the common parameters:
276 for (Eigen::Index iCol = 0; iCol < n_common; ++iCol) {
277 const auto [beta, mu] = computeHouseholderVec<T>(R_nn(iCol, iCol), A_common.col(iCol));
278 if (beta == 0) {
279 continue;
280 }
281
282 R_nn(iCol, iCol) = mu;
283
284 const auto& v2m = A_common.col(iCol);
285 for (Eigen::Index jCol_common = iCol + 1; jCol_common < n_common; ++jCol_common) {
286 applyHouseholderTransformation<T>(
287 beta, v2m, R_nn_(iCol, jCol_common), A_common.col(jCol_common));
288 }
289
290 // Apply the Householder vector to the rhs.
291 applyHouseholderTransformation<T>(beta, v2m, y_n_(iCol), b);
292 }
293}

295template <typename T>
296typename OnlineBlockHouseholderQR<T>::MatrixType OnlineBlockHouseholderQR<T>::R_dense() const {
297 const auto nCol_common = R_nn_.cols();
298
299 Eigen::Index nCol_diag = 0;
300 for (const auto& R_ii : R_ii_) {
301 nCol_diag += R_ii.cols();
302 }
303
304 const Eigen::Index nCol_total = nCol_diag + nCol_common;
305
306 MatrixType result = MatrixType::Zero(nCol_total, nCol_total);
307 auto offset_cur = 0;
308 for (size_t iBlock = 0; iBlock < R_ii_.size(); ++iBlock) {
309 const auto& R_ii = R_ii_[iBlock];
310 const auto& R_in = R_in_[iBlock];
311 MT_CHECK(R_ii.rows() == R_in.rows());
312 MT_CHECK(R_in.cols() == nCol_common);
313 result.block(offset_cur, offset_cur, R_ii.rows(), R_ii.cols()) = R_ii;
314 result.block(offset_cur, nCol_diag, R_in.rows(), nCol_common) = R_in;
315 offset_cur += R_ii.rows();
316 }
317 result.block(offset_cur, offset_cur, nCol_common, nCol_common) = R_nn_;
318 return result;
319}

321template <typename T>
322typename OnlineBlockHouseholderQR<T>::MatrixType OnlineBlockHouseholderQR<T>::y_dense() const {
323 const auto nRow_common = y_n_.rows();
324
325 Eigen::Index nRow_diag = 0;
326 for (const auto& y_i : y_i_) {
327 nRow_diag += y_i.rows();
328 }
329
330 const Eigen::Index nRow_total = nRow_diag + nRow_common;
331
332 auto offset_cur = 0;
333 VectorType result = VectorType::Zero(nRow_total);
334 for (size_t iBlock = 0; iBlock < R_ii_.size(); ++iBlock) {
335 const auto& y_i = y_i_[iBlock];
336 result.segment(offset_cur, y_i.rows()) = y_i;
337 offset_cur += y_i.rows();
338 }
339
340 result.segment(offset_cur, nRow_common) = y_n_;
341 return result;
342}

344template <typename T>
345typename OnlineBlockHouseholderQR<T>::VectorType OnlineBlockHouseholderQR<T>::x_dense() const {
346 const auto nBlocks = R_ii_.size();
347 Eigen::Index nRows_diag = 0;
348 for (const auto& block : R_ii_) {
349 nRows_diag += block.cols();
350 }
351 const Eigen::Index nRows_common = R_nn_.cols();
352 const Eigen::Index nRows_total = nRows_diag + nRows_common;
353
354 VectorType result(nRows_total);
355 result.segment(nRows_diag, nRows_common) =
356 R_nn_.template triangularView<Eigen::Upper>().solve(y_n_);
357
358 const auto& x_n = result.segment(nRows_diag, nRows_common);
359 Eigen::Index offset = 0;
360 for (size_t i = 0; i < nBlocks; ++i) {
361 const auto nRows_cur = R_ii_[i].cols();
362 result.segment(offset, nRows_cur) =
363 R_ii_[i].template triangularView<Eigen::Upper>().solve(y_i_[i] - R_in_[i] * x_n);
364 offset += nRows_cur;
365 }
366
367 return result;
368}

370template <typename T>
371typename OnlineBlockHouseholderQR<T>::VectorType OnlineBlockHouseholderQR<T>::x_n() const {
372 return R_nn_.template triangularView<Eigen::Upper>().solve(y_n_);

377 size_t iBlock) const {

389template <typename T>
390typename OnlineBlockHouseholderQR<T>::VectorType OnlineBlockHouseholderQR<T>::At_times_b_n() const {
391 VectorType result = R_nn_.transpose() * y_n_;
392 for (size_t iBlock = 0; iBlock < R_ii_.size(); ++iBlock) {
393 result.noalias() += R_in_[iBlock].transpose() * y_i_[iBlock];
394 }
395 return result;

398template <typename T>
399typename OnlineBlockHouseholderQR<T>::VectorType OnlineBlockHouseholderQR<T>::At_times_b_dense()
400 const {
401 Eigen::Index nRows_total = y_n_.rows();
402 for (const auto& y_i : y_i_) {
403 nRows_total += y_i.rows();
404 }
405
406 VectorType result(nRows_total);
407 Eigen::Index offset = 0;
408 for (size_t iBlock = 0; iBlock < R_ii_.size(); ++iBlock) {
409 const auto nRows_cur = y_i_[iBlock].rows();
410 result.segment(offset, nRows_cur).noalias() = R_ii_[iBlock].transpose() * y_i_[iBlock];
411 offset += nRows_cur;
412 }
413
414 result.segment(offset, y_n_.rows()).noalias() = R_nn_.transpose() * y_n_;
415 for (size_t iBlock = 0; iBlock < R_ii_.size(); ++iBlock) {
416 result.segment(offset, y_n_.rows()).noalias() += R_in_[iBlock].transpose() * y_i_[iBlock];
417 }
418
419 return result;
420}

441template <typename T>
442OnlineBandedHouseholderQR<T>::OnlineBandedHouseholderQR(
443 const Eigen::Index n_band,
444 Eigen::Index n_common,
445 const Eigen::Index bandwidth,
446 T lambda)
447 : lambda_(lambda),
448 R_common_{Eigen::MatrixX<T>::Zero(n_band + n_common, n_common)},
449 R_band_{Eigen::MatrixX<T>::Zero(bandwidth, n_band)},
450 y_{Eigen::VectorX<T>::Zero(n_band + n_common)} {
451 // Bandwidth is only allowed to be zero if there are no parameters in the banded part,
452 // in which case it falls back to behaving the same as a regular Householder QR (for
453 // the common part).
454 MT_CHECK(n_band == 0 || bandwidth > 0);
455
456 initializeDiagonal();
457}

461 const auto n_common = R_common_.cols();
462
463 if (n_common != 0) {

465 }
466
467 if (R_band_.rows() > 0 && R_band_.cols() > 0) {
468 R_band_.template bottomRows<1>().setConstant(lambda_);
469 }
470}

491template <typename T>
492void OnlineBandedHouseholderQR<T>::zeroBandedPart(
493 const Eigen::Index iCol_offset,
494 ColumnIndexedMatrix<MatrixType> A_band,
495 ColumnIndexedMatrix<MatrixType> A_common,
496 Eigen::Ref<VectorType> b) {
497 const auto n_common = R_common_.cols();
498
499 MT_CHECK(A_band.cols() <= R_band_.rows());
500 MT_CHECK(iCol_offset + A_band.cols() <= R_band_.cols());
501
502 // TODO what to assert about A_band?
503 MT_CHECK(A_common.cols() == n_common);
504
505 for (Eigen::Index iCol_local = 0; iCol_local < A_band.cols(); ++iCol_local) {
506 const Eigen::Index iCol_global = iCol_local + iCol_offset;
507
508 // At this point, we have zeroed out all the columns to the
509 // left of iCol. The current matrix looks like this.
510 // iCol
511 // \|/
512 // [ r r r r ]
513 // [ 0 r r r ]
514 // [ 0 0 r r ]
515 // [ 0 0 0 r ]
516 // [ 0 0 0 0 ]
517 // [ ... ]
518 // [ 0 0 a a ]
519 // [ 0 0 a a ]
520 // We will apply a Householder reflection (I - beta*v*v^T) * A
521 // This will zero out the ith column.
522 const auto [beta, mu] =
523 computeHouseholderVec<T>(R_band_entry(iCol_global, iCol_global), A_band.col(iCol_local));
524 if (beta == 0) {
525 continue;
526 }
527
528 // computeHouseholderVec guarantees that applying the transform to iCol
529 // results in a vector [mu 0 ... 0].
530 R_band_entry(iCol_global, iCol_global) = mu;
531
532 // Apply the Householder vector to the remaining columns.
533 const auto& v2m = A_band.col(iCol_local);
534 for (Eigen::Index jCol_local = iCol_local + 1; jCol_local < A_band.cols(); ++jCol_local) {
535 const auto jCol_global = jCol_local + iCol_offset;
536
537 applyHouseholderTransformation<T>(
538 beta, v2m, R_band_entry(iCol_global, jCol_global), A_band.col(jCol_local));
539 }
540
541 for (Eigen::Index jCol_common = 0; jCol_common < n_common; ++jCol_common) {
542 applyHouseholderTransformation<T>(
543 beta, v2m, R_common_(iCol_global, jCol_common), A_common.col(jCol_common));
544 }
545
546 applyHouseholderTransformation<T>(beta, v2m, y_(iCol_global), b);
547 }
548}

555 const auto n_band = R_band_.cols();

580template <typename T>
581typename OnlineBandedHouseholderQR<T>::MatrixType OnlineBandedHouseholderQR<T>::R_dense() const {
582 const auto n_common = R_common_.cols();
583 const auto n_band = R_band_.cols();
584 const auto bandwidth = R_band_.rows();
585
586 const auto n_total = n_band + n_common;
587
588 MatrixType result = MatrixType::Zero(n_total, n_total);
589
590 for (Eigen::Index iCol_band = 0; iCol_band < n_band; ++iCol_band) {
591 for (Eigen::Index jRow = iCol_band; (iCol_band - jRow) < bandwidth && jRow >= 0; --jRow) {
592 result(jRow, iCol_band) = R_band_entry(jRow, iCol_band);
593 }
594 }
595
596 for (Eigen::Index jRow = 0; jRow < n_total; ++jRow) {
597 for (Eigen::Index iCol_common = 0; iCol_common < n_common; ++iCol_common) {
598 result(jRow, n_band + iCol_common) = R_common_(jRow, iCol_common);
599 }
600 }
601
602 return result;
603}

605template <typename T>
606typename OnlineBandedHouseholderQR<T>::VectorType OnlineBandedHouseholderQR<T>::x_dense() const {
607 VectorType result = VectorType::Zero(R_band_.cols() + R_common_.cols());
608
609 const auto n_common = R_common_.cols();
610 const auto n_band = R_band_.cols();
611 const auto bandwidth = R_band_.rows();
612
613 if (n_common > 0) {
614 result.tail(n_common) =
615 R_common_.bottomRows(n_common).template triangularView<Eigen::Upper>().solve(
616 y_.tail(n_common));
617 }
618
619 if (n_band > 0) {
620 for (Eigen::Index iRow = n_band - 1; iRow >= 0; --iRow) {
621 T dotProd = n_common > 0 ? R_common_.row(iRow).dot(result.tail(n_common)) : T(0);
622
623 const Eigen::Index bandwidth_cur = std::min(bandwidth, n_band - iRow);
624 for (Eigen::Index jColOffset = 1; jColOffset < bandwidth_cur; ++jColOffset) {
625 dotProd += R_band_entry(iRow, iRow + jColOffset) * result(iRow + jColOffset);
626 }
627 result(iRow) = (y_(iRow) - dotProd) / R_band_entry(iRow, iRow);
628 }
629 }
630
631 return result;
632}

634template <typename T>
635typename OnlineBandedHouseholderQR<T>::VectorType OnlineBandedHouseholderQR<T>::At_times_b() const {
636 // As noted above, A^T*b = R^T*y
637 VectorType result = VectorType::Zero(R_band_.cols() + R_common_.cols());
638
639 const auto n_common = R_common_.cols();
640 const auto n_band = R_band_.cols();
641 const auto bandwidth = R_band_.rows();
642
643 for (Eigen::Index iRow = 0; iRow < n_band; ++iRow) {
644 const Eigen::Index bandwidth_cur = std::min(bandwidth, n_band - iRow);
645 for (Eigen::Index jColOffset = 0; jColOffset < bandwidth_cur; ++jColOffset) {
646 const Eigen::Index jCol = iRow + jColOffset;
647 result(jCol) += R_band_entry(iRow, jCol) * y_(iRow);
648 }
649 }
650
651 result.tail(n_common).noalias() = R_common_.transpose() * y_;
652
653 return result;
654}
