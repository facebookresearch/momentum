48 const auto nFullParams = this->parameters_.size();
49 this->enabledParameters_.clear();
50 this->enabledParameters_.reserve(nFullParams);
51 for (int iFullParam = 0; iFullParam < nFullParams; ++iFullParam) {

69template <typename T>
70void SubsetGaussNewtonSolverT<T>::doIteration() {
71 MT_PROFILE_FUNCTION();
72
73 // get the jacobian and residual
74 size_t jacobianRows = 0;
75 this->error_ = this->solverFunction_->getJacobian(
76 this->parameters_, this->jacobian_, this->residual_, jacobianRows);
77 const double error_orig = this->error_;
78
79 const int nSubsetParams = gsl::narrow_cast<int>(this->enabledParameters_.size());
80
81 // Fill in J for this subset
82
83 // Move the solved columns of the Jacobian into place.
84 // Doing this in-place in the jacobian minimizes the total work.
85 // For the common case where all the "dropped" parameters at the end,
86 // this will effectively be a no-op, since enabledParameters_[iSubsetParam] = iFullParam
87 // for all parameters prior to the dropped ones.
88 for (int iSubsetParam = 0; iSubsetParam < nSubsetParams; ++iSubsetParam) {
89 // Only move columns that need it:
90 if (this->enabledParameters_[iSubsetParam] > iSubsetParam) {
91 this->jacobian_.col(iSubsetParam) =
92 this->jacobian_.col(this->enabledParameters_[iSubsetParam]);
93 }
94 }
95
96 const auto JBlock = this->jacobian_.topLeftCorner(jacobianRows, nSubsetParams);
97
98 // Compute grad = (J^T * r) and hess ~= (J^T * J) for the actual solved subset:
99 // The gradient is just Jt * r but only keeping `jacobianRows` which can be smaller than the
100 // number of rows of the `jacobian` matrix.
101 this->subsetGradient_.noalias() = this->residual_.head(jacobianRows).transpose() * JBlock;
102
103 // The Hessian takes a regularizer on top of Jt * J
104 this->subsetHessian_.template triangularView<Eigen::Lower>() = JBlock.transpose() * JBlock;
105 this->subsetHessian_.diagonal().array() += this->regularization_;
106
107 // LLT only reads the lower triangular part
108 subsetDelta_.noalias() = llt_.compute(this->subsetHessian_).solve(subsetGradient_);
109
110 // We don't need to set this->delta_ to zero because it is already set to zero when
111 // enabledParameters_ is updated.
112 for (int iSubsetParam = 0; iSubsetParam < nSubsetParams; ++iSubsetParam) {
113 const int iFullParam = this->enabledParameters_[iSubsetParam];
114 delta_(iFullParam) = subsetDelta_(iSubsetParam);
115 }
116 // update the parameters
117 if (doLineSearch_) {
118 const double innerProd = -this->subsetGradient_.dot(subsetDelta_);
119
120 // Line search:
121 const float c_1 = 1e-4F;
122 const float tau = 0.5F;
123 float alpha = 1.0F;
124
125 const Eigen::VectorX<T> parameters_orig = this->parameters_;
126 for (size_t kStep = 0; kStep < 10 && std::fpclassify(alpha) == FP_NORMAL; ++kStep) {
127 // update the this->parameters_
128 this->parameters_ = parameters_orig;
129 this->solverFunction_->updateParameters(this->parameters_, alpha * delta_);
130
131 const double error_new = this->solverFunction_->getError(this->parameters_);
132
133 if ((error_orig - error_new) >= c_1 * alpha * -innerProd) {
134 break;
135 }
136
137 // Reduce step size:
138 alpha = alpha * tau;
139 }
140 } else {
141 this->solverFunction_->updateParameters(this->parameters_, delta_);
142 }
143}
