20size_t markerCount(const std::vector<std::vector<Marker>>& frames) {
21 size_t maxCountMarkers = 0;
22 for (const auto& fr : frames) {
23 const size_t markerCount =
24 std::count_if(fr.begin(), fr.end(), [](const Marker& marker) { return !marker.occluded; });
25 maxCountMarkers = std::max(markerCount, maxCountMarkers);
26 }
27 return maxCountMarkers;
28}

31std::vector<MarkerSequence> loadMarkers(const std::string& filename, UpVector up) {
32 const std::string ext = filesystem::path(filename).extension().string();
33 try {
34 if (ext == ".c3d") {
35 return loadC3d(filename, up);
36 } else if (ext == ".trc") {
37 return {loadTrc(filename, up)};
38 } else if (ext == ".glb") {
39 return {loadMarkerSequence(filename)};
40 } else {
41 MT_LOGE("{} Unknown marker file type {}", __func__, filename);
42 return {};
43 }
44 } catch (...) {
45 MT_LOGE("{}: Unknown error in {}", __func__, filename);
46 return {};
47 }
48}

50std::optional<MarkerSequence> loadMarkersForMainSubject(const std::string& filename, UpVector up) {
51 const std::vector<MarkerSequence> markerSequences = loadMarkers(filename, up);
52 const int subjectID = findMainSubjectIndex(markerSequences);

54 return {};

60int findMainSubjectIndex(gsl::span<const MarkerSequence> markerSequences) {
61 if (markerSequences.empty()) {
62 return -1;
63 }
64
65 // We define the main subject as one with the most markers. Ideally, the main subject is the one
66 // with a proper name, because unlabeled markers are usually the largest number, except when there
67 // is only one actor.
68 // XXX Note that the logic could still break, eg. when a person is unnamed but there are named
69 // objects.
70 //
71 // We will go through the sequence to find the max number of visible markers. This is quite slow
72 // but more robust.
73 const size_t numActors = markerSequences.size();
74 std::vector<int> maxMarkers(numActors, 0);
75 // count markers for each actor
76 for (size_t iActor = 0; iActor < numActors; ++iActor) {
77 maxMarkers.at(iActor) = markerCount(markerSequences[iActor].frames);
78 }
79
80 // Special case when there's only one actor.
81 if (numActors == 1) {
82 if (maxMarkers.at(0) > 0) {
83 return 0;
84 } else {
85 return -1;
86 }
87 }
88
89 // find the max marker count with a name
90 size_t maxCount = 0;
91 size_t actorID = 0; // default to be the first actor
92 for (size_t iActor = 0; iActor < numActors; ++iActor) {
93 const size_t count = maxMarkers.at(iActor);
94 auto actorName = markerSequences[iActor].name;
95 std::transform(actorName.begin(), actorName.end(), actorName.begin(), [](unsigned char c) {
96 return std::tolower(c);
97 });
98 if (count > maxCount && !actorName.empty() && actorName != "unlabeled") {
99 maxCount = count;
100 actorID = iActor;
101 }
102 }
103
104 if (maxCount > 0) {
105 return actorID;
106 } else {
107 return -1;
108 }
109}
