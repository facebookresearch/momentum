58template <typename T>
59double VertexVertexDistanceErrorFunctionT<T>::getError(
60 const ModelParametersT<T>& modelParameters,
61 const SkeletonStateT<T>& state) {
62 MT_PROFILE_FUNCTION();
63
64 updateMeshes(modelParameters, state);
65
66 double error = 0.0;
67
68 for (const auto& constraint : constraints_) {
69 const auto& pos1 = posedMesh_->vertices[constraint.vertexIndex1];
70 const auto& pos2 = posedMesh_->vertices[constraint.vertexIndex2];
71
72 const T actualDistance = (pos1 - pos2).norm();
73 const T distanceDiff = actualDistance - constraint.targetDistance;
74
75 error += constraint.weight * distanceDiff * distanceDiff;
76 }
77
78 return error * this->weight_;
79}

81template <typename T>
82void VertexVertexDistanceErrorFunctionT<T>::updateMeshes(
83 const ModelParametersT<T>& modelParameters,
84 const SkeletonStateT<T>& state) {
85 MT_PROFILE_FUNCTION();
86
87 // Update rest mesh with blend shapes if present
88 bool doUpdateNormals = false;
89 if (this->character_.blendShape) {
90 const BlendWeightsT<T> blendWeights =
91 extractBlendWeights(this->parameterTransform_, modelParameters);
92 this->character_.blendShape->computeShape(blendWeights, this->restMesh_->vertices);
93 doUpdateNormals = true;
94 }
95
96 if (doUpdateNormals) {
97 this->restMesh_->updateNormals();
98 }
99
100 // Apply skinning to get the posed mesh
101 applySSD(
102 cast<T>(character_.inverseBindPose),
103 *this->character_.skinWeights,
104 *this->restMesh_,
105 state,
106 *this->posedMesh_);
107}

127template <typename T>
128double VertexVertexDistanceErrorFunctionT<T>::getJacobian(
129 const ModelParametersT<T>& modelParameters,
130 const SkeletonStateT<T>& state,
131 Eigen::Ref<Eigen::MatrixX<T>> jacobian,
132 Eigen::Ref<Eigen::VectorX<T>> residual,
133 int& usedRows) {
134 MT_PROFILE_FUNCTION();
135
136 MT_CHECK(
137 jacobian.cols() == static_cast<Eigen::Index>(this->parameterTransform_.transform.cols()));
138 MT_CHECK(jacobian.rows() >= static_cast<Eigen::Index>(constraints_.size()));
139 MT_CHECK(residual.rows() >= static_cast<Eigen::Index>(constraints_.size()));
140
141 updateMeshes(modelParameters, state);
142
143 double error = 0.0;
144
145 for (size_t i = 0; i < constraints_.size(); i++) {
146 T residualValue;
147 error += calculateJacobian(
148 modelParameters,
149 state,
150 constraints_[i],
151 jacobian.block(i, 0, 1, modelParameters.size()),
152 residualValue);
153 residual(i) = residualValue;
154 }
155
156 usedRows = static_cast<int>(constraints_.size());
157 return error;
158}

165template <typename T>
166double VertexVertexDistanceErrorFunctionT<T>::calculateJacobian(
167 const ModelParametersT<T>& modelParameters,
168 const SkeletonStateT<T>& state,
169 const VertexVertexDistanceConstraintT<T>& constraint,
170 Eigen::Ref<Eigen::MatrixX<T>> jacobian,
171 T& residual) const {
172 MT_PROFILE_FUNCTION();
173
174 const auto& pos1 = posedMesh_->vertices[constraint.vertexIndex1];
175 const auto& pos2 = posedMesh_->vertices[constraint.vertexIndex2];
176
177 const Eigen::Vector3<T> diff = pos1 - pos2;
178 const T actualDistance = diff.norm();
179
180 // Handle degenerate case where vertices are at the same position
181 if (actualDistance == T(0)) {
182 residual = T(0);
183 return T(0); // No meaningful jacobian when distance is zero
184 }
185
186 const T distanceDiff = actualDistance - constraint.targetDistance;
187 const Eigen::Vector3<T> distanceGradient = diff / actualDistance; // normalized difference vector
188
189 // Weight for the jacobian: sqrt(weight * this->weight_)
190 const T wgt = std::sqrt(constraint.weight * this->weight_);
191
192 // Set residual: wgt * distanceDiff
193 residual = wgt * distanceDiff;
194
195 // Calculate jacobian contribution from vertex1 (positive contribution)
196 calculateVertexJacobian(
197 modelParameters, state, constraint.vertexIndex1, wgt * distanceGradient, jacobian);
198
199 // Calculate jacobian contribution from vertex2 (negative contribution)
200 calculateVertexJacobian(
201 modelParameters, state, constraint.vertexIndex2, -wgt * distanceGradient, jacobian);
202
203 return wgt * wgt * distanceDiff * distanceDiff;
204}

206template <typename T>
207void VertexVertexDistanceErrorFunctionT<T>::calculateVertexJacobian(
208 const ModelParametersT<T>& /*modelParameters*/,
209 const SkeletonStateT<T>& state,
210 int vertexIndex,
211 const Eigen::Vector3<T>& jacobianDirection,
212 Eigen::Ref<Eigen::MatrixX<T>> jacobian) const {
213 MT_PROFILE_FUNCTION();
214
215 SkinningWeightIteratorT<T> skinningIter(this->character_, *this->restMesh_, state, vertexIndex);
216
217 // Handle derivatives wrt joint parameters
218 while (!skinningIter.finished()) {
219 size_t jointIndex = 0;
220 T boneWeight;
221 Eigen::Vector3<T> pos;
222 std::tie(jointIndex, boneWeight, pos) = skinningIter.next();
223
224 // Check for valid index
225 MT_CHECK(jointIndex < this->skeleton_.joints.size());
226
227 const auto& jointState = state.jointState[jointIndex];
228 const size_t paramIndex = jointIndex * kParametersPerJoint;
229 const Eigen::Vector3<T> posd = pos - jointState.translation();
230
231 // Calculate derivatives based on active joints
232 for (size_t d = 0; d < 3; d++) {
233 if (this->activeJointParams_[paramIndex + d]) {
234 // Jacobian wrt translation:
235 jacobian_jointParams_to_modelParams<T>(
236 boneWeight * jacobianDirection.dot(jointState.getTranslationDerivative(d)),
237 paramIndex + d,
238 this->parameterTransform_,
239 jacobian);
240 }
241 if (this->activeJointParams_[paramIndex + 3 + d]) {
242 // Jacobian wrt rotation:
243 jacobian_jointParams_to_modelParams<T>(
244 boneWeight * jacobianDirection.dot(jointState.getRotationDerivative(d, posd)),
245 paramIndex + 3 + d,
246 this->parameterTransform_,
247 jacobian);
248 }
249 }
250 if (this->activeJointParams_[paramIndex + 6]) {
251 // Jacobian wrt scale:
252 jacobian_jointParams_to_modelParams<T>(
253 boneWeight * jacobianDirection.dot(jointState.getScaleDerivative(posd)),
254 paramIndex + 6,
255 this->parameterTransform_,
256 jacobian);
257 }
258 }
259
260 // Handle derivatives wrt blend shape parameters
261 if (this->character_.blendShape) {
262 for (Eigen::Index iBlendShape = 0;
263 iBlendShape < this->parameterTransform_.blendShapeParameters.size();
264 ++iBlendShape) {
265 const auto paramIdx = this->parameterTransform_.blendShapeParameters[iBlendShape];
266 if (paramIdx < 0) {
267 continue;
268 }
269
270 const Eigen::Vector3<T> d_restPos =
271 this->character_.blendShape->getShapeVectors()
272 .template block<3, 1>(3 * vertexIndex, iBlendShape, 3, 1)
273 .template cast<T>();
274 Eigen::Vector3<T> d_worldPos = Eigen::Vector3<T>::Zero();
275 calculateDWorldPos(state, vertexIndex, d_restPos, d_worldPos);
276
277 jacobian(0, paramIdx) += jacobianDirection.dot(d_worldPos);
278 }
279 }
280
281 // Handle derivatives wrt face expression blend shape parameters
282 if (this->character_.faceExpressionBlendShape) {
283 for (Eigen::Index iBlendShape = 0;
284 iBlendShape < this->parameterTransform_.faceExpressionParameters.size();
285 ++iBlendShape) {
286 const auto paramIdx = this->parameterTransform_.faceExpressionParameters[iBlendShape];
287 if (paramIdx < 0) {
288 continue;
289 }
290
291 const Eigen::Vector3<T> d_restPos =
292 this->character_.faceExpressionBlendShape->getShapeVectors()
293 .template block<3, 1>(3 * vertexIndex, iBlendShape, 3, 1)
294 .template cast<T>();
295 Eigen::Vector3<T> d_worldPos = Eigen::Vector3<T>::Zero();
296 calculateDWorldPos(state, vertexIndex, d_restPos, d_worldPos);
297
298 jacobian(0, paramIdx) += jacobianDirection.dot(d_worldPos);
299 }
300 }
301}

303template <typename T>
304double VertexVertexDistanceErrorFunctionT<T>::calculateGradient(
305 const ModelParametersT<T>& modelParameters,
306 const SkeletonStateT<T>& state,
307 const VertexVertexDistanceConstraintT<T>& constraint,
308 Eigen::Ref<Eigen::VectorX<T>> gradient) const {
309 MT_PROFILE_FUNCTION();
310
311 const auto& pos1 = posedMesh_->vertices[constraint.vertexIndex1];
312 const auto& pos2 = posedMesh_->vertices[constraint.vertexIndex2];
313
314 const Eigen::Vector3<T> diff = pos1 - pos2;
315 const T actualDistance = diff.norm();
316
317 // Handle degenerate case where vertices are at the same position
318 if (actualDistance == T(0)) {
319 return T(0); // No meaningful gradient when distance is zero
320 }
321
322 const T distanceDiff = actualDistance - constraint.targetDistance;
323 const Eigen::Vector3<T> distanceGradient = diff / actualDistance; // normalized difference vector
324
325 // Weight for the gradient: 2 * weight * distanceDiff * this->weight_
326 const T wgt = T(2) * constraint.weight * distanceDiff * this->weight_;
327
328 // Calculate gradient contribution from vertex1 (positive contribution)
329 calculateVertexGradient(
330 modelParameters, state, constraint.vertexIndex1, wgt * distanceGradient, gradient);
331
332 // Calculate gradient contribution from vertex2 (negative contribution)
333 calculateVertexGradient(
334 modelParameters, state, constraint.vertexIndex2, -wgt * distanceGradient, gradient);
335
336 return constraint.weight * distanceDiff * distanceDiff;
337}

436template <typename T>
437void VertexVertexDistanceErrorFunctionT<T>::calculateDWorldPos(
438 const SkeletonStateT<T>& state,
439 int vertexIndex,
440 const Eigen::Vector3<T>& d_restPos,
441 Eigen::Vector3<T>& d_worldPos) const {
442 const auto& skinWeights = *character_.skinWeights;
443
444 for (uint32_t i = 0; i < kMaxSkinJoints; ++i) {
445 const auto w = skinWeights.weight(vertexIndex, i);
446 const auto parentBone = skinWeights.index(vertexIndex, i);
447 if (w > 0) {
448 d_worldPos += w *
449 (state.jointState[parentBone].transform.toLinear() *
450 (character_.inverseBindPose[parentBone].linear().template cast<T>() * d_restPos));
451 }
452 }
