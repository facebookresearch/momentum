40template <typename T>
41void CollisionErrorFunctionStatelessT<T>::updateCollisionPairs() {
42 // clear collisions
43 collisionPairs.clear();
44
45 // create zero state
46 const SkeletonStateT<T> state(
47 this->parameterTransform_.zero().template cast<T>(), this->skeleton_);
48 collisionState.update(state, collisionGeometry);
49
50 // go over all possible pairs
51 for (size_t i = 0; i < collisionGeometry.size(); ++i) {
52 for (size_t j = i + 1; j < collisionGeometry.size(); ++j) {
53 // check if pair naturally intersects
54 T distance;
55 Vector2<T> cp;
56 T overlap;
57 if (overlaps(
58 collisionState.origin[i],
59 collisionState.direction[i],
60 collisionState.radius[i],
61 collisionState.delta[i],
62 collisionState.origin[j],
63 collisionState.direction[j],
64 collisionState.radius[j],
65 collisionState.delta[j],
66 distance,
67 cp,
68 overlap)) {
69 continue;
70 }
71
72 // no, it doesn't. check
73
74 // walk down the this->skeleton_ hierarchy for the collision pair until we find a common
75 // ancestor
76 size_t p0 = collisionGeometry[i].parent;
77 size_t p1 = collisionGeometry[j].parent;
78 size_t count = 0;
79
80 // Find the common ancestor. We can always just walk up the higher joint index because the
81 // parent always has to have a lower index than the child joint.
82 while (p0 != p1) {
83 if (p0 > p1) {
84 p0 = this->skeleton_.joints[p0].parent;
85 } else {
86 p1 = this->skeleton_.joints[p1].parent;
87 }
88 count++;
89 }
90
91 const Vector3i collisionPair(
92 gsl::narrow_cast<int>(i), gsl::narrow_cast<int>(j), gsl::narrow_cast<int>(p0));
93
94 // Store collision pairs excluding those with the same parent or adjacent joints
95 if (count > 1) {
96 collisionPairs.push_back(collisionPair);
97 }
98 }
99 }
100
101 collisionActive.setConstant(collisionPairs.size(), false);
102}

104template <typename T>
105std::vector<Vector2i> CollisionErrorFunctionStatelessT<T>::getCollisionPairs() const {
106 std::vector<Vector2i> result;
107
108 for (size_t i = 0; i < collisionPairs.size(); i++) {
109 if (collisionActive[i]) {
110 result.emplace_back(collisionPairs[i].template head<2>());
111 }
112 }
113

117template <typename T>
118double CollisionErrorFunctionStatelessT<T>::getError(
119 const ModelParametersT<T>& /*unused*/,
120 const SkeletonStateT<T>& state) {
121 if (state.jointState.empty()) {
122 return 0.0;
123 }
124
125 // check all is valid
126 MT_CHECK(
127 state.jointParameters.size() ==
128 gsl::narrow<Eigen::Index>(this->skeleton_.joints.size() * kParametersPerJoint));
129
130 // update collision state
131 collisionState.update(state, collisionGeometry);
132 collisionActive.setConstant(false);
133
134 // loop over all pairs of collision objects and check
135 double error = 0.0;
136
137 for (size_t i = 0; i < collisionPairs.size(); i++) {
138 const auto& pr = collisionPairs[i];
139 T distance;
140 Vector2<T> cp;
141 T overlap;
142 if (!overlaps(
143 collisionState.origin[pr[0]],
144 collisionState.direction[pr[0]],
145 collisionState.radius[pr[0]],
146 collisionState.delta[pr[0]],
147 collisionState.origin[pr[1]],
148 collisionState.direction[pr[1]],
149 collisionState.radius[pr[1]],
150 collisionState.delta[pr[1]],
151 distance,
152 cp,
153 overlap)) {
154 continue;
155 }
156
157 collisionActive[i] = true;
158 error += sqr(overlap) * kCollisionWeight * this->weight_;
159 }
160
161 // return error
162 return error;

165template <typename T>
166double CollisionErrorFunctionStatelessT<T>::getGradient(
167 const ModelParametersT<T>& /*unused*/,
168 const SkeletonStateT<T>& state,
169 Ref<VectorX<T>> gradient) {
170 if (state.jointState.empty()) {
171 return 0.0;
172 }
173
174 // check all is valid
175 MT_CHECK(
176 state.jointParameters.size() ==
177 gsl::narrow<Eigen::Index>(this->skeleton_.joints.size() * kParametersPerJoint));
178
179 // update collision state
180 collisionState.update(state, collisionGeometry);
181 collisionActive.setConstant(false);
182
183 double error = 0;
184 for (size_t i = 0; i < collisionPairs.size(); ++i) {
185 const auto& pr = collisionPairs[i];
186 T distance;
187 Vector2<T> cp;
188 T overlap;
189 if (!overlaps(
190 collisionState.origin[pr[0]],
191 collisionState.direction[pr[0]],
192 collisionState.radius[pr[0]],
193 collisionState.delta[pr[0]],
194 collisionState.origin[pr[1]],
195 collisionState.direction[pr[1]],
196 collisionState.radius[pr[1]],
197 collisionState.delta[pr[1]],
198 distance,
199 cp,
200 overlap)) {
201 continue;
202 }
203
204 collisionActive[i] = true;
205 error += sqr(overlap) * kCollisionWeight * this->weight_;
206
207 // calculate collision resolve direction. this is what we need to push joint parent i in.
208 // the direction for joint parent j is the inverse
209 const Vector3<T> position_i =
210 (collisionState.origin[pr[0]] + collisionState.direction[pr[0]] * cp[0]);
211 const Vector3<T> position_j =
212 (collisionState.origin[pr[1]] + collisionState.direction[pr[1]] * cp[1]);
213 const Vector3<T> direction = position_i - position_j;
214 const T overlapFraction = overlap / distance;
215
216 // calculate weight
217 const T wgt = -T(2) * kCollisionWeight * this->weight_ * overlapFraction;
218
219 // -----------------------------------
220 // process first joint
221 // -----------------------------------
222 size_t jointIndex = collisionGeometry[pr[0]].parent;
223 while (jointIndex != kInvalidIndex && jointIndex != gsl::narrow_cast<size_t>(pr[2])) {
224 const auto& jointState = state.jointState[jointIndex];
225 const size_t paramIndex = jointIndex * kParametersPerJoint;
226 const Vector3<T> posd = position_i - jointState.translation();
227
228 // calculate derivatives based on active joints
229 for (size_t d = 0; d < 3; d++) {
230 if (this->activeJointParams_[paramIndex + d]) {
231 // calculate joint gradient
232 const T val = direction.dot(jointState.getTranslationDerivative(d)) * wgt;
233 // explicitly multiply with the parameter transform to generate parameter space
234 // gradients
235 for (auto index = this->parameterTransform_.transform.outerIndexPtr()[paramIndex + d];
236 index < this->parameterTransform_.transform.outerIndexPtr()[paramIndex + d + 1];
237 ++index) {
238 gradient[this->parameterTransform_.transform.innerIndexPtr()[index]] +=
239 val * this->parameterTransform_.transform.valuePtr()[index];
240 }
241 }
242 if (this->activeJointParams_[paramIndex + 3 + d]) {
243 // calculate joint gradient
244 const T val = direction.dot(jointState.getRotationDerivative(d, posd)) * wgt;
245 // explicitly multiply with the parameter transform to generate parameter space
246 // gradients
247 for (auto index = this->parameterTransform_.transform.outerIndexPtr()[paramIndex + 3 + d];
248 index < this->parameterTransform_.transform.outerIndexPtr()[paramIndex + d + 3 + 1];
249 ++index) {
250 gradient[this->parameterTransform_.transform.innerIndexPtr()[index]] +=
251 val * this->parameterTransform_.transform.valuePtr()[index];
252 }
253 }
254 }
255 if (this->activeJointParams_[paramIndex + 6]) {
256 // calculate joint gradient
257 const T val = direction.dot(jointState.getScaleDerivative(posd)) * wgt;
258 // explicitly multiply with the parameter transform to generate parameter space
259 // gradients
260 for (auto index = this->parameterTransform_.transform.outerIndexPtr()[paramIndex + 6];
261 index < this->parameterTransform_.transform.outerIndexPtr()[paramIndex + 6 + 1];
262 ++index) {
263 gradient[this->parameterTransform_.transform.innerIndexPtr()[index]] +=
264 val * this->parameterTransform_.transform.valuePtr()[index];
265 }
266 }
267
268 // go to the next joint
269 jointIndex = this->skeleton_.joints[jointIndex].parent;
270 }
271
272 // -----------------------------------
273 // process second joint
274 // -----------------------------------
275 jointIndex = collisionGeometry[pr[1]].parent;
276 while (jointIndex != kInvalidIndex && jointIndex != gsl::narrow_cast<size_t>(pr[2])) {
277 const auto& jointState = state.jointState[jointIndex];
278 const size_t paramIndex = jointIndex * kParametersPerJoint;
279 const Vector3<T> posd = position_j - jointState.translation();
280
281 // calculate derivatives based on active joints
282 for (size_t d = 0; d < 3; d++) {
283 if (this->activeJointParams_[paramIndex + d]) {
284 // calculate joint gradient
285 const T val = direction.dot(jointState.getTranslationDerivative(d)) * -wgt;
286 // explicitly multiply with the parameter transform to generate parameter space
287 // gradients
288 for (auto index = this->parameterTransform_.transform.outerIndexPtr()[paramIndex + d];
289 index < this->parameterTransform_.transform.outerIndexPtr()[paramIndex + d + 1];
290 ++index) {
291 gradient[this->parameterTransform_.transform.innerIndexPtr()[index]] +=
292 val * this->parameterTransform_.transform.valuePtr()[index];
293 }
294 }
295 if (this->activeJointParams_[paramIndex + 3 + d]) {
296 // calculate joint gradient
297 const T val = direction.dot(jointState.getRotationDerivative(d, posd)) * -wgt;
298 // explicitly multiply with the parameter transform to generate parameter space
299 // gradients
300 const auto maxIndex =
301 this->parameterTransform_.transform.outerIndexPtr()[paramIndex + d + 3 + 1];
302 for (auto index = this->parameterTransform_.transform.outerIndexPtr()[paramIndex + 3 + d];
303 index < maxIndex;
304 ++index) {
305 gradient[this->parameterTransform_.transform.innerIndexPtr()[index]] +=
306 val * this->parameterTransform_.transform.valuePtr()[index];
307 }
308 }
309 }
310 if (this->activeJointParams_[paramIndex + 6]) {
311 // calculate joint gradient
312 const T val = direction.dot(jointState.getScaleDerivative(posd)) * -wgt;
313 // explicitly multiply with the parameter transform to generate parameter space
314 // gradients
315 const auto maxIndex =
316 this->parameterTransform_.transform.outerIndexPtr()[paramIndex + 6 + 1];
317 for (auto index = this->parameterTransform_.transform.outerIndexPtr()[paramIndex + 6];
318 index < maxIndex;
319 ++index) {
320 gradient[this->parameterTransform_.transform.innerIndexPtr()[index]] +=
321 val * this->parameterTransform_.transform.valuePtr()[index];
322 }
323 }
324
325 // go to the next joint
326 jointIndex = this->skeleton_.joints[jointIndex].parent;
327 }
328 }
329
330 // return error
331 return error;
332}

334template <typename T>
335double CollisionErrorFunctionStatelessT<T>::getJacobian(
336 const ModelParametersT<T>& /*unused*/,
337 const SkeletonStateT<T>& state,
338 Ref<MatrixX<T>> jacobian,
339 Ref<VectorX<T>> residual,
340 int& usedRows) {
341 MT_PROFILE_FUNCTION();
342 MT_CHECK(
343 jacobian.rows() >= gsl::narrow<Eigen::Index>(collisionPairs.size()),
344 "Jacobian rows mismatch: Actual {}, Expected {}",
345 jacobian.rows(),
346 gsl::narrow<Eigen::Index>(collisionPairs.size()));
347
348 MT_CHECK(
349 residual.rows() >= gsl::narrow<Eigen::Index>(collisionPairs.size()),
350 "Residual rows mismatch: Actual {}, Expected {}",
351 residual.rows(),
352 gsl::narrow<Eigen::Index>(collisionPairs.size()));
353
354 if (state.jointState.empty()) {
355 return 0.0;
356 }
357
358 // check all is valid
359 MT_CHECK(
360 state.jointParameters.size() ==
361 gsl::narrow<Eigen::Index>(this->skeleton_.joints.size() * kParametersPerJoint));
362
363 // update collision state
364 {
365 MT_PROFILE_EVENT("Collision: updateState");
366
367 collisionState.update(state, collisionGeometry);
368 collisionActive.setConstant(false);
369 }
370
371 const T wgt = std::sqrt(kCollisionWeight * this->weight_);
372
373 int pos = 0;
374 double error = 0;
375
376 // loop over all pairs of collision objects and check
377 for (size_t i = 0; i < collisionPairs.size(); ++i) {
378 const auto& pr = collisionPairs[i];
379 T distance;
380 Vector2<T> cp;
381 T overlap;
382 if (!overlaps(
383 collisionState.origin[pr[0]],
384 collisionState.direction[pr[0]],
385 collisionState.radius[pr[0]],
386 collisionState.delta[pr[0]],
387 collisionState.origin[pr[1]],
388 collisionState.direction[pr[1]],
389 collisionState.radius[pr[1]],
390 collisionState.delta[pr[1]],
391 distance,
392 cp,
393 overlap)) {
394 continue;
395 }
396
397 collisionActive[i] = true;
398 error += sqr(overlap) * kCollisionWeight * this->weight_;
399
400 // calculate collision resolve direction. this is what we need to push joint parent i in.
401 // the direction for joint parent j is the inverse
402 const Vector3<T> position_i =
403 (collisionState.origin[pr[0]] + collisionState.direction[pr[0]] * cp[0]);
404 const Vector3<T> position_j =
405 (collisionState.origin[pr[1]] + collisionState.direction[pr[1]] * cp[1]);
406 const Vector3<T> direction = position_i - position_j;
407 const T inverseDistance = T(1) / distance;
408
409 // calculate constant factor
410 const T fac = T(2) * inverseDistance * wgt;
411
412 // get position in jacobian
413 const int row = pos++;
414
415 // -----------------------------------
416 // process first joint
417 // -----------------------------------
418 size_t jointIndex = collisionGeometry[pr[0]].parent;
419 while (jointIndex != kInvalidIndex && jointIndex != gsl::narrow_cast<size_t>(pr[2])) {
420 const auto& jointState = state.jointState[jointIndex];
421 const size_t paramIndex = jointIndex * kParametersPerJoint;
422 const Vector3<T> posd = position_i - jointState.translation();
423
424 // calculate derivatives based on active joints
425 for (size_t d = 0; d < 3; d++) {
426 if (this->activeJointParams_[paramIndex + d]) {
427 // calculate joint gradient
428 const T val = direction.dot(jointState.getTranslationDerivative(d)) * -fac;
429 // explicitly multiply with the parameter transform to generate parameter space
430 // gradients
431 for (auto index = this->parameterTransform_.transform.outerIndexPtr()[paramIndex + d];
432 index < this->parameterTransform_.transform.outerIndexPtr()[paramIndex + d + 1];
433 ++index) {
434 jacobian(row, this->parameterTransform_.transform.innerIndexPtr()[index]) +=
435 val * this->parameterTransform_.transform.valuePtr()[index];
436 }
437 }
438 if (this->activeJointParams_[paramIndex + 3 + d]) {
439 // calculate joint gradient
440 const T val = direction.dot(jointState.getRotationDerivative(d, posd)) * -fac;
441 // explicitly multiply with the parameter transform to generate parameter space
442 // gradients
443 for (auto index = this->parameterTransform_.transform.outerIndexPtr()[paramIndex + 3 + d];
444 index < this->parameterTransform_.transform.outerIndexPtr()[paramIndex + d + 3 + 1];
445 ++index) {
446 jacobian(row, this->parameterTransform_.transform.innerIndexPtr()[index]) +=
447 val * this->parameterTransform_.transform.valuePtr()[index];
448 }
449 }
450 }
451 if (this->activeJointParams_[paramIndex + 6]) {
452 // calculate joint gradient
453 const T val = direction.dot(jointState.getScaleDerivative(posd)) * -fac;
454 // explicitly multiply with the parameter transform to generate parameter space
455 // gradients
456 for (auto index = this->parameterTransform_.transform.outerIndexPtr()[paramIndex + 6];
457 index < this->parameterTransform_.transform.outerIndexPtr()[paramIndex + 6 + 1];
458 ++index) {
459 jacobian(row, this->parameterTransform_.transform.innerIndexPtr()[index]) +=
460 val * this->parameterTransform_.transform.valuePtr()[index];
461 }
462 }
463
464 // go to the next joint
465 jointIndex = this->skeleton_.joints[jointIndex].parent;
466 }
467
468 // -----------------------------------
469 // process second joint
470 // -----------------------------------
471 jointIndex = collisionGeometry[pr[1]].parent;
472 while (jointIndex != kInvalidIndex && jointIndex != gsl::narrow_cast<size_t>(pr[2])) {
473 const auto& jointState = state.jointState[jointIndex];
474 const size_t paramIndex = jointIndex * kParametersPerJoint;
475 const Vector3<T> posd = position_j - jointState.translation();
476
477 // calculate derivatives based on active joints
478 for (size_t d = 0; d < 3; d++) {
479 if (this->activeJointParams_[paramIndex + d]) {
480 // calculate joint gradient
481 const T val = direction.dot(jointState.getTranslationDerivative(d)) * fac;
482 // explicitly multiply with the parameter transform to generate parameter space
483 // gradients
484 for (auto index = this->parameterTransform_.transform.outerIndexPtr()[paramIndex + d];
485 index < this->parameterTransform_.transform.outerIndexPtr()[paramIndex + d + 1];
486 ++index) {
487 jacobian(row, this->parameterTransform_.transform.innerIndexPtr()[index]) +=
488 val * this->parameterTransform_.transform.valuePtr()[index];
489 }
490 }
491 if (this->activeJointParams_[paramIndex + 3 + d]) {
492 // calculate joint gradient
493 const T val = direction.dot(jointState.getRotationDerivative(d, posd)) * fac;
494 // explicitly multiply with the parameter transform to generate parameter space
495 // gradients
496 for (auto index = this->parameterTransform_.transform.outerIndexPtr()[paramIndex + 3 + d];
497 index < this->parameterTransform_.transform.outerIndexPtr()[paramIndex + d + 3 + 1];
498 ++index) {
499 jacobian(row, this->parameterTransform_.transform.innerIndexPtr()[index]) +=
500 val * this->parameterTransform_.transform.valuePtr()[index];
501 }
502 }
503 }
504 if (this->activeJointParams_[paramIndex + 6]) {
505 // calculate joint gradient
506 const T val = direction.dot(jointState.getScaleDerivative(posd)) * fac;
507 // explicitly multiply with the parameter transform to generate parameter space
508 // gradients
509 for (auto index = this->parameterTransform_.transform.outerIndexPtr()[paramIndex + 6];

512 jacobian(row, this->parameterTransform_.transform.innerIndexPtr()[index]) +=

519 }

522 }
