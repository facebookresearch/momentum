42template <typename T>
43double ModelParametersErrorFunctionT<T>::getError(
44 const ModelParametersT<T>& parameters,
45 const SkeletonStateT<T>& /* state */) {
46 // ignore if we don't have any reasonable data
47 if (targetParameters_.size() != parameters.size() || targetWeights_.size() != parameters.size()) {
48 return 0.0;
49 }
50
51 // calculate difference between parameters and desired parameters
52 double error = 0;
53 for (Eigen::Index i = 0; i < parameters.size(); ++i) {
54 if (this->enabledParameters_.test(i)) {
55 const auto pdiff = targetWeights_(i) * (parameters(i) - targetParameters_(i));
56 error += pdiff * pdiff;
57 }
58 }
59
60 // return error
61 return error * this->weight_ * kMotionWeight;
62}

64template <typename T>
65double ModelParametersErrorFunctionT<T>::getGradient(
66 const ModelParametersT<T>& parameters,
67 const SkeletonStateT<T>& /* state */,
68 Ref<Eigen::VectorX<T>> gradient) {
69 // ignore if we don't have any reasonable data
70 if (targetParameters_.size() != parameters.size() || targetWeights_.size() != parameters.size()) {
71 return 0.0;
72 }
73
74 double error = 0;
75 for (Eigen::Index i = 0; i < parameters.size(); ++i) {
76 if (this->enabledParameters_.test(i)) {
77 const auto pdiff = targetWeights_(i) * (parameters(i) - targetParameters_(i));
78 gradient(i) += 2.0f * targetWeights_(i) * pdiff * this->weight_ * kMotionWeight;
79 error += pdiff * pdiff;
80 }
81 }
82
83 // return error
84 return error * this->weight_ * kMotionWeight;

92template <typename T>
93double ModelParametersErrorFunctionT<T>::getJacobian(
94 const ModelParametersT<T>& parameters,
95 const SkeletonStateT<T>& /* state */,
96 Ref<Eigen::MatrixX<T>> jacobian,
97 Ref<Eigen::VectorX<T>> residual,
98 int& usedRows) {
99 MT_PROFILE_FUNCTION();
100
101 usedRows = 0;
102
103 // ignore if we don't have any reasonable data
104 if (targetParameters_.size() != parameters.size() || targetWeights_.size() != parameters.size()) {
105 return 0.0;
106 }
107
108 const float sWeight = std::sqrt(this->weight_ * kMotionWeight);
109
110 Eigen::Index out = 0;
111
112 // calculate difference between parameters and desired parameters
113 double error = 0;
114 for (Eigen::Index i = 0; i < parameters.size(); ++i) {
115 if (this->enabledParameters_.test(i) && targetWeights_(i) > 0) {
116 const auto pdiff = targetWeights_(i) * (parameters(i) - targetParameters_(i));
117 error += pdiff * pdiff;
118 residual(out) = pdiff * sWeight;
119 jacobian(out, i) = sWeight * targetWeights_(i);
120 ++out;
121 }
122 }
123
124 usedRows = gsl::narrow_cast<int>(out);
125
126 // return error
127 return error * this->weight_ * kMotionWeight;
128}
