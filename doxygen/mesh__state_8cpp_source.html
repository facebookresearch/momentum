52template <typename T>
53void MeshStateT<T>::update(
54 const ModelParametersT<T>& parameters,
55 const SkeletonStateT<T>& state,
56 const Character& character) {
57 MT_PROFILE_FUNCTION();
58
59 // Initialize mesh objects on first call if character has mesh data
60 if (!restMesh_ && character.mesh) {
61 neutralMesh_ = std::make_unique<MeshT<T>>(character.mesh->template cast<T>());
62 restMesh_ = std::make_unique<MeshT<T>>(character.mesh->template cast<T>());
63 posedMesh_ = std::make_unique<MeshT<T>>(character.mesh->template cast<T>());
64 }
65
66 if (!restMesh_ || !posedMesh_) {
67 return; // No mesh to update
68 }
69
70 // Update rest mesh with blend shapes and face expressions
71 bool doUpdateNormals = false;
72
73 // Apply blend shapes if available
74 if (character.blendShape) {
75 const BlendWeightsT<T> blendWeights =
76 extractBlendWeights(character.parameterTransform, parameters);
77 character.blendShape->computeShape(blendWeights, restMesh_->vertices);
78 doUpdateNormals = true;
79 }
80
81 // Apply face expression blend shapes if available
82 if (character.faceExpressionBlendShape) {
83 if (!character.blendShape) {
84 // Set restMesh back to neutral, removing potential previous expressions.
85 // Note that if the character comes with (shape) blendShape, the previous if block already
86 // takes care of this step.
87 Eigen::Map<Eigen::VectorX<T>> outputVec(
88 &restMesh_->vertices[0][0], restMesh_->vertices.size() * 3);
89 const Eigen::Map<Eigen::VectorX<T>> baseVec(
90 &neutralMesh_->vertices[0][0], neutralMesh_->vertices.size() * 3);
91 outputVec = baseVec.template cast<T>();
92 }
93 const BlendWeightsT<T> faceExpressionBlendWeights =
94 extractFaceExpressionBlendWeights(character.parameterTransform, parameters);
95 character.faceExpressionBlendShape->applyDeltas(
96 faceExpressionBlendWeights, restMesh_->vertices);
97 doUpdateNormals = true;
98 }
99
100 // Update normals if blend shapes were applied
101 if (doUpdateNormals) {
102 restMesh_->updateNormals();
103 }
104
105 // Apply skinning to get the posed mesh
106 applySSD(
107 cast<T>(character.inverseBindPose), *character.skinWeights, *restMesh_, state, *posedMesh_);
108}
