22template <typename T>
23double ProjectionErrorFunctionT<T>::getError(
24 const ModelParametersT<T>& /*parameters*/,
25 const SkeletonStateT<T>& skeletonState,
26 const MeshStateT<T>& /* meshState */) {
27 double error = 0;
28
29 const auto& jointState = skeletonState.jointState;
30
31 const auto nCons = constraints_.size();
32 for (size_t iCons = 0; iCons < nCons; ++iCons) {
33 const auto& cons = constraints_[iCons];
34 const auto& js = jointState[cons.parent];
35 const Eigen::Vector3<T> p_world_cm = js.transform * cons.offset;
36 const Eigen::Vector3<T> p_projected_cm =
37 constraints_[iCons].projection * p_world_cm.homogeneous();
38
39 // Behind camera:
40 if (p_projected_cm.z() < _nearClip) {
41 continue;
42 }
43
44 const Eigen::Vector2<T> diff = p_projected_cm.hnormalized().template head<2>() - cons.target;
45 error += cons.weight * kProjectionWeight * this->weight_ * diff.squaredNorm();
46 }
47
48 return error;
49}

56template <typename T>
57double ProjectionErrorFunctionT<T>::getGradient(
58 const momentum::ModelParametersT<T>& /*params*/,
59 const momentum::SkeletonStateT<T>& skeletonState,
60 const momentum::MeshStateT<T>& /* meshState */,
61 Eigen::Ref<Eigen::VectorX<T>> gradient) {
62 double error = 0;
63
64 for (size_t iCons = 0; iCons < constraints_.size(); ++iCons) {
65 const auto& cons = constraints_[iCons];
66
67 const auto& jsCons = skeletonState.jointState[cons.parent];
68 const Eigen::Vector3<T> p_world_cm = jsCons.transform * cons.offset;
69 const Eigen::Vector3<T> p_projected_cm = cons.projection * p_world_cm.homogeneous();
70
71 // Behind camera:
72 if (p_projected_cm.z() < _nearClip) {
73 continue;
74 }
75
76 const Eigen::Vector2<T> p_res = p_projected_cm.hnormalized() - cons.target;
77
78 error += cons.weight * kProjectionWeight * this->weight_ * p_res.squaredNorm();
79 const T wgt = 2.0f * cons.weight * kProjectionWeight * this->weight_;
80
81 const T z = p_projected_cm(2);
82 const T z_sqr = sqr(z);
83 const T x_zz = p_projected_cm(0) / z_sqr;
84 const T y_zz = p_projected_cm(1) / z_sqr;
85
86 auto addGradient = [&](const size_t jFullBodyDOF, const Eigen::Vector3<T>& d_p_world_cm) {
87 const Eigen::Vector3<T> d_p_projected =
88 cons.projection.template topLeftCorner<3, 3>() * d_p_world_cm;
89 const T dx = d_p_projected(0);
90 const T dy = d_p_projected(1);
91 const T dz = d_p_projected(2);
92
93 const Eigen::Vector2<T> d_p_res(dx / z - x_zz * dz, dy / z - y_zz * dz);
94 const T gradFull = d_p_res.dot(p_res);
95
96 // multiply by the parameter transform:
97 for (momentum::SparseRowMatrixf::InnerIterator it(
98 this->parameterTransform_.transform, (int)jFullBodyDOF);
99 it;
100 ++it) {
101 assert(static_cast<size_t>(it.row()) == jFullBodyDOF);
102 const auto jReducedDOF = it.col();
103 gradient(jReducedDOF) += it.value() * wgt * gradFull;
104 }
105 };
106
107 // loop over all joints the constraint is attached to and calculate gradient
108 size_t jointIndex = cons.parent;
109 while (jointIndex != kInvalidIndex) {
110 // check for valid index
111 assert(jointIndex < this->skeleton_.joints.size());
112
113 const auto& jointState = skeletonState.jointState[jointIndex];
114 const size_t paramIndex = jointIndex * 7;
115 const Eigen::Vector3<T> posd_cm = p_world_cm - jointState.translation();
116
117 // calculate derivatives based on active joints
118 for (size_t d = 0; d < 3; d++) {
119 if (this->activeJointParams_[paramIndex + d]) {
120 addGradient(paramIndex + d, jointState.getTranslationDerivative(d));
121 }
122 if (this->activeJointParams_[paramIndex + 3 + d]) {
123 addGradient(paramIndex + 3 + d, jointState.getRotationDerivative(d, posd_cm));
124 }
125 }
126 if (this->activeJointParams_[paramIndex + 6]) {
127 addGradient(paramIndex + 6, jointState.getScaleDerivative(posd_cm));
128 }
129
130 // go to the next joint
131 jointIndex = this->skeleton_.joints[jointIndex].parent;
132 }
133 }
134
135 return error;
136}

138template <typename T>
139double ProjectionErrorFunctionT<T>::getJacobian(
140 const ModelParametersT<T>& /*parameters*/,
141 const SkeletonStateT<T>& skeletonState,
142 const MeshStateT<T>& /* meshState */,
143 Eigen::Ref<Eigen::MatrixX<T>> jacobian,
144 Eigen::Ref<Eigen::VectorX<T>> residual,
145 int& usedRows) {
146 double error = 0;
147
148 for (size_t iCons = 0; iCons < constraints_.size(); ++iCons) {
149 const auto& cons = constraints_[iCons];
150
151 const auto& jsCons = skeletonState.jointState[cons.parent];
152 const Eigen::Vector3<T> p_world_cm = jsCons.transform * cons.offset;
153 const Eigen::Vector3<T> p_projected_cm = cons.projection * p_world_cm.homogeneous();
154
155 // Behind camera:
156 if (p_projected_cm.z() < _nearClip) {
157 continue;
158 }
159
160 const Eigen::Vector2<T> p_res = p_projected_cm.hnormalized() - cons.target;
161 error += cons.weight * kProjectionWeight * this->weight_ * p_res.squaredNorm();
162
163 const T z = p_projected_cm(2);
164 const T z_sqr = sqr(z);
165 const T x_zz = p_projected_cm(0) / z_sqr;
166 const T y_zz = p_projected_cm(1) / z_sqr;
167
168 const T wgt = std::sqrt(cons.weight * kProjectionWeight * this->weight_);
169 residual.template segment<2>(2 * iCons) = (wgt * p_res).template cast<T>();
170
171 auto addJacobian = [&](const size_t jFullBodyDOF, const Eigen::Vector3<T>& d_p_world_cm) {
172 const Eigen::Vector3<T> d_p_projected =
173 cons.projection.template topLeftCorner<3, 3>() * d_p_world_cm;
174 const T dx = d_p_projected(0);
175 const T dy = d_p_projected(1);
176 const T dz = d_p_projected(2);
177
178 const Eigen::Vector2<T> d_p_res(dx / z - x_zz * dz, dy / z - y_zz * dz);
179
180 // multiply by the parameter transform:
181 for (momentum::SparseRowMatrixf::InnerIterator it(
182 this->parameterTransform_.transform, (int)jFullBodyDOF);
183 it;
184 ++it) {
185 assert(static_cast<size_t>(it.row()) == jFullBodyDOF);
186 const auto jReducedDOF = it.col();
187 jacobian.template block<2, 1>(2 * iCons, jReducedDOF) += it.value() * wgt * d_p_res;
188 }
189 };
190
191 // loop over all joints the constraint is attached to and calculate gradient
192 size_t jointIndex = cons.parent;
193 while (jointIndex != kInvalidIndex) {
194 // check for valid index
195 assert(jointIndex < this->skeleton_.joints.size());
196
197 const auto& jointState = skeletonState.jointState[jointIndex];
198 const size_t paramIndex = jointIndex * 7;
199 const Eigen::Vector3<T> posd_cm = p_world_cm - jointState.translation();
200
201 // calculate derivatives based on active joints
202 for (size_t d = 0; d < 3; d++) {
203 if (this->activeJointParams_[paramIndex + d]) {
204 addJacobian(paramIndex + d, jointState.getTranslationDerivative(d));
205 }
206 if (this->activeJointParams_[paramIndex + 3 + d]) {
207 addJacobian(paramIndex + 3 + d, jointState.getRotationDerivative(d, posd_cm));
208 }
209 }
210 if (this->activeJointParams_[paramIndex + 6]) {
211 addJacobian(paramIndex + 6, jointState.getScaleDerivative(posd_cm));
212 }
213
214 // go to the next joint
215 jointIndex = this->skeleton_.joints[jointIndex].parent;
216 }
217 }
218
219 usedRows = static_cast<int>(jacobian.rows());
220
221 return error;
222}
